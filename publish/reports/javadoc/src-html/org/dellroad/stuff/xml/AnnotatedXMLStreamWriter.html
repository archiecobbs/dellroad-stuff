006package org.dellroad.stuff.xml;
007
008import javax.xml.stream.XMLStreamException;
009import javax.xml.stream.XMLStreamWriter;
010
011/**
012 * {@link XMLStreamWriter} that adds an extra annotation element to an XML document as it is written.
013 * The annotation element will be added as the first element inside the top-level document element.
014 *
015 * <p>
016 * This class can be used in combination with {@link AnnotatedXMLStreamReader} to transparently annotate XML documents.
017 * </p>
018 *
019 * @see AnnotatedXMLStreamReader
020 */
021public abstract class AnnotatedXMLStreamWriter extends StreamWriterDelegate {
022
023    private final StringBuilder trailingSpace = new StringBuilder();
024
025    // State:
026    //  0 = before document element
027    //  1 = after document element but before annotation element
028    //  2 = after annotation element
029    private byte state;
030    private boolean needEnding;
031
032    public AnnotatedXMLStreamWriter(XMLStreamWriter parent) {
033        super(parent);
034    }
035
036    @Override
037    public void writeStartElement(String localName) throws XMLStreamException {
038        this.checkStartElement();
039        super.writeStartElement(localName);
040    }
041
042    @Override
043    public void writeStartElement(String namespaceURI, String localName) throws XMLStreamException {
044        this.checkStartElement();
045        super.writeStartElement(namespaceURI, localName);
046    }
047
048    @Override
049    public void writeStartElement(String prefix, String namespaceURI, String localName) throws XMLStreamException {
050        this.checkStartElement();
051        super.writeStartElement(prefix, namespaceURI, localName);
052    }
053
054    @Override
055    public void writeEmptyElement(String namespaceURI, String localName) throws XMLStreamException {
056        if (this.state == 0)
057            this.writeStartElement(namespaceURI, localName);            // super.writeEndDocument() will add closing tag for us
058        else
059            super.writeEmptyElement(namespaceURI, localName);
060    }
061
062    @Override
063    public void writeEmptyElement(String prefix, String namespaceURI, String localName) throws XMLStreamException {
064        if (this.state == 0)
065            this.writeStartElement(prefix, namespaceURI, localName);    // super.writeEndDocument() will add closing tag for us
066        else
067            super.writeEmptyElement(prefix, namespaceURI, localName);
068    }
069
070    @Override
071    public void writeEmptyElement(String localName) throws XMLStreamException {
072        if (this.state == 0)
073            this.writeStartElement(localName);                          // super.writeEndDocument() will add closing tag for us
074        else
075            super.writeEmptyElement(localName);
076    }
077
078    @Override
079    public void writeComment(String data) throws XMLStreamException {
080        this.checkNonWhitespace();
081        super.writeComment(data);
082    }
083
084    @Override
085    public void writeProcessingInstruction(String target) throws XMLStreamException {
086        this.checkNonWhitespace();
087        super.writeProcessingInstruction(target);
088    }
089
090    @Override
091    public void writeProcessingInstruction(String target, String data) throws XMLStreamException {
092        this.checkNonWhitespace();
093        super.writeProcessingInstruction(target, data);
094    }
095
096    @Override
097    public void writeCData(String data) throws XMLStreamException {
098        this.checkNonWhitespace();
099        super.writeCData(data);
100    }
101
102    @Override
103    public void writeDTD(String dtd) throws XMLStreamException {
104        this.checkNonWhitespace();
105        super.writeDTD(dtd);
106    }
107
108    @Override
109    public void writeEntityRef(String name) throws XMLStreamException {
110        this.checkNonWhitespace();
111        super.writeEntityRef(name);
112    }
113
114    @Override
115    public void writeCharacters(String text) throws XMLStreamException {
116        this.checkWhitespace(text);
117        super.writeCharacters(text);
118    }
119
120    @Override
121    public void writeCharacters(char[] text, int start, int len) throws XMLStreamException {
122        this.checkWhitespace(new String(text, start, len));
123        super.writeCharacters(text, start, len);
124    }
125
126    @Override
127    public void writeEndDocument() throws XMLStreamException {
128        this.checkNonWhitespace();
129        super.writeEndDocument();
130    }
131
132    @Override
133    public void writeEndElement() throws XMLStreamException {
134        this.checkNonWhitespace();
135        super.writeEndElement();
136    }
137
138    private void checkStartElement() throws XMLStreamException {
139        switch (this.state) {
140        case 0:
141            this.state++;
142            break;
143        case 1:
144            this.checkNonWhitespace();
145            break;
146        default:
147            break;
148        }
149    }
150
151    private void checkNonWhitespace() throws XMLStreamException {
152        if (this.state != 1)
153            return;
154        this.state++;
155        this.addAnnotationElement(this.getParent());
156        if (this.trailingSpace.length() > 0)
157            super.writeCharacters(this.trailingSpace.toString());
158    }
159
160    private void checkWhitespace(String text) throws XMLStreamException {
161        if (this.state != 1)
162            return;
163        if (text.trim().length() == 0)
164            this.trailingSpace.append(text);
165        else
166            this.checkNonWhitespace();
167    }
168
169    /**
170     * Get the whitespace found between the opening document tag and the first non-space child.
171     */
172    protected String getTrailingSpace() {
173        return this.trailingSpace.toString();
174    }
175
176    /**
177     * Add the annotation element.
178     *
179     * <p>
180     * This method should write the start element, followed by by any nested content, and then lastly the
181     * end element for the annotation element.
182     * </p>
183     *
184     * @param writer output to which the annotation element should be written
185     */
186    protected abstract void addAnnotationElement(XMLStreamWriter writer) throws XMLStreamException;
187}

006package org.dellroad.stuff.xml;
007
008import javax.xml.transform.ErrorListener;
009import javax.xml.transform.TransformerException;
010
011import org.slf4j.Logger;
012
013/**
014 * {@link ErrorListener} implementation that logs the messages to a configured {@link Logger}
015 * and throws exceptions in cases of errors and fatal errors.
016 *
017 * <p>
018 * This class also optionally works around some stupid Xalan-J bugs:
019 * <ul>
020 * <li>Throw exceptions in the case of {@link #warning} also; this is required because Xalan-J
021 *  reports even <code>&lt;message terminate="yes"&gt;</code> messages as warnings</li>
022 * <li>When throwing exceptions, wrap them in {@link RuntimeException}s to avoid being swallowed;
023 *  otherwise Xalan-J will not terminate on a <code>&lt;message terminate="yes"&gt;</code></li>
024 * </ul>
025 */
026public class TransformErrorListener implements ErrorListener {
027
028    protected final Logger log;
029    protected final boolean xalanWorkarounds;
030
031    public TransformErrorListener(Logger log, boolean xalanWorkarounds) {
032        this.log = log;
033        this.xalanWorkarounds = xalanWorkarounds;
034    }
035
036    @Override
037    public void warning(TransformerException e) throws TransformerException {
038        this.log.warn(this.getLogMessageFor(e));
039        if (this.xalanWorkarounds)
040            this.rethrow(e);
041    }
042
043    @Override
044    public void error(TransformerException e) throws TransformerException {
045        this.log.error(this.getLogMessageFor(e));
046        this.rethrow(e);
047    }
048
049    @Override
050    public void fatalError(TransformerException e) throws TransformerException {
051        this.log.error(this.getLogMessageFor(e));
052        this.rethrow(e);
053    }
054
055    protected String getLogMessageFor(TransformerException e) {
056        return e.getMessageAndLocation();
057    }
058
059    protected void rethrow(TransformerException e) throws TransformerException {
060        if (!this.xalanWorkarounds)
061            throw e;
062        throw new RuntimeException("exception from XSL transform", e);
063    }
064}

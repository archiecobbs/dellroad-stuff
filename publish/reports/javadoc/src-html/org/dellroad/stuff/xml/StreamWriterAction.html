010import javax.xml.namespace.NamespaceContext;
011import javax.xml.stream.XMLStreamConstants;

014import javax.xml.stream.XMLStreamWriter;

018/**
019 * Represents some XML output event(s) that can be applied to an {@link XMLStreamWriter}.
020 */
021public abstract class StreamWriterAction {
022
023    protected StreamWriterAction() {
024    }
025
026    /**
027     * Apply this action to the given {@link XMLStreamWriter} by invoking method(s) to effect some XML output event(s).
028     *
029     * @param writer XML writer
030     * @throws NullPointerException if {@code writer} is null
031     */
032    public abstract void apply(XMLStreamWriter writer) throws XMLStreamException;
033
034    /**
035     * Get an instance that corresponds to the current event state of the given {@link XMLStreamReader}, such that
036     * a sequence of such instances created from consecutive input events will will recreate the input XML.
037     *
038     * <p>
039     * This method does not advance or modify {@code reader}.
040     *
041     * @param reader XML input
042     * @throws XMLStreamException if an error occurs
043     * @throws NullPointerException if {@code reader} is null
044     */
045    public static StreamWriterAction of(XMLStreamReader reader) throws XMLStreamException {
046        switch (reader.getEventType()) {
047        case XMLStreamConstants.START_ELEMENT:
048            return new StartElementAction(reader);
049        case XMLStreamConstants.END_ELEMENT:
050            return new EndElementAction();
051        case XMLStreamConstants.PROCESSING_INSTRUCTION:
052            return new PIAction(reader);
053        case XMLStreamConstants.CHARACTERS:
054        case XMLStreamConstants.SPACE:
055            return new CharactersAction(reader);
056        case XMLStreamConstants.COMMENT:
057            return new CommentAction(reader);
058        case XMLStreamConstants.START_DOCUMENT:
059            return new StartDocumentAction(reader);
060        case XMLStreamConstants.END_DOCUMENT:
061            return new EndDocumentAction();
062        case XMLStreamConstants.ENTITY_REFERENCE:
063            return new EntityRefAction(reader);
064        case XMLStreamConstants.DTD:
065            return new DTDAction(reader);
066        case XMLStreamConstants.CDATA:
067            return new CDataAction(reader);
068        case XMLStreamConstants.ATTRIBUTE:
069        case XMLStreamConstants.NAMESPACE:
070            return new EmptyAction();
071        case XMLStreamConstants.NOTATION_DECLARATION:
072        case XMLStreamConstants.ENTITY_DECLARATION:
073            return new EmptyAction();                       // ???
074        default:
075            throw new XMLStreamException("unknown event type " + reader.getEventType());
076        }
077    }

081    private static class StartElementAction extends StreamWriterAction {
082
083        final String prefix;
084        final String localName;
085        final String namespaceURI;
086        final NamespaceContext namespaceContext;
087        final String[] nsPrefix;
088        final String[] nsURI;
089        final String[] attrPrefix;
090        final String[] attrNamespace;
091        final String[] attrLocalName;
092        final String[] attrValue;
093
094        public StartElementAction(XMLStreamReader reader) {
095            this.prefix = reader.getPrefix();
096            this.localName = reader.getLocalName();
097            this.namespaceURI = reader.getNamespaceURI();
098            this.namespaceContext = reader.getNamespaceContext();
099            final int numNS = reader.getNamespaceCount();
100            this.nsPrefix = new String[numNS];
101            this.nsURI = new String[numNS];
102            for (int i = 0; i < numNS; i++) {
103                this.nsPrefix[i] = reader.getNamespacePrefix(i);
104                this.nsURI[i] = reader.getNamespaceURI(i);
105            }
106            final int numAttrs = reader.getAttributeCount();
107            this.attrPrefix = new String[numAttrs];
108            this.attrNamespace = new String[numAttrs];
109            this.attrLocalName = new String[numAttrs];
110            this.attrValue = new String[numAttrs];
111            for (int i = 0; i < numAttrs; i++) {
112                this.attrPrefix[i] = reader.getAttributePrefix(i);
113                this.attrNamespace[i] = reader.getAttributeNamespace(i);
114                this.attrLocalName[i] = reader.getAttributeLocalName(i);
115                this.attrValue[i] = reader.getAttributeValue(i);
116            }
117        }
118
119        @Override
120        public void apply(XMLStreamWriter writer) throws XMLStreamException {
121            if (this.namespaceURI != null)
122                writer.writeStartElement(this.prefix, this.localName, this.namespaceURI);
123            else
124                writer.writeStartElement(this.localName);
125            writer.setNamespaceContext(namespaceContext);
126            for (int i = 0; i < this.nsPrefix.length; i++)
127                writer.writeNamespace(this.nsPrefix[i], this.nsURI[i]);
128            for (int i = 0; i < this.attrPrefix.length; i++) {
129                if (this.attrNamespace[i] != null)
130                    writer.writeAttribute(this.attrPrefix[i], this.attrNamespace[i], this.attrLocalName[i], this.attrValue[i]);
131                else
132                    writer.writeAttribute(this.attrLocalName[i], this.attrValue[i]);
133            }
134        }
135
136        @Override
137        public String toString() {
138            final StringBuilder buf = new StringBuilder();
139            buf.append("StartElement[");
140            buf.append("prefix=\"").append(this.prefix).append("\"");
141            buf.append(",localName=\"").append(this.localName).append("\"");
142            buf.append(",namespaceURI=\"").append(this.namespaceURI).append("\"");
143            buf.append(",namespaceContext=\"").append(this.namespaceContext).append("\"");
144            buf.append(",attrPrefix=").append(Arrays.asList(this.attrPrefix));
145            buf.append(",attrNamespace=").append(Arrays.asList(this.attrNamespace));
146            buf.append(",attrLocalName=").append(Arrays.asList(this.attrLocalName));
147            buf.append(",attrValue=").append(Arrays.asList(this.attrValue));
148            buf.append("]");
149            return buf.toString();
150        }
151    }

153    private static class EndElementAction extends StreamWriterAction {
154
155        @Override
156        public void apply(XMLStreamWriter writer) throws XMLStreamException {
157            writer.writeEndElement();
158        }
159
160        @Override
161        public String toString() {
162            return "EndElement";
163        }
164    }

166    private static class PIAction extends StreamWriterAction {
167
168        private final String target;
169        private final String data;
170
171        public PIAction(XMLStreamReader reader) {
172            this.target = reader.getPITarget();
173            this.data = reader.getPIData();
174        }
175
176        @Override
177        public void apply(XMLStreamWriter writer) throws XMLStreamException {
178            writer.writeProcessingInstruction(this.target, this.data);
179        }
180
181        @Override
182        public String toString() {
183            return "PI[target=\"" + this.target + "\",data=\"" + this.data + "\"]";
184        }
185    }
186
187    private static class CharactersAction extends StreamWriterAction {
188
189        protected final String text;
190
191        public CharactersAction(XMLStreamReader reader) {
192            this.text = reader.getText();
193        }
194
195        public void apply(XMLStreamWriter writer) throws XMLStreamException {
196            writer.writeCharacters(this.text);
197        }
198
199        @Override
200        public String toString() {
201            return this.getClass().getSimpleName().replaceAll("Action$", "") + "[text="
202              + StringEncoder.enquote(this.text) + "]";
203        }
204    }

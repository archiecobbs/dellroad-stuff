006package org.dellroad.stuff.xml;
007
008import java.util.Arrays;
009
010import javax.xml.stream.XMLStreamConstants;
011import javax.xml.stream.XMLStreamException;
012import javax.xml.stream.XMLStreamWriter;
013
014/**
015 * Wrapper for an underlying {@link XMLStreamWriter} that "pretty-prints" the XML by replacing the whitespace between XML tags
016 * so that the result is properly indented.
017 *
018 * <p>
019 * This class will also fixup a missing/incomplete initial XML declaration.
020 */
021public class IndentXMLStreamWriter extends StreamWriterDelegate {
022
023    /**
024     * Initial value for the {@linkplain #setDefaultVersion default XML version}.
025     */
026    public static final String DEFAULT_VERSION = "1.0";
027
028    /**
029     * Initial value for the {@linkplain #setDefaultEncoding default character encoding}.
030     */
031    public static final String DEFAULT_ENCODING = "UTF-8";
032
033    /**
034     * Default number of spaces corresponding to one indent level.
035     */
036    public static final int DEFAULT_INDENT = 4;
037
038    private final String newline = System.getProperty("line.separator", "\\n");
039    private final StringBuilder whitespaceBuffer = new StringBuilder();
040    private final int indent;
041
042    private boolean addMissingXmlDeclaration = true;
043    private boolean indentAfterXmlDeclaration = true;
044    private String defaultVersion = DEFAULT_VERSION;
045    private String defaultEncoding = DEFAULT_ENCODING;
046
047    private boolean started;
048    private boolean newlineAfter;
049    private int lastEvent = -1;
050    private int depth;
051
052    /**
053     * Default constructor. Sets the indent to {@link #DEFAULT_INDENT}.
054     * The parent must be configured via {@link #setParent setParent()}.
055     */
056    public IndentXMLStreamWriter() {
057        this.indent = DEFAULT_INDENT;
058    }
059
060    /**
061     * Convenience constructor. Equivalent to:
062     * <blockquote>
063     *  {@link #IndentXMLStreamWriter(XMLStreamWriter, int) IndentXMLStreamWriter}{@code (writer, }{@link #DEFAULT_INDENT}{@code )};
064     * </blockquote>
065     *
066     * @param writer underlying writer
067     */
068    public IndentXMLStreamWriter(XMLStreamWriter writer) {
069        this(writer, DEFAULT_INDENT);
070    }
071
072    /**
073     * Primary constructor.
074     *
075     * @param writer underlying writer
076     * @param indent number of spaces corresponding to one indent level, or negative for no inter-tag whitespace at all
077     */
078    public IndentXMLStreamWriter(XMLStreamWriter writer, int indent) {
079        super(writer);
080        this.indent = indent;
081    }
082
083    /**
084     * Set whether to add an XML declaration, if missing.
085     *
086     * <p>
087     * Default is true.
088     */
089    public void setAddMissingXmlDeclaration(boolean addMissingXmlDeclaration) {
090        this.addMissingXmlDeclaration = addMissingXmlDeclaration;
091    }
092
093    /**
094     * Set whether to "indent" (actually, just output a newline) after the XML declaration if necessary.
095     * In some cases, such as transforming into a DOM, this behavior must be disabled to avoid hierarchy
096     * exceptions due to characters not being allowed before the document element.
097     *
098     * <p>
099     * Default is true.
100     */
101    public void setIndentAfterXmlDeclaration(boolean indentAfterXmlDeclaration) {
102        this.indentAfterXmlDeclaration = indentAfterXmlDeclaration;
103    }
104
105    /**
106     * Set the version for the XML declaration in case it's not already specified.
107     *
108     * <p>
109     * Default is {@link #DEFAULT_VERSION} ({@value #DEFAULT_VERSION}).
110     *
111     * @param defaultVersion XML version
112     */
113    public void setDefaultVersion(String defaultVersion) {
114        this.defaultVersion = defaultVersion;
115    }
116
117    /**
118     * Set the character encoding for the XML declaration in case it's not already specified.
119     *
120     * <p>
121     * Default is {@link #DEFAULT_ENCODING} ({@value #DEFAULT_ENCODING}).
122     *
123     * @param defaultEncoding character encoding name
124     */
125    public void setDefaultEncoding(String defaultEncoding) {
126        this.defaultEncoding = defaultEncoding;
127    }
128
129    @Override
130    public void writeStartDocument(String encoding, String version) throws XMLStreamException {
131        if (this.started)
132            return;
133        this.started = true;
134        this.handleOther(XMLStreamConstants.START_DOCUMENT);
135        super.writeStartDocument(encoding, version);
136    }
137
138    @Override
139    public void writeStartDocument(String version) throws XMLStreamException {
140        this.writeStartDocument(this.defaultEncoding, version);
141    }
142
143    @Override
144    public void writeStartDocument() throws XMLStreamException {
145        this.writeStartDocument(this.defaultEncoding, this.defaultVersion);
146    }
147
148    @Override
149    public void writeStartElement(String localName) throws XMLStreamException {
150        this.handleStartElement(false);
151        super.writeStartElement(localName);
152    }
153
154    @Override
155    public void writeStartElement(String namespaceURI, String localName) throws XMLStreamException {
156        this.handleStartElement(false);
157        super.writeStartElement(namespaceURI, localName);
158    }
159
160    @Override
161    public void writeStartElement(String prefix, String namespaceURI, String localName) throws XMLStreamException {
162        this.handleStartElement(false);
163        super.writeStartElement(prefix, namespaceURI, localName);
164    }
165
166    @Override
167    public void writeEmptyElement(String namespaceURI, String localName) throws XMLStreamException {
168        this.handleStartElement(true);
169        super.writeEmptyElement(namespaceURI, localName);
170    }
171
172    @Override
173    public void writeEmptyElement(String prefix, String namespaceURI, String localName) throws XMLStreamException {
174        this.handleStartElement(true);
175        super.writeEmptyElement(prefix, namespaceURI, localName);
176    }
177
178    @Override
179    public void writeEmptyElement(String localName) throws XMLStreamException {
180        this.handleStartElement(true);
181        super.writeEmptyElement(localName);
182    }
183
184    @Override
185    public void writeComment(String data) throws XMLStreamException {
186        this.handleOther(XMLStreamConstants.COMMENT);
187        super.writeComment(data);
188    }
189
190    @Override
191    public void writeProcessingInstruction(String target) throws XMLStreamException {
192        this.handleOther(XMLStreamConstants.PROCESSING_INSTRUCTION);
193        super.writeProcessingInstruction(target);
194    }
195
196    @Override
197    public void writeProcessingInstruction(String target, String data) throws XMLStreamException {
198        this.handleOther(XMLStreamConstants.PROCESSING_INSTRUCTION);
199        super.writeProcessingInstruction(target, data);
200    }
201
202    @Override
203    public void writeCData(String data) throws XMLStreamException {
204        this.handleOther(XMLStreamConstants.CDATA);
205        super.writeCData(data);
206    }
207
208    @Override
209    public void writeDTD(String dtd) throws XMLStreamException {
210        this.handleOther(XMLStreamConstants.DTD);
211        super.writeDTD(dtd);
212    }
213
214    @Override
215    public void writeEntityRef(String name) throws XMLStreamException {
216        this.handleOther(XMLStreamConstants.ENTITY_REFERENCE);
217        super.writeEntityRef(name);
218    }
219
220    @Override
221    public void writeCharacters(String text) throws XMLStreamException {
222        this.handleCharacters(text);
223    }
224
225    @Override
226    public void writeCharacters(char[] text, int start, int len) throws XMLStreamException {
227        this.writeCharacters(new String(text, start, len));
228    }
229
230    @Override
231    public void writeEndDocument() throws XMLStreamException {
232        while (this.depth > 0) {
233            this.writeEndElement();
234            this.depth--;
235        }
236        super.writeEndDocument();
237    }
238
239    @Override
240    public void writeEndElement() throws XMLStreamException {
241        this.depth--;
242        if (this.lastEvent == XMLStreamConstants.END_ELEMENT)
243            this.indent();
244        this.handleOther(XMLStreamConstants.END_ELEMENT);
245        super.writeEndElement();
246    }
247
248    private void handleStartElement(boolean selfClosing) throws XMLStreamException {
249        this.writeStartDocumentIfNecessary();
250        if (this.lastEvent == XMLStreamConstants.START_ELEMENT || this.lastEvent == XMLStreamConstants.END_ELEMENT
251          || (this.indentAfterXmlDeclaration && this.lastEvent == XMLStreamConstants.START_DOCUMENT))
252            this.indent();
253        this.handleOther(selfClosing ? XMLStreamConstants.END_ELEMENT : XMLStreamConstants.START_ELEMENT);
254        if (!selfClosing)
255            this.depth++;
256    }
257
258    private void handleCharacters(String text) throws XMLStreamException {
259        this.writeStartDocumentIfNecessary();
260        if ((this.lastEvent == XMLStreamConstants.START_ELEMENT || this.lastEvent == XMLStreamConstants.END_ELEMENT)
261          && text.trim().length() == 0)
262            this.whitespaceBuffer.append(text);
263        else {
264            this.handleOther(XMLStreamConstants.CHARACTERS);
265            super.writeCharacters(text);
266        }
267    }
268
269    private void handleOther(int eventType) throws XMLStreamException {
270        this.writeStartDocumentIfNecessary();
271        if (this.whitespaceBuffer.length() > 0) {
272            super.writeCharacters(this.whitespaceBuffer.toString());
273            this.whitespaceBuffer.setLength(0);
274        }
275        this.lastEvent = eventType;
276    }
277
278    private void writeStartDocumentIfNecessary() throws XMLStreamException {
279        if (!this.started && this.addMissingXmlDeclaration)
280            this.writeStartDocument();
281    }
282
283    /**
284     * Replace existing content of whitespaceBuffer with newline(s) followed by indentation to the current depth.
285     */
286    private void indent() {
287
288        // Are we doing any indenting?
289        if (this.indent < 0) {
290            this.whitespaceBuffer.setLength(0);
291            return;
292        }
293
294        // Count how many initial newlines there were in the original stream
295        int newlines = 0;
296        for (int i = 0; i < this.whitespaceBuffer.length(); i++) {
297            if (this.whitespaceBuffer.charAt(i) == '\n'
298              || (this.whitespaceBuffer.charAt(i++) == '\r'
299               && i < this.whitespaceBuffer.length() && this.whitespaceBuffer.charAt(i) == '\n')) {
300                newlines++;
301                continue;
302            }
303            break;
304        }
305        this.whitespaceBuffer.setLength(0);
306
307        // Add back that many newline(s) (at least one) followed by indent
308        newlines = Math.max(newlines, 1);
309        final char[] buf = new char[newlines + Math.max(this.depth, 0) * this.indent];
310        Arrays.fill(buf, 0, newlines, '\n');
311        Arrays.fill(buf, newlines, buf.length, ' ');
312        this.whitespaceBuffer.append(buf);
313    }

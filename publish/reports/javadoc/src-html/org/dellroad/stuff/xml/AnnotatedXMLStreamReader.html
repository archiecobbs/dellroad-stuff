006package org.dellroad.stuff.xml;
007
008import javax.xml.stream.XMLStreamException;
009import javax.xml.stream.XMLStreamReader;
010import javax.xml.stream.util.StreamReaderDelegate;
011
012/**
013 * {@link XMLStreamReader} that reads and removes an initial annotation element from an XML document.
014 * The annotation element, if present, must be the first element inside the top-level document element.
015 * When the annotation element is encountered, {@link #readAnnotationElement readAnnotationElement()} will be invoked.
016 *
017 * <p>
018 * This class can be used in combination with {@link AnnotatedXMLStreamWriter} to transparently annotate XML documents.
019 *
020 * @see AnnotatedXMLStreamWriter
021 */
022public abstract class AnnotatedXMLStreamReader extends StreamReaderDelegate {
023
024    // State:
025    //  0 = before document element
026    //  1 = after document element but before annotation element (if any)
027    //  2 = after annotation element (if any)
028    private byte state;
029
030    /**
031     * Constructor.
032     *
033     * @param parent parent reader
034     */
035    public AnnotatedXMLStreamReader(XMLStreamReader parent) {
036        super(parent);
037    }
038
039    @Override
040    public int next() throws XMLStreamException {
041        int eventType = super.next();
042        switch (this.state) {
043        case 0:
044            if (eventType == START_ELEMENT)
045                this.state++;
046            break;
047        case 1:
048            eventType = this.checkAnnotation(eventType);
049            break;
050        default:
051            break;
052        }
053        return eventType;
054    }
055
056    @Override
057    public int nextTag() throws XMLStreamException {
058        int eventType = this.next();
059        while ((eventType == CHARACTERS && this.isWhiteSpace())
060          || (eventType == CDATA && this.isWhiteSpace())
061          || eventType == SPACE || eventType == PROCESSING_INSTRUCTION || eventType == COMMENT)
062            eventType = this.next();
063        if (eventType != START_ELEMENT && eventType != END_ELEMENT)
064            throw new XMLStreamException("expected start or end tag", this.getLocation());
065        return eventType;
066    }
067
068    /**
069     * Determine if the given {@link XMLStreamReader} is positioned at the annotation element, and if so, read it.
070     *
071     * <p>
072     * When this method is invoked, the {@code reader}'s current event has type {@link #START_ELEMENT} and may represent
073     * the start of the annotation element. If it doesn't, this method should not read any events and immediately return false.
074     * Otherwise, this method should read subsequent events up <i>through</i> the corresponding {@link #END_ELEMENT}
075     * event and return true. Upon return, the {@code reader}'s current event will have type {@link #END_ELEMENT}.
076     *
077     * @param reader source from which the the annotation element is to be read, if found
078     * @return true if the current event is the start of the annotation element and has been read, false otherwise
079     * @throws XMLStreamException if an XML parse error occurs
080     */
081    protected abstract boolean readAnnotationElement(XMLStreamReader reader) throws XMLStreamException;
082
083    private int checkAnnotation(int eventType) throws XMLStreamException {
084
085        // Sanity check
086        assert this.state == 1;
087
088        if (this.getParent().isWhiteSpace() || eventType == COMMENT || eventType == PROCESSING_INSTRUCTION)
089            return eventType;
090
091        // Anything else means we either we have found the annotation element or it is not there
092        this.state++;
093        if (eventType != START_ELEMENT || !this.readAnnotationElement(this.getParent()))
094            return eventType;
095        if (super.getEventType() != END_ELEMENT)
096            throw new XMLStreamException("readAnnotationElement() did not stop on an END_ELEMENT event", this.getLocation());
097
098        // Skip whitespace after annotation element
099        do
100            eventType = super.next();
101        while (super.isWhiteSpace());
102
103        // Done
104        return eventType;
105    }
106}

006package org.dellroad.stuff.xml;
007
008import javax.xml.namespace.NamespaceContext;
009import javax.xml.stream.XMLEventReader;
010import javax.xml.stream.XMLEventWriter;
011import javax.xml.stream.XMLStreamException;
012import javax.xml.stream.events.XMLEvent;
013
014/**
015 * Support superclass for filtering {@link XMLEventWriter} implementations.
016 * All methods delegate to the configured parent.
017 */
018public class EventWriterDelegate implements XMLEventWriter {
019
020    private XMLEventWriter parent;
021
022    /**
023     * Default constructor. The parent must be configured via {@link #setParent}.
024     */
025    public EventWriterDelegate() {
026    }
027
028    /**
029     * Constructor.
030     *
031     * @param parent underlying writer wrapped by this instance
032     */
033    public EventWriterDelegate(XMLEventWriter parent) {
034        this.parent = parent;
035    }
036
037    /**
038     * Get parent instance.
039     */
040    public XMLEventWriter getParent() {
041        return this.parent;
042    }
043
044    /**
045     * Set parent instance.
046     */
047    public void setParent(XMLEventWriter parent) {
048        this.parent = parent;
049    }
050
051    @Override
052    public void add(XMLEvent event) throws XMLStreamException {
053        this.parent.add(event);
054    }
055
056    @Override
057    public void flush() throws XMLStreamException {
058        this.parent.flush();
059    }
060
061    @Override
062    public void close() throws XMLStreamException {
063        this.parent.close();
064    }
065
066    @Override
067    public void add(XMLEventReader reader) throws XMLStreamException {
068        this.parent.add(reader);
069    }
070
071    @Override
072    public String getPrefix(String uri) throws XMLStreamException {
073        return this.parent.getPrefix(uri);
074    }
075
076    @Override
077    public void setPrefix(String prefix, String uri) throws XMLStreamException {
078        this.parent.setPrefix(prefix, uri);
079    }
080
081    @Override
082    public void setDefaultNamespace(String uri) throws XMLStreamException {
083        this.parent.setDefaultNamespace(uri);
084    }
085
086    @Override
087    public void setNamespaceContext(NamespaceContext context) throws XMLStreamException {
088        this.parent.setNamespaceContext(context);
089    }
090
091    @Override
092    public NamespaceContext getNamespaceContext() {
093        return this.parent.getNamespaceContext();
094    }
095}

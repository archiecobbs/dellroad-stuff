006package org.dellroad.stuff.xml;
007
008import java.util.Arrays;
009
010import javax.xml.stream.XMLEventFactory;
011import javax.xml.stream.XMLEventReader;
012import javax.xml.stream.XMLEventWriter;
013import javax.xml.stream.XMLStreamException;
014import javax.xml.stream.events.XMLEvent;
015
016/**
017 * Wrapper for an underlying {@link XMLEventWriter} that automatically adds indentation to the event stream.
018 */
019public class IndentXMLEventWriter extends EventWriterDelegate {
020
021    /**
022     * The configured event factory for this instance.
023     */
024    protected final XMLEventFactory factory;
025
026    private final int indent;
027
028    private int lastEvent = -1;
029    private int depth;
030
031    /**
032     * Constructor.
033     *
034     * @param writer underlying writer
035     * @param factory event factory
036     * @param indent indent amount, or negative to not add any whitespace
037     * @throws IllegalArgumentException if {@code writer} or {@code factory} is null
038     */
039    public IndentXMLEventWriter(XMLEventWriter writer, XMLEventFactory factory, int indent) {
040        super(writer);
041        if (factory == null)
042            throw new IllegalArgumentException("null factory");
043        this.factory = factory;
044        this.indent = indent;
045    }
046
047    /**
048     * Convenience constructor. Equivalent to:
049     * <blockquote>
050     *  {@link #IndentXMLEventWriter(XMLEventWriter, XMLEventFactory, int)
051     *      IndentXMLEventWriter(writer, XMLEventFactory.newFactory(), indent)}
052     * </blockquote>
053     *
054     * @param writer underlying writer
055     * @param indent indent amount, or negative to not add any whitespace
056     * @throws IllegalArgumentException if {@code writer} or {@code factory} is null
057     */
058    public IndentXMLEventWriter(XMLEventWriter writer, int indent) {
059        this(writer, XMLEventFactory.newFactory(), indent);
060    }
061
062    @Override
063    public void add(XMLEvent event) throws XMLStreamException {
064        switch (event.getEventType()) {
065        case XMLEvent.START_ELEMENT:
066            switch (lastEvent) {
067            case XMLEvent.START_ELEMENT:
068            case XMLEvent.END_ELEMENT:
069                this.indent(this.depth);
070                break;
071            default:
072                break;
073            }
074            this.depth++;
075            break;
076        case XMLEvent.END_ELEMENT:
077            this.depth--;
078            switch (lastEvent) {
079            case XMLEvent.START_ELEMENT:
080                break;
081            case XMLEvent.END_ELEMENT:
082                this.indent(this.depth);
083                break;
084            default:
085                break;
086            }
087            break;
088        default:
089            break;
090        }
091        this.lastEvent = event.getEventType();
092        super.add(event);
093    }
094
095    @Override
096    public void add(XMLEventReader reader) throws XMLStreamException {
097        while (reader.hasNext())
098            this.add(reader.nextEvent());
099    }
100
101    /**
102     * Emit a newline followed by indentation to the given depth.
103     */
104    protected void indent(int depth) throws XMLStreamException {
105        if (this.indent < 0)
106            return;
107        char[] buf = new char[1 + depth * this.indent];
108        Arrays.fill(buf, ' ');
109        buf[0] = '\n';
110        super.add(this.factory.createIgnorableSpace(new String(buf)));
111    }
112}

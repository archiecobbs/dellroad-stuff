006package org.dellroad.stuff.xml;
007
008import javax.xml.namespace.NamespaceContext;
009import javax.xml.stream.XMLStreamException;
010import javax.xml.stream.XMLStreamWriter;
011
012/**
013 * Support superclass for filtering {@link XMLStreamWriter} implementations.
014 * All methods delegate to the configured parent.
015 */
016public class StreamWriterDelegate implements XMLStreamWriter {
017
018    private XMLStreamWriter parent;
019
020    /**
021     * Default constructor. The parent must be configured via {@link #setParent}.
022     */
023    public StreamWriterDelegate() {
024    }
025
026    /**
027     * Constructor.
028     *
029     * @param parent underlying writer wrapped by this instance
030     */
031    public StreamWriterDelegate(XMLStreamWriter parent) {
032        this.parent = parent;
033    }
034
035    /**
036     * Get parent instance.
037     */
038    public XMLStreamWriter getParent() {
039        return this.parent;
040    }
041
042    /**
043     * Set parent instance.
044     */
045    public void setParent(XMLStreamWriter parent) {
046        this.parent = parent;
047    }
048
049    @Override
050    public void writeStartElement(String localName) throws XMLStreamException {
051        this.parent.writeStartElement(localName);
052    }
053
054    @Override
055    public void writeStartElement(String namespaceURI, String localName) throws XMLStreamException {
056        this.parent.writeStartElement(namespaceURI, localName);
057    }
058
059    @Override
060    public void writeStartElement(String prefix, String namespaceURI, String localName) throws XMLStreamException {
061        this.parent.writeStartElement(prefix, namespaceURI, localName);
062    }
063
064    @Override
065    public void writeEmptyElement(String namespaceURI, String localName) throws XMLStreamException {
066        this.parent.writeEmptyElement(namespaceURI, localName);
067    }
068
069    @Override
070    public void writeEmptyElement(String prefix, String namespaceURI, String localName) throws XMLStreamException {
071        this.parent.writeEmptyElement(prefix, namespaceURI, localName);
072    }
073
074    @Override
075    public void writeEmptyElement(String localName) throws XMLStreamException {
076        this.parent.writeEmptyElement(localName);
077    }
078
079    @Override
080    public void writeEndElement() throws XMLStreamException {
081        this.parent.writeEndElement();
082    }
083
084    @Override
085    public void writeEndDocument() throws XMLStreamException {
086        this.parent.writeEndDocument();
087    }
088
089    @Override
090    public void close() throws XMLStreamException {
091        this.parent.close();
092    }
093
094    @Override
095    public void flush() throws XMLStreamException {
096        this.parent.flush();
097    }
098
099    @Override
100    public void writeAttribute(String localName, String value) throws XMLStreamException {
101        this.parent.writeAttribute(localName, value);
102    }
103
104    @Override
105    public void writeAttribute(String prefix, String namespaceURI, String localName, String value) throws XMLStreamException {
106        this.parent.writeAttribute(prefix, namespaceURI, localName, value);
107    }
108
109    @Override
110    public void writeAttribute(String namespaceURI, String localName, String value) throws XMLStreamException {
111        this.parent.writeAttribute(namespaceURI, localName, value);
112    }
113
114    @Override
115    public void writeNamespace(String prefix, String namespaceURI) throws XMLStreamException {
116        this.parent.writeNamespace(prefix, namespaceURI);
117    }
118
119    @Override
120    public void writeDefaultNamespace(String namespaceURI) throws XMLStreamException {
121        this.parent.writeDefaultNamespace(namespaceURI);
122    }
123
124    @Override
125    public void writeComment(String data) throws XMLStreamException {
126        this.parent.writeComment(data);
127    }
128
129    @Override
130    public void writeProcessingInstruction(String target) throws XMLStreamException {
131        this.parent.writeProcessingInstruction(target);
132    }
133
134    @Override
135    public void writeProcessingInstruction(String target, String data) throws XMLStreamException {
136        this.parent.writeProcessingInstruction(target, data);
137    }
138
139    @Override
140    public void writeCData(String data) throws XMLStreamException {
141        this.parent.writeCData(data);
142    }
143
144    @Override
145    public void writeDTD(String dtd) throws XMLStreamException {
146        this.parent.writeDTD(dtd);
147    }
148
149    @Override
150    public void writeEntityRef(String name) throws XMLStreamException {
151        this.parent.writeEntityRef(name);
152    }
153
154    @Override
155    public void writeStartDocument() throws XMLStreamException {
156        this.parent.writeStartDocument();
157    }
158
159    @Override
160    public void writeStartDocument(String version) throws XMLStreamException {
161        this.parent.writeStartDocument(version);
162    }
163
164    @Override
165    public void writeStartDocument(String encoding, String version) throws XMLStreamException {
166        this.parent.writeStartDocument(encoding, version);
167    }
168
169    @Override
170    public void writeCharacters(String text) throws XMLStreamException {
171        this.parent.writeCharacters(text);
172    }
173
174    @Override
175    public void writeCharacters(char[] text, int start, int len) throws XMLStreamException {
176        this.parent.writeCharacters(text, start, len);
177    }
178
179    @Override
180    public String getPrefix(String uri) throws XMLStreamException {
181        return this.parent.getPrefix(uri);
182    }
183
184    @Override
185    public void setPrefix(String prefix, String uri) throws XMLStreamException {
186        this.parent.setPrefix(prefix, uri);
187    }
188
189    @Override
190    public void setDefaultNamespace(String uri) throws XMLStreamException {
191        this.parent.setDefaultNamespace(uri);
192    }
193
194    @Override
195    public void setNamespaceContext(NamespaceContext context) throws XMLStreamException {
196        this.parent.setNamespaceContext(context);
197    }
198
199    @Override
200    public NamespaceContext getNamespaceContext() {
201        return this.parent.getNamespaceContext();
202    }
203
204    @Override
205    public Object getProperty(String name) {
206        return this.parent.getProperty(name);
207    }
208}

006package org.dellroad.stuff.xml;
007
008import javax.xml.stream.XMLEventReader;
009import javax.xml.stream.XMLStreamException;
010import javax.xml.stream.events.Comment;
011import javax.xml.stream.events.XMLEvent;
012import javax.xml.stream.util.EventReaderDelegate;
013
014/**
015 * {@link XMLEventReader} that reads and removes an initial annotation element from an XML document.
016 * The annotation element, if present, must be the first element inside the top-level document element.
017 * {@link #readAnnotationElement readAnnotationElement()} must be provided by the subclass to determine
018 * whether the first non-top element is the expected annotation element, and read it if so.
019 * Whoever reads from this {@link AnnotatedXMLEventReader} instance class will not see the annotation element.
020 *
021 * <p>
022 * This class can be used in combination with {@link AnnotatedXMLEventWriter} to transparently annotate XML documents.
023 *
024 * @see AnnotatedXMLEventWriter
025 */
026public abstract class AnnotatedXMLEventReader extends EventReaderDelegate {
027
028    // State:
029    //  0 = before document element
030    //  1 = after document element but before annotation element (if any)
031    //  2 = after annotation element (if any)
032    private byte state;
033
034    /**
035     * Constructor.
036     *
037     * @param parent parent reader
038     */
039    public AnnotatedXMLEventReader(XMLEventReader parent) {
040        super(parent);
041    }
042
043    @Override
044    public Object next() {
045        try {
046            return this.nextEvent();
047        } catch (XMLStreamException e) {
048            throw new RuntimeException(e);
049        }
050    }
051
052    @Override
053    public XMLEvent peek() throws XMLStreamException {
054        XMLEvent event = super.peek();
055        if (this.state == 1)
056            event = this.checkAnnotation(event);
057        return event;
058    }
059
060    @Override
061    public XMLEvent nextEvent() throws XMLStreamException {
062        switch (this.state) {
063        case 0:
064            XMLEvent event = super.nextEvent();
065            if (event.isStartElement())
066                this.state++;
067            return event;
068        case 1:
069            this.peek();
070            return super.nextEvent();
071        default:
072            return super.nextEvent();
073        }
074    }
075
076    /**
077     * Determine if the next event from the given {@link XMLEventReader} is the annotation element, and if so, read it.
078     *
079     * <p>
080     * This method should invoke {@link #peek reader.peek()} to determine if the next event is the annotation
081     * {@link javax.xml.stream.events.StartElement}; if it's not, this method should not read any events and immediately
082     * return false. Otherwise, it should read the annotation {@link javax.xml.stream.events.StartElement} and all
083     * subsequent events up <i>through</i> the matching {@link javax.xml.stream.events.EndElement} and return true.
084     *
085     * @param reader source from which the rest of the annotation element is to be read
086     * @return false if the annotation element is not seen, true if seen and fully read
087     */
088    protected abstract boolean readAnnotationElement(XMLEventReader reader) throws XMLStreamException;
089
090    /**
091     * Scan for the annotation. We only allow whitespace and comments between the first {@link StartElement} and the
092     * annotation {@link StartElement}. Whitespace after the annotation {@link javax.xml.stream.events.EndElement} is removed.
093     *
094     * Pre-condition:
095     *  - Given event comes after the document element and is before or equal to updates start element
096     *  - Given event is the current "peek event" for this stream
097     *
098     * Post-condition:
099     *  - If event was updates start element, updates were consumed, else no change
100     *  - If updates were consumed:
101     *      - New "peek event" is whatever follows updates and trailing whitespace
102     *      - State has advanced
103     *  - Returned event is the (possibly new) "peek event"
104     */
105    private XMLEvent checkAnnotation(XMLEvent event) throws XMLStreamException {
106
107        // Sanity check
108        assert this.state == 1;
109
110        // Ignore leading whitespace, comments, and PI's
111        if (this.isWhiteSpace(event) || event instanceof Comment || event.isProcessingInstruction())
112            return event;
113
114        // Anything else means we either we have found the annotation element or it is not there
115        this.state++;
116        if (!event.isStartElement() || !this.readAnnotationElement(this.getParent()))
117            return event;
118
119        // Skip whitespace after annotation element
120        for (event = super.peek(); this.isWhiteSpace(event); event = super.peek())
121            super.nextEvent();
122
123        // Done
124        return event;
125    }
126
127    private boolean isWhiteSpace(XMLEvent event) {
128        return event.isCharacters() && event.asCharacters().isWhiteSpace();
129    }
130}

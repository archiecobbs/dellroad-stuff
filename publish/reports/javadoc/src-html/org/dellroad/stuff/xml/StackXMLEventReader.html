006package org.dellroad.stuff.xml;
007
008import java.util.ArrayList;
009import java.util.List;
010
011import javax.xml.stream.XMLEventReader;
012import javax.xml.stream.XMLStreamException;
013import javax.xml.stream.events.StartElement;
014import javax.xml.stream.events.XMLEvent;
015import javax.xml.stream.util.EventReaderDelegate;
016
017/**
018 * Wrapper for a {@link XMLEventReader} that keeps track of the current element(s) being parsed as a stack
019 * of {@link StartElement} events.
020 *
021 * <p>
022 * Each time a {@link StartElement} is read from the underlying {@link XMLEventReader}, it is pushed onto
023 * the stack; each time a {@link javax.xml.stream.events.EndElement} is read from the underlying {@link XMLEventReader},
024 * the previously push {@link StartElement} is popped off.
025 *
026 * <p>
027 * Invocations of {@link #peek} do not affect the element stack.
028 *
029 * <p>
030 * If an {@link XMLStreamException} is thrown at any point, the element stack is no longer guaranteed to track properly.
031 */
032public class StackXMLEventReader extends EventReaderDelegate {
033
034    private final ArrayList<StartElement> stack = new ArrayList<StartElement>();
035
036    /**
037     * Constructor.
038     *
039     * @param reader underlying reader
040     * @throws IllegalArgumentException if {@code reader} is null
041     */
042    public StackXMLEventReader(XMLEventReader reader) {
043        super(reader);
044        if (reader == null)
045            throw new IllegalArgumentException("null reader");
046    }
047
048    /**
049     * Get the top-most {@link StartElement} event on the stack. This corresponds to the innermost XML
050     * element currently being read.
051     *
052     * @return current element, or null if the current parse position is outside of the document element
053     */
054    public StartElement getTopElement() {
055        return !this.stack.isEmpty() ? this.stack.get(this.stack.size() - 1) : null;
056    }
057
058    /**
059     * Get the entire stack of {@link StartElement}s.
060     *
061     * <p>
062     * A copy is returned; changes to the returned list do not affect this instance.
063     *
064     * @return element stack; the first element in the list is the document element
065     */
066    public List<StartElement> getElementStack() {
067        return new ArrayList<StartElement>(this.stack);
068    }
069
070    @Override
071    public String getElementText() throws XMLStreamException {
072        String text = super.getElementText();
073        this.stack.remove(this.stack.size() - 1);
074        return text;
075    }
076
077    @Override
078    public Object next() {
079        try {
080            return this.nextEvent();
081        } catch (XMLStreamException e) {
082            throw new RuntimeException(e);
083        }
084    }
085
086    @Override
087    public XMLEvent nextEvent() throws XMLStreamException {
088        return this.adjust(super.nextEvent());
089    }
090
091    @Override
092    public XMLEvent nextTag() throws XMLStreamException {
093        return this.adjust(super.nextTag());
094    }
095
096    /**
097     * Adjust the stack after reading the given event.
098     *
099     * @param event newly read event
100     * @return {@code event}
101     */
102    protected XMLEvent adjust(XMLEvent event) {
103        switch (event.getEventType()) {
104        case XMLEvent.START_ELEMENT:
105            this.stack.add(event.asStartElement());
106            break;
107        case XMLEvent.END_ELEMENT:
108            this.stack.remove(this.stack.size() - 1);
109            break;
110        default:
111            break;
112        }
113        return event;
114    }
115}

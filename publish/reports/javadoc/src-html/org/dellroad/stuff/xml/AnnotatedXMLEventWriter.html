006package org.dellroad.stuff.xml;
007
008import javax.xml.stream.XMLEventFactory;
009import javax.xml.stream.XMLEventReader;
010import javax.xml.stream.XMLEventWriter;
011import javax.xml.stream.XMLStreamException;
012import javax.xml.stream.events.XMLEvent;
013
014/**
015 * {@link XMLEventWriter} that adds an extra annotation element to an XML document as it is written.
016 * The annotation element will be added as the first element inside the top-level document element.
017 *
018 * <p>
019 * This class can be used in combination with {@link AnnotatedXMLEventReader} to transparently annotate XML documents.
020 *
021 * @see AnnotatedXMLEventReader
022 */
023public abstract class AnnotatedXMLEventWriter extends EventWriterDelegate {
024
025    protected final XMLEventFactory xmlEventFactory = XMLEventFactory.newFactory();
026
027    private final StringBuilder trailingSpace = new StringBuilder();
028
029    // State:
030    //  0 = before document element
031    //  1 = after document element but before annotation element
032    //  2 = after annotation element
033    private byte state;
034
035    public AnnotatedXMLEventWriter(XMLEventWriter inner) {
036        super(inner);
037    }
038
039    @Override
040    public void add(XMLEvent event) throws XMLStreamException {
041        switch (this.state) {
042        case 0:
043            if (event.isStartElement())
044                this.state++;
045            super.add(event);
046            break;
047        case 1:
048            if (event.isNamespace() || event.isAttribute()) {
049                super.add(event);
050                break;
051            }
052            if (event.isCharacters() && event.asCharacters().isWhiteSpace()) {
053                this.trailingSpace.append(event.asCharacters().getData());
054                super.add(event);
055                break;
056            }
057            this.state++;
058            this.addAnnotationElement(this.getParent());
059            if (this.trailingSpace.length() > 0)
060                super.add(this.xmlEventFactory.createCharacters(this.trailingSpace.toString()));
061            super.add(event);
062            break;
063        case 2:
064            super.add(event);
065            break;
066        default:
067            throw new RuntimeException("internal error");
068        }
069    }
070
071    @Override
072    public void add(XMLEventReader reader) throws XMLStreamException {
073        if (reader == null)
074            throw new XMLStreamException("null reader");
075        while (reader.hasNext())
076            this.add(reader.nextEvent());
077    }
078
079    /**
080     * Get the whitespace found between the opening document tag and the first non-space child.
081     */
082    protected String getTrailingSpace() {
083        return this.trailingSpace.toString();
084    }
085
086    /**
087     * Add the annotation element.
088     *
089     * <p>
090     * This method should {@link #add add()} the {@link javax.xml.stream.events.StartElement} for the annotation element, followed
091     * by any nested content, and then lastly the {@link javax.xml.stream.events.EndElement} for the annotation element.
092     *
093     * @param writer output to which the annotation element should be written
094     */
095    protected abstract void addAnnotationElement(XMLEventWriter writer) throws XMLStreamException;
096}

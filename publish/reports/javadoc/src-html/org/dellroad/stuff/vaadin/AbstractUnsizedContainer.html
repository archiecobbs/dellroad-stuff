013/**
014 * A specialization of {@link AbstractQueryContainer} that doesn't require up-front knowledge of the underlying data's size.
015 *
016 * <p>
017 * Subclasses only need support retrieving a contiguous window of fixed size from the underlying data
018 * via {@link #queryWindow queryWindow()}; when the requested window goes beyond the end of the underlying data,
019 * a short or empty result is returned. Based on just this information, this class maintains an estimate of
020 * the size of the underlying data. Each time that size estimate changes, {@link #handleSizeChange} is invoked
021 * to schedule a (non-reentrant) property set change notification. Once the end of the underlying data is reached,
022 * the size is known.
023 * </p>
024 *
025 * <p>
026 * If the actual size of the underlying data is constant, this class will eventually find it. If the
027 * actual size of the underlying data can change (either up or down), this class will adapt accordingly,
028 * but only when it learns of the new size through an invocation of {@link #queryWindow queryWindow()};
029 * as this depends on how the container is used, this may not occur for a long time.
030 * </p>
031 *
032 * <p>
033 * When used to back a Vaadin table, the user will see a table that automatically grows as the user scrolls
034 * downward, until the actual end of the data is detected.
035 * </p>
036 *
037 * @param <T> the type of the Java objects that back each {@link com.vaadin.data.Item} in the container
038 * @see AbstractQueryContainer
039 */
040@SuppressWarnings("serial")
041public abstract class AbstractUnsizedContainer<T> extends AbstractQueryContainer<T> {
042
043    public static final int DEFAULT_WINDOW_SIZE = 50;
044
045    private final int windowSize;
046
047    private long size;
048    private boolean sizeIsKnown;

242    @Override
243    protected QueryList<T> query(long hint) {
244
245        // Clip hint value
246        hint = Math.min(hint, this.size - 1);
247        hint = Math.max(hint, 0);
248
249        // Query underlying data for the window of items centered at "hint"
250        final long offset = Math.max(0, hint - this.windowSize / 2);
251        final List<? extends T> window = this.queryWindow(offset, this.windowSize);
252        final int querySize = window.size();
253
254        // Handle the case where we got an empty window - so the underlying data must have shrunk
255        if (querySize <= 0) {
256
257            // If our starting offset was zero, then we know the size must be zero
258            if (offset == 0) {
259                this.sizeIsKnown = true;
260                this.size = 0;
261            } else {
262
263                // Guess at what the smaller size is and return a QueryList that will throw an InvalidQueryListException,
264                // thereby trigging another probe with a smaller offset. This cycle will repeat until the size is found.
265                this.sizeIsKnown = false;
266                this.size = this.getSmallerEstimate(offset);
267                return new AlwaysInvalidQueryList<T>(this.size);
268            }
269        }
270
271        // Handle the case where we overlapped the end; now we know the exact size
272        if (querySize < this.windowSize) {                  // we overlapped the end; now we know the exact size
273            this.size = offset + querySize;
274            this.sizeIsKnown = true;
275            return new WindowQueryList<T>(offset, window, this.size);
276        }
277
278        // Handle the case where we are somewhere in the middle
279        if (!this.sizeIsKnown) {
280
281            // Get lower bound on size estimate
282            final long lowerBound = offset + querySize;
283
284            // If current size is less than lowerBound + 1, need to expand (the +1 is to leave room to trigger a requery)
285            if (this.size < lowerBound + 1) {
286
287                // Get estimate of new larger size
288                long largerSize = this.getLargerEstimate(lowerBound);
289
290                // Ensure that it is at least large enough to avoid causing another, redundant subsequent resize. This also
291                // ensures that it is strictly larger than the last item in our window so we can always trigger another query.
292                largerSize = Math.max(largerSize, lowerBound + this.windowSize);
293
294                // Increase size estimate (maybe)
295                this.size = Math.max(this.size, largerSize);
296            }
297        } else {                                            // we are somewhere in the middle and size is known
298
299            // Detect if the size has grown
300            this.size = Math.max(this.size, offset + querySize);
301        }
302
303        // Return QueryList based on the window of data we found
304        return new WindowQueryList<T>(offset, window, this.size);
305    }

321    /**
322     * Handle the case where the underlying data's size has suddenly shrunk, so we need to estimate the new size.
323     * All we know is that the actual new size is {@code upperBound} or less.
324     * This method should guess at the new value and, when called repeatedly, should converge rapidly to zero.
325     *
326     * <p>
327     * Note: this situation will not occur if the underlying data's size never decreases.
328     * </p>
329     *
330     * <p>
331     * The implementation in {@link AbstractUnsizedContainer} returns {@code upperBound * 0.75}.
332     * </p>
333     *
334     * @param upperBound an upper bound on the size of the underlying data
335     * @return estimate of the actual size of the underlying data
336     */
337    protected long getSmallerEstimate(long upperBound) {
338        return (upperBound >> 1) + (upperBound >> 2);
339    }
340
341    /**
342     * Estimate the size of the underlying data given that {@code lowerBound} is a lower bound.
343     * This effectively determines how much data will appear to be "beyond" the current window.
344     *
345     * <p>
346     * The implementation in {@link AbstractUnsizedContainer} returns {@code lowerBound * 1.25}.
347     * </p>
348     *
349     * @param lowerBound a lower bound on the size of the underlying data
350     * @return estimate of the actual size of the underlying data
351     */
352    protected long getLargerEstimate(long lowerBound) {
353        return lowerBound + (lowerBound >> 2);
354    }
355
356    @Override
357    protected void fireItemSetChange(Container.ItemSetChangeEvent event) {
358        this.sizeIsKnown = false;
359        super.fireItemSetChange(event);
360    }
361
362    /**
363     * Query the underlying data for a window of items in the given range. This should return the "window" of underlying
364     * data items starting at offset {@code offset} and having at least length {@code length}, or else however many remain.
365     * If {@code offset} is greater than or equal to the size of the underlying data, an empty list should be returned.
366     *
367     * @param offset starting offset for window
368     * @param length window size, always greater than zero
369     * @return list containing at least {@code length} items in the window starting at offset {@code offset}; or less than
370     *  {@code length} items if {@code offset + length} is greater than the size of the underlying data
371     */
372    protected abstract List<? extends T> queryWindow(long offset, int length);
373
374    /**
375     * Emit a property set change notification.
376     *
377     * <p>
378     * Subclasses are required to implement this so that size changes are detected.
379     * </p>
380     *
381     * <p>
382     * Note: to avoid re-entrancy problems, this method should not send out any notifications itself;
383     * instead, it must schedule notifications to be delivered later (perhaps in a different thread).
384     * </p>
385     */
386    @Override
387    protected abstract void handleSizeChange();

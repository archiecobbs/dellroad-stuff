006package org.dellroad.stuff.vaadin;
007
008/**
009 * Cacheable list of Java objects backing an {@link AbstractQueryContainer}.
010 * Instances contain a list of Java objects (or some portion thereof), and may become invalid over time.
011 *
012 * @see AbstractQueryContainer
013 */
014public interface QueryList<T> {
015
016    /**
017     * Get the total size of this list.
018     *
019     * <p>
020     * For any given {@link QueryList} instance, this method is expected to return a the same value if invoked multiple times.
021     * Therefore, callers may safely choose to invoke it only once on a given instance and cache the result.
022     * </p>
023     */
024    long size();
025
026    /**
027     * Get an item in the list, or throw an exception if this instance is no longer valid
028     * or cannot provide the item.
029     *
030     * @param index index of the item (zero-based)
031     * @throws IndexOutOfBoundsException if {@code index} is less than zero or greater than or equal to {@link #size}
032     * @throws InvalidQueryListException if this list has become invalid or cannot provide the item
033     */
034    T get(long index) throws InvalidQueryListException;
035}

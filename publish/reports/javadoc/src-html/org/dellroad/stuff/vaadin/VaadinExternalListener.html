006package org.dellroad.stuff.vaadin;
007
008/**
009 * Support superclass customized for use by listeners that are part of a Vaadin application when listening
010 * to non-Vaadin ("external") event sources.
011 *
012 * <p>
013 * Listeners that are part of a Vaadin application should use this superclass if they are going to be registered
014 * with non-Vaadin event sources, and use the methods {@link #register()} and {@link #unregister()} to control listener
015 * registration. Subclasses implement {@link #register(Object) register(S)} and {@link #unregister(Object) register(S)}
016 * to perform the actual listener registration/unregister operations.
017 *
018 * <p>
019 * Use of this class will ensure two things:
020 * <ul>
021 *  <li>Events can be delivered {@linkplain ContextApplication#invoke in the proper Vaadin application context}; and</li>
022 *  <li>The listener is automatically unregistered from the external event source when the Vaadin application is closed;
023 *      this avoids a memory leak</li>
024 * </ul>
025 *
026 * <p>
027 * <b>Important:</b> subclass listener methods must use {@link #handleEvent handleEvent()} when handling events.
028 * This will ensure proper locking to avoid race conditions.
029 *
030 * <p>
031 * Note: when listening to event sources that are scoped to specific Vaadin application instances and already originate events
032 * within the proper Vaadin application context (i.e., event sources that are not external to the Vaadin application),
033 * then the use of this superclass is not necessary (however, it also doesn't hurt to use it anyway).
034 *
035 * @param <S> The type of the event source
036 * @see ContextApplication#invoke
037 * @see VaadinApplicationScope
038 * @see VaadinApplicationListener
039 * @see SpringContextApplication
040 */
041public abstract class VaadinExternalListener<S> {
042
043    private final S eventSource;
044    private final ContextApplication application;
045    private final CloseListener closeListener = new CloseListener();
046
047    /**
048     * Convenience constructor. Equivalent to:
049     * <blockquote>
050     *  {@link #VaadinExternalListener(Object, ContextApplication) VaadinExternalListener(eventSource, ContextApplication.get())}
051     * </blockquote>
052     *
053     * @param eventSource the event source on which this listener will register
054     * @throws IllegalArgumentException if {@code eventSource} is null
055     * @throws IllegalStateException if there is no {@link ContextApplication} associated with the current thread
056     */
057    protected VaadinExternalListener(S eventSource) {
058        this(eventSource, ContextApplication.get());
059    }
060
061    /**
062     * Primary constructor.
063     *
064     * @param eventSource the event source on which this listener will register
065     * @param application the associated Vaadin application
066     * @throws IllegalArgumentException if either parameter is null
067     */
068    protected VaadinExternalListener(S eventSource, ContextApplication application) {
069        if (eventSource == null)
070            throw new IllegalArgumentException("null eventSource");
071        if (application == null)
072            throw new IllegalArgumentException("null application");
073        this.eventSource = eventSource;
074        this.application = application;
075    }
076
077    /**
078     * Register as a listener on configured event source.
079     *
080     * <p>
081     * This also registers a {@link ContextApplication.CloseListener} on the
082     * {@linkplain #getApplication configured Vaadin application} so that when the application
083     * closes we can unregister this instance from the event source to avoid a memory leak.
084     */
085    public void register() {
086        this.application.addListener(this.closeListener);
087        this.register(this.eventSource);
088    }
089
090    /**
091     * Un-register as a listener on configured event source.
092     *
093     * <p>
094     * This also unregisters the {@link ContextApplication.CloseListener} registered by {@link #register}.
095     */
096    public void unregister() {
097        this.application.removeListener(this.closeListener);
098        this.unregister(this.eventSource);
099    }
100
101    /**
102     * Get the {@link ContextApplication} with which this instance is associated.
103     */
104    public final ContextApplication getApplication() {
105        return this.application;
106    }
107
108    /**
109     * Get the event source with which this instance is (or was) registered as a listener.
110     */
111    public final S getEventSource() {
112        return this.eventSource;
113    }
114
115    /**
116     * Execute the given action using the {@link ContextApplication} with which this instance is associated.
117     * Subclass listener methods should handle events using this method to ensure they are handled
118     * {@linkplain ContextApplication#invoke in the proper Vaadin application context}.
119     *
120     * @param action action to perform
121     */
122    protected void handleEvent(Runnable action) {
123        this.getApplication().invoke(action);
124    }
125
126    /**
127     * Register as a listener on the given event source.
128     *
129     * <p>
130     * Subclass must implement this to perform the actual listener registration.
131     *
132     * @param eventSource event source, never null
133     */
134    protected abstract void register(S eventSource);
135
136    /**
137     * Register as a listener from the given event source.
138     *
139     * <p>
140     * Subclass must implement this to perform the actual listener registration.
141     *
142     * @param eventSource event source, never null
143     */
144    protected abstract void unregister(S eventSource);
145
146// Application close listener
147
148    private final class CloseListener implements ContextApplication.CloseListener {
149
150        @Override
151        public void applicationClosed(ContextApplication.CloseEvent closeEvent) {
152            VaadinExternalListener.this.unregister();
153        }
154    }
155}

006package org.dellroad.stuff.vaadin;
007
008import com.vaadin.ui.AbstractSelect;
009import com.vaadin.ui.ComboBox;
010
011/**
012 * {@link ComboBox} that chooses an {@link Enum} value.
013 *
014 * @see EnumContainer
015 */
016@SuppressWarnings("serial")
017public class EnumComboBox extends ComboBox {
018
019    /**
020     * Default constructor.
021     *
022     * <p>
023     * Caller must separately invoke {@link #setEnumDataSource}.
024     */
025    public EnumComboBox() {
026        this(null);
027    }
028
029    /**
030     * Convenience constructor.
031     *
032     * <p>
033     * Equivalent to:
034     *  <blockquote><code>
035     *  EnumComboBox(enumClass, EnumContainer.TO_STRING_PROPERTY.getName(), false);
036     *  </code></blockquote>
037     *
038     * @param enumClass enum type, or null to leave data source unset
039     */
040    public <T extends Enum<T>> EnumComboBox(Class<T> enumClass) {
041        this(enumClass, false);
042    }
043
044    /**
045     * Convenience constructor.
046     *
047     * <p>
048     * Equivalent to:
049     *  <blockquote><code>
050     *  EnumComboBox(enumClass, EnumContainer.TO_STRING_PROPERTY, allowNull);
051     *  </code></blockquote>
052     *
053     * @param enumClass enum type, or null to leave data source unset
054     * @param allowNull true to allow a null selection, false otherwise
055     */
056    public <T extends Enum<T>> EnumComboBox(Class<T> enumClass, boolean allowNull) {
057        this(enumClass, EnumContainer.TO_STRING_PROPERTY, allowNull);
058    }
059
060    /**
061     * Convenience constructor.
062     *
063     * <p>
064     * Equivalent to:
065     *  <blockquote><code>
066     *  EnumComboBox(enumClass, displayPropertyName, false);
067     *  </code></blockquote>
068     *
069     * @param enumClass enum type
070     * @param displayPropertyName container property to display in the combo box
071     */
072    public <T extends Enum<T>> EnumComboBox(Class<T> enumClass, String displayPropertyName) {
073        this(enumClass, displayPropertyName, false);
074    }
075
076    /**
077     * Convenience constructor.
078     *
079     * <p>
080     * Equivalent to:
081     *  <blockquote><code>
082     *  EnumComboBox(enumClass != null ? new EnumContainer&lt;T&gt;(enumClass) : null, displayPropertyName, allowNull);
083     *  </code></blockquote>
084     *
085     * @param enumClass enum type, or null to leave data source unset
086     * @param displayPropertyName container property to display in the combo box
087     * @param allowNull true to allow a null selection, false otherwise
088     */
089    public <T extends Enum<T>> EnumComboBox(Class<T> enumClass, String displayPropertyName, boolean allowNull) {
090        this(enumClass != null ? new EnumContainer<T>(enumClass) : null, displayPropertyName, allowNull);
091    }
092
093    /**
094     * Primary constructor.
095     *
096     * <p>
097     * This instance is configured for item caption {@link AbstractSelect#ITEM_CAPTION_MODE_PROPERTY} mode, with
098     * {@code displayPropertyName} as the {@linkplain #setItemCaptionPropertyId item caption property}.
099     *
100     * @param container container data source, or null to leave data source unset
101     * @param displayPropertyName container property to display in the combo box
102     * @param allowNull true to allow a null selection, false otherwise
103     */
104    public EnumComboBox(EnumContainer<?> container, String displayPropertyName, boolean allowNull) {
105        if (container != null)
106            this.setContainerDataSource(container);
107        this.setNewItemsAllowed(false);
108        this.setTextInputAllowed(false);
109        this.setFilteringMode(AbstractSelect.Filtering.FILTERINGMODE_OFF);
110        this.setItemCaptionMode(AbstractSelect.ITEM_CAPTION_MODE_PROPERTY);
111        this.setItemCaptionPropertyId(displayPropertyName);
112
113        // Set up whether null selection is allowed
114        this.setNullSelectionAllowed(allowNull);
115        if (!allowNull && !this.getContainerDataSource().getItemIds().isEmpty())
116            this.setValue(this.getContainerDataSource().getItemIds().iterator().next());
117    }
118
119    /**
120     * Set the {@link Enum} type whose instances serve as this instance's data source.
121     *
122     * @throws IllegalArgumentException if {@code enumClass} is null
123     */
124    public <T extends Enum<T>> void setEnumDataSource(Class<T> enumClass) {
125        this.setContainerDataSource(new EnumContainer<T>(enumClass));
126        if (!this.isNullSelectionAllowed() && !this.getContainerDataSource().getItemIds().isEmpty())
127            this.setValue(this.getContainerDataSource().getItemIds().iterator().next());
128    }
129}

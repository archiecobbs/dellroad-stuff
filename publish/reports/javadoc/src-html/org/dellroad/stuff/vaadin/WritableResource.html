006package org.dellroad.stuff.vaadin;
007
008import com.vaadin.terminal.StreamResource;
009
010import java.io.IOException;
011import java.io.InputStream;
012import java.io.OutputStream;
013
014import org.dellroad.stuff.io.NullModemInputStream;
015import org.dellroad.stuff.io.WriteCallback;
016
017/**
018 * Support superclass for {@link StreamResource} implementations that can be more easily implemented
019 * by writing to an {@link OutputStream} than providing an {@link InputStream}.
020 *
021 * @see <a href="http://dev.vaadin.com/ticket/5145">Vaadin Ticket #5145</a>
022 */
023@SuppressWarnings("serial")
024public abstract class WritableResource extends StreamResource implements WriteCallback {
025
026    /**
027     * Constructor.
028     *
029     * @param filename resource file name
030     */
031    protected WritableResource(final String filename) {
032        super(null, filename, ContextApplication.currentApplication());
033        this.setStreamSource(new StreamResource.StreamSource() {
034            @Override
035            public InputStream getStream() {
036                return new NullModemInputStream(WritableResource.this, "WritableResource for \"" + filename + "\"");
037            }
038        });
039    }
040
041    /**
042     * Convenience constructor that also sets the MIME type.
043     *
044     * @param filename resource file name
045     * @param mimeType resource MIME type
046     */
047    protected WritableResource(String filename, String mimeType) {
048        this(filename);
049        this.setMIMEType(mimeType);
050    }
051
052    /**
053     * Write the resource contents to the given {@link OutputStream} and then close it.
054     *
055     * <p>
056     * Note that the {@link com.vaadin.server.VaadinSession} associated with this resource will <b>not</b> be locked
057     * when this method is invoked.
058     * </p>
059     *
060     * @param output resource output stream
061     * @throws IOException if an I/O error occurs
062     */
063    @Override
064    public abstract void writeTo(OutputStream output) throws IOException;
065}

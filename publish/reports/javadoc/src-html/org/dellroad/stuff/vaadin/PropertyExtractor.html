006package org.dellroad.stuff.vaadin;
007
008/**
009 * Classes that can extract {@link com.vaadin.data.Property} values from Java objects.
010 *
011 * @param <T> target object type for extraction
012 */
013public interface PropertyExtractor<T> {
014
015    /**
016     * Read the value of the property defined by {@code propertyDef} from the given object.
017     *
018     * @param <V> property value type
019     * @param obj Java object
020     * @param propertyDef definition of which property to read
021     * @return property value
022     * @throws NullPointerException if either parameter is null
023     */
024    <V> V getPropertyValue(T obj, PropertyDef<V> propertyDef);
025}

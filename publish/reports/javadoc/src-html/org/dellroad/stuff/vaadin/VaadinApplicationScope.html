006package org.dellroad.stuff.vaadin;
007
008import java.util.HashMap;
009
010import org.springframework.beans.factory.ObjectFactory;
011import org.springframework.beans.factory.config.BeanFactoryPostProcessor;
012import org.springframework.beans.factory.config.ConfigurableListableBeanFactory;
013import org.springframework.beans.factory.config.Scope;
014
015/**
016 * A Spring custom {@link Scope} for Vaadin applications.
017 *
018 * <p>
019 * This works for applications that subclass {@link ContextApplication}; objects will be scoped to each
020 * {@link ContextApplication} instance. Spring {@linkplain org.springframework.beans.factory.DisposableBean#destroy destroy-methods}
021 * will be invoked when the {@link ContextApplication} is closed.
022 *
023 * <p>
024 * To enable this scope, simply add this bean to your application context as a singleton (it will register itself):
025 * <blockquote><pre>
026 *  &lt;!-- Enable the "vaadinApplication" custom scope --&gt;
027 *  &lt;bean class="org.dellroad.stuff.vaadin.VaadinApplicationScope"/&gt;
028 * </pre></blockquote>
029 * Then declare scoped beans normally using the scope name {@code "vaadinApplication"}.
030 */
031public class VaadinApplicationScope implements Scope, BeanFactoryPostProcessor, ContextApplication.CloseListener {
032
033    /**
034     * Key to the current application instance. For use by {@link #resolveContextualObject}.
035     */
036    public static final String APPLICATION_KEY = "application";
037
038    /**
039     * The name of this scope (i.e., <code>{@value}</code>).
040     */
041    public static final String SCOPE_NAME = "vaadinApplication";
042
043    private final HashMap<ContextApplication, ApplicationBeanHolder> beanHolders
044      = new HashMap<ContextApplication, ApplicationBeanHolder>();
045
046// BeanFactoryPostProcessor methods
047
048    @Override
049    public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) {
050        beanFactory.registerScope(VaadinApplicationScope.SCOPE_NAME, this);
051    }
052
053// ContextApplication.CloseListener methods
054
055    @Override
056    public void applicationClosed(ContextApplication.CloseEvent closeEvent) {
057        ApplicationBeanHolder beanHolder;
058        synchronized (this) {
059            beanHolder = this.beanHolders.remove(closeEvent.getContextApplication());
060        }
061        if (beanHolder != null)
062            beanHolder.close();
063    }
064
065// Scope methods
066
067    @Override
068    public synchronized Object get(String name, ObjectFactory<?> objectFactory) {
069        return this.getApplicationBeanHolder(true).getBean(name, objectFactory);
070    }
071
072    @Override
073    public synchronized Object remove(String name) {
074        ApplicationBeanHolder beanHolder = this.getApplicationBeanHolder(false);
075        return beanHolder != null ? beanHolder.remove(name) : null;
076    }
077
078    @Override
079    public synchronized void registerDestructionCallback(String name, Runnable callback) {
080        this.getApplicationBeanHolder(true).registerDestructionCallback(name, callback);
081    }
082
083    @Override
084    public String getConversationId() {
085        ContextApplication application = ContextApplication.currentApplication();
086        if (application == null)
087            return null;
088        return application.getClass().getName() + "@" + System.identityHashCode(application);
089    }
090
091    @Override
092    public Object resolveContextualObject(String key) {
093        if (APPLICATION_KEY.equals(key))
094            return ContextApplication.currentApplication();
095        return null;
096    }
097
098// Internal methods
099
100    private synchronized ApplicationBeanHolder getApplicationBeanHolder(boolean create) {
101        ContextApplication application = ContextApplication.get();
102        application.addListener(this);
103        ApplicationBeanHolder beanHolder = this.beanHolders.get(application);
104        if (beanHolder == null && create) {
105            beanHolder = new ApplicationBeanHolder();
106            this.beanHolders.put(application, beanHolder);
107        }
108        return beanHolder;
109    }
110
111// Bean holder class corresponding to a single Application instance
112
113    private static class ApplicationBeanHolder {
114
115        private final HashMap<String, Object> beans = new HashMap<String, Object>();
116        private final HashMap<String, Runnable> destructionCallbacks = new HashMap<String, Runnable>();
117
118        public Object getBean(String name, ObjectFactory<?> objectFactory) {
119            Object bean = this.beans.get(name);
120            if (bean == null) {
121                bean = objectFactory.getObject();
122                this.beans.put(name, bean);
123            }
124            return bean;
125        }
126
127        public Object remove(String name) {
128            this.destructionCallbacks.remove(name);
129            return this.beans.remove(name);
130        }
131
132        public void registerDestructionCallback(String name, Runnable callback) {
133            this.destructionCallbacks.put(name, callback);
134        }
135
136        public void close() {
137            for (Runnable callback : this.destructionCallbacks.values())
138                callback.run();
139            this.beans.clear();
140            this.destructionCallbacks.clear();
141        }
142    }
143}

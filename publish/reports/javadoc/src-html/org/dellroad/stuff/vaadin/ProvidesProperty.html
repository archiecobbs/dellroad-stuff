006package org.dellroad.stuff.vaadin;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Retention;
011import java.lang.annotation.RetentionPolicy;
012import java.lang.annotation.Target;
013
014/**
015 * Declares that a Java method provides a read-only Vaadin {@link com.vaadin.data.Property} value in a Java class
016 * whose instances back the items in a {@link com.vaadin.data.Container}.
017 *
018 * <p>
019 * {@link ProvidesProperty &#64;ProvidesProperty} and {@link ProvidesPropertySort &#64;ProvidesPropertySort} method annotations
020 * can be used to automatically generate a list of {@link PropertyDef}s and a {@link PropertyExtractor} using a
021 * {@link ProvidesPropertyScanner}. This happens automatically when using the appropriate constructors of the various
022 * {@link com.vaadin.data.Container} classes in this package.
023 *
024 * <p>
025 * This annotation indicates that a read-only Vaadin {@link com.vaadin.data.Property} having the {@linkplain #value specified name}
026 * and type derived from the method's return value is accessible by reading that method.
027 * Annotated methods must have zero parameters.
028 *
029 * <p>
030 * For example:
031 * <blockquote><pre>
032 * // Container backing object class
033 * public class User {
034 *
035 *     public static final String USERNAME_PROPERTY = "username";
036 *     public static final String REAL_NAME_PROPERTY = "realName";
037 *
038 *     private String username;
039 *     private String realName;
040 *
041 *     public String getUsername() {
042 *         return this.username;
043 *     }
044 *     public void setUsername(String username) {
045 *         this.username = username;
046 *     }
047 *
048 *     <b>&#64;ProvidesProperty</b>                     // property name "realName" is implied by method name
049 *     public String getRealName() {
050 *         return this.realName;
051 *     }
052 *     public void setRealName(String realName) {
053 *         this.realName = realName;
054 *     }
055 *
056 *     <b>&#64;ProvidesProperty(USERNAME_PROPERTY)</b>  // display usernames in fixed-width font
057 *     private Label usernameProperty() {
058 *         return new Label("&lt;code&gt;" + StringUtil.escapeHtml(this.username) + "&lt;/code&gt;", ContentMode.HTML);
059 *     }
060 * }
061 *
062 * // User container class
063 * public class UserContainer extends SimpleKeyedContainer&lt;String, User&gt; {
064 *
065 *     public UserContainer() {
066 *         super(<b>User.class</b>);
067 *     }
068 *
069 *     &#64;Override
070 *     public String getKeyFor(User user) {
071 *         return user.getUsername();
072 *     }
073 * }
074 *
075 * // Create container holding all users
076 * UserContainer container = new UserContainer();
077 * container.load(this.userDAO.getAllUsers());
078 *
079 * // Build table showing users
080 * Table table = new Table();
081 * table.setColumnHeader(User.USERNAME_PROPERTY, "User");
082 * table.setColumnHeader(User.REAL_NAME_PROPERTY, "Name");
083 * table.setVisibleColumns(User.USERNAME_PROPERTY, User.REAL_NAME_PROPERTY);
084 *
085 * // Select user "jsmith" in the table
086 * table.setValue("jsmith");
087 * ...
088 * </pre></blockquote>
089 *
090 * <p>
091 * Some details regarding {@link ProvidesProperty &#64;ProvidesProperty} annotations on methods:
092 *  <ul>
093 *  <li>Only non-void methods taking zero parameters are supported; {@link ProvidesProperty &#64;ProvidesProperty}
094 *      annotations on other methods are ignored</li>
095 *  <li>Protected, package private, and private methods are supported.</li>
096 *  <li>{@link ProvidesProperty &#64;ProvidesProperty} annotations on interface methods are supported</li>
097 *  <li>If a method and the superclass or superinterface method it overrides are both annotated with
098 *      {@link ProvidesProperty &#64;ProvidesProperty}, then the overridding method's annotation takes precedence.
099 *  <li>If two methods with different names are annotated with {@link ProvidesProperty &#64;ProvidesProperty} for the same
100 *      {@linkplain #value property name}, then the declaration in the class which is a sub-type of the other
101 *      wins (if the two methods are delcared in the same class, an exception is thrown). This allows subclasses
102 *      to "override" which method supplies a given property.</li>
103 *  </ul>
104 *
105 * <p>
106 * To control how properties are sorted (e.g., in tables), see {@link ProvidesPropertySort &#64;ProvidesPropertySort}.
107 *
108 * @see ProvidesPropertySort
109 * @see ProvidesPropertyScanner
110 */
111@Retention(RetentionPolicy.RUNTIME)
112@Target({ ElementType.METHOD, ElementType.FIELD })
113@Documented
114public @interface ProvidesProperty {
115
116    /**
117     * Get the name of the Vaadin property. If this is left unset (empty string), then the
118     * bean property name of the annotated bean property "getter" method is used.
119     */
120    String value() default "";
121}

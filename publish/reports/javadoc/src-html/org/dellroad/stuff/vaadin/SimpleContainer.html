006package org.dellroad.stuff.vaadin;
007
008import java.util.ArrayList;
009import java.util.Collection;
010
011/**
012 * An {@link AbstractSimpleContainer} with {@link Integer} item IDs.
013 *
014 * <p>
015 * Use {@link #load} to (re)load the container.
016 *
017 * @param <T> the type of the Java objects that back each {@link com.vaadin.data.Item} in the container
018 * @see AbstractSimpleContainer
019 */
020@SuppressWarnings("serial")
021public class SimpleContainer<T> extends AbstractSimpleContainer<Integer, T> {
022
023    private ArrayList<T> items = new ArrayList<T>(0);
024
025    /**
026     * Constructor.
027     *
028     * <p>
029     * After using this constructor, subsequent invocations of {@link #setPropertyExtractor setPropertyExtractor()}
030     * and {@link #setProperties setProperties()} are required to define the properties of this container
031     * and how to extract them.
032     */
033    public SimpleContainer() {
034    }
035
036    /**
037     * Constructor.
038     *
039     * <p>
040     * After using this constructor, a subsequent invocation of {@link #setProperties setProperties()} is required
041     * to define the properties of this container.
042     *
043     * @param propertyExtractor used to extract properties from the underlying Java objects;
044     *  may be null but then container is not usable until one is configured via
045     * {@link #setPropertyExtractor setPropertyExtractor()}
046     */
047    public SimpleContainer(PropertyExtractor<? super T> propertyExtractor) {
048        this(propertyExtractor, null);
049    }
050
051    /**
052     * Constructor.
053     *
054     * <p>
055     * After using this constructor, a subsequent invocation of {@link #setPropertyExtractor setPropertyExtractor()} is required
056     * to define how to extract the properties of this container; alternately, subclasses can override
057     * {@link #getPropertyValue getPropertyValue()}.
058     *
059     * @param propertyDefs container property definitions; null is treated like the empty set
060     */
061    protected SimpleContainer(Collection<? extends PropertyDef<?>> propertyDefs) {
062        this(null, propertyDefs);
063    }
064
065    /**
066     * Constructor.
067     *
068     * @param propertyExtractor used to extract properties from the underlying Java objects;
069     *  may be null but then container is not usable until one is configured via
070     * {@link #setPropertyExtractor setPropertyExtractor()}
071     * @param propertyDefs container property definitions; null is treated like the empty set
072     */
073    public SimpleContainer(PropertyExtractor<? super T> propertyExtractor, Collection<? extends PropertyDef<?>> propertyDefs) {
074        super(propertyExtractor, propertyDefs);
075    }
076
077    /**
078     * Constructor.
079     *
080     * <p>
081     * Properties will be determined by the {@link ProvidesProperty &#64;ProvidesProperty} and
082     * {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods in the given class.
083     *
084     * @param type class to introspect for annotated methods
085     * @throws IllegalArgumentException if {@code type} is null
086     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}
087     *  or {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods for the same property
088     * @throws IllegalArgumentException if a {@link ProvidesProperty &#64;ProvidesProperty}-annotated method with no
089     *  {@linkplain ProvidesProperty#value property name specified} has a name which cannot be interpreted as a bean
090     *  property "getter" method
091     * @see ProvidesProperty
092     * @see ProvidesPropertySort
093     * @see ProvidesPropertyScanner
094     */
095    protected SimpleContainer(Class<? super T> type) {
096        super(type);
097    }
098
099    @Override
100    public T getJavaObject(Object itemId) {
101        if (!(itemId instanceof Integer))
102            return null;
103        int index = ((Integer)itemId).intValue();
104        if (index < 0 || index >= this.items.size())
105            return null;
106        return this.items.get(index);
107    }
108
109    @Override
110    protected void resetItemIds() {
111        this.items = new ArrayList<T>();
112    }
113
114    @Override
115    protected Integer generateItemId(T obj) {
116        int itemId = this.items.size();
117        this.items.add(obj);
118        return itemId;
119    }
120}

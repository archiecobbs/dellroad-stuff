028/**
029 * Vaadin application implementation that manages an associated Spring {@link WebApplicationContext}.
030 *
031 * <h3>Overview</h3>
032 *
033 * <p>
034 * Each Vaadin application instance is given its own Spring application context, and all such
035 * application contexts share the same parent context, which is the one associated with the overal servlet web context
036 * (i.e., the one created by Spring's {@link org.springframework.web.context.ContextLoaderListener ContextLoaderListener}).
037 * A context is created when a new Vaadin application instance is initialized, and destroyed when it is closed.
038 *
039 * <p>
040 * This setup is analogous to how Spring's {@link org.springframework.web.servlet.DispatcherServlet DispatcherServlet}
041 * creates per-servlet application contexts that are children of the overall servlet web context.
042 *
043 * <h3>Application Context XML File Location</h3>
044 *
045 * <p>
046 * For each Vaadin application {@code com.example.FooApplication} that subclasses this class, there should exist an XML
047 * file named {@code FooApplication.xml} in the {@code WEB-INF/} directory that defines the per-Vaadin application Spring
048 * application context. This naming scheme {@linkplain #getApplicationName can be overriden}, or explicit config file
049 * location(s) can be specified by setting the {@link #VAADIN_CONTEXT_LOCATION_PARAMETER} servlet parameter.
050 *
051 * <h3>Vaadin Application as BeanFactory singleton</h3>
052 *
053 * <p>
054 * This {@link SpringContextApplication} instance can itself be exposed in, and configured by, the associated Spring
055 * application context. Simply create a bean definition that invokes {@link ContextApplication#get}:
056 * <blockquote><pre>
057 *  &lt;bean id="myVaadinApplication" class="org.dellroad.stuff.vaadin.ContextApplication" factory-method="get"/&gt;
058 * </pre></blockquote>
059 *
060 * <p>
061 * This then allows you to autowire the {@link SpringContextApplication} and other UI components together, e.g.:
062 * <blockquote><pre>
063 *  public class MyApplication extends SpringContextApplication {
064 *
065 *      &#64;Autowired
066 *      private MainPanel mainPanel;
067 *
068 *      &#64;Override
069 *      public void initSpringApplication(ConfigurableWebApplicationContext context) {
070 *          Window mainWindow = new Window("MyApplication", this.mainPanel);
071 *          this.setMainWindow(mainWindow);
072 *      }
073 *
074 *      ...
075 *  }
076 *
077 *  &#64;Component
078 *  public class MainPanel extends VerticalLayout {
079 *
080 *      &#64;Autowired
081 *      private MyApplication application;
082 *
083 *      ...
084 *  }
085 * </pre></blockquote>
086 *
087 * <p>
088 * Even if you don't explicitly define the {@link SpringContextApplication} bean in your Spring application context,
089 * it will still be available as a dependency for autowiring into other beans (this is accomplished using
090 * {@link ConfigurableListableBeanFactory#registerResolvableDependency
091 * ConfigurableListableBeanFactory.registerResolvableDependency()}). Of course, in this case the
092 * {@link SpringContextApplication} bean won't itself be autowired or configured.
093 *
094 * <h3><code>@VaadinConfigurable</code> Beans</h3>
095 *
096 * <p>
097 * It is also possible to configure beans outside of this application context using AOP, so that any invocation of
098 * {@code new FooBar()}, where the class {@code FooBar} is marked {@link VaadinConfigurable @VaadinConfigurable},
099 * will automagically cause the new {@code FooBar} object to be configured by the application context associated with
100 * the {@linkplain ContextApplication#get() currently running application instance}. In effect, this does for
101 * Vaadin application beans what Spring's {@link org.springframework.beans.factory.annotation.Configurable @Configurable}
102 * does for regular beans.
103 *
104 * <p>
105 * Note however that Spring {@linkplain org.springframework.beans.factory.DisposableBean#destroy destroy methods}
106 * will not be invoked on application close for these beans, since their lifecycle is controlled outside of the
107 * Spring application context (this is also the case with
108 * {@link org.springframework.beans.factory.annotation.Configurable @Configurable} beans). Instead, these beans
109 * can register as a {@link ContextApplication.CloseListener} for shutdown notification.
110 *
111 * <p>
112 * For the this annotation to do anything, {@link VaadinConfigurable @VaadinConfigurable} classes must be woven
113 * (either at build time or runtime) using the
114 * <a href="http://www.eclipse.org/aspectj/doc/released/faq.php#compiler">AspectJ compiler</a> with the
115 * {@code VaadinConfigurableAspect} aspect (included in the <code>dellroad-stuff</code> JAR file).
116 *
117 * @see ContextApplication#get
118 * @see <a href="https://github.com/archiecobbs/dellroad-stuff-vaadin-spring-demo3">Example Code on GitHub</a>
119 */
120@SuppressWarnings("serial")
121public abstract class SpringContextApplication extends ContextApplication {
122
123    /**
124     * Optional servlet initialization parameter (<code>{@value #VAADIN_CONTEXT_LOCATION_PARAMETER}</code>) used to specify
125     * the location(s) of the application context XML file(s). Multiple XML files may be separated by whitespace.
126     * If this parameter is not defined, then the XML file location is built using {@code /WEB-INF/} as a prefix,
127     * the string from {@link #getApplicationName}, and {@code .xml} as a suffix.
128     */
129    public static final String VAADIN_CONTEXT_LOCATION_PARAMETER = "vaadinContextConfigLocation";
130
131    private static final AtomicLong UNIQUE_INDEX = new AtomicLong();
132
133    private transient ConfigurableWebApplicationContext context;
134
135    /**
136     * Get this instance's associated Spring application context.
137     */
138    public ConfigurableWebApplicationContext getApplicationContext() {
139        return this.context;
140    }
141
142    /**
143     * Get the {@link SpringContextApplication} instance associated with the current thread or throw an exception if there is none.
144     *
145     * <p>
146     * Works just like {@link ContextApplication#get()} but returns this narrower type.
147     *
148     * @return the {@link SpringContextApplication} associated with the current thread
149     * @throws IllegalStateException if the current thread is not servicing a Vaadin web request
150     *  or the current Vaadin {@link com.vaadin.Application} is not a {@link SpringContextApplication}
151     */
152    public static SpringContextApplication get() {
153        return ContextApplication.get(SpringContextApplication.class);
154    }
155
156    /**
157     * Initializes the associated {@link ConfigurableWebApplicationContext}.
158     *
159     * <p>
160     * After initializing the associated Spring application context, this method delegates to {@link #initSpringApplication}.
161     */
162    @Override
163    protected final void initApplication() {
164
165        // Load the context
166        this.loadContext();
167
168        // Initialize subclass
169        this.initSpringApplication(context);
170    }
171
172    /**
173     * Initialize the application. Sub-classes of {@link SpringContextApplication} must implement this method.
174     *
175     * @param context the associated {@link WebApplicationContext} just created and refreshed
176     * @see #destroySpringApplication
177     */
178    protected abstract void initSpringApplication(ConfigurableWebApplicationContext context);
179
180    /**
181     * Perform any application-specific shutdown work. This will be invoked at shutdown after this Vaadin application and the
182     * associated {@link WebApplicationContext} have both been closed.
183     *
184     * <p>
185     * The implementation in {@link SpringContextApplication} does nothing. Subclasses may override as necessary.
186     *
187     * <p>
188     * Note that if a {@link SpringContextApplication} instance is exposed in the application context and configured
189     * with a Spring {@linkplain org.springframework.beans.factory.DisposableBean#destroy destroy method}, then that
190     * method will also be invoked when the application is closed. In such cases overriding this method is not necessary.
191     *
192     * @see #initSpringApplication
193     */
194    protected void destroySpringApplication() {
195    }
196
197    /**
198     * Post-process the given {@link WebApplicationContext} after initial creation but before the initial
199     * {@link org.springframework.context.ConfigurableApplicationContext#refresh refresh()}.
200     *
201     * <p>
202     * The implementation in {@link SpringContextApplication} does nothing. Subclasses may override as necessary.
203     *
204     * @param context the associated {@link WebApplicationContext} just refreshed
205     * @see #onRefresh
206     * @see ConfigurableWebApplicationContext#refresh()
207     */
208    protected void postProcessWebApplicationContext(ConfigurableWebApplicationContext context) {
209    }
210
211    /**
212     * Perform any application-specific work after a successful application context refresh.
213     *
214     * <p>
215     * The implementation in {@link SpringContextApplication} does nothing. Subclasses may override as necessary.
216     *
217     * @param context the associated {@link WebApplicationContext} just refreshed
218     * @see #postProcessWebApplicationContext
219     * @see org.springframework.context.ConfigurableApplicationContext#refresh
220     */
221    protected void onRefresh(ApplicationContext context) {
222    }
223
224    /**
225     * Get the name for this application. This is used as the name of the XML file in {@code WEB-INF/} that
226     * defines the Spring application context associated with this instance, unless the
227     * {@link #VAADIN_CONTEXT_LOCATION_PARAMETER} servlet parameter is set.
228     *
229     * <p>
230     * The implementation in {@link SpringContextApplication} returns this instance's class'
231     * {@linkplain Class#getSimpleName simple name}.
232     */
233    protected String getApplicationName() {
234        return this.getClass().getSimpleName();
235    }
236
237// ApplicationContext setup
238
239    private void loadContext() {
240
241        // Logging
242        this.log.info("loading application context for Vaadin application " + this.getApplicationName());
243
244        // Sanity check
245        if (this.context != null)
246            throw new IllegalStateException("context already loaded");
247
248        // Find the application context associated with the servlet; it will be the parent
249        String contextPath = ContextApplication.currentRequest().getContextPath() + "/";
250        ServletContext servletContext = ContextApplication.currentRequest().getSession().getServletContext();
251        WebApplicationContext parent = WebApplicationContextUtils.getWebApplicationContext(servletContext);
252
253        // Create and configure a new application context for this Application instance
254        this.context = new XmlWebApplicationContext();
255        this.context.setId(ConfigurableWebApplicationContext.APPLICATION_CONTEXT_ID_PREFIX
256          + contextPath + this.getApplicationName() + "-" + UNIQUE_INDEX.incrementAndGet());
257        if (parent != null)
258            context.setParent(parent);
259        if (servletContext != null)
260            context.setServletContext(servletContext);
261        //context.setServletConfig(??);
262        this.context.setNamespace(this.getApplicationName());
263
264        // Set explicit config location(s) if set by parameter
265        String configLocationValue = this.getProperty(VAADIN_CONTEXT_LOCATION_PARAMETER);
266        if (configLocationValue != null)
267            this.context.setConfigLocation(configLocationValue);
268
269        // Register listener so we can notify subclass on refresh events
270        this.context.addApplicationListener(new SourceFilteringListener(this.context, new RefreshListener()));
271
272        // Register this instance as an implicitly resolvable dependency
273        this.context.addBeanFactoryPostProcessor(new BeanFactoryPostProcessor() {
274            @Override
275            public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) {
276                beanFactory.registerResolvableDependency(Application.class, SpringContextApplication.this);
277            }
278        });
279
280        // Invoke any subclass setup
281        this.postProcessWebApplicationContext(context);
282
283        // Refresh context
284        this.context.refresh();
285
286        // Get notified of application shutdown so we can shut down the context as well
287        this.addListener(new ContextCloseListener());
288    }
289
290// Serialization
291
292    private void readObject(ObjectInputStream input) throws IOException, ClassNotFoundException {
293        input.defaultReadObject();
294        this.loadContext();
295    }
296
297// Nested classes
298
299    // My refresh listener
300    private class RefreshListener implements ApplicationListener<ContextRefreshedEvent>, Serializable {
301        @Override
302        public void onApplicationEvent(ContextRefreshedEvent event) {
303            SpringContextApplication.this.onRefresh(event.getApplicationContext());
304        }
305    }
306
307    // My close listener
308    private class ContextCloseListener implements CloseListener, Serializable {
309        @Override
310        public void applicationClosed(CloseEvent closeEvent) {
311            SpringContextApplication.this.log.info("closing application context associated with Vaadin application "
312              + SpringContextApplication.this.getApplicationName());
313            SpringContextApplication.this.context.close();
314            SpringContextApplication.this.destroySpringApplication();
315        }
316    }

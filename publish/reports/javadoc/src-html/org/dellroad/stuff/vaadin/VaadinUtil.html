006package org.dellroad.stuff.vaadin;
007
008/**
009 * Miscellaneous utility methods.
010 */
011public final class VaadinUtil {
012
013    private VaadinUtil() {
014    }
015
016    /**
017     * Verify that we are running in the context of the given {@link ContextApplication} and that it is locked.
018     * This method can be used by any code that manipulates Vaadin state to assert that the proper Vaadin
019     * locking has been performed.
020     *
021     * @param application the {@link ContextApplication} we are supposed to be running with
022     * @throws IllegalArgumentException if {@code application} is null
023     * @throws IllegalStateException if there is no {@link ContextApplication} associated with the current thread
024     * @throws IllegalStateException if the {@link ContextApplication} associated with the current thread is not {@code application}
025     * @throws IllegalStateException if the {@link ContextApplication} associated with the current thread is not locked
026     */
027    public static void assertApplication(ContextApplication application) {
028        if (application == null)
029            throw new IllegalArgumentException("null application");
030        final ContextApplication currentApplication = ContextApplication.currentApplication();
031        if (currentApplication == null)
032            throw new IllegalStateException("there is no ContextApplication associated with the current thread");
033        if (currentApplication != application) {
034            throw new IllegalStateException("the ContextApplication associated with the current thread " + currentApplication
035              + " is not the same application as the given one " + application);
036        }
037        if (!Thread.holdsLock(application)) {
038            throw new IllegalStateException("the ContextApplication associated with the current thread " + currentApplication
039              + " is not locked by this thread");
040        }
041    }
042}

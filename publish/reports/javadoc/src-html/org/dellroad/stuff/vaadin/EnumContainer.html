006package org.dellroad.stuff.vaadin;
007
008import java.util.List;
009
010import org.dellroad.stuff.java.EnumUtil;
011
012/**
013 * Container backed by the instances of an {@link Enum} type.
014 *
015 * <p>
016 * Instances will auto-detect properties defined by {@link ProvidesProperty &#64;ProvidesProperty} annotations (if any),
017 * and will also pre-define the following properties:
018 *  <ul>
019 *  <li>{@value #NAME_PROPERTY} - {@link String} property derived from {@link Enum#name Enum.name()}</li>
020 *  <li>{@value #VALUE_PROPERTY} - type {@code T} property that returns the {@link Enum} instance value itself</li>
021 *  <li>{@value #ORDINAL_PROPERTY} - {@link Integer} property derived from {@link Enum#ordinal Enum.ordinal()}</li>
022 *  <li>{@value #TO_STRING_PROPERTY} - {@link String} property derived from {@link Enum#toString}</li>
023 *  </ul>
024 * </p>
025 *
026 * @param <T> enum type
027 * @see EnumComboBox
028 */
029@SuppressWarnings("serial")
030public class EnumContainer<T extends Enum<T>> extends SelfKeyedContainer<T> {
031
032    public static final String NAME_PROPERTY = "name";
033    public static final String VALUE_PROPERTY = "value";
034    public static final String ORDINAL_PROPERTY = "ordinal";
035    public static final String TO_STRING_PROPERTY = "toString";
036
037    private static final PropertyDef<String> NAME_PROPERTY_DEF = new PropertyDef<String>(NAME_PROPERTY, String.class);
038    @SuppressWarnings("rawtypes")
039    private static final PropertyDef<Enum> VALUE_PROPERTY_DEF = new PropertyDef<Enum>(VALUE_PROPERTY, Enum.class);
040    private static final PropertyDef<Integer> ORDINAL_PROPERTY_DEF = new PropertyDef<Integer>(ORDINAL_PROPERTY, Integer.class);
041    private static final PropertyDef<String> TO_STRING_PROPERTY_DEF = new PropertyDef<String>(TO_STRING_PROPERTY, String.class);
042
043    /**
044     * Constructor.
045     *
046     * @param type enum type
047     * @throws IllegalArgumentException if {@code type} is null
048     * @throws IllegalArgumentException if {@code type} has a {@link ProvidesProperty &#64;ProvidesProperty}-annotated method
049     *  with no {@linkplain ProvidesProperty#value property name specified} has a name which cannot be interpreted as a bean
050     *  property "getter" method
051     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}-annotated
052     *  fields or methods with the same {@linkplain ProvidesProperty#value property name}
053     */
054    public EnumContainer(final Class<T> type) {
055        super(type);
056        this.setProperty(NAME_PROPERTY_DEF);
057        this.setProperty(VALUE_PROPERTY_DEF);
058        this.setProperty(ORDINAL_PROPERTY_DEF);
059        this.setProperty(TO_STRING_PROPERTY_DEF);
060        this.load(this.getExposedValues(type));
061    }
062
063    /**
064     * Get the {@link Enum} values to expose in this container in the desired order.
065     *
066     * <p>
067     * The implementation in {@link EnumContainer} returns all values in their natural order.
068     * Subclasses may override to filter and/or re-order.
069     * </p>
070     */
071    protected List<T> getExposedValues(Class<T> type) {
072        return EnumUtil.getValues(type);
073    }
074
075    @Override
076    public <V> V getPropertyValue(T value, PropertyDef<V> propertyDef) {
077        if (propertyDef.equals(VALUE_PROPERTY_DEF))
078            return propertyDef.getType().cast(value);
079        if (propertyDef.equals(VALUE_PROPERTY_DEF))
080            return propertyDef.getType().cast(value);
081        if (propertyDef.equals(ORDINAL_PROPERTY_DEF))
082            return propertyDef.getType().cast(value.ordinal());
083        if (propertyDef.equals(TO_STRING_PROPERTY_DEF))
084            return propertyDef.getType().cast(value.toString());
085        return super.getPropertyValue(value, propertyDef);
086    }
087}

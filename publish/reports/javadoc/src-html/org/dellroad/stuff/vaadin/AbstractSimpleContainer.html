024/**
025 * Support superclass for simple read-only, in-memory {@link Container} implementations where each
026 * {@link Item} in the container is backed by a Java object.
027 *
028 * <p>
029 * This {@link Container}'s {@link Property}s are defined via {@link PropertyDef}s, and a {@link PropertyExtractor}
030 * is used to actually extract the property values from each underlying object (alternately, subclasses can override
031 * {@link #getPropertyValue getPropertyValue()}). However, the easist way to configure the container {@link Property}s
032 * is to pass a {@link ProvidesProperty &#64;ProvidesProperty}-annotated Java class to the {@link #AbstractSimpleContainer(Class)}
033 * constructor.
034 *
035 * <p>
036 * Use {@link #load load()} to load or reload the container.
037 *
038 * @param <I> the item ID type
039 * @param <T> the type of the Java objects that back each {@link Item} in the container
040 *
041 * @see SimpleItem
042 * @see SimpleProperty
043 */
044@SuppressWarnings("serial")
045public abstract class AbstractSimpleContainer<I, T> extends AbstractInMemoryContainer<I, String, SimpleItem<T>>
046  implements PropertyExtractor<T>, Container.Filterable, Container.SimpleFilterable, Container.Sortable {
047
048    private final HashMap<String, PropertyDef<?>> propertyMap = new HashMap<String, PropertyDef<?>>();
049    private PropertyExtractor<? super T> propertyExtractor;
050
051// Constructors
052
053    /**
054     * Constructor.
055     *
056     * <p>
057     * After using this constructor, subsequent invocations of {@link #setPropertyExtractor setPropertyExtractor()}
058     * and {@link #setProperties setProperties()} are required to define the properties of this container
059     * and how to extract them.
060     */
061    protected AbstractSimpleContainer() {
062        this((PropertyExtractor<? super T>)null);
063    }
064
065    /**
066     * Constructor.
067     *
068     * <p>
069     * After using this constructor, a subsequent invocation of {@link #setProperties setProperties()} is required
070     * to define the properties of this container.
071     *
072     * @param propertyExtractor used to extract properties from the underlying Java objects;
073     *  may be null but then container is not usable until one is configured via
074     * {@link #setPropertyExtractor setPropertyExtractor()}
075     */
076    protected AbstractSimpleContainer(PropertyExtractor<? super T> propertyExtractor) {
077        this(propertyExtractor, null);
078    }

080    /**
081     * Constructor.
082     *
083     * <p>
084     * After using this constructor, a subsequent invocation of {@link #setPropertyExtractor setPropertyExtractor()} is required
085     * to define how to extract the properties of this container; alternately, subclasses can override
086     * {@link #getPropertyValue getPropertyValue()}.
087     *
088     * @param propertyDefs container property definitions; null is treated like the empty set
089     */
090    protected AbstractSimpleContainer(Collection<? extends PropertyDef<?>> propertyDefs) {
091        this(null, propertyDefs);
092    }
093
094    /**
095     * Constructor.
096     *
097     * @param propertyExtractor used to extract properties from the underlying Java objects;
098     *  may be null but then container is not usable until one is configured via
099     * {@link #setPropertyExtractor setPropertyExtractor()}
100     * @param propertyDefs container property definitions; null is treated like the empty set
101     */
102    protected AbstractSimpleContainer(PropertyExtractor<? super T> propertyExtractor,
103      Collection<? extends PropertyDef<?>> propertyDefs) {
104        this.setItemSorter(new SimpleItemSorter());
105        this.setPropertyExtractor(propertyExtractor);
106        this.setProperties(propertyDefs);
107    }
108
109    /**
110     * Constructor.
111     *
112     * <p>
113     * Properties will be determined by the {@link ProvidesProperty &#64;ProvidesProperty} and
114     * {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods in the given class.
115     *
116     * @param type class to introspect for annotated methods
117     * @throws IllegalArgumentException if {@code type} is null
118     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}
119     *  or {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods for the same property
120     * @throws IllegalArgumentException if a {@link ProvidesProperty &#64;ProvidesProperty}-annotated method with no
121     *  {@linkplain ProvidesProperty#value property name specified} has a name which cannot be interpreted as a bean
122     *  property "getter" method
123     * @see ProvidesProperty
124     * @see ProvidesPropertySort
125     * @see ProvidesPropertyScanner
126     */
127    @SuppressWarnings({ "unchecked", "rawtypes" })
128    protected AbstractSimpleContainer(Class<? super T> type) {
129        // Why the JLS forces this stupid cast:
130        //  http://stackoverflow.com/questions/4902723/why-cant-a-java-type-parameter-have-a-lower-bound
131        final ProvidesPropertyScanner<? super T> propertyReader
132          = (ProvidesPropertyScanner<? super T>)new ProvidesPropertyScanner(type);
133        this.setItemSorter(new SimpleItemSorter());
134        this.setPropertyExtractor(propertyReader.getPropertyExtractor());
135        this.setProperties(propertyReader.getPropertyDefs());
136    }

160    /**
161     * Read the value of the property defined by {@code propertyDef} from the given object.
162     *
163     * <p>
164     * The implementation in {@link AbstractSimpleContainer} just delegates to the {@linkplain #setPropertyExtractor configured}
165     * {@link PropertyExtractor}; subclasses may override to customize property extraction.
166     *
167     * @param obj Java object
168     * @param propertyDef definition of which property to read
169     * @throws NullPointerException if either parameter is null
170     * @throws IllegalStateException if no {@link PropertyExtractor} is configured for this container
171     */
172    @Override
173    public <V> V getPropertyValue(T obj, PropertyDef<V> propertyDef) {
174        if (this.propertyExtractor == null)
175            throw new IllegalStateException("no PropertyExtractor is configured for this container");
176        return this.propertyExtractor.getPropertyValue(obj, propertyDef);
177    }
178
179    /**
180     * Change the configured properties of this container.
181     *
182     * @param propertyDefs container property definitions; null is treated like the empty set
183     * @throws IllegalArgumentException if {@code propertyDefs} contains a property with a duplicate name
184     */
185    public void setProperties(Collection<? extends PropertyDef<?>> propertyDefs) {
186        if (propertyDefs == null)
187            propertyDefs = Collections.<PropertyDef<?>>emptySet();
188        this.propertyMap.clear();
189        for (PropertyDef<?> propertyDef : propertyDefs) {
190            if (this.propertyMap.put(propertyDef.getName(), propertyDef) != null)
191                throw new IllegalArgumentException("duplicate property name `" + propertyDef.getName() + "'");
192        }
193        this.fireContainerPropertySetChange();
194    }
195
196    /**
197     * Add or replace a configured property of this container.
198     *
199     * @param propertyDef new container property definitions
200     * @throws IllegalArgumentException if {@code propertyDef} is null
201     */
202    public void setProperty(PropertyDef<?> propertyDef) {
203        if (propertyDef == null)
204            throw new IllegalArgumentException("null propertyDef");
205        this.propertyMap.put(propertyDef.getName(), propertyDef);
206        this.fireContainerPropertySetChange();
207    }

209    /**
210     * Change this container's contents.
211     *
212     * @param contents new container contents
213     * @throws IllegalArgumentException if {@code contents} or any item in {@code contents} is null
214     */
215    public void load(Iterable<? extends T> contents) {
216
217        // Sanity check
218        if (contents == null)
219            throw new IllegalArgumentException("null contents");
220
221        // Reset item IDs
222        this.resetItemIds();
223        this.internalRemoveAllItems();
224
225        // Bulk load and register items with id's 0, 1, 2, ...
226        int index = 0;
227        for (T obj : contents) {
228            if (obj == null)
229                throw new IllegalArgumentException("null item in contents at index " + index);
230            SimpleItem<T> item = new SimpleItem<T>(obj, this.propertyMap, this);
231            this.internalAddItemAtEnd(this.generateItemId(obj), item, false);
232            index++;
233        }
234
235        // Apply filters
236        this.filterAll();
237
238        // Notify subclass
239        this.afterReload();
240
241        // Fire event
242        this.fireItemSetChange();
243    }

245    /**
246     * Get the container item ID corresponding to the given underlying Java object which is wrapped by this container.
247     * Objects are tested for equality using {@link Object#equals Object.equals()}.
248     *
249     * <p>
250     * The implementation in {@link AbstractSimpleContainer} requires a linear search of the container.
251     * Some subclasses may provide a more efficient implementation.
252     *
253     * <p>
254     * This method is not used by this class but is defined as a convenience for subclasses.
255     *
256     * <p>
257     * Note: items that are filtered out will not be found.
258     *
259     * @param obj underlying container object
260     * @return item ID corresponding to {@code object}, or null if {@code object} is not found in this container
261     * @throws IllegalArgumentException if {@code object} is null
262     * @see #getItemIdForSame
263     */
264    public I getItemIdFor(T obj) {
265        if (obj == null)
266            throw new IllegalArgumentException("null object");
267        for (I itemId : this.getItemIds()) {
268            T candidate = this.getJavaObject(itemId);
269            if (obj.equals(candidate))
270                return itemId;
271        }
272        return null;
273    }
274
275    /**
276     * Get the container item ID corresponding to the given underlying Java object which is wrapped by this container.
277     * Objects are tested for equality using object equality, not {@link Object#equals Object.equals()}.
278     *
279     * <p>
280     * The implementation in {@link AbstractSimpleContainer} requires a linear search of the container.
281     * Some subclasses may provide a more efficient implementation.
282     *
283     * <p>
284     * This method is not used by this class but is defined as a convenience for subclasses.
285     *
286     * <p>
287     * Note: items that are filtered out will not be found.
288     *
289     * @param obj underlying container object
290     * @return item ID corresponding to {@code object}, or null if {@code object} is not found in this container
291     * @throws IllegalArgumentException if {@code object} is null
292     * @see #getItemIdFor
293     */
294    public I getItemIdForSame(T obj) {
295        if (obj == null)
296            throw new IllegalArgumentException("null object");
297        for (I itemId : this.getItemIds()) {
298            T candidate = this.getJavaObject(itemId);
299            if (obj == candidate)
300                return itemId;
301        }
302        return null;
303    }

416    @Override
417    public Collection<?> getSortableContainerPropertyIds() {
418        final ArrayList<String> propertyIds = new ArrayList<String>(this.propertyMap.size());
419        for (Map.Entry<String, PropertyDef<?>> entry : this.propertyMap.entrySet()) {
420            if (this.propertyExtractor instanceof SortingPropertyExtractor) {
421                final SortingPropertyExtractor<? super T> sortingPropertyExtractor
422                  = (SortingPropertyExtractor<? super T>)this.propertyExtractor;
423                if (sortingPropertyExtractor.canSort(entry.getValue())) {
424                    propertyIds.add(entry.getKey());
425                    continue;
426                }
427            }
428            if (entry.getValue().isSortable())
429                propertyIds.add(entry.getKey());
430        }
431        return propertyIds;
432    }

436    /**
437     * {@link com.vaadin.data.util.ItemSorter} implementation used by {@link AbstractSimpleContainer}.
438     */
439    private class SimpleItemSorter extends DefaultItemSorter {
440
441        @Override
442        @SuppressWarnings("unchecked")
443        protected int compareProperty(Object propertyId, boolean ascending, Item item1, Item item2) {
444
445            // Get property definition
446            final PropertyDef<?> propertyDef = AbstractSimpleContainer.this.propertyMap.get(propertyId);
447            if (propertyDef == null)
448                return super.compareProperty(propertyId, ascending, item1, item2);
449
450            // Ask SortingPropertyExtractor if we have one
451            if (AbstractSimpleContainer.this.propertyExtractor instanceof SortingPropertyExtractor) {
452                final SortingPropertyExtractor<? super T> sortingPropertyExtractor
453                  = (SortingPropertyExtractor<? super T>)AbstractSimpleContainer.this.propertyExtractor;
454                if (sortingPropertyExtractor.canSort(propertyDef)) {
455                    final T obj1 = ((SimpleItem<T>)item1).getObject();
456                    final T obj2 = ((SimpleItem<T>)item2).getObject();
457                    final int diff = sortingPropertyExtractor.sort(propertyDef, obj1, obj2);
458                    return ascending ? diff : -diff;
459                }
460            }
461
462            // Ask property definition
463            if (propertyDef.isSortable()) {
464                final int diff = this.sort(propertyDef, item1, item2);
465                return ascending ? diff : -diff;
466            }
467
468            // Defer to superclass
469            return super.compareProperty(propertyId, ascending, item1, item2);
470        }
471
472        // This method exists only to allow the generic parameter <V> to be bound
473        private <V> int sort(PropertyDef<V> propertyDef, Item item1, Item item2) {
474            return propertyDef.sort(propertyDef.read(item1), propertyDef.read(item2));
475        }

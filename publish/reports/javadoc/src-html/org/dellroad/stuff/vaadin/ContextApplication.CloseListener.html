032/**
033 * {@link Application} subclass that provides some basic infrastructure for Vaadin applications:
034 * <ul>
035 *  <li>Access to the currently running Vaadin application (via a {@link ContextApplication#get()})</li>
036 *  <li>A way to safely interact with a Vaadin application from a background thread (via {@link ContextApplication#invoke})</li>
037 *  <li>Support for Vaadin {@linkplain ContextApplication#addListener application close event notifications}</li>
038 *  <li>Displays any exceptions thrown in an overlay error window</li>
039 *  <li>A {@link Logger} to use</li>
040 * </ul>
041 *
042 * @since 1.0.134
043 */
044@SuppressWarnings("serial")
045public abstract class ContextApplication extends Application implements Executor, HttpServletRequestListener {
046
047    /**
048     * Default notification linger time for error notifications (in milliseconds): Value is {@value}ms.
049     */
050    public static final int DEFAULT_NOTIFICATION_DELAY = 30000;

052    private static final ThreadLocal<ContextApplication> CURRENT_APPLICATION = new ThreadLocal<ContextApplication>();
053    private static final ThreadLocal<HttpServletRequest> CURRENT_REQUEST = new ThreadLocal<HttpServletRequest>();
054    private static final ThreadLocal<HttpServletResponse> CURRENT_RESPONSE = new ThreadLocal<HttpServletResponse>();
055
056    protected final Logger log = LoggerFactory.getLogger(this.getClass());
057
058    private final HashSet<CloseListener> closeListeners = new HashSet<CloseListener>();

062// Initialization

064    /**
065     * Initialize the application.
066     *
067     * <p>
068     * The implementation in {@link ContextApplication} delegates to {@link #initApplication}.
069     * </p>
070     */
071    @Override
072    public final synchronized void init() {             // synchronized to work around http://dev.vaadin.com/ticket/9946
073
074        // Set current application
075        CURRENT_APPLICATION.set(this);
076
077        // Create executor service
078        this.executorService = Executors.newSingleThreadExecutor();
079
080        // Initialize application
081        boolean initialized = false;
082        try {
083            this.initApplication();
084            initialized = true;
085        } finally {
086            if (!initialized)
087                this.shutdownExecutorService();
088        }
089    }

098    /**
099     * Handle an uncaugt exception thrown by a Vaadin HTTP request.
100     *
101     * <p>
102     * The implementation in {@link ContextApplication} logs the error and displays in on the
103     * user's screen via {@link #showError(String, Throwable)}.
104     */
105    @Override
106    public void terminalError(Terminal.ErrorEvent event) {
107
108        // Delegate to superclass
109        super.terminalError(event);
110
111        // Get exception; ignore client "hangups"
112        final Throwable t = event.getThrowable();
113        if (t instanceof SocketException)
114            return;
115
116        // Notify user and log it
117        this.showError("Internal Error", "" + t);
118        this.log.error("error within Vaadin operation", t);
119    }
120
121    /**
122     * Display an error message to the user.
123     */
124    public void showError(String title, String description) {
125
126        // Get window
127        Window window = this.getMainWindow();
128        if (window == null)
129            return;
130
131        // Show error
132        Window.Notification notification = new Window.Notification(title, description, Window.Notification.TYPE_ERROR_MESSAGE);
133        notification.setStyleName("warning");
134        notification.setDelayMsec(this.getNotificationDelay());
135        window.showNotification(notification);
136    }
137
138    /**
139     * Display an error message to the user caused by an exception.
140     */
141    public void showError(String title, Throwable t) {
142        for (int i = 0; i < 100 && t.getCause() != null; i++)
143            t = t.getCause();
144        this.showError(title, this.getErrorMessage(t));
145    }

166    /**
167     * Set this instance as the "current application" (if not set already) while invoking the given callback.
168     *
169     * <p>
170     * This method is useful for situations in which non-Vaadin threads need to call into Vaadin code
171     * that expects to successfully retrieve the current application via {@link #currentApplication}.
172     * </p>
173     *
174     * <p>
175     * This method also synchronizes on this {@link ContextApplication} instance as required by Vaadin for thread safety.
176     * </p>
177     *
178     * @param action action to perform
179     * @throws IllegalStateException if a different {@link ContextApplication} is already set as the current application
180     *  associated with the current thread
181     * @see #invokeLater invokeLater()
182     */
183    public void invoke(Runnable action) {
184        final ContextApplication previous = ContextApplication.CURRENT_APPLICATION.get();
185        if (previous != null && previous != this)
186            throw new IllegalStateException("there is already a current application for this thread");
187        ContextApplication.CURRENT_APPLICATION.set(this);
188        try {
189            synchronized (this) {
190                action.run();
191            }
192        } finally {
193            ContextApplication.CURRENT_APPLICATION.set(previous);
194        }
195    }
196
197    /**
198     * Set this instance as the "current application" and invoke the given callback from within another thread.
199     *
200     * <p>
201     * This method functions like {@link #invoke invoke()} except that {@code action} will be invoked from within
202     * a separate thread dedicated to this application instance. This is useful to reduce Vaadin application lock
203     * contention, by performing Vaadin-related actions in a separate, dedicated thread. Actions are executed
204     * in the order they are given to this method.
205     * </p>
206     *
207     * <p>
208     * The returned {@link Future}'s {@link Future#get get()} method will return null upon successful completion.
209     * This method itself always returns immediately.
210     * </p>
211     *
212     * @param action action to perform
213     * @return a {@link Future} representing the pending results of {@code action}
214     * @throws IllegalStateException if this instance is not initialized or has been closed
215     * @see #invoke invoke()
216     */
217    public Future<?> invokeLater(Runnable action) {
218        ExecutorService executor = this.executorService;
219        if (executor == null)
220            throw new IllegalStateException("application instance is either not initialized or already closed");
221        return executor.submit(new LaterRunnable(action));
222    }

224    /**
225     * {@link Executor} interface method, equivalent to {@link #invoke invoke()}.
226     *
227     * @see #invoke invoke()
228     */
229    public void execute(Runnable action) {
230        this.invoke(action);
231    }

233    /**
234     * Handle the start of a request.
235     *
236     * <p>
237     * The implementation in {@link ContextApplication} delegates to {@link #doOnRequestStart}.
238     * </p>
239     */
240    @Override
241    public final void onRequestStart(HttpServletRequest request, HttpServletResponse response) {
242        ContextApplication.CURRENT_APPLICATION.set(this);
243        ContextApplication.CURRENT_REQUEST.set(request);
244        ContextApplication.CURRENT_RESPONSE.set(response);
245        this.doOnRequestStart(request, response);
246    }
247
248    /**
249     * Handle the end of a request.
250     *
251     * <p>
252     * The implementation in {@link ContextApplication} delegates to {@link #doOnRequestEnd}.
253     * </p>
254     */
255    @Override
256    public final void onRequestEnd(HttpServletRequest request, HttpServletResponse response) {
257        try {
258            this.doOnRequestEnd(request, response);
259        } finally {
260            ContextApplication.CURRENT_APPLICATION.remove();
261            ContextApplication.CURRENT_REQUEST.remove();
262            ContextApplication.CURRENT_RESPONSE.remove();
263        }
264    }

384    /**
385     * Close this instance.
386     *
387     * <p>
388     * The implementation in {@link ContextApplication} first delegates to the superclass and then
389     * notifies any registered {@link CloseListener}s.
390     * </p>
391     */
392    @Override
393    public void close() {
394
395        // Invoke superclass
396        super.close();
397
398        // Notify listeners
399        CloseEvent closeEvent = new CloseEvent(this);
400        for (CloseListener closeListener : this.getCloseListeners()) {
401            try {
402                closeListener.applicationClosed(closeEvent);
403            } catch (ThreadDeath t) {
404                throw t;
405            } catch (Throwable t) {
406                this.log.error("exception thrown by CloseListener " + closeListener, t);
407            }
408        }
409
410        // Shutdown ExecutorService
411        this.shutdownExecutorService();
412    }
413
414    private void shutdownExecutorService() {
415
416        // Already shutdown?
417        if (this.executorService == null)
418            return;
419
420        // Shut it down; waiting up to 1 second to finish
421        this.executorService.shutdown();
422        boolean terminated = false;
423        try {
424            terminated = this.executorService.awaitTermination(1, TimeUnit.SECONDS);
425        } catch (InterruptedException e) {
426            Thread.currentThread().interrupt();
427        }
428
429        // Log warnings if it didn't finish
430        if (!terminated) {
431            this.log.warn("forcibly terminating outstanding tasks for closed Vaadin application " + this);
432            List<Runnable> list = this.executorService.shutdownNow();
433            if (!list.isEmpty())
434                this.log.warn(list.size() + " outstanding task(s) remain for closed Vaadin application " + this + ": " + list);
435        }
436
437        // Done
438        this.executorService = null;
439    }

441    /**
442     * Add a {@link CloseListener} to be notified when this instance is closed.
443     */
444    public void addListener(CloseListener listener) {
445        synchronized (this.closeListeners) {
446            this.closeListeners.add(listener);
447        }
448    }
449
450    /**
451     * Remove a {@link CloseListener}.
452     */
453    public void removeListener(CloseListener listener) {
454        synchronized (this.closeListeners) {
455            this.closeListeners.remove(listener);
456        }
457    }
458
459    private HashSet<CloseListener> getCloseListeners() {
460        synchronized (this.closeListeners) {
461            return new HashSet<CloseListener>(this.closeListeners);
462        }
463    }

484            super(application);
485        }

495    // Used by invokeLater()
496    private class LaterRunnable implements Runnable {
497
498        private final Runnable action;
499
500        LaterRunnable(Runnable action) {
501            this.action = action;
502        }
503
504        @Override
505        public void run() {
506            try {
507                ContextApplication.this.invoke(this.action);
508            } catch (ThreadDeath t) {
509                throw t;
510            } catch (Throwable e) {
511                ContextApplication.this.log.error("exception thrown by invokeLater() action " + this.action, e);
512            }
513        }
514
515        @Override
516        public String toString() {
517            return this.action.toString();
518        }
519    }

523    private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
524        in.defaultReadObject();
525        if (in.readBoolean())
526            this.executorService = Executors.newSingleThreadExecutor();
527    }
528
529    private void writeObject(ObjectOutputStream out) throws IOException {
530        out.defaultWriteObject();
531        out.writeBoolean(this.executorService != null);
532    }

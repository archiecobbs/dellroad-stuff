006package org.dellroad.stuff.vaadin;
007
008import org.springframework.context.ApplicationEvent;
009import org.springframework.context.event.ApplicationEventMulticaster;
010import org.springframework.context.event.SmartApplicationListener;
011
012/**
013 * A Spring {@link org.springframework.context.ApplicationListener} support superclass customized for use by
014 * listeners that are part of a Vaadin application when listening to non-Vaadin application event sources.
015 *
016 * <p>
017 * Listeners that are part of a Vaadin application should use this superclass if they are going to be registered
018 * with non-Vaadin event multicasters. This will ensure that events are delivered {@linkplain ContextApplication#invoke
019 * in the proper Vaadin application context} and memory leaks are avoided.
020 * </p>
021 *
022 * @param <E> The type of the event
023 * @see VaadinExternalListener
024 * @see ContextApplication#invoke
025 * @see VaadinApplicationScope
026 * @see SpringContextApplication
027 */
028public abstract class VaadinApplicationListener<E extends ApplicationEvent>
029  extends VaadinExternalListener<ApplicationEventMulticaster> implements SmartApplicationListener {
030
031    private final Class<E> eventType;
032
033    /**
034     * Convenience constructor. Equivalent to:
035     * <blockquote>
036     *  {@link #VaadinApplicationListener(ApplicationEventMulticaster, Class, ContextApplication)
037     *      VaadinApplicationListener(multicaster, eventType, ContextApplication.get())}
038     * </blockquote>
039     *
040     * @param multicaster the application event multicaster on which this listener will register
041     * @param eventType type of events this instance should receive (others will be ignored)
042     * @throws IllegalArgumentException if {@code eventType} is null
043     * @throws IllegalStateException if there is no {@link ContextApplication} associated with the current thread
044     */
045    public VaadinApplicationListener(ApplicationEventMulticaster multicaster, Class<E> eventType) {
046        this(multicaster, eventType, ContextApplication.get());
047    }
048
049    /**
050     * Primary constructor. This results in this instance being registered as a listener on {@code multicaster}.
051     *
052     * @param multicaster the application event multicaster on which this listener will register
053     * @param eventType type of events this instance should receive (others will be ignored)
054     * @param application the associated Vaadin application
055     * @throws IllegalArgumentException if either parameter is null
056     */
057    public VaadinApplicationListener(ApplicationEventMulticaster multicaster, Class<E> eventType, ContextApplication application) {
058        super(multicaster, application);
059        if (eventType == null)
060            throw new IllegalArgumentException("null eventType");
061        this.eventType = eventType;
062    }
063
064    /**
065     * Get the event type that this listener listens for.
066     */
067    public final Class<E> getEventType() {
068        return this.eventType;
069    }
070
071    @Override
072    protected void register(ApplicationEventMulticaster eventSource) {
073        this.getEventSource().addApplicationListener(this);
074    }
075
076    @Override
077    protected void unregister(ApplicationEventMulticaster eventSource) {
078        this.getEventSource().removeApplicationListener(this);
079    }
080
081    @Override
082    public final void onApplicationEvent(ApplicationEvent event) {
083        E castEvent;
084        try {
085            castEvent = this.eventType.cast(event);
086        } catch (ClassCastException e) {
087            // should not happen
088            return;
089        }
090        final E castEvent2 = castEvent;
091        this.handleEvent(new Runnable() {
092            @Override
093            public void run() {
094                VaadinApplicationListener.this.onApplicationEventInternal(castEvent2);
095            }
096        });
097    }
098
099    /**
100     * Handle a listener event. When this method is invoked, it will already be within the context
101     * of the {@link ContextApplication} with which this listener is associated.
102     * The current {@link ContextApplication} is also available via {@link ContextApplication#get}.
103     *
104     * @see ContextApplication#get
105     */
106    protected abstract void onApplicationEventInternal(E event);
107
108    /**
109     * Determine whether this listener actually supports the given event type.
110     *
111     * <p>
112     * The implementation in {@link VaadinApplicationListener} tests whether {@code eventType}
113     * is assignable to the type given in the constructor. Subclasses may override as desired.
114     */
115    @Override
116    public boolean supportsEventType(Class<? extends ApplicationEvent> eventType) {
117        return this.eventType.isAssignableFrom(eventType);
118    }
119
120    /**
121     * Determine whether this listener actually supports the given source type.
122     *
123     * <p>
124     * The implementation in {@link VaadinApplicationListener} always returns true. Subclasses may override as desired.
125     */
126    @Override
127    public boolean supportsSourceType(Class<?> sourceType) {
128        return true;
129    }
130
131    /**
132     * Get ordering value.
133     *
134     * <p>
135     * The implementation in {@link VaadinApplicationListener} always returns zero. Subclasses may override as desired.
136     */
137    @Override
138    public int getOrder() {
139        return 0;
140    }
141}

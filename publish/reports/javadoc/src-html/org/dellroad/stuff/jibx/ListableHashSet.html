006package org.dellroad.stuff.jibx;
007
008import java.util.Collection;
009import java.util.LinkedHashSet;
010
011import org.jibx.runtime.JiBXException;
012
013/**
014 * {@link java.util.Set} implementation with these properties which make it suitable for use with JiBX:
015 * <ul>
016 * <li>Iteration order reflects addition order (this property is inherited from {@link LinkedHashSet})</li>
017 * <li>An {@link #addUnique} method that throws {@link JiBXException} if the item is already in the set
018 *  (suitable for use as a JiBX {@code add-method})</li>
019 * </ul>
020 *
021 * @since 1.0.64
022 */
023@SuppressWarnings("serial")
024public class ListableHashSet<E> extends LinkedHashSet<E> {
025
026    public ListableHashSet() {
027    }
028
029    public ListableHashSet(Collection<? extends E> c) {
030        super(c);
031    }
032
033    public ListableHashSet(int initialCapacity) {
034        super(initialCapacity);
035    }
036
037    public ListableHashSet(int initialCapacity, float loadFactor) {
038        super(initialCapacity, loadFactor);
039    }
040
041    /**
042     * Add an item to a set while verifying that the item is not already in the set.
043     *
044     * @throws JiBXException if item is already in the set
045     */
046    public void addUnique(E item) throws JiBXException {
047        if (this.contains(item))
048            throw new JiBXException("duplicate item in set: " + item);
049        this.add(item);
050    }
051}

037    /**
038     * Read in an object encoded as XML.
039     * This method assumes there is exactly one binding for the given class.
040     *
041     * <p>
042     * This method runs within a new invocation of {@link IdGenerator#run(Callable) IdGenerator.run()} to support object references
043     * (see {@link IdMapper}).
044     *
045     * <p>
046     * The {@code input} is not closed by this method.
047     *
048     * @param <T> type of object to read
049     * @param targetClass target class
050     * @param input source for the XML document
051     * @return decoded object
052     * @throws JiBXException if there is a JiBX parse error
053     * @throws IOException if an error occurs reading from {@code input}
054     */
055    public static <T> T readObject(Class<T> targetClass, InputStream input) throws JiBXException, IOException {
056        return JiBXUtil.readObject(targetClass, null, input);
057    }

059    /**
060     * Read in an object encoded as XML.
061     *
062     * <p>
063     * This method runs within a new invocation of {@link IdGenerator#run(Callable) IdGenerator.run()} to support object references
064     * (see {@link IdMapper}).
065     *
066     * <p>
067     * The {@code input} is not closed by this method.
068     *
069     * @param <T> type of object to read
070     * @param targetClass target class
071     * @param bindingName binding name, or null to choose the only one
072     * @param input source for the XML document
073     * @return decoded object
074     * @throws JiBXException if there is a JiBX parse error
075     * @throws IOException if an error occurs reading from {@code input}
076     */
077    public static <T> T readObject(final Class<T> targetClass, String bindingName, final InputStream input)
078      throws JiBXException, IOException {
079        IBindingFactory bindingFactory = bindingName != null ?
080          BindingDirectory.getFactory(bindingName, targetClass) : BindingDirectory.getFactory(targetClass);
081        final IUnmarshallingContext unmarshallingContext = bindingFactory.createUnmarshallingContext();
082        try {
083            return IdGenerator.run(new Callable<T>() {
084                @Override
085                public T call() throws Exception {
086                    return targetClass.cast(unmarshallingContext.unmarshalDocument(input, null));
087                }
088            });
089        } catch (Exception e) {
090            JiBXUtil.unwrapException(e);
091            return null;            // not reached
092        }
093    }

174     *
175     * <p>
176     * This method runs within a new invocation of {@link IdGenerator#run(Callable) IdGenerator.run()} to support object references
177     * (see {@link IdMapper}).
178     *
179     * @param <T> type of object to read
180     * @param targetClass target class
181     * @param url source for the XML document
182     * @param bindingName binding name, or null to choose the only one
183     * @return decoded object
184     * @throws JiBXException if there is a JiBX parse error
185     * @throws IOException if an error occurs reading the referenced document

187    public static <T> T readObject(Class<T> targetClass, String bindingName, URL url) throws JiBXException, IOException {
188        InputStream in = url.openStream();
189        try {
190            return JiBXUtil.readObject(targetClass, bindingName, in);
191        } finally {
192            try {
193                in.close();
194            } catch (IOException e) {
195                // ignore
196            }
197        }
198    }

254    /**
255     * Write out the given instance encoded as an XML document with "UTF-8" as the declared encoding.
256     *
257     * <p>
258     * This method runs within a new invocation of {@link IdGenerator#run(Callable) IdGenerator.run()} to support object references
259     * (see {@link IdMapper}).
260     *
261     * @param obj object to write
262     * @param bindingName binding name, or null to choose the only one
263     * @param writer output destination; will <b>not</b> be closed by this method
264     * @throws JiBXException if there is a JiBX encoding error
265     * @throws IOException if an error occurs writing to {@code writer}
266     */
267    public static void writeObject(final Object obj, String bindingName, final Writer writer) throws JiBXException, IOException {
268        IBindingFactory bindingFactory = bindingName != null ?
269          BindingDirectory.getFactory(bindingName, obj.getClass()) : BindingDirectory.getFactory(obj.getClass());
270        final IMarshallingContext marshallingContext = bindingFactory.createMarshallingContext();
271        try {
272            IdGenerator.run(new Callable<Void>() {
273                @Override
274                public Void call() throws Exception {
275                    BufferedWriter bufferedWriter = new BufferedWriter(writer);
276                    marshallingContext.setIndent(4);
277                    marshallingContext.setOutput(bufferedWriter);
278                    marshallingContext.startDocument(XML_ENCODING, null);
279                    ((IMarshallable)obj).marshal(marshallingContext);
280                    marshallingContext.getXmlWriter().flush();
281                    bufferedWriter.newLine();
282                    bufferedWriter.flush();
283                    return null;
284                }
285            });
286        } catch (Exception e) {
287            JiBXUtil.unwrapException(e);
288        }
289    }

291    /**
292     * Encode the given instance as an XML document and return it as a {@link String}.
293     * This method assumes there is exactly one binding for the given class.
294     *
295     * <p>
296     * This method runs within a new invocation of {@link IdGenerator#run(Callable) IdGenerator.run()} to support object references
297     * (see {@link IdMapper}).
298     *
299     * @param obj object to encode
300     * @return XML encoding of {@code obj}
301     * @throws JiBXException if there is a JiBX encoding error
302     */
303    public static String toString(Object obj) throws JiBXException {
304        return JiBXUtil.toString(obj, null);
305    }
306
307    /**
308     * Encode the given instance as an XML document and return it as a {@link String}.
309     *
310     * <p>
311     * This method runs within a new invocation of {@link IdGenerator#run(Callable) IdGenerator.run()} to support object references
312     * (see {@link IdMapper}).
313     *
314     * @param obj object to encode
315     * @param bindingName binding name, or null to choose the only one
316     * @return XML encoding of {@code obj}
317     * @throws JiBXException if there is a JiBX encoding error
318     */
319    public static String toString(Object obj, String bindingName) throws JiBXException {
320        final StringWriter w = new StringWriter();
321        try {
322            JiBXUtil.writeObject(obj, bindingName, w);
323            w.close();
324        } catch (IOException e) {
325            throw new JiBXException("unexpected exception", e);
326        }
327        return w.toString();
328    }
329
330    private static void unwrapException(Exception e) throws JiBXException, IOException {
331        if (e instanceof JiBXException)
332            throw (JiBXException)e;
333        if (e instanceof IOException)
334            throw (IOException)e;
335        if (e instanceof RuntimeException)
336            throw (RuntimeException)e;
337        throw new RuntimeException(e);
338    }

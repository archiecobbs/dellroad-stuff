006package org.dellroad.stuff.jibx;
007
008import java.util.Iterator;
009import java.util.Map;
010
011import org.jibx.runtime.JiBXParseException;
012
013/**
014 * Utility class that makes it slightly easier to model {@link Map} properties in JiBX.
015 * This class can be used to represent entries in the map, each of which is modeled in XML as a separate XML element.
016 *
017 * <p>
018 * For example, suppose you have a class {@code Company} and want to add a {@code directory} property that has
019 * type {@code Map<String, Person>}:
020 * <blockquote><pre>
021 * public class Company {
022 *     private Map&lt;String, Person&gt; directory = new HashMap&lt;String, Person&gt;();
023 *
024 *     // Getter and setter for the "directory" property
025 *     public Map&lt;String, Person&gt; getDirectory() {
026 *         return this.directory;
027 *     }
028 *     public void setDirectory(Map&lt;String, Person&gt; directory) {
029 *         this.directory = directory;
030 *     }
031 * }
032 * </pre></blockquote>
033 *
034 * <p>
035 * Because the JiBX binding process modifies class files, you first need to create your own subclass of {@link MapEntry}
036 * that can be modified. In this example, we'll use an inner class of {@code Company}. In addition, you also need to add
037 * JiBX "add-method" and "iter-method" helper methods. The resulting new code might look like this:
038 * <blockquote><pre>
039 *     // JiBX holder for a single entry in the Directory map
040 *     public static class DirectoryEntry extends MapEntry&lt;String, Person&gt; {
041 *        public String getKey()   { return super.getKey();   }   // JiBX requires exact return types
042 *        public Person getValue() { return super.getValue(); }   // JiBX requires exact return types
043 *     }
044 *
045 *     // JiBX "add-method" that adds a new entry to the directory
046 *     void addDirectoryEntry(DirectoryEntry entry) throws JiBXParseException {
047 *         MapEntry.add(this.directory, entry);
048 *     }
049 *
050 *     // JiBX "iter-method" that iterates all entries in the directory
051 *     Iterator&lt;DirectoryEntry&gt; iterateDirectoryEntries() {
052 *         return MapEntry.iterate(this.directory, DirectoryEntry.class);
053 *     }
054 * </pre></blockquote>
055 *
056 * <p>
057 * Then in your JiBX binding definition, you would do something like this:
058 * <blockquote><pre>
059 * &lt;binding package="com.example"&gt;
060 *
061 *     &lt;!-- Include XML mapping definition for a Person object (having type-name "person") --&gt;
062 *     &lt;include path="person.xml"/&gt;
063 *
064 *     &lt;!-- Define the XML mapping for one entry in the "directory" map --&gt;
065 *     &lt;mapping abstract="true" type-name="directory_entry" class="com.example.Company$DirectoryEntry"&gt;
066 *         &lt;value name="name" get-method="getKey" set-method="setKey" type="java.lang.String" style="attribute"/&gt;
067 *         &lt;structure name="Person" get-method="getValue" set-method="setValue" map-as="person"/&gt;
068 *     &lt;/mapping&gt;
069 *
070 *     &lt;!-- Define XML mapping for a Company object --&gt;
071 *     &lt;mapping abstract="true" type-name="company" class="com.example.Company"&gt;
072 *         &lt;collection name="Directory" item-type="com.example.Company$DirectoryEntry"
073 *           add-method="addDirectoryEntry" iter-method="iterateDirectoryEntries"&gt;
074 *             &lt;structure name="DirectoryEntry" map-as="directory_entry"/&gt;
075 *         &lt;/collection&gt;
076 *         &lt;!-- other properties... --&gt;
077 *     &lt;/mapping&gt;
078 * &lt;/binding&gt;
079 * </pre></blockquote>
080 *
081 * Then the resulting XML would end up looking something like this:
082 * <blockquote><pre>
083 * &lt;Company&gt;
084 *     &lt;Directory&gt;
085 *         &lt;DirectoryEntry name="George Washington"&gt;
086 *             &lt;Person&gt;
087 *                  &lt;!-- properties of George Washington... --&gt;
088 *             &lt;/Person&gt;
089 *         &lt;/DirectoryEntry&gt;
090 *         &lt;DirectoryEntry name="Betsy Ross"&gt;
091 *             &lt;Person&gt;
092 *                  &lt;!-- properties of Betsy Ross... --&gt;
093 *             &lt;/Person&gt;
094 *         &lt;/DirectoryEntry&gt;
095 *     &lt;/Directory&gt;
096 *     &lt;!-- other properties... --&gt;
097 * &lt;/Company&gt;
098 * </pre></blockquote>
099 *
100 * <p>
101 * Note that during unmarshalling, the <code>Map</code> itself is not created; it is expected to already exist
102 * and be empty. This will be the case if you provide a field initializer as in the example above.
103 *
104 * <p>
105 * The map keys are not constrained to being simple values: for complex keys, just adjust the mapping for the
106 * {@code DirectoryEntry} structure accordingly.
107 */
108public class MapEntry<K, V> {
109
110    private K key;
111    private V value;
112
113    /**
114     * Get this map entry's key.
115     *
116     * @return map entry key
117     */
118    public K getKey() {
119        return this.key;
120    }
121    public void setKey(K key) {
122        this.key = key;
123    }
124
125    /**
126     * Get this map entry's value.
127     *
128     * @return map entry value
129     */
130    public V getValue() {
131        return this.value;
132    }
133    public void setValue(V value) {
134        this.value = value;
135    }
136
137    /**
138     * Helper method intended to be used by a custom JiBX "iter-method".
139     * This method returns an iterator that iterates over all entries in the given map.
140     *
141     * @param <K> type of map keys
142     * @param <V> type of map values
143     * @param <E> map entry type
144     * @param map map to iterate
145     * @param entryClass the subclass of {@link MapEntry} used for iterated elements; must have a default constructor
146     * @return map entry iterator
147     */
148    public static <K, V, E extends MapEntry<K, V>> Iterator<E> iterate(Map<K, V> map, final Class<E> entryClass) {
149        final Iterator<Map.Entry<K, V>> entryIterator = map.entrySet().iterator();
150        return new Iterator<E>() {
151
152            @Override
153            public boolean hasNext() {
154                return entryIterator.hasNext();
155            }
156
157            @Override
158            public E next() {
159                Map.Entry<K, V> entry = entryIterator.next();
160                E mapEntry;
161                try {
162                    mapEntry = entryClass.newInstance();
163                } catch (Exception e) {
164                    throw new RuntimeException("unexpected exception", e);
165                }
166                mapEntry.setKey(entry.getKey());
167                mapEntry.setValue(entry.getValue());
168                return mapEntry;
169            }
170
171            @Override
172            public void remove() {
173                entryIterator.remove();
174            }
175        };
176    }
177
178    /**
179     * Helper method intended to be used by a custom JiBX "add-method".
180     * If there is an existing entry with the same key, a {@link JiBXParseException} is thrown.
181     *
182     * @param <K> type of map keys
183     * @param <V> type of map values
184     * @param map map to which to add an new entry
185     * @param entry new entry to add
186     * @throws JiBXParseException if the map already contains an entry with the given key
187     */
188    public static <K, V> void add(Map<K, V> map, MapEntry<? extends K, ? extends V> entry) throws JiBXParseException {
189        MapEntry.add(map, entry, false);
190    }
191
192    /**
193     * Helper method intended to be used by a custom JiBX "add-method".
194     *
195     * @param <K> type of map keys
196     * @param <V> type of map values
197     * @param map map to which to add an new entry
198     * @param entry new entry to add
199     * @param allowDuplicate {@code true} to replace any existing entry having the same key,
200     *  or {@code false} to throw a {@link JiBXParseException} if there is an existing entry
201     * @throws JiBXParseException if {@code allowDuplicate} is {@code false} and an entry
202     *  with the same key already exists in {@code map}
203     */
204    public static <K, V> void add(Map<K, V> map, MapEntry<? extends K, ? extends V> entry, boolean allowDuplicate)
205      throws JiBXParseException {
206        K key = entry.getKey();
207        V value = entry.getValue();
208        if (!allowDuplicate && map.containsKey(key))
209            throw new JiBXParseException("duplicate key in map", "" + key);
210        map.put(key, value);
211    }
212}

115    /**
116     * Deserialize an object by reference.
117     *
118     * <p>
119     * Invoke this method from your own custom deserializer to produce an result of the correct type.
120     *
121     * <p>
122     * The object must have been unmarshalled already and had its ID registered via {@link IdMapper#setId}.
123     *
124     * @param <T> expected type
125     * @param string XML encoding created by {@link #serializeReference}
126     * @param type expected type
127     * @return decoded value
128     * @throws IllegalArgumentException if {@code string} cannot be parsed
129     * @throws JiBXParseException if the reference has not been registered to any object yet (e.g., forward reference)
130     * @throws JiBXParseException if the referenced object is not an instance of {@code type}
131     * @see #serializeReference
132     * @see IdMapper
133     */
134    public static <T> T deserializeReference(String string, Class<T> type) throws JiBXParseException {
135        if (string == null)
136            return null;
137        long id;
138        try {
139            id = IdMapper.parseId(string);
140        } catch (IllegalArgumentException e) {
141            throw new JiBXParseException("invalid object reference", string, e);
142        }
143        Object obj = IdGenerator.get().getObject(id);
144        if (obj == null)
145            throw new JiBXParseException("unregistered object reference `" + string + "'; possible forward reference", string);
146        try {
147            return type.cast(obj);
148        } catch (ClassCastException e) {
149            throw new JiBXParseException("object reference `" + string + "' is assigned to an instance of "
150              + obj.getClass() + " which is not assignable to " + type, string);
151        }
152    }

154    /**
155     * Serialize an object by reference.
156     *
157     * <p>
158     * The object must have been marshalled already and had its ID assigned via {@link IdMapper#getId}.
159     *
160     * @param obj object to encode by reference for XML
161     * @return encoded reference
162     * @throws IllegalArgumentException if the object has not been registered yet (e.g., forward reference)
163     * @see #deserializeReference
164     * @see IdMapper
165     */
166    public static String serializeReference(Object obj) {
167        if (obj == null)
168            return null;
169        long id = IdGenerator.get().checkId(obj);
170        if (id == 0)
171            throw new IllegalArgumentException("unregistered object; possible forward reference to " + obj);
172        return IdMapper.formatId(id);
173    }

193    /**
194     * Derialize a millisecond-granularity time interval, e.g., "30s", "1d12h", "0.250s", etc.
195     *
196     * @param string XML encoding created by {@link #serializeTimeInterval}
197     * @return decoded value
198     * @throws JiBXParseException if the parse fails
199     * @see #serializeTimeInterval
200     */
201    public static long deserializeTimeInterval(String string) throws JiBXParseException {
202
203        // Apply regular expression
204        Matcher m = Pattern.compile(TIME_INTERVAL_PATTERN).matcher(string);
205        if (string.length() == 0 || !m.matches())
206            throw new JiBXParseException("invalid time interval", string);
207
208        // Parse regex groups
209        Object[][] groups = new Object[][] {
210            { 2, 24 * 60 * 60 * 1000 },
211            { 4,      60 * 60 * 1000 },
212            { 6,           60 * 1000 },
213            { 8,                1000 },
214        };
215        long value = 0;
216        for (int i = 0; i < groups.length; i++) {
217            String group = m.group((Integer)groups[i][0]);
218            if (group == null || group.length() == 0)
219                continue;
220            double num;
221            try {
222                num = Double.parseDouble(group);
223            } catch (NumberFormatException e) {
224                throw new JiBXParseException("invalid time interval", string);
225            }
226            value += Math.round(num * (Integer)groups[i][1]);
227            if (value < 0)
228                throw new JiBXParseException("time interval is too large", string);
229        }
230
231        // Done
232        return value;
233    }

235    /**
236     * Serialize a millisecond-granularity time interval, e.g., "30s", "1d12h", "0.250s", etc.
237     *
238     * @param value millisecond value to encode
239     * @return encoded value
240     * @see #deserializeTimeInterval
241     */
242    public static String serializeTimeInterval(long value) {
243        if (value < 0)
244            throw new IllegalArgumentException("negative value");
245        StringBuilder b = new StringBuilder(32);
246        long days = value / (24 * 60 * 60 * 1000);
247        value = value % (24 * 60 * 60 * 1000);
248        if (days > 0)
249            b.append(days).append('d');
250        long hours = value / (60 * 60 * 1000);
251        value = value % (60 * 60 * 1000);
252        if (hours > 0)
253            b.append(hours).append('h');
254        long minutes = value / (60 * 1000);
255        value = value % (60 * 1000);
256        if (minutes > 0)
257            b.append(minutes).append('m');
258        long millis = value;
259        if (millis != 0 || b.length() == 0) {
260            if (millis >= 1000 && millis % 1000 == 0)
261                b.append(String.format("%ds", millis / 1000));
262            else
263                b.append(String.format("%.3fs", (double)millis / 1000.0));
264        }
265        return b.toString();
266    }

295    /**
296     * Deserialize a byte array using MAC address notation (colon-separated). Each byte must have two digits.
297     *
298     * @param string XML encoding created by {@link #serializeByteArrayWithColons}
299     * @return decoded value
300     * @throws JiBXParseException if the parse fails
301     * @see #serializeByteArrayWithColons
302     */
303    public static byte[] deserializeByteArrayWithColons(String string) throws JiBXParseException {
304        if (string.length() == 0)
305            return new byte[0];
306        if (string.length() % 3 != 2)
307            throw new JiBXParseException("invalid byte array", string);
308        char[] nocolons = new char[((string.length() + 1) / 3) * 2];
309        int j = 0;
310        for (int i = 0; i < string.length(); i++) {
311            if (i % 3 == 2) {
312                if (string.charAt(i) != ':')
313                    throw new JiBXParseException("invalid byte array", string);
314                continue;
315            }
316            nocolons[j++] = string.charAt(i);
317        }
318        try {
319            return ByteArrayEncoder.decode(new String(nocolons));
320        } catch (IllegalArgumentException e) {
321            throw new JiBXParseException("invalid byte array", string, e);
322        }
323    }
324
325    /**
326     * Serialize a byte array using MAC address notation (colon-separated). Each byte will have two digits.
327     *
328     * @param array array to encode
329     * @return encoded array
330     * @see #deserializeByteArrayWithColons
331     */
332    public static String serializeByteArrayWithColons(byte[] array) {
333        char[] colons = new char[array.length * 3 - 1];
334        String nocolons = ByteArrayEncoder.encode(array);
335        int j = 0;
336        for (int i = 0; i < nocolons.length(); i += 2) {
337            colons[j++] = nocolons.charAt(i);
338            colons[j++] = nocolons.charAt(i + 1);
339            if (j < colons.length)
340                colons[j++] = ':';
341        }
342        return new String(colons);
343    }

535    /**
536     * Boolean parser that allows "yes" and "no" as well as the usual "true", "false", "0", "1".
537     * Comparisons are case-insensitive.
538     *
539     * @param string XML encoding of boolean value
540     * @return decoded value
541     * @throws JiBXParseException if the value is not recognizable as a boolean
542     * @see #deserializeBooleanStrictly
543     */
544    public static boolean deserializeBoolean(String string) throws JiBXParseException {
545        for (String s : BOOLEAN_TRUES) {
546            if (string.equalsIgnoreCase(s))
547                return true;
548        }
549        for (String s : BOOLEAN_FALSES) {
550            if (string.equalsIgnoreCase(s))
551                return false;
552        }
553        throw new JiBXParseException("invalid Boolean value", string);
554    }
555
556    /**
557     * Deserialize a boolean strictly, only allowing the values {@code true} or {@code false}.
558     *
559     * @param string XML encoding created by {@link Boolean#toString}
560     * @return decoded value
561     * @throws JiBXParseException if the parse fails
562     * @see #deserializeBoolean
563     */
564    public static boolean deserializeBooleanStrictly(String string) throws JiBXParseException {
565        if ("true".equals(string))
566            return true;
567        if ("false".equals(string))
568            return false;
569        throw new JiBXParseException("invalid boolean value; must be `true' or `false'", string);
570    }

572    /**
573     * Deserialize an array of integers separated by commas and/or whitespace.
574     *
575     * @param string XML encoding created by {@link #serializeIntArray}
576     * @return decoded integer array
577     * @throws JiBXParseException if the parse fails
578     * @see #serializeIntArray
579     */
580    public static int[] deserializeIntArray(String string) throws JiBXParseException {
581        String[] ints = string.trim().split("(\\s*,\\s*|\\s+)");
582        if (ints.length == 0 || ints[0].length() == 0)
583            return new int[0];
584        int[] array = new int[ints.length];
585        try {
586            for (int i = 0; i < array.length; i++)
587                array[i] = Integer.parseInt(ints[i]);
588        } catch (NumberFormatException e) {
589            throw new JiBXParseException("invalid integer list", string, e);
590        }
591        return array;
592    }

594    /**
595     * Serialize an array of integers. Example: "1, 2, 3".
596     *
597     * @param array integer array
598     * @return string encoding of {@code array}
599     * @see #deserializeIntArray
600     */
601    public static String serializeIntArray(int[] array) {
602        StringBuilder buf = new StringBuilder(array.length * 2);
603        for (int i = 0; i < array.length; i++) {
604            if (i > 0)
605                buf.append(", ");
606            buf.append(array[i]);
607        }
608        return buf.toString();
609    }

006package org.dellroad.stuff.jibx;
007
008import java.io.BufferedOutputStream;
009import java.io.IOException;
010import java.io.OutputStream;
011
012import org.dellroad.stuff.io.OutputStreamWriter;
013import org.jibx.runtime.JiBXException;
014
015/**
016 * {@link OutputStream} over which XML documents are passed. This class is a companion to {@link XMLDocumentInputStream}.
017 *
018 * <p>
019 * XML documents are created from Java objects via {@link JiBXUtil#writeObject(Object, OutputStream) JiBXUtil.writeObject()}.
020 *
021 * <p>
022 * Instances of this class are thread-safe.
023 *
024 * @param <T> XML document type
025 * @see XMLDocumentInputStream
026 */
027public class XMLDocumentOutputStream<T> {
028
029    private final OutputStreamWriter output;
030
031    /**
032     * Constructor.
033     *
034     * @param output data destination
035     */
036    public XMLDocumentOutputStream(OutputStream output) {
037        if (output == null)
038            throw new IllegalArgumentException("null output");
039        this.output = new OutputStreamWriter(new BufferedOutputStream(output));
040    }
041
042    /**
043     * Write the object encoded as XML to the underlying output stream.
044     * The underlying output stream is flushed.
045     *
046     * @param obj object to write
047     * @throws IOException if an I/O error occurs
048     * @throws JiBXException if JiBX encoding fails
049     */
050    public synchronized void write(T obj) throws IOException, JiBXException {
051        this.output.start();
052        JiBXUtil.writeObject(obj, this.output);
053        this.output.stop();
054    }
055
056    /**
057     * Close the underlying output stream.
058     *
059     * @throws IOException if an I/O error occurs
060     */
061    public void close() throws IOException {
062        this.output.close();
063    }
064}

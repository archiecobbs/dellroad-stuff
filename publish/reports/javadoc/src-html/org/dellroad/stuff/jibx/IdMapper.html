006package org.dellroad.stuff.jibx;
007
008import java.util.Map;
009
010import org.dellroad.stuff.java.IdGenerator;
011import org.jibx.extras.IdDefRefMapperBase;
012import org.jibx.runtime.IMarshallable;
013import org.jibx.runtime.IMarshallingContext;
014import org.jibx.runtime.JiBXException;
015import org.jibx.runtime.impl.MarshallingContext;
016
017/**
018 * JiBX Marshaller/Unmarshaller that assigns unique ID's to each object and
019 * replaces duplicate appearances of the same object with an IDREF reference.
020 *
021 * <p>
022 * This class allows for easy ID/IDREF handling for existing classes, with minimal
023 * modifications to those classes and no custom (un)marshaller subclasses.
024 *
025 * <h3>JiBX Mapping</h3>
026 *
027 * <p>
028 * Suppose you have a class {@code Person.java} with a single {@code name} property
029 * and you want to add ID/IDREF support to it.
030 *
031 * <p>
032 * First add the following two pseudo-bean property methods to the classes:
033 *
034 * <blockquote><pre>
035 *  import org.dellroad.stuff.jibx.IdMapper;
036 *
037 *  public class Person {
038 *
039 *      private String name;
040 *
041 *      public String getName() {
042 *          return this.name;
043 *      }
044 *      public void setName(String name) {
045 *          this.name = name;
046 *      }
047 *
048 *      <b>// JiBX methods
049 *      private String getJiBXId() {
050 *          return IdMapper.getId(this);
051 *      }
052 *      private void setJiBXId(String id) {
053 *          // do nothing
054 *      }</b>
055 *  }
056 * </pre></blockquote>
057 * Note: if you subclass {@code Person.java} from a different sub-package, you may need
058 * to change the access privileges of those methods from {@code private} to {@code protected}.
059 *
060 * <p>
061 * Next, define a concrete mapping for {@code Person.java} and add the {@code id} attribute:
062 * <blockquote><pre>
063 *  &lt;mapping name="Person" class="com.example.Person"&gt;
064 *      <b>&lt;value name="id" style="attribute" ident="def"
065 *        get-method="getJiBXId" set-method="setJiBXId"/&gt;</b>
066 *      &lt;value name="name" field="name"/&gt;
067 *  &lt;/mapping&gt;
068 * </pre></blockquote>
069 *
070 * <p>
071 * Finally, use {@link IdMapper} as the custom marshaller and unmarshaller wherever a {@code Person} appears, e.g.:
072 * <blockquote><pre>
073 *  &lt;mapping name="Company" class="com.example.Company"&gt;
074 *      &lt;collection name="Employees" field="employees" create-type="java.util.ArrayList"&gt;
075 *          &lt;structure name="Person" type="com.example.Person"
076 *            <b>marshaller="org.dellroad.stuff.jibx.IdMapper"
077 *            unmarshaller="org.dellroad.stuff.jibx.IdMapper"</b>/&gt;
078 *      &lt;/collection&gt;
079 *      &lt;structure name="EmployeeOfTheWeek"&gt;
080 *          &lt;structure name="Person" field="employeeOfTheWeek"
081 *            <b>marshaller="org.dellroad.stuff.jibx.IdMapper"
082 *            unmarshaller="org.dellroad.stuff.jibx.IdMapper"</b>/&gt;
083 *      &lt;/structure&gt;
084 *  &lt;/mapping&gt;
085 * </pre></blockquote>
086 * Note the {@code EmployeeOfTheWeek} "wrapper" element for the {@code employeeOfTheWeek} field; this is required
087 * in order to use an XML name for this field other than {@code Person} (see limitations below).
088 *
089 * <p>
090 * Now the first appearance of any {@code Person} will contain the full XML structure with an additional <code>id="..."</code>
091 * attribute, while all subsequent appearances will contain just a reference of the form <code>&lt;Person idref="..."/&gt;</code>.
092 * Conversely, when unmarshalled all {@code Person} XML elements that refer to the same original {@code Person} will
093 * re-use the same unmarshalled {@code Person} object.
094 *
095 * <p>
096 * So the resulting XML might look like:
097 * <blockquote><pre>
098 *  &lt;Company&gt;
099 *      &lt;Employees&gt;
100 *          &lt;Person id="N00001"&gt;
101 *              &lt;name&gt;Aardvark, Annie&lt;/name&gt;
102 *          &lt;/Person&gt;
103 *          &lt;Person id="N00002"&gt;
104 *              &lt;name&gt;Appleby, Arnold&lt;/name&gt;
105 *          &lt;/Person&gt;
106 *          ...
107 *      &lt;/Employees&gt;
108 *      &lt;EmployeeOfTheWeek&gt;
109 *          &lt;Person idref="N00001"/&gt;
110 *      &lt;/EmployeeOfTheWeek&gt;
111 *  &lt;/Company&gt;
112 * </pre></blockquote>
113 *
114 * <h3>Limitations</h3>
115 *
116 * <p>
117 * JiBX and this class impose some limitations:
118 * <ul>
119 * <li>JiBX marshalling must be performed within an invocation of {@link IdGenerator#run IdGenerator.run()}
120 *      so that an {@link IdGenerator} is available to generate the unique IDs (when using Spring, consider using
121 *      {@link IdMappingMarshaller}; otherwise, the {@link JiBXUtil} methods all satisfy this requirement).</li>
122 * <li>Classes that use ID/IDREF must have concrete JiBX mappings.</li>
123 * <li>All occurences of the class must use the XML element name of the concrete mapping, so the use of
124 *      a "wrapper" element is required when a different element name is desired.</li>
125 * </ul>
126 *
127 * <h3>A Simpler Approach</h3>
128 *
129 * The above approach is useful when you don't want to keep track of which instance of an object will appear first
130 * in the XML encoding: the first one will always fully define the object, while subsequent ones will just reference it.
131 *
132 * <p>
133 * If this flexibility is not needed, i.e., if you can identify where in your mapping the first occurrence of an object
134 * will appear, then the following simpler approach works without the above approach's limitations (other than requiring
135 * that marshalling be peformed within an invocation of {@link IdGenerator#run IdGenerator.run()}):
136 *
137 * <p>
138 * First, replace the <code>// do nothing</code> in the example above with call to {@link IdMapper#setId IdMapper.setId()},
139 * and add a custom deserializer delegating to {@link ParseUtil#deserializeReference ParseUtil.deserializeReference()} to
140 * <blockquote><pre>
141 *      private void setJiBXId(String id) {
142 *          IdMapper.setId(this, id);
143 *      }
144 *
145 *      public static Employee deserializeEmployeeReference(String string) throws JiBXParseException {
146 *          return ParseUtil.deserializeReference(string, Employee.class);
147 *      }
148 * </pre></blockquote>
149 *
150 * <p>
151 * Then, map the first occurrence of an object exactly as in the concrete mapping above, exposing the <code>JiBXId</code> property.
152 * In all subsequent occurrences of the object, expose the reference to the object as a simple property using the custom
153 * serializer/deserializer pair {@link ParseUtil#serializeReference ParseUtil.serializeReference()} and
154 * {@code Employee.deserializeEmployeeReference()}.
155 *
156 * <p>
157 * For example, the following binding would yeild the same XML encoding as before:
158 * <blockquote><pre>
159 *  &lt;mapping abstract="true" type-name="person" class="com.example.Person"&gt;
160 *      <b>&lt;value name="id" style="attribute" get-method="getJiBXId" set-method="setJiBXId"/&gt;</b>
161 *      &lt;value name="name" field="name"/&gt;
162 *  &lt;/mapping&gt;
163 *
164 *  &lt;mapping name="Company" class="com.example.Company"&gt;
165 *      &lt;collection name="Employees" field="employees" create-type="java.util.ArrayList"&gt;
166 *          &lt;structure name="Person" map-as="person"/&gt;    &lt;!-- first occurences of all these objects --&gt;
167 *      &lt;/collection&gt;
168 *      &lt;structure name="EmployeeOfTheWeek"&gt;
169 *          &lt;structure name="Person"&gt;
170 *              <b>&lt;value name="idref" style="attribute" field="employeeOfTheWeek"
171 *                serializer="org.dellroad.stuff.jibx.ParseUtil.serializeReference"
172 *                deserializer="com.example.Employee.deserializeEmployeeReference"</b>/&gt;
173 *          &lt;/structure&gt;
174 *      &lt;/structure&gt;
175 *  &lt;/mapping&gt;
176 * </pre></blockquote>
177 *
178 * <p>
179 * If you want the reference to be optionally <code>null</code>, then you'll also need to add a <code>test-method</code>:
180 * <blockquote><pre>
181 *      <b>private boolean hasEmployeeOfTheWeek() {
182 *          return this.getEmployeeOfTheWeek() != null;
183 *      }</b>
184 *
185 *      &lt;structure name="EmployeeOfTheWeek" <b>usage="optional" test-method="hasEmployeeOfTheWeek"</b>&gt;
186 *          &lt;structure name="Person"&gt;
187 *              &lt;value name="idref" style="attribute" field="employeeOfTheWeek"
188 *                serializer="org.dellroad.stuff.jibx.ParseUtil.serializeReference"
189 *                deserializer="com.example.Employee.deserializeEmployeeReference"/&gt;
190 *          &lt;/structure&gt;
191 *      &lt;/structure&gt;
192 * </pre></blockquote>
193 * This approach causes the whole <code>&lt;EmployeeOfTheWeek&gt;</code> element to disappear when there is no
194 * such employee. Alternately, you can avoid the need for the <code>test-method</code> if you want to allow
195 * just the attribute to disappear, or you could even change from <code>style="attribute"</code> to <code>style="element"</code>;
196 * in both cases you would be making the reference itself optional instead of the containing element.
197 *
198 * @see IdMappingMarshaller
199 */
200public class IdMapper extends IdDefRefMapperBase {
201
202    private final int index;
203    private final String name;
204    // This is here to work around bogus JiBX binding error
205    private IdMapper() {
206        super(null, 0, null);
207        throw new UnsupportedOperationException();
208    }
209
210    public IdMapper(String uri, int index, String name, String className) {
211        super(uri, index, name);
212        this.index = index;
213        this.name = name;
214    }
215
216    /**
217     * Get the unique ID value for the given object.
218     *
219     * <p>
220     * The implementation in {@link IdMapper} formats an ID of the form <code>N012345</code>
221     * using the {@link IdGenerator} acquired from {@link IdGenerator#get}.
222     *
223     * @param obj any object
224     * @return unique ID for the object
225     */
226    public static String getId(Object obj) {
227        return IdMapper.formatId(IdGenerator.get().getId(obj));
228    }
229
230    /**
231     * Set the unique ID value for the given object.
232     *
233     * <p>
234     * The implementation in {@link IdMapper} expects an ID of the form <code>N012345</code>,
235     * then associates the parsed {@code long} value with the given object
236     * using the {@link IdGenerator} acquired from {@link IdGenerator#get}.
237     *
238     * @param obj object to register
239     * @param idref string ID assigned to the object
240     * @throws IllegalArgumentException if {@code idref} is not of the form <code>N012345</code>
241     * @throws IllegalArgumentException if {@code idref} is already associated with a different object
242     */
243    public static void setId(Object obj, String idref) {
244        IdGenerator.get().setId(obj, IdMapper.parseId(idref));
245    }
246
247    /**
248     * Format the unique ID.
249     *
250     * @param id ID value
251     * @return formatted idref
252     */
253    public static String formatId(long id) {
254        return String.format("N%05d", id);
255    }
256
257    /**
258     * Parse the unique ID value assigned to the given object by {@link #getId getId()}.
259     *
260     * @param idref ID value assigned to the object
261     * @return parse ID number
262     * @throws IllegalArgumentException if {@code idref} is not of the form <code>N012345</code>
263     */
264    public static long parseId(String idref) {
265        if (idref == null || idref.length() == 0 || !idref.matches("N-?\\d+"))
266            throw new IllegalArgumentException("invalid id value `" + idref + "'");
267        try {
268            return Long.parseLong(idref.substring(1), 10);
269        } catch (NumberFormatException e) {
270            throw new IllegalArgumentException("invalid id value `" + idref + "'");
271        }
272    }
273
274    /**
275     * Get the unique ID for the given object. Delegates to {@link #getId getId()}.
276     */
277    @Override
278    protected String getIdValue(Object obj) {
279        return IdMapper.getId(obj);
280    }
281
282    /**
283     * Get the ID reference attribute name. Default is <code>"idref"</code>.
284     */
285    @Override
286    protected String getAttributeName() {
287        return "idref";
288    }
289
290    /**
291     * Overrides superclass to use object equality instead of {@code Object.equals()} for sanity checking.
292     */
293    @Override
294    @SuppressWarnings("unchecked")
295    public void marshal(Object obj, IMarshallingContext ictx) throws JiBXException {
296
297        // Sanity check
298        if (obj == null)
299            return;
300        if (!(ictx instanceof MarshallingContext))
301            throw new JiBXException("Invalid context type for marshaller");
302
303        // Check if ID already defined
304        MarshallingContext ctx = (MarshallingContext)ictx;
305        Map<String, Object> map = (Map<String, Object>)ctx.getIdMap();
306        String id = this.getIdValue(obj);
307        Object value = map.get(id);
308
309        // New object? Output normally
310        if (value == null) {
311            if (!(obj instanceof IMarshallable))
312                throw new JiBXException("instance of " + obj.getClass() + " is not marshallable");
313            map.put(id, obj);
314            ((IMarshallable)obj).marshal(ctx);
315            return;
316        }
317
318        // Sanity check what we got
319        if (value != obj)
320            throw new JiBXException("encountered two objects with the same ID " + id);
321
322        // Emit a reference
323        ctx.startTagAttributes(this.index, this.name);
324        ctx.attribute(0, this.getAttributeName(), id);
325        ctx.closeStartEmpty();
326    }
327}

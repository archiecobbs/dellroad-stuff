006package org.dellroad.stuff.jibx;
007
008import java.io.BufferedInputStream;
009import java.io.IOException;
010import java.io.InputStream;
011
012import org.dellroad.stuff.io.InputStreamReader;
013import org.jibx.runtime.JiBXException;
014
015/**
016 * {@link InputStream} over which XML documents are passed. This class is a companion to {@link XMLDocumentOutputStream}.
017 *
018 * <p>
019 * XML documents are converted into Java objects via {@link JiBXUtil#readObject(Class, InputStream) JiBXUtil.readObject()}.
020 * </p>
021 *
022 * <p>
023 * Instances of this class are thread-safe.
024 * </p>
025 *
026 * @param <T> XML document type
027 * @see XMLDocumentOutputStream
028 */
029public class XMLDocumentInputStream<T> {
030
031    private final Class<T> type;
032    private final InputStreamReader input;
033
034    /**
035     * Constructor.
036     *
037     * @param type Java type for XML documents
038     * @param input data source
039     */
040    public XMLDocumentInputStream(Class<T> type, InputStream input) {
041        if (type == null)
042            throw new IllegalArgumentException("null type");
043        if (input == null)
044            throw new IllegalArgumentException("null input");
045        this.type = type;
046        this.input = new InputStreamReader(new BufferedInputStream(input));
047    }
048
049    /**
050     * Read the next XML document, parsed and objectified.
051     *
052     * @return decoded object or {@code null} on EOF
053     */
054    public T read() throws IOException, JiBXException {
055        InputStream xml = this.input.read();
056        if (xml == null)
057            return null;
058        try {
059            return JiBXUtil.readObject(this.type, xml);
060        } finally {
061            try {
062                xml.close();
063            } catch (IOException e) {
064                // ignore
065            }
066        }
067    }
068
069    public void close() throws IOException {
070        this.input.close();
071    }
072}

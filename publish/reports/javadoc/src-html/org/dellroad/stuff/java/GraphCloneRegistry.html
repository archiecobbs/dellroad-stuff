006package org.dellroad.stuff.java;
007
008import java.util.ArrayDeque;
009import java.util.IdentityHashMap;
010
011/**
012 * A registry for copies of {@link GraphCloneable} objects used during <i>graph cloning</i> operations.
013 * A <i>graph clone</i> of a graph of objects is a deep copy without duplicates and preserving reference topology.
014 *
015 * <p>
016 * See {@link GraphCloneable} for a complete description.
017 * </p>
018 *
019 * @see GraphCloneable
020 */
021public class GraphCloneRegistry {
022
023    private final IdentityHashMap<Object, Object> map = new IdentityHashMap<Object, Object>();
024
025    private final ArrayDeque<GraphCloneable> stack = new ArrayDeque<GraphCloneable>();
026
027    /**
028     * Get the unique clone of a given value during a graph clone operation, creating it if necessary.
029     *
030     * <p>
031     * If the {@code value} has already been cloned and registered with this instance, it is returned.
032     * Otherwise, a new clone of {@code value} is created and registered with this instance by invoking
033     * {@link GraphCloneable#createGraphClone value.createGraphClone()}, and then returned.
034     * </p>
035     *
036     * <p>
037     * If the {@code value} is null, null is returned.
038     * <p>
039     *
040     * @param value original value
041     * @return unique clone of {@code value}, or null if {@code value} is null
042     * @throws IllegalStateException if {@link GraphCloneable#createGraphClone value.createGraphClone()}
043     *  fails to invoke {@link #setGraphClone setGraphClone()}
044     * @throws IllegalStateException if {@link GraphCloneable#createGraphClone value.createGraphClone()}
045     *  invokes {@link #setGraphClone setGraphClone()} more than once
046     * @throws IllegalStateException if {@link GraphCloneable#createGraphClone value.createGraphClone()}
047     *  fails to invoke {@link #setGraphClone setGraphClone()} prior to recursing on other {@link GraphCloneable} fields
048     */
049    @SuppressWarnings("unchecked")
050    public <T extends GraphCloneable> T getGraphClone(final T value) {
051
052        // Verify parent clone registered itself before recursing on children
053        final GraphCloneable parent = this.stack.peekFirst();
054        if (parent != null && !this.map.containsKey(parent))
055            throw new IllegalStateException("must invoke setGraphClone() prior to recursing via getGraphClone()");
056
057        // Handle null
058        if (value == null)
059            return null;
060
061        // Have we already cloned the value?
062        if (this.map.containsKey(value))
063            return (T)this.map.get(value);
064
065        // Need to recurse
066        this.stack.addFirst(value);
067        try {
068
069            // Create new clone
070            try {
071                value.createGraphClone(this);
072            } catch (CloneNotSupportedException e) {
073                throw new RuntimeException(e);
074            }
075
076            // Verify it was registered
077            if (!this.map.containsKey(value))
078                throw new IllegalStateException("createGraphClone() failed to invoke setGraphClone() for value " + value);
079
080            // Return clone
081            return (T)this.map.get(value);
082        } finally {
083            this.stack.removeFirst();
084        }
085    }
086
087    /**
088     * Register the given clone as the unique instance associated with the object currently being cloned.
089     *
090     * <p>
091     * The "object currently being cloned" is the {@link GraphCloneable} object associated with the top-most invocation
092     * of {@link GraphCloneable#createGraphClone GraphCloneable.createGraphClone()} on the Java execution stack.
093     * </p>
094     *
095     * <p>
096     * This method should only be invoked within implementations of
097     * {@link GraphCloneable#createGraphClone GraphCloneable.createGraphClone()}.
098     * </p>
099     *
100     * @param clone the clone associated with the current object being graph cloned; may be null (but that would be weird)
101     * @throws IllegalStateException if the current thread is not executing within
102     *  an invocation of {@link GraphCloneable#createGraphClone GraphCloneable.createGraphClone()}
103     */
104    public void setGraphClone(GraphCloneable clone) {
105
106        // Sanity check
107        final GraphCloneable current = this.stack.peekFirst();
108        if (current == null)
109            throw new IllegalStateException("not executing within an invocation of GraphCloneable.createGraphClone()");
110        if (this.map.containsKey(current))
111            throw new IllegalStateException("duplicate invocation of setGraphClone()");
112
113        // Register clone
114        this.map.put(current, clone);
115    }
116}

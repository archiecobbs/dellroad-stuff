015/**
016 * Simple utility enumeration for working Java {@link Class} instances representing primitive types (including {@code void}).
017 *
018 * <p>
019 * Instances of this class represent the eight Java primitive types plus {@code void}.
020 * <p/>
021 *
022 * @param <T> Primitive wrapper type
023 */
024public abstract class Primitive<T> implements Comparator<T> {
025
026    /**
027     * Void type. The {@link #getDefaultValue}, {@link #compare compare()}, and {@link #parseValue parseValue()}
028     * methods always throw exceptions.
029     */
030    public static final Primitive<Void> VOID = new Primitive<Void>(Void.TYPE, Void.class, 'V', "") {
031        @Override
032        public <R> R visit(PrimitiveSwitch<R> pswitch) {
033            return pswitch.caseVoid();
034        }
035        @Override
036        public Void getDefaultValue() {
037            throw new UnsupportedOperationException();
038        }
039        @Override
040        public int compare(Void value1, Void value2) {
041            throw new NullPointerException();
042        }
043        @Override
044        protected Void doParseValue(String string) {
045            throw new IllegalArgumentException();
046        }
047    };

231    private static HashMap<Class<?>, Primitive<?>> classMap;
232
233    private final Class<T> primType;
234    private final Class<T> wrapType;
235    private final char letter;
236    private final Pattern parsePattern;
237
238    Primitive(Class<T> primType, Class<T> wrapType, char letter, String parsePattern) {
239        this.primType = primType;
240        this.wrapType = wrapType;
241        this.letter = letter;
242        this.parsePattern = parsePattern != null ? Pattern.compile(parsePattern) : null;
243        if (Primitive.classMap == null)
244            Primitive.classMap = new HashMap<Class<?>, Primitive<?>>();
245        Primitive.classMap.put(primType, this);
246        Primitive.classMap.put(wrapType, this);
247    }

279     * Get the {@link Class} object representing this primitive type, e.g., {@code Integer.TYPE}.
280     */
281    public Class<T> getType() {
282        return this.primType;
283    }
284
285    /**
286     * Get the wrapper {@link Class} object for this primitive type, e.g., {@code Integer.class}.
287     */
288    public Class<T> getWrapperType() {
289        return this.wrapType;
290    }
291
292    /**
293     * Get a regular expression that matches all allowed inputs to {@link #parseValue parseValue()}.
294     * The returned pattern may also accept some values that {@link #parseValue parseValue()} rejects,
295     * such as {@code 32768} for a {@code short}.
296     */
297    public Pattern getParsePattern() {
298        if (this.parsePattern == null)
299            throw new UnsupportedOperationException();
300        return this.parsePattern;
301    }
302
303    /**
304     * Get the wrapper class' "unwrap" method for this primitive type, e.g., {@code Integer.intValue()}.
305     */
306    public Method getUnwrapMethod() {
307        try {
308            return this.wrapType.getMethod(getName() + "Value");
309        } catch (NoSuchMethodException e) {
310            throw new RuntimeException(e);
311        }
312    }

314    /**
315     * Compare two primitive values.
316     *
317     * @throws NullPointerException if either value is null
318     */
319    @Override
320    public abstract int compare(T value1, T value2);
321
322    /**
323     * Parse a string-encoded value of this instance.
324     * This method is guaranteed to accept all possible return values from the primitive's {@link #toString},
325     * and possibly other reasonable inputs (e.g., hex and octal values for integral types).
326     *
327     * @param string string representation of a value
328     * @throws IllegalArgumentException if {@code string} is null
329     * @throws IllegalArgumentException if {@code string} does not match the {@linkplain #getParsePattern parse pattern}
330     */
331    public T parseValue(String string) {
332        if (string == null)
333            throw new IllegalArgumentException("null string");
334        return this.doParseValue(string);
335    }
336
337    abstract T doParseValue(String string);
338
339    /**
340     * Get the instance with the given name (e.g., {@code "void"}, {@code "int"}, etc.).
341     *
342     * @param name instance short name
343     * @return corresponding instance, or null if name does not correspond to any {@link Primitive} instance
344     * @throws IllegalArgumentException if {@code name} is null
345     */
346    public static Primitive<?> forName(String name) {
347        if (name == null)
348            throw new IllegalArgumentException("null name");
349        for (Primitive<?> primitive : Primitive.values()) {
350            if (name.equals(primitive.getName()))
351                return primitive;
352        }
353        return null;
354    }
355
356    /**
357     * Get all instances of this class.
358     *
359     * @return a modifiable set containing all instances; modifications have no effect on this class
360     */
361    public static Set<Primitive<?>> values() {
362        return new HashSet<Primitive<?>>(Primitive.classMap.values());
363    }
364
365    /**
366     * Get the value corresponding to the given Java primitive or primitive wrapper type.
367     *
368     * @return the {@link Primitive} corresponding to {@code c}, or {@code null}
369     *         if {@code c} is not a primitive or primitive wrapper type
370     */
371    public static Primitive<?> get(Class<?> c) {
372        return Primitive.classMap.get(c);
373    }

375    // This is put into an inner class to avoid initialization ordering problems
376    private static final class DoubleFormat {
377
378        private static final String DIGITS = "(\\p{Digit}+)";
379
380        private static final String HEX_DIGITS = "(\\p{XDigit}+)";
381
382        // an exponent is 'e' or 'E' followed by an optionally
383        // signed decimal integer.
384        private static final String EXPONENT = "[eE][+-]?" + DIGITS;
385
386        private static final String REGEX =
387          "[+-]?("              // Optional sign character
388          + "NaN|"              // "NaN" string
389          + "Infinity|"         // "Infinity" string
390
391          // A decimal floating-point string representing a finite positive
392          // number without a leading sign has at most five basic pieces:
393          // Digits . Digits ExponentPart FloatTypeSuffix
394          //
395          // Since this method allows integer-only strings as input
396          // in addition to strings of floating-point literals, the
397          // two sub-patterns below are simplifications of the grammar
398          // productions from the Java Language Specification, 2nd
399          // edition, section 3.10.2.
400
401          // Digits ._opt Digits_opt ExponentPart_opt FloatTypeSuffix_opt
402          + "(((" + DIGITS + "(\\.)?(" + DIGITS + "?)(" + EXPONENT + ")?)|"
403
404          // . Digits ExponentPart_opt FloatTypeSuffix_opt
405          + "(\\.(" + DIGITS + ")(" + EXPONENT + ")?)|"
406
407          // Hexadecimal strings
408          + "(("
409            // 0[xX] HexDigits ._opt BinaryExponent FloatTypeSuffix_opt
410            + "(0[xX]" + HEX_DIGITS + "(\\.)?)|"
411
412            // 0[xX] HexDigits_opt . HexDigits BinaryExponent FloatTypeSuffix_opt
413            + "(0[xX]" + HEX_DIGITS + "?(\\.)" + HEX_DIGITS + ")"
414             + ")[pP][+-]?" + DIGITS + "))" + "))";
415
416        private DoubleFormat() {
417        }
418    }
419
420    @Override
421    public String toString() {
422        return this.wrapType.getSimpleName().toUpperCase();
423    }

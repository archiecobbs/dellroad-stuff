006package org.dellroad.stuff.java;
007
008import com.google.common.cache.CacheBuilder;
009import com.google.common.cache.CacheLoader;
010import com.google.common.cache.LoadingCache;
011
012import java.util.Comparator;
013import java.util.concurrent.atomic.AtomicLong;
014
015/**
016 * A {@link Comparator} that creates a stable, total ordering of all Java objects.
017 *
018 * <p>
019 * Instances are compared first using {@linkplain System#identityHashCode identity hash codes}, and then using
020 * an internally generated unique identifier; identifiers are indexed using weak keys to avoid memory leaks.
021 * Any two distict Java objects will always compare as non-equal.
022 *
023 * <p>
024 * Note: this ordering is only <i>consistent with equals</i> (see {@link Comparable}) for classes whose
025 * {@link #equals equals()} method is implemented in terms of object equality (e.g., not overridden at all).
026 *
027 * <p>
028 * Note: while each instance creates a stable sort ordering, distinct instances of this class may sort objects differently.
029 *
030 * <p>
031 * Null values are supported and always sort last. Instances of this class are thread safe.
032 *
033 * <p>
034 * This class requires <a href="https://github.com/google/guava">Google Guava</a>.
035 */
036public class ObjectComparator implements Comparator<Object> {
037
038    private final AtomicLong nextId = new AtomicLong();
039    private final LoadingCache<Object, Long> idMap;
040
041    /**
042     * Default constructor.
043     */
044    public ObjectComparator() {
045        this(CacheBuilder.newBuilder());
046    }
047
048    /**
049     * Constructor.
050     *
051     * @param concurrencyLevel guides the allowed concurrency among update operations; used as a hint for internal sizing
052     * @throws IllegalArgumentException if {@code concurrencyLevel} is zero or less
053     * @see CacheBuilder#concurrencyLevel
054     */
055    public ObjectComparator(int concurrencyLevel) {
056        this(CacheBuilder.newBuilder().concurrencyLevel(concurrencyLevel));
057    }
058
059    /**
060     * Internal constructor.
061     *
062     * @param cacheBuilder object unique ID cache builder
063     * @throws NullPointerException if {@code cacheBuilder} is null
064     */
065    protected ObjectComparator(CacheBuilder<Object, Object> cacheBuilder) {
066        this.idMap = cacheBuilder.weakKeys().<Object, Long>build(new CacheLoader<Object, Long>() {
067            @Override
068            public Long load(Object obj) {
069                return ObjectComparator.this.nextId.getAndIncrement();
070            }
071        });
072    }
073
074    @Override
075    public int compare(Object obj1, Object obj2) {
076
077        // Handle the equality case first to avoid creating unique ID's for no reason
078        if (obj1 == obj2)
079            return 0;
080
081        // Handle one of the objects being null
082        if (obj1 == null)
083            return 1;
084        if (obj2 == null)
085            return -1;
086
087        // Compare hash values
088        final int hash1 = System.identityHashCode(obj1);
089        final int hash2 = System.identityHashCode(obj2);
090        if (hash1 < hash2)
091            return -1;
092        if (hash1 > hash2)
093            return 1;
094
095        // Compare unique serial numbers (it should be very rare that we get to this point)
096        final long id1 = this.idMap.getUnchecked(obj1);
097        final long id2 = this.idMap.getUnchecked(obj2);
098        assert id1 != id2;
099        return Long.compare(id1, id2);
100    }
101}

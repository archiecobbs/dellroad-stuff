006package org.dellroad.stuff.java;
007
008import java.util.concurrent.Callable;
009
010/**
011 * Manages a thread local whose lifetime matches the duration of some method call.
012 *
013 * <p>
014 * This class is useful for this common pattern:
015 * <ul>
016 *  <li>A thread local variable is instantiated by some initial method call and has an intended
017 *      lifetime matching the duration of that method call;</li>
018 *  <li>The thread local variable is accessible from some other nested method calls in the same thread,
019 *      as long as the initial method call is still executing;</li>
020 *  <li>The thread local variable is removed (and optionally cleaned up) when the initial method call exits,
021 *      whether successfully or not.</li>
022 * </ul>
023 * </p>
024 *
025 * <p>
026 * Example:
027 * <blockquote><pre>
028 * public class Activity {
029 *
030 *     private static final ThreadLocalHolder&lt;Activity&gt; CURRENT_ACTIVITY = new ThreadLocalHolder&lt;Activity&gt;();
031 *
032 *     public void perform(final Object parameter) {
033 *         CURRENT_ACTIVITY.invoke(this, new Runnable() {
034 *             &#64;Override
035 *             public void run() {
036 *                 // do whatever with parameter
037 *             }
038 *         });
039 *     }
040 *
041 *     /**
042 *      * Get the current activity being performed.
043 *      *
044 *      * @throws IllegalStateException if there is no current activity
045 *      *&#47;
046 *     public static Activity currentActivity() {
047 *         return CURRENT_ACTIVITY.require();
048 *     }
049 * }
050 * </pre></blockquote>
051 * </p>
052 *
053 * @param <T> the type of the thread local variable
054 */
055public class ThreadLocalHolder<T> {
056
057    private final ThreadLocal<T> threadLocal;
058
059    /**
060     * Conveninece constructor.
061     *
062     * <p>
063     * Equivalent to:
064     *  <blockquote><code>new ThreadLocalHolder&lt;T&gt;(new ThreadLocal&lt;T&gt;())</code></blockquote>
065     */
066    public ThreadLocalHolder() {
067        this(new ThreadLocal<T>());
068    }
069
070    /**
071     * Primary constructor.
072     *
073     * @param threadLocal the thread local to use
074     * @throws IllegalArgumentException if {@code threadLocal} is null
075     */
076    public ThreadLocalHolder(ThreadLocal<T> threadLocal) {
077        if (threadLocal == null)
078            throw new IllegalArgumentException("null threadLocal");
079        this.threadLocal = threadLocal;
080    }
081
082    /**
083     * Invoke the given action while making the given thread local variable available via {@link #get} and {@link #require}.
084     *
085     * <p>
086     * If there is already a thread local variable set for the current thread (i.e., we are already executing within
087     * an invocation of <code>ThreadLocalHolder.invoke()</code>), then if {@code value} is the exact same Java object
088     * (using object equality, not <code>equals()</code>), execution proceeds normally, otherwise an exception is thrown.
089     *
090     * @param value value for the thread local variable
091     * @param action action to invoke
092     * @throws IllegalArgumentException if either {@code action} or {@code value} is null
093     * @throws IllegalStateException if there is already a thread local variable <code>previous</code>
094     *  associated with the current thread and <code>value != previous</code>
095     */
096    public void invoke(final T value, Runnable action) {
097        if (action == null)
098            throw new IllegalArgumentException("null action");
099        if (value == null)
100            throw new IllegalArgumentException("null value");
101        final T previousValue = this.threadLocal.get();
102        final boolean topLevel = previousValue == null;
103        if (!topLevel) {
104            if (value != previousValue) {
105                throw new IllegalStateException("already executing within an invocation of ThreadLocalHolder.invoke()"
106                  + " but with a different value");
107            }
108        } else
109            this.threadLocal.set(value);
110        try {
111            action.run();
112        } finally {
113            if (topLevel) {
114                this.threadLocal.remove();
115                this.destroy(value);
116            }
117        }
118    }
119
120    /**
121     * Invoke the given action while making the given thread local variable available via {@link #get} and {@link #require}.
122     *
123     * <p>
124     * If there is already a thread local variable set for the current thread (i.e., we are already executing within
125     * an invocation of <code>ThreadLocalHolder.invoke()</code>), then if {@code value} is the exact same Java object
126     * (using object equality, not <code>equals()</code>), execution proceeds normally, otherwise an exception is thrown.
127     *
128     * @param value value for the thread local variable
129     * @param action action to invoke
130     * @throws IllegalArgumentException if either {@code action} or {@code value} is null
131     * @throws IllegalStateException if there is already a thread local variable <code>previous</code>
132     *  associated with the current thread and <code>value != previous</code>
133     * @throws Exception if {@code action} throws an {@link Exception}
134     */
135    public <R> R invoke(final T value, Callable<R> action) throws Exception {
136        if (action == null)
137            throw new IllegalArgumentException("null action");
138        if (value == null)
139            throw new IllegalArgumentException("null value");
140        final T previousValue = this.threadLocal.get();
141        final boolean topLevel = previousValue == null;
142        if (!topLevel) {
143            if (value != previousValue) {
144                throw new IllegalStateException("already executing within an invocation of ThreadLocalHolder.invoke()"
145                  + " but with a different value");
146            }
147        } else
148            this.threadLocal.set(value);
149        try {
150            return action.call();
151        } finally {
152            if (topLevel) {
153                this.threadLocal.remove();
154                this.destroy(value);
155            }
156        }
157    }
158
159    /**
160     * Get the thread local value associated with the current thread, if any.
161     *
162     * @return the current thread local variable value, or null if not executing
163     *  within an invocation of <code>ThreadLocalHolder.invoke()</code>
164     */
165    public T get() {
166        return this.threadLocal.get();
167    }
168
169    /**
170     * Get the thread local value associated with the current thread; there must be one.
171     *
172     * @return the current thread local variable value, never null
173     * @throws IllegalStateException if the current thread is not running
174     *  within an invocation of <code>ThreadLocalHolder.invoke()</code>
175     */
176    public T require() {
177        T value = this.threadLocal.get();
178        if (value == null) {
179            throw new IllegalStateException("no value associated with the current thread;"
180              + " are we executing within an invocation of ThreadLocalHolder.invoke()?");
181        }
182        return value;
183    }
184
185    /**
186     * Clean up the thread local value when no longer needed.
187     *
188     * <p>
189     * The implementation in {@link ThreadLocalHolder} does nothing. Subclasses may override if necessary.
190     */
191    protected void destroy(T value) {
192    }
193}

006package org.dellroad.stuff.java;
007
008import org.slf4j.LoggerFactory;
009
010/**
011 * Enumeration of possible actions to take when an error of some kind is detected.
012 */
013public enum ErrorAction {
014
015    /**
016     * Ignore the error: do nothing.
017     */
018    IGNORE {
019        @Override
020        public void execute(String message) {
021        }
022    },
023
024    /**
025     * Log the error.
026     */
027    LOG {
028        @Override
029        public void execute(String message) {
030            LoggerFactory.getLogger(this.getClass()).error(message);
031        }
032    },
033
034    /**
035     * Raise an assertion error (if assertions are enabled).
036     */
037    ASSERT {
038        @Override
039        public void execute(String message) {
040            assert false : message;
041        }
042    },
043
044    /**
045     * Throw a {@link RuntimeException}.
046     */
047    EXCEPTION {
048        @Override
049        public void execute(String message) {
050            throw new RuntimeException(message);
051        }
052    };
053
054    /**
055     * Take the action appropriate for this instance.
056     *
057     * @param message a message describing the error
058     * @throws AssertionError if this instance is {@link #ASSERT} and assertions are enabled
059     * @throws RuntimeException if this instance is {@link #EXCEPTION}
060     */
061    public abstract void execute(String message);
062}

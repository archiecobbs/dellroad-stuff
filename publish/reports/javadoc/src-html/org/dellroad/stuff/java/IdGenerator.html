006package org.dellroad.stuff.java;
007
008import java.lang.ref.Reference;
009import java.lang.ref.ReferenceQueue;
010import java.lang.ref.WeakReference;
011import java.util.HashMap;
012import java.util.concurrent.Callable;
013
014/**
015 * Registry of unique IDs for objects.
016 *
017 * <p>
018 * Instances support creating unique {@code long} ID numbers for objects, as well as setting the unique ID
019 * to a specific value for any unregistered object.
020 *
021 * <p>
022 * This class uses object identity, not {@link Object#equals Object.equals()}, to distinguish objects.
023 *
024 * <p>
025 * Weak references are used to ensure that registered objects can be garbage collected normally.
026 *
027 * <p>
028 * New {@code long} ID numbers are issued serially; after 2<sup>64</sup>-1 invocations of {@link #getId getId()},
029 * an {@link IllegalStateException} will be thrown.
030 *
031 * @see org.dellroad.stuff.jibx.IdMapper
032 */
033public class IdGenerator {
034
035    private static final ThreadLocalHolder<IdGenerator> CURRENT = new ThreadLocalHolder<IdGenerator>();
036
037    private final HashMap<Ref, Long> idMap = new HashMap<Ref, Long>();
038    private final HashMap<Long, Ref> refMap = new HashMap<Long, Ref>();
039    private final ReferenceQueue<Object> queue = new ReferenceQueue<Object>();
040
041    private long next = 1;
042
043    /**
044     * Get a unique ID for the given object.
045     *
046     * <p>
047     * If this method has been previously invoked on this instance with the same {@code obj} parameter (where "same" means
048     * object identity, not {@link Object#equals Object.equals()} identity), then the same ID value will be returned.
049     * Otherwise a new ID value will be returned.
050     *
051     * <p>
052     * New IDs are assigned sequentially starting at {@code 1}. No conflict avoidance with IDs assigned
053     * via {@link #setId setId()} is performed; if there is a conflict, an exception is thrown.
054     *
055     * @throws IllegalArgumentException if {@code obj} is null
056     * @throws IllegalStateException if the next sequential ID has already been assigned to a different object
057     *  via {@link #setId setId()}
058     * @throws IllegalStateException if all 2<sup>64</sup>-1 values have been used up
059     * @return a non-zero, unique identifier for {@code obj}
060     */
061    public synchronized long getId(Object obj) {
062        if (obj == null)
063            throw new IllegalArgumentException("null obj");
064        this.flush();
065        Long id = this.idMap.get(new Ref(obj));         // don't register on queue just yet
066        if (id == null) {
067            if (this.next == 0)
068                throw new IllegalStateException("no more identifiers left!");
069            Ref ref = new Ref(obj, this.queue);             // now register on queue
070            id = this.next++;
071            this.idMap.put(ref, id);
072            this.refMap.put(id, ref);
073        }
074        return id;
075    }
076
077    /**
078     * Test whether the given object is already registered with this instance.
079     *
080     * @throws IllegalArgumentException if {@code obj} is null
081     * @return a non-zero, unique identifier for {@code obj} if already registered, otherwise zero
082     */
083    public synchronized long checkId(Object obj) {
084        if (obj == null)
085            throw new IllegalArgumentException("null obj");
086        this.flush();
087        Long id = this.idMap.get(new Ref(obj));
088        return id != null ? id : 0;
089    }
090
091    /**
092     * Assign a unique ID to the given object. Does nothing if the object and ID number are already associated.
093     *
094     * @param obj object to assign
095     * @param id unique ID number to assign
096     * @throws IllegalArgumentException if {@code obj} is null
097     * @throws IllegalArgumentException if {@code id} has already been assigned to some other object
098     */
099    public synchronized void setId(Object obj, long id) {
100        if (obj == null)
101            throw new IllegalArgumentException("null obj");
102        this.flush();
103        Ref ref = this.refMap.get(id);
104        if (ref != null) {
105            if (ref.get() != obj)
106                throw new IllegalArgumentException("id " + id + " is already assigned to another object");
107            return;
108        }
109        ref = new Ref(obj, this.queue);
110        this.idMap.put(ref, id);
111        this.refMap.put(id, ref);
112    }
113
114    /**
115     * Get the object assigned to the given ID.
116     *
117     * @param id unique ID
118     * @return object associated with that ID, or null if no object is assigned to {@code id}
119     */
120    public synchronized Object getObject(long id) {
121        this.flush();
122        Ref ref = this.refMap.get(id);
123        return ref != null ? ref.get() : null;
124    }
125
126    /**
127     * Flush any cleared weak references.
128     *
129     * <p>
130     * This operation is invoked by {@link #getId getId()} and {@link #setId setId()}, so it's usually not necessary
131     * to explicitly invoke it. However, if a lot of previously ID'd objects have been garbage collected since the
132     * last call to {@link #getId getId()}, then invoking this method may free up some additional memory.
133     */
134    public synchronized void flush() {
135        Reference<? extends Object> entry;
136        while ((entry = this.queue.poll()) != null) {
137            Ref ref = (Ref)entry;
138            Long id = this.idMap.get(ref);
139            this.idMap.remove(ref);
140            this.refMap.remove(id);
141        }
142    }
143
144    /**
145     * Create a new {@link IdGenerator} and make it available via {@link #get()} for the duration of the given operation.
146     *
147     * <p>
148     * This method is re-entrant: nested invocations of this method in the same thread re-use the same {@link IdGenerator}
149     * instance.
150     *
151     * @param action action to perform, and which may successfully invoke {@link #get}
152     * @throws NullPointerException if {@code action} is null
153     */
154    public static void run(final Runnable action) {
155        IdGenerator current = IdGenerator.CURRENT.get();
156        if (current == null)
157            current = new IdGenerator();
158        IdGenerator.CURRENT.invoke(current, action);
159    }
160
161    /**
162     * Create a new {@link IdGenerator} and make it available via {@link #get()} for the duration of the given operation.
163     *
164     * <p>
165     * This method is re-entrant: nested invocations of this method in the same thread re-use the same {@link IdGenerator}
166     * instance.
167     *
168     * @param action action to perform, and which may successfully invoke {@link #get}
169     * @return result of invoking {@code action}
170     * @throws NullPointerException if {@code action} is null
171     */
172    public static <R> R run(final Callable<R> action) throws Exception {
173        IdGenerator current = IdGenerator.CURRENT.get();
174        if (current == null)
175            current = new IdGenerator();
176        return IdGenerator.CURRENT.invoke(current, action);
177    }
178
179    /**
180     * Get the {@link IdGenerator} associated with the current thread.
181     * This method only works when the current thread is running within an invocation of {@link #run run()};
182     * otherwise, an {@link IllegalStateException} is thrown.
183     *
184     * @return the {@link IdGenerator} created in the most recent, still running invocation of {@link #run} in this thread
185     * @throws IllegalStateException if there is not such instance
186     */
187    public static IdGenerator get() {
188        return IdGenerator.CURRENT.require();
189    }
190
191// Reference to a registered object that weakly references the actual object
192
193    private static final class Ref extends WeakReference<Object> {
194
195        private final int hashCode;
196
197        Ref(Object obj, ReferenceQueue<Object> queue) {
198            super(obj, queue);
199            if (obj == null)
200                throw new IllegalArgumentException("null obj");
201            this.hashCode = System.identityHashCode(obj);
202        }
203
204        Ref(Object obj) {
205            this(obj, null);
206        }
207
208        @Override
209        public boolean equals(Object obj) {
210            if (obj == this)
211                return true;
212            if (obj == null || obj.getClass() != this.getClass())
213                return false;
214            Ref that = (Ref)obj;
215            obj = this.get();
216            return obj != null ? obj == that.get() : false;
217        }
218
219        @Override
220        public int hashCode() {
221            return this.hashCode;
222        }
223    }
224}

006package org.dellroad.stuff.java;
007
008/**
009 * Wraps checked exceptions so they can be thrown across API methods that don't declare them.
010 */
011@SuppressWarnings("serial")
012public class CheckedExceptionWrapper extends RuntimeException {
013
014    private final Exception exception;
015
016    /**
017     * Constructor.
018     *
019     * @param exception the checked exception to wrap
020     * @throws IllegalArgumentException if {@code exception} is {@code null}
021     */
022    public CheckedExceptionWrapper(Exception exception) {
023        if (exception == null)
024            throw new IllegalArgumentException("null exception");
025        this.exception = exception;
026    }
027
028    /**
029     * Get the wrapped exception.
030     *
031     * @return the wrapped checked exception
032     */
033    public Exception getException() {
034        return this.exception;
035    }
036
037    /**
038     * Throw the wrapped exception.
039     *
040     * @throws Exception always
041     */
042    public void throwException() throws Exception {
043        throw this.exception;
044    }
045}

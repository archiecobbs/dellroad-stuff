006package org.dellroad.stuff.java;
007
008/**
009 * Implemented by objects in an object graph that support <i>graph cloning</i>, i.e., deep copies without
010 * duplicates and preserving reference topology.
011 *
012 * <p>
013 * Graph cloning creates a copy of an object graph that, like a normal deep copy, contains no references to objects in the
014 * original object graph (except for immutables), but in addition, each {@link GraphCloneable} object is only copied once.
015 * As a result the copy graph has the same reference topology as the original graph (with respect to all the
016 * {@link GraphCloneable}s). In particular, reference cycles among {@link GraphCloneable}s are preserved and do not
017 * cause infinite loops.
018 *
019 * <p>
020 * Graph cloning operates similar to a deep copy operation, except that graph cloning uses a {@link GraphCloneRegistry} to keep
021 * track of new objects as they are created. When each object's {@link GraphCloneable} references are copied, the
022 * {@link GraphCloneRegistry} is used to check whether the referred-to objects have already been copied, and if so, the existing
023 * copy is used. This requires that implementations {@linkplain GraphCloneRegistry#setGraphClone register} their clones
024 * prior to {@linkplain GraphCloneRegistry#getGraphClone recursing} on any {@link GraphCloneable} fields.
025 *
026 * <p>
027 * The net effect is equivalent to serializing and then deserializing the entire object graph, but without the overhead,
028 * losing {@code transient} values, and other issues.
029 *
030 * <p>
031 * Here is an example of a class properly implementing this interface:
032 *  <pre>
033 *  // We implement Cloneable so super.clone() will work
034 *  public class Person implements Cloneable, GraphCloneable {
035 *
036 *      // Regular fields
037 *      private int age;
038 *      private String lastName;
039 *      private String firstName;
040 *      private List&lt;String&gt; nicknames = new ArrayList&lt;String&gt;();
041 *
042 *      // GraphCloneable fields - values may be null and/or even refer back to me
043 *      private Person spouse;
044 *      private List&lt;Person&gt; friends = new ArrayList&lt;Person&gt;();
045 *
046 *      // Getters &amp; setters go here...
047 *
048 *      // Our implementation of the GraphCloneable interface
049 *      &#64;Override
050 *      public void createGraphClone(GraphCloneRegistry registry) throws CloneNotSupportedException {
051 *
052 *          // Create clone and register it with the registry
053 *          final Person clone = (Person)super.clone();         // this copies all of the simple fields
054 *          registry.setGraphClone(clone);                      // let the registry know who our clone is
055 *
056 *          // Deep copy any non-GraphCloneable fields not already handled by super.clone()
057 *          clone.nicknames = new ArrayList&lt;String&gt;(this.nicknames);
058 *
059 *          // Now copy GraphCloneable fields using registry.getGraphClone()
060 *          clone.spouse = registry.getGraphClone(this.spouse);
061 *          clone.friends = new ArrayList&lt;Person&gt;(this.friends.size());
062 *          for (Person friend : this.friends)
063 *              clone.friends.add(registry.getGraphClone(friend));
064 *      }
065 *  }
066 *  </pre>
067 *
068 * <p>
069 * To graph clone any object graph rooted at {@code root}, you would do this:
070 *  <pre>
071 *      new GraphCloneRegistry().getGraphClone(root);
072 *  </pre>
073 *
074 * @see GraphCloneRegistry
075 */
076public interface GraphCloneable {
077
078    /**
079     * Create a graph clone of this instance and register it with the given {@link GraphCloneRegistry}.
080     *
081     * <p>
082     * This method should perform a normal "deep copy" operation, but with the following changes:
083     * <ul>
084     *  <li>
085     *      The new clone must be {@linkplain GraphCloneRegistry#setGraphClone registered} with the provided
086     *      {@link GraphCloneRegistry} before this method returns.
087     *  </li>
088     *  <li>
089     *      The {@linkplain GraphCloneRegistry#setGraphClone registration} in the previous step must occur
090     *      prior to {@linkplain GraphCloneRegistry#getGraphClone recursing} on any {@link GraphCloneable} fields.
091     *  </li>
092     *  <li>
093     *      All {@link GraphCloneable} fields must be copied via {@link GraphCloneRegistry#getGraphClone}.
094     *  </li>
095     * </ul>
096     *
097     * <p>
098     * The most efficient implementation of this method often involves declaring the class to implement {@link Cloneable}
099     * and starting with an invocation of {@link Object#clone super.clone()} (the class need not actually override
100     * {@link Object#clone}). For that reason, this method is declared to throw {@link CloneNotSupportedException}
101     * as a coding convenience; if a {@link CloneNotSupportedException} is actually thrown, it will trigger a
102     * {@link RuntimeException}.
103     *
104     * <p>
105     * This method will only be invoked once for any instance during a graph cloning operation.
106     *
107     * @param registry registry for clones
108     * @throws NullPointerException if {@code registry} is null
109     * @throws CloneNotSupportedException declared so implementors can invoke {@link Object#clone super.clone()}
110     *  directly; if thrown, it will trigger a {@link RuntimeException}
111     */
112    void createGraphClone(GraphCloneRegistry registry) throws CloneNotSupportedException;
113}

006package org.dellroad.stuff.servlet;
007
008import java.sql.Driver;
009import java.sql.DriverManager;
010import java.util.Enumeration;
011
012import javax.servlet.ServletContextEvent;
013
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016import org.springframework.web.util.IntrospectorCleanupListener;
017
018/**
019 * Clears references that can cause container memory leaks on Web container class unloading.
020 *
021 * @see <a href="http://opensource.atlassian.com/confluence/spring/pages/viewpage.action?pageId=2669">Spring Wiki Discussion of
022 *      ClassLoader Memory Leaks</a>
023 * @see <a href="http://bugs.mysql.com/bug.php?id=65909">MySQL Bug #65909</a>
024 * @since 1.0.49
025 */
026public class ContextCleanupListener extends IntrospectorCleanupListener {
027
028    protected final Logger log = LoggerFactory.getLogger(getClass());
029
030    /**
031     * Unregisters any JDBC drivers registered under the current class loader after
032     * first invoking the superclass version of this method.
033     */
034    @Override
035    public void contextDestroyed(ServletContextEvent event) {
036        try {
037            super.contextDestroyed(event);
038
039            // Get my class loader
040            final ClassLoader thisClassLoader = this.getClass().getClassLoader();
041
042            // Work around MySQL bug http://bugs.mysql.com/bug.php?id=65909
043            try {
044                Class.forName("com.mysql.jdbc.AbandonedConnectionCleanupThread", false,
045                  thisClassLoader).getMethod("shutdown").invoke(null);
046            } catch (Exception e) {
047                // ignore
048            }
049
050            // Unregister JDBC drivers
051            for (Enumeration<Driver> e = DriverManager.getDrivers(); e.hasMoreElements(); ) {
052                final Driver driver = e.nextElement();
053                if (driver.getClass().getClassLoader() == thisClassLoader)
054                    DriverManager.deregisterDriver(driver);
055            }
056        } catch (Throwable e) {
057            log.error("exception cleaning up servlet context", e);
058        }
059    }
060}

006package org.dellroad.stuff.servlet;
007
008import java.io.IOException;
009import java.net.SocketException;
010
011import javax.servlet.FilterChain;
012import javax.servlet.ServletException;
013import javax.servlet.http.HttpServletRequest;
014import javax.servlet.http.HttpServletResponse;
015
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018import org.springframework.web.filter.OncePerRequestFilter;
019
020/**
021 * Servlet filter that ensures any exceptions which are not handled by the servlet get logged somewhere.
022 */
023public class ExceptionLoggingFilter extends OncePerRequestFilter {
024
025    private final Logger defaultLogger = LoggerFactory.getLogger(this.getClass());
026
027    /**
028     * Process the request. If an exception is thrown, it will be (possibly) logged and re-thrown.
029     *
030     * @see #shouldLogException shouldLogException()
031     */
032    @Override
033    public void doFilterInternal(HttpServletRequest request, HttpServletResponse response, FilterChain filterChain)
034      throws IOException, ServletException {
035        try {
036            filterChain.doFilter(request, response);
037        } catch (IOException e) {
038            if (this.shouldLogException(request, response, e))
039                this.logException(request, response, e);
040            throw e;
041        } catch (ServletException e) {
042            if (this.shouldLogException(request, response, e))
043                this.logException(request, response, e);
044            throw e;
045        } catch (RuntimeException e) {
046            if (this.shouldLogException(request, response, e))
047                this.logException(request, response, e);
048            throw e;
049        } catch (Error e) {
050            if (this.shouldLogException(request, response, e))
051                this.logException(request, response, e);
052            throw e;
053        } catch (Throwable e) {
054            if (this.shouldLogException(request, response, e))
055                this.logException(request, response, e);
056           throw new ServletException(e);
057        }
058    }
059
060    /**
061     * Determine if an exception caught by this instance should be logged.
062     *
063     * <p>
064     * The implementation in {@link ExceptionLoggingFilter} returns {@code true} except for
065     * {@link SocketException} (typically caused by the client disconnecting) and {@link ThreadDeath}
066     * (typically caused by virtual machine shutdown). Subclasses should override if necessary.
067     * </p>
068     *
069     * @param request HTTP request
070     * @param response HTTP response
071     * @param t exception caught by this instance
072     */
073    protected boolean shouldLogException(HttpServletRequest request, HttpServletResponse response, Throwable t) {
074        for (Throwable e = t; e != null; e = e.getCause()) {
075            if (e instanceof SocketException)
076                return false;
077            if (e.getClass().getName().equals("org.apache.catalina.connector.ClientAbortException"))
078                return false;
079            if (e instanceof ThreadDeath)
080                return false;
081        }
082        return true;
083    }
084
085    /**
086     * Log an exception caught by this instance and determined to be loggable by {@link #shouldLogException shouldLogException()}.
087     *
088     * <p>
089     * The implementation in {@link ExceptionLoggingFilter} logs the exception as an error via the logger
090     * returned by {@link #getLogger getLogger()}. Subclasses should override if necessary.
091     * </p>
092     *
093     * @param request HTTP request
094     * @param response HTTP response
095     * @param t exception caught by this instance
096     */
097    protected void logException(HttpServletRequest request, HttpServletResponse response, Throwable t) {
098        this.getLogger(request, response, t).error("exception within servlet", t);
099    }
100
101    /**
102     * Get the logging destination.
103     *
104     * <p>
105     * The implementation in {@link ExceptionLoggingFilter} uses the {@link Logger} returned by
106     * {@link LoggerFactory#getLogger} when passed this instance's class as the parameter.
107     * Subclasses should override if necessary.
108     * </p>
109     *
110     * @param request HTTP request
111     * @param response HTTP response
112     * @param t the exception about to be logged
113     */
114    protected Logger getLogger(HttpServletRequest request, HttpServletResponse response, Throwable t) {
115        return this.defaultLogger;
116    }
117}

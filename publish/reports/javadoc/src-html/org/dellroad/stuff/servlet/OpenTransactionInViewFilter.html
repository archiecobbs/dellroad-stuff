006package org.dellroad.stuff.servlet;
007
008import java.io.IOException;
009
010import javax.servlet.FilterChain;
011import javax.servlet.ServletException;
012import javax.servlet.http.HttpServletRequest;
013import javax.servlet.http.HttpServletResponse;
014
015import org.slf4j.Logger;
016import org.slf4j.LoggerFactory;
017import org.springframework.transaction.PlatformTransactionManager;
018import org.springframework.transaction.TransactionStatus;
019import org.springframework.transaction.support.DefaultTransactionDefinition;
020import org.springframework.transaction.support.TransactionCallback;
021import org.springframework.transaction.support.TransactionTemplate;
022import org.springframework.web.context.WebApplicationContext;
023import org.springframework.web.context.support.WebApplicationContextUtils;
024import org.springframework.web.filter.OncePerRequestFilter;
025
026/**
027 * Servlet filter that wraps execution in a transaction. A {@link TransactionTemplate} must exist in the
028 * associated {@link WebApplicationContext} and is identified by name via {@link #setTransactionManagerBeanName
029 * setTransactionManagerBeanName()} (by default, {@link #DEFAULT_TRANSACTION_MANAGER_BEAN_NAME}).
030 *
031 * <p>
032 * Transaction properties are configurable via filter <code>&lt;init-param&gt;</code>'s {@code isolation},
033 * {@code propagation}, and {@code readOnly}.
034 */
035public class OpenTransactionInViewFilter extends OncePerRequestFilter {
036
037    public static final String DEFAULT_TRANSACTION_MANAGER_BEAN_NAME = "transactionManager";
038
039    protected final Logger log = LoggerFactory.getLogger(getClass());
040
041    private final DefaultTransactionDefinition transactionDefinition = new DefaultTransactionDefinition();
042
043    private WebApplicationContext webApplicationContext;
044    private String transactionManagerBeanName = DEFAULT_TRANSACTION_MANAGER_BEAN_NAME;
045    private PlatformTransactionManager transactionManager;
046
047    @Override
048    protected void initFilterBean() throws ServletException {
049        super.initFilterBean();
050        log.debug("finding containing WebApplicationContext");
051        try {
052            this.webApplicationContext = WebApplicationContextUtils.getRequiredWebApplicationContext(getServletContext());
053        } catch (IllegalStateException e) {
054            throw new ServletException("could not locate containing WebApplicationContext");
055        }
056    }
057
058    public void setIsolation(String isolation) {
059        this.transactionDefinition.setIsolationLevelName(isolation);
060    }
061
062    public void setPropagation(String propagation) {
063        this.transactionDefinition.setPropagationBehaviorName(propagation);
064    }
065
066    public void setReadOnly(boolean readOnly) {
067        this.transactionDefinition.setReadOnly(readOnly);
068    }
069
070    public void setTransactionManagerBeanName(String transactionManagerBeanName) {
071        this.transactionManagerBeanName = transactionManagerBeanName;
072    }
073
074    protected synchronized PlatformTransactionManager getTransactionManager() {
075        if (this.transactionManager == null) {
076            this.transactionManager = this.webApplicationContext.getBean(
077              this.transactionManagerBeanName, PlatformTransactionManager.class);
078        }
079        return this.transactionManager;
080    }
081
082    @Override
083    protected void doFilterInternal(final HttpServletRequest request, final HttpServletResponse response,
084      final FilterChain filterChain) throws ServletException, IOException {
085        TransactionTemplate transactionTemplate = new TransactionTemplate(getTransactionManager(), this.transactionDefinition);
086        try {
087            transactionTemplate.execute(new TransactionCallback<Void>() {
088
089                @Override
090                public Void doInTransaction(TransactionStatus status) {
091                    try {
092                        filterChain.doFilter(request, response);
093                    } catch (IOException e) {
094                        throw new RuntimeException(e);
095                    } catch (ServletException e) {
096                        throw new RuntimeException(e);
097                    }
098                    return null;
099                }
100            });
101        } catch (RuntimeException e) {
102            Throwable nested = e.getCause();
103            if (nested instanceof IOException)
104                throw (IOException)nested;
105            if (nested instanceof ServletException)
106                throw (ServletException)nested;
107            throw e;
108        }
109    }
110}

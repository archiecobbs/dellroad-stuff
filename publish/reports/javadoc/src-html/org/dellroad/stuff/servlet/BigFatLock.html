006package org.dellroad.stuff.servlet;
007
008import java.io.IOException;
009import java.util.concurrent.Callable;
010
011import javax.servlet.FilterChain;
012import javax.servlet.ServletException;
013import javax.servlet.http.HttpServletRequest;
014import javax.servlet.http.HttpServletResponse;
015
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018import org.springframework.web.filter.OncePerRequestFilter;
019
020/**
021 * Class for serializing processing using a big fat lock.
022 *
023 * <p>
024 * The lock can be acquired using {@link #runWithLock}. This class can also be used as a servlet filter.
025 *
026 * <p>
027 * Trace-level logging for displaying lock acquire and release events is included.
028 */
029public final class BigFatLock extends OncePerRequestFilter {
030
031    private static final Object BIG_FAT_LOCK = new BigFatLock();
032    private static final Logger LOG = LoggerFactory.getLogger(BigFatLock.class);
033
034    @Override
035    protected void doFilterInternal(HttpServletRequest request, HttpServletResponse response,
036      FilterChain filterChain) throws ServletException, IOException {
037        if (LOG.isTraceEnabled()) {
038            LOG.trace("thread " + Thread.currentThread() + " waiting for " + BIG_FAT_LOCK);
039            synchronized (BIG_FAT_LOCK) {
040                LOG.trace("thread " + Thread.currentThread() + " acquired " + BIG_FAT_LOCK);
041                try {
042                    filterChain.doFilter(request, response);
043                } finally {
044                    LOG.trace("thread " + Thread.currentThread() + " releasing " + BIG_FAT_LOCK);
045                }
046            }
047        } else {
048            synchronized (BIG_FAT_LOCK) {
049                filterChain.doFilter(request, response);
050            }
051        }
052    }
053
054    /**
055     * Invoke the given action after acquiring the big fat lock.
056     *
057     * @param action action to perform while locked
058     */
059    public static void runWithLock(Runnable action) {
060        if (LOG.isTraceEnabled()) {
061            LOG.trace("thread " + Thread.currentThread() + " waiting for " + BIG_FAT_LOCK);
062            synchronized (BIG_FAT_LOCK) {
063                LOG.trace("thread " + Thread.currentThread() + " acquired " + BIG_FAT_LOCK);
064                try {
065                    action.run();
066                } finally {
067                    LOG.trace("thread " + Thread.currentThread() + " releasing " + BIG_FAT_LOCK);
068                }
069            }
070        } else {
071            synchronized (BIG_FAT_LOCK) {
072                action.run();
073            }
074        }
075    }
076
077    /**
078     * Invoke the given action after acquiring the big fat lock.
079     *
080     * @param <V> action return type
081     * @param action action to perform while locked
082     * @return result from {@code action}
083     * @throws Exception if {@code action} throws an exception
084     */
085    public static <V> V runWithLock(Callable<V> action) throws Exception {
086        if (LOG.isTraceEnabled()) {
087            LOG.trace("thread " + Thread.currentThread() + " waiting for " + BIG_FAT_LOCK);
088            synchronized (BIG_FAT_LOCK) {
089                LOG.trace("thread " + Thread.currentThread() + " acquired " + BIG_FAT_LOCK);
090                try {
091                    return action.call();
092                } finally {
093                    LOG.trace("thread " + Thread.currentThread() + " releasing " + BIG_FAT_LOCK);
094                }
095            }
096        } else {
097            synchronized (BIG_FAT_LOCK) {
098                return action.call();
099            }
100        }
101    }
102
103    /**
104     * Verify that the current thread is holding the big fat lock.
105     *
106     * @return true if lock is held
107     */
108    public static boolean isLockHeld() {
109        return Thread.holdsLock(BIG_FAT_LOCK);
110    }
111
112    @Override
113    public String toString() {
114        return "BigFatLock@" + System.identityHashCode(this);
115    }
116}

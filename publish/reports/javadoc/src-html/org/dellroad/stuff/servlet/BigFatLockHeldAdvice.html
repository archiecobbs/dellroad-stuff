006package org.dellroad.stuff.servlet;
007
008import java.lang.reflect.Method;
009
010import org.springframework.aop.aspectj.AspectInstanceFactory;
011import org.springframework.aop.aspectj.AspectJExpressionPointcut;
012import org.springframework.aop.aspectj.AspectJMethodBeforeAdvice;
013
014/**
015 * AspectJ method "before advice" that verifies that the {@link BigFatLock} is held.
016 */
017public class BigFatLockHeldAdvice extends AspectJMethodBeforeAdvice {
018
019    /**
020     * Constructor.
021     *
022     * @param aspectJBeforeAdviceMethod method to advise
023     * @param pointcut pointcut
024     * @param aif factory for aspect instance
025     */
026    public BigFatLockHeldAdvice(Method aspectJBeforeAdviceMethod, AspectJExpressionPointcut pointcut, AspectInstanceFactory aif) {
027        super(aspectJBeforeAdviceMethod, pointcut, aif);
028    }
029
030    /**
031     * Verifies the {@link BigFatLock} is held. If not, {@link #lockNotHeld} is invoked.
032     */
033    @Override
034    public void before(Method method, Object[] args, Object target) {
035        if (!BigFatLock.isLockHeld())
036            this.lockNotHeld();
037    }
038
039    /**
040     * Invoked in cases where the current thread does not hold the {@link BigFatLock}.
041     *
042     * <p>
043     * The implementation in {@link BigFatLockHeldAdvice} throws a {@link BigFatLockNotHeldException}.
044     * Subclasses can override to suit.
045     */
046    protected void lockNotHeld() {
047        throw new BigFatLockNotHeldException();
048    }
049}

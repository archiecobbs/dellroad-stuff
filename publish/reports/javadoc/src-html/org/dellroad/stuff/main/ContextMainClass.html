006package org.dellroad.stuff.main;
007
008import org.springframework.context.support.ClassPathXmlApplicationContext;
009
010/**
011 * Support superclass for {@link MainClass} implementations that wish to execute
012 * with an associated Spring application context.
013 */
014public abstract class ContextMainClass extends MainClass {
015
016    protected ClassPathXmlApplicationContext context;
017
018    private void openContext() {
019        String path = getContextLocation();
020        this.log.info("opening application context " + path);
021        this.context = new ClassPathXmlApplicationContext(path, getClass());
022    }
023
024    private void closeContext() {
025        this.log.info("closing application context");
026        this.context.close();
027        this.context = null;
028    }
029
030    /**
031     * Get the classpath location of this instance's associated XML application context.
032     * The returned string will resolved on the classpath relative to this instance's class.
033     *
034     * <p>
035     * The implementation in {@link ContextMainClass} returns {@code getClass().getSimpleName() + ".xml"},
036     * which will locate an XML file in the same package and with the same name.
037     *
038     * @return application context XML file location
039     */
040    protected String getContextLocation() {
041        return getClass().getSimpleName() + ".xml";
042    }
043
044    /**
045     * Autowire this instance using its associated application context.
046     * This may be invoked by {@link #runInContext} to autowire this bean using its associated context.
047     *
048     * <p>
049     * For this to work, the application context must have autowiring enabled, e.g., via
050     * {@code <context:annotation-config/>}.
051     */
052    protected void autowire() {
053        this.log.info("autowiring instance of " + this.getClass() + " using " + this.context.getAutowireCapableBeanFactory());
054        this.context.getAutowireCapableBeanFactory().autowireBean(this);
055    }
056
057    @Override
058    public int run(final String[] args) throws Exception {
059
060        // Open context
061        openContext();
062
063        // Invoke subclass
064        try {
065            return this.runInContext(args);
066        } catch (Exception e) {
067            this.log.error("caught exception during execution", e);
068            throw e;
069        } finally {
070            closeContext();
071        }
072    }
073
074    /**
075     * Execute the main method. The application context will be open when this method is invoked.
076     *
077     * @param args command line arguments
078     * @return process exit value
079     * @throws Exception if an error occurs; will result in the process exiting with an exit value of one
080     */
081    protected abstract int runInContext(String[] args) throws Exception;
082}

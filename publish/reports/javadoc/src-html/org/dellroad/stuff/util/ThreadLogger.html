006package org.dellroad.stuff.util;
007
008import java.io.IOException;
009import java.io.Writer;
010
011import org.apache.log4j.WriterAppender;
012
013/**
014 * A Log4J appender that can be configured on a per-thread basis to direct the log messages
015 * to an arbitrary {@link Writer} destination.
016 *
017 * <p>
018 * Normally this appender does nothing. However, if a thread invokes {@link #setLogDestination setLogDestination()}
019 * with a non-null {@link Writer}, then any messages logged by <i>that particular thread</i> will go to that {@link Writer}.
020 * This allows individual threads to copy and/or redirect their own log messages as they see fit.
021 * </p>
022 *
023 * <p>
024 * Configure this appender just as you would any other. For example:
025 *
026 *  <blockquote><pre>
027 *  &lt;!-- Per-thread logger --&gt;
028 *  &lt;appender name="threadlog" class="org.dellroad.stuff.util.ThreadLogger"&gt;
029 *      &lt;param name="Threshold" value="debug"/&gt;
030 *      &lt;layout class="org.apache.log4j.PatternLayout"&gt;
031 *          &lt;param name="ConversionPattern" value="%d{ISO8601} %p: %m%n"/&gt;
032 *      &lt;/layout&gt;
033 *  &lt;/appender&gt;
034 *
035 *  ...
036 *
037 *  &lt;root&gt;
038 *      ...
039 *      &lt;appender-ref ref="threadlog"&gt;
040 *  &lt;/root&gt;
041 *  </pre></blockquote>
042 * </p>
043 */
044public class ThreadLogger extends WriterAppender {
045
046    private static final InheritableThreadLocal<WriterInfo> CURRENT_WRITER = new InheritableThreadLocal<WriterInfo>() {
047        @Override
048        protected WriterInfo childValue(WriterInfo writerInfo) {
049            return writerInfo != null && writerInfo.isInherit() ? writerInfo : null;
050        }
051    };
052
053    public ThreadLogger() {
054        this.setWriter(new ThreadWriter());
055    }
056
057    /**
058     * Configure the logging output destination for the current thread.
059     * Optionally also applies to all of the current thread's descendant threads.
060     *
061     * @param writer current thread logging destination, or null for none
062     * @param inherit whether {@code writer} (if not null) should be inherited by descendant threads
063     */
064    public static void setLogDestination(Writer writer, boolean inherit) {
065        CURRENT_WRITER.set(writer != null ? new WriterInfo(writer, inherit) : null);
066    }
067
068    /**
069     * Configure the logging output destination for the current thread and all its descendant threads.
070     * This is a convenience method, equivalent to
071     * {@link #setLogDestination(Writer, boolean) setLogDestination}{@code (writer, true)}.
072     *
073     * @param writer current thread logging destination, or null for none
074     */
075    public static void setLogDestination(Writer writer) {
076        ThreadLogger.setLogDestination(writer, true);
077    }
078
079    /**
080     * Get the currently configured logging output destination for the current thread.
081     *
082     * @return current thread logging destination, or null if there is none
083     */
084    public static Writer getLogDestination() {
085        final WriterInfo writerInfo = CURRENT_WRITER.get();
086        return writerInfo != null ? writerInfo.getWriter() : null;
087    }
088
089    @Override
090    protected boolean checkEntryConditions() {
091        return CURRENT_WRITER.get() != null && super.checkEntryConditions();
092    }
093
094    // Per-thread info
095    private static class WriterInfo {
096
097        private final Writer writer;
098        private final boolean inherit;
099
100        WriterInfo(Writer writer, boolean inherit) {
101            this.writer = writer;
102            this.inherit = inherit;
103        }
104
105        public Writer getWriter() {
106            return this.writer;
107        }
108
109        public boolean isInherit() {
110            return this.inherit;
111        }
112    }
113
114    // Wrapper writer
115    private static class ThreadWriter extends Writer {
116
117        @Override
118        public void write(char[] buf, int off, int len) throws IOException {
119            final WriterInfo writerInfo = CURRENT_WRITER.get();
120            if (writerInfo != null)
121                writerInfo.getWriter().write(buf, off, len);
122        }
123
124        @Override
125        public void close() throws IOException {
126            final WriterInfo writerInfo = CURRENT_WRITER.get();
127            if (writerInfo != null)
128                writerInfo.getWriter().close();
129        }
130
131        @Override
132        public void flush() throws IOException {
133            final WriterInfo writerInfo = CURRENT_WRITER.get();
134            if (writerInfo != null)
135                writerInfo.getWriter().flush();
136        }
137    }
138}

028/**
029 * Outputs the result of an SQL query as XML.
030 *
031 * <p>
032 * For example, the query <code>SELECT ID, LAST_NAME AS LAST, FIRST_NAME AS FIRST FROM EMPLOYEE</code> might become:
033 * <pre>
034 *  &lt;result-set&gt;
035 *      &lt;query&gt;&lt;![CDATA[SELECT ID, LAST_NAME, FIRST_NAME FROM EMPLOYEE]]&gt;&lt;/query&gt;
036 *      &lt;columns&gt;
037 *          &lt;column index="1" name="ID" label="ID" precision="20" type="BIGINT" typeName="BIGINT" nullable="false"/&gt;
038 *          &lt;column index="2" name="LAST_NAME" label="LAST" precision="255" type="VARCHAR" typeName="VARCHAR" nullable="false"/&gt;
039 *          &lt;column index="3" name="FIRST_NAME" label="FIRST" precision="255" type="VARCHAR" typeName="VARCHAR" nullable="false"/&gt;
040 *      &lt;/columns&gt;
041 *      &lt;data&gt;
042 *          &lt;row&gt;
043 *              &lt;column index="1"&gt;1302&lt;/column&gt;
044 *              &lt;column index="2"&gt;Washington&lt;/column&gt;
045 *              &lt;column index="3"&gt;George&lt;/column&gt;
046 *          &lt;/row&gt;
047 *          &lt;row&gt;
048 *              &lt;column index="1"&gt;1303&lt;/column&gt;
049 *              &lt;column index="2"&gt;Lincoln&lt;/column&gt;
050 *              &lt;column index="3"&gt;Abraham&lt;/column&gt;
051 *          &lt;/row&gt;
052 *          ...
053 *      &lt;/data&gt;
054 *  &lt;/result-set&gt;
055 * </pre>
056 *
057 * <p>
058 * If you turn on column name tags via {@link #setColumnNameTags setColumnNameTags()}, the output would look like this:
059 * <pre>
060 *  &lt;result-set&gt;
061 *      &lt;query&gt;&lt;![CDATA[SELECT ID, LAST_NAME, FIRST_NAME FROM EMPLOYEE]]&gt;&lt;/query&gt;
062 *      &lt;columns&gt;
063 *          &lt;column index="1" name="ID" label="ID" precision="20" type="BIGINT" typeName="BIGINT" nullable="false"/&gt;
064 *          &lt;column index="2" name="LAST_NAME" label="LAST" precision="255" type="VARCHAR" typeName="VARCHAR" nullable="false"/&gt;
065 *          &lt;column index="3" name="FIRST_NAME" label="FIRST" precision="255" type="VARCHAR" typeName="VARCHAR" nullable="false"/&gt;
066 *      &lt;/columns&gt;
067 *      &lt;data&gt;
068 *          &lt;row&gt;
069 *              &lt;ID index="1"&gt;1302&lt;/ID&gt;
070 *              &lt;LAST index="2"&gt;Washington&lt;/LAST&gt;
071 *              &lt;FIRST index="3"&gt;George&lt;/FIRST&gt;
072 *          &lt;/row&gt;
073 *          &lt;row&gt;
074 *              &lt;ID index="1"&gt;1303&lt;/ID&gt;
075 *              &lt;LAST index="2"&gt;Lincoln&lt;/LAST&gt;
076 *              &lt;FIRST index="3"&gt;Abraham&lt;/FIRST&gt;
077 *          &lt;/row&gt;
078 *          ...
079 *      &lt;/data&gt;
080 *  &lt;/result-set&gt;
081 * </pre>
082 */
083public class XMLResultSetWriter {
084
085    /**
086     * Mapping from result set {@link Types} values to their corresponding names.
087     */
088    public static final Map<Integer, String> TYPE_NAMES;
089    static {
090        HashMap<Integer, String> map = new HashMap<Integer, String>();
091        for (Field field : Types.class.getDeclaredFields()) {
092            if (field.getType() != Integer.TYPE || field.getModifiers() != (Modifier.PUBLIC | Modifier.STATIC | Modifier.FINAL))
093                continue;
094            try {
095                map.put(field.getInt(null), field.getName());
096            } catch (IllegalAccessException e) {
097                throw new RuntimeException(e);
098            }
099        }
100        TYPE_NAMES = Collections.unmodifiableMap(map);
101    }
102
103    private static final String DATE_FORMAT = "yyyy-MM-dd";
104    private static final String TIMESTAMP_FORMAT = DATE_FORMAT + " HH:mm:ss";
105
106    private final XMLStreamWriter writer;
107    private final String indentSpace;
108
109    private State state = State.LAST_CLOSE;
110    private boolean columnNameTags;
111    private int nesting;
112
113    // Tracks state of the output
114    private enum State {
115        LAST_OPEN,
116        LAST_CLOSE,
117        LAST_TEXT
118    };
119
120    /**
121     * Constructor.
122     *
123     * @param writer where to write the XML
124     * @param indent number of spaces for indentation
125     * @throws IllegalArgumentException if {@code writer} is null
126     */
127    public XMLResultSetWriter(XMLStreamWriter writer, int indent) {
128        if (writer == null)
129            throw new IllegalArgumentException("null writer");
130        this.writer = writer;
131        char[] spaces = new char[indent];
132        Arrays.fill(spaces, ' ');
133        this.indentSpace = new String(spaces);
134    }

136    /**
137     * Set whether to use the SQL column labels as the XML element names for the data columns
138     * instead of <code>&lt;column&gt;</code>.
139     * Default is false.
140     *
141     * @param columnNameTags true to name XML elements using the corresponding SQL column names
142     */
143    public void setColumnNameTags(boolean columnNameTags) {
144        this.columnNameTags = columnNameTags;
145    }
146
147    /**
148     * Write the given {@link ResultSet} out within a <code>&lt;result-set&gt;</code> XML element.
149     *
150     * <p>
151     * This is a convenience method, equivalent to:
152     *  <blockquote><code>
153     *      write(null, resultSet)
154     *  </code></blockquote>
155     * element is included.
156     *
157     * @param resultSet query result set
158     * @throws SQLException if accessing {@code resultSet} throws an exception
159     * @throws XMLStreamException if XML output fails
160     */
161    public void write(ResultSet resultSet) throws SQLException, XMLStreamException {
162        this.write(null, resultSet);
163    }
164
165    /**
166     * Write the given {@link ResultSet} out within a <code>&lt;result-set&gt;</code> XML element including the query.
167     * The given SQL query will be present as a <code>&lt;query&gt;</code> first child element,
168     * followed by a <code>&lt;columns&gt;</code> element containing column meta-data, and finally
169     * the <code>&lt;data&gt;</code> element containing each returned row.
170     *
171     * <p>
172     * This method just outputs an XML element. To output an entire XML document, you would do something like this:
173     * <pre>
174     *  XMLResultSetWriter xmlResultSetWriter = new XMLResultSetWriter(writer, 4);
175     *  writer.writeStartDocument("UTF-8", "1.0");
176     *  xmlResultSetWriter.write(query, resultSet);
177     *  writer.writeEndDocument();
178     * </pre>
179     *
180     * <p>
181     * The result set is iterated in a forward direction only.
182     *
183     * @param query the SQL query, or null to omit
184     * @param resultSet result of the query
185     * @throws XMLStreamException if there is an error writing out the XML
186     * @throws SQLException if there is an error reading from {@code resultSet}
187     */
188    public void write(String query, ResultSet resultSet) throws SQLException, XMLStreamException {
189
190        // Start result set
191        this.openTag("result-set");
192
193        // Emit query
194        if (query != null) {
195            this.openTag("query");
196            this.emitCData(query);
197            this.closeTag();
198        }
199
200        // Emit column info
201        final ResultSetMetaData metaData = resultSet.getMetaData();
202        final int numColumns = metaData.getColumnCount();
203        String[] columnNames = new String[numColumns];
204        this.openTag("columns");
205        for (int col = 1; col <= numColumns; col++) {
206            this.openTag("column");
207            this.writer.writeAttribute("index", "" + col);
208            String name = metaData.getColumnName(col);
209            if (name != null)
210                this.writer.writeAttribute("name", name);
211            String label = metaData.getColumnLabel(col);
212            if (label != null)
213                this.writer.writeAttribute("label", label);
214            if (this.columnNameTags)
215                columnNames[col - 1] = label != null ? label : name != null ? name : null;
216            int precision = metaData.getPrecision(col);
217            if (precision != 0)
218                this.writer.writeAttribute("precision", "" + precision);
219            int scale = metaData.getScale(col);
220            if (scale != 0)
221                this.writer.writeAttribute("scale", "" + scale);
222            int type = metaData.getColumnType(col);
223            final String typeName = XMLResultSetWriter.TYPE_NAMES.get(type);
224            this.writer.writeAttribute("type", typeName != null ? typeName : "" + type);
225            this.writer.writeAttribute("typeName", metaData.getColumnTypeName(col));
226            String nullableDesc;
227            switch (metaData.isNullable(col)) {
228            case ResultSetMetaData.columnNullable:
229                nullableDesc = "true";
230                break;
231            case ResultSetMetaData.columnNoNulls:
232                nullableDesc = "false";
233                break;
234            default:
235                nullableDesc = "unknown";
236                break;
237            }
238            this.writer.writeAttribute("nullable", nullableDesc);
239            this.closeTag();
240        }
241        this.closeTag();
242
243        // Emit result data
244        this.openTag("data");
245        while (resultSet.next()) {
246            this.openTag("row");
247            for (int col = 1; col <= numColumns; col++)
248                this.writeDataColumn(resultSet, metaData.getColumnType(col), col, columnNames[col - 1]);
249            this.closeTag();
250        }
251        this.closeTag();
252
253        // End result set
254        this.closeTag();
255    }

257// Writing a data row
258
259    private void writeDataColumn(ResultSet resultSet, int columnType, int col, String columnName)
260      throws SQLException, XMLStreamException {
261        Object value = null;
262        switch (columnType) {
263        case Types.BIT:
264        case Types.BOOLEAN:
265        {
266            boolean b = resultSet.getBoolean(col);
267            if (!resultSet.wasNull())
268                value = "" + b;
269            break;
270        }
271        case Types.CLOB:
272        case Types.NCLOB:
273        case Types.LONGVARCHAR:
274        case Types.LONGNVARCHAR:
275        case Types.VARCHAR:
276        case Types.NVARCHAR:
277        case Types.CHAR:
278        case Types.NCHAR:
279        {
280            Reader reader = resultSet.getCharacterStream(col);
281            if (reader == null)
282                break;
283            int len;
284            char[] buf = new char[1024];
285            this.openColumnTag(col, columnName);
286            try {
287                while ((len = reader.read(buf)) != -1)
288                    this.emitText(buf, 0, len);
289                reader.close();
290            } catch (IOException e) {
291                throw (SQLException)new SQLException("error reading from CLOB").initCause(e);
292            }
293            this.closeTag();
294            return;
295        }
296        case Types.BIGINT:
297        case Types.DECIMAL:
298        case Types.DOUBLE:
299        case Types.FLOAT:
300        case Types.REAL:
301        case Types.NUMERIC:
302            value = resultSet.getBigDecimal(col);
303            break;
304        case Types.INTEGER:
305        case Types.TINYINT:
306        case Types.SMALLINT:
307            value = resultSet.getInt(col);
308            break;
309        case Types.DATE:
310        {
311            Date date = resultSet.getDate(col);
312            if (date != null)
313                value = new SimpleDateFormat(DATE_FORMAT).format(date);
314            break;
315        }
316        case Types.TIME:
317            value = resultSet.getTime(col);
318            break;
319        case Types.TIMESTAMP:
320        {
321            Timestamp timestamp = resultSet.getTimestamp(col);
322            if (timestamp != null)
323                value = new SimpleDateFormat(TIMESTAMP_FORMAT).format(timestamp);
324            break;
325        }
326        case Types.BLOB:
327        case Types.BINARY:
328        case Types.VARBINARY:
329        case Types.LONGVARBINARY:
330        {
331            InputStream input = resultSet.getBinaryStream(col);
332            if (input == null)
333                break;
334            this.openColumnTag(col, columnName);
335            try {
336                this.writeBinary(input);
337                input.close();
338            } catch (IOException e) {
339                throw (SQLException)new SQLException("error reading from BLOB").initCause(e);
340            }
341            this.closeTag();
342            return;
343        }
344        case Types.ROWID:
345        {
346            try {
347                value = resultSet.getRowId(col);
348            } catch (AbstractMethodError e) {
349                value = resultSet.getObject(col);
350            }
351            break;
352        }
353        // Unhandled cases...
354        case Types.ARRAY:
355        case Types.DATALINK:
356        case Types.DISTINCT:
357        case Types.JAVA_OBJECT:
358        case Types.NULL:
359        case Types.OTHER:
360        case Types.REF:
361        case Types.SQLXML:
362        case Types.STRUCT:
363        default:
364            value = resultSet.getObject(col);
365            break;
366        }
367
368        // Handle null by omitting the tag entirely
369        if (value == null || resultSet.wasNull())
370            return;
371
372        // Emit value
373        this.openColumnTag(col, columnName);
374        this.emitText(value.toString());
375        this.closeTag();
376    }
377
378    private void writeBinary(InputStream input) throws IOException, XMLStreamException {
379        int len;
380        byte[] buf = new byte[1024];
381        char[] ch = new char[2];
382        while ((len = input.read(buf)) != -1) {
383            for (int i = 0; i < len; i++) {
384                int val = buf[i] & 0xff;
385                ch[0] = Character.forDigit(val >> 4, 16);
386                ch[1] = Character.forDigit(val & 0xf, 16);
387                this.emitText(ch, 0, 2);
388            }
389        }
390    }

392// XML output formatting
393
394    private void openColumnTag(int col, String columnName) throws XMLStreamException {
395        this.openTag(columnName != null ? this.xmlify(columnName) : "column");
396        this.emitAttr("index", "" + col);
397    }
398
399    private String xmlify(String name) {
400        StringBuilder buf = new StringBuilder(name.length());
401        for (int i = 0; i < name.length(); i++) {
402            final char ch = name.charAt(i);
403            if (Character.isLetter(ch) || ch == '_' || ch == ':') {
404                buf.append(ch);
405                continue;
406            }
407            if (i > 0 && (Character.isDigit(ch) || ch == '.' || ch == '-')) {
408                buf.append(ch);
409                continue;
410            }
411            buf.append('_');
412        }
413        return buf.toString();
414    }
415
416    private void openTag(String name) throws XMLStreamException {
417        switch (this.state) {
418        case LAST_OPEN:
419            this.writer.writeCharacters("\n");
420            break;
421        case LAST_CLOSE:
422            break;
423        case LAST_TEXT:
424        default:
425            throw new RuntimeException("mixed element content not supported");
426        }
427        for (int i = 0; i < this.nesting; i++)
428            this.writer.writeCharacters(this.indentSpace);
429        this.writer.writeStartElement(name);
430        this.nesting++;
431        this.state = State.LAST_OPEN;
432    }
433
434    private void emitAttr(String name, String value) throws XMLStreamException {
435        this.writer.writeAttribute(name, value);
436    }
437
438    private void emitText(String content) throws XMLStreamException {
439        this.writer.writeCharacters(content);
440        this.state = State.LAST_TEXT;
441    }
442
443    private void emitText(char[] buf, int off, int len) throws XMLStreamException {
444        this.writer.writeCharacters(buf, off, len);
445        this.state = State.LAST_TEXT;
446    }
447
448    private void emitCData(String content) throws XMLStreamException {
449        this.writer.writeCData(content);
450        this.state = State.LAST_TEXT;
451    }
452
453    private void closeTag() throws XMLStreamException {
454        this.nesting--;
455        if (this.state == State.LAST_CLOSE) {
456            for (int i = 0; i < this.nesting; i++)
457                this.writer.writeCharacters(this.indentSpace);
458        }
459        this.writer.writeEndElement();
460        if (this.nesting > 0)
461            this.writer.writeCharacters("\n");
462        this.state = State.LAST_CLOSE;
463    }

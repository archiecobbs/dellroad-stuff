036/**
037 * Merges multiple {@link ResultSet}s into a single {@link ResultSet}. The merge is performed in an online fashion
038 * and supports preserving sort order for sorted {@link ResultSet}s.
039 *
040 * <p>
041 * The given {@link ResultSet}s must have the same column geometry. If the {@link ResultSet}s are sorted, then in order
042 * to preserve that sorting the {@link ResultSet}s must be sorted on the same columns, and the sort columns and corresponding
043 * sort orderings must be provided to the appropriate constructor.
044 *
045 * <p>
046 * This class provides a {@link ResultSet} of type {@link #TYPE_FORWARD_ONLY} and concurrency mode {@link #CONCUR_READ_ONLY}.
047 * Non-row specific information, such as is returned by {@link #getMetaData}, {@link #getStatement}, {@link #getFetchSize},
048 * {@link #getHoldability}, and {@link #findColumn findColumn()} is derived from the first {@link ResultSet} provided.
049 */
050@SuppressWarnings("deprecation")
051public class MergedResultSet implements ResultSet {
052

054    private final ArrayList<ResultSet> resultSets;
055    private final PriorityQueue<ResultSet> remaining;

110     * @param sortColumns the names of the columns on which the {@link ResultSet}s are sorted
111     * @param sortOrders the ordering corresponding to {@code sortColumns} (true for ascending); if {@code sortOrders}
112     *  is shorter than {@code sortColumns}, then ascending is assumed for the unspecified columns
113     * @throws IllegalArgumentException if any parameter is null
114     * @throws IllegalArgumentException if {@code resultSets} is empty or contains a null element
115     */
116    public MergedResultSet(List<ResultSet> resultSets, String[] sortColumns, boolean[] sortOrders) {

120    /**
121     * Constructor.
122     *
123     * @param resultSets the {@link ResultSet}s to be merged
124     * @param sortColumns the indicies (1-based) of the columns on which the {@link ResultSet}s are sorted
125     * @param sortOrders the ordering corresponding to {@code sortColumns} (true for ascending); if {@code sortOrders}
126     *  is shorter than {@code sortColumns}, then ascending is assumed for the unspecified columns
127     * @throws IllegalArgumentException if any parameter is null
128     * @throws IllegalArgumentException if {@code resultSets} is empty or contains a null element
129     */
130    public MergedResultSet(List<ResultSet> resultSets, final int[] sortColumns, boolean[] sortOrders) {
131        if (resultSets == null)
132            throw new IllegalArgumentException("null resultSets");
133        if (resultSets.isEmpty())
134            throw new IllegalArgumentException("empty resultSets");
135        for (ResultSet rs : resultSets) {
136            if (rs == null)
137                throw new IllegalArgumentException("null resultSet");
138        }
139        if (sortColumns == null)
140            throw new IllegalArgumentException("null sortColumns");
141        if (sortOrders == null)
142            throw new IllegalArgumentException("null sortOrders");
143        if (sortOrders.length < sortColumns.length) {
144            final boolean[] newSortOrders = new boolean[sortColumns.length];
145            Arrays.fill(newSortOrders, true);
146            System.arraycopy(sortOrders, 0, newSortOrders, 0, Math.min(sortOrders.length, newSortOrders.length));
147            sortOrders = newSortOrders;
148        }
149        this.resultSets = new ArrayList<ResultSet>(resultSets);
150        this.representative = this.resultSets.get(0);
151        final boolean[] sortOrders2 = sortOrders;
152        this.remaining = new PriorityQueue<ResultSet>(this.resultSets.size(), new Comparator<ResultSet>() {
153            @Override
154            @SuppressWarnings("unchecked")
155            public int compare(ResultSet rs1, ResultSet rs2) {
156                for (int i = 0; i < sortColumns.length; i++) {
157                    Object value1;
158                    Object value2;
159                    try {
160                        value1 = rs1.getObject(sortColumns[i]);
161                        value2 = rs2.getObject(sortColumns[i]);
162                    } catch (SQLException e) {
163                        throw new RuntimeException(e);
164                    }
165                    if (value1 == null && value2 != null)
166                        return -1;
167                    if (value1 != null && value2 == null)
168                        return 1;
169                    if (value1 == null && value2 == null)
170                        return 0;
171                    int diff = ((Comparable)value1).compareTo(value2);
172                    if (diff != 0)
173                        return sortOrders2[i] ? diff : -diff;
174                }
175                return 0;
176            }
177        });
178    }

180    // Convert column names into column indexes
181    private static int[] determineSortColumns(List<ResultSet> resultSets, String[] sortColumnNames) {
182        if (resultSets.isEmpty())
183            throw new IllegalArgumentException("empty resultSets");
184        final int[] sortColumnNums = new int[sortColumnNames.length];
185        for (int i = 0; i < sortColumnNums.length; i++) {
186            try {
187                sortColumnNums[i] = resultSets.get(0).findColumn(sortColumnNames[i]);
188            } catch (SQLException e) {
189                throw new IllegalArgumentException("column `" + sortColumnNames[i] + "' not found", e);
190            }
191        }
192        return sortColumnNums;
193    }

195    private ResultSet getCurrent() throws SQLException {
196        this.checkClosed();
197        if (this.remaining.isEmpty())
198            throw new SQLException("there is no current row");
199        return this.remaining.peek();
200    }
201
202    private void checkClosed() throws SQLException {
203        if (this.closed)
204            throw new SQLException("ResultSet is closed");
205    }
206
207    @Override
208    public boolean next() throws SQLException {
209        try {
210            if (this.currentRow == 0) {                 // we are before the first row
211
212                // Perform the initial load
213                for (ResultSet resultSet : this.resultSets) {
214                    if (resultSet.next())
215                        this.remaining.add(resultSet);
216                }
217            } else if (!this.remaining.isEmpty()) {   // we are somewhere in the middle
218
219                // Advance the ResultSet we just read from and re-insert it into the priority queue
220                final ResultSet resultSet = this.remaining.remove();
221                if (resultSet.next())
222                    this.remaining.add(resultSet);
223            }
224
225            // Are all ResultSets completely exhausted?
226            if (this.remaining.isEmpty()) {
227                this.currentRow = -1;               // indicates we are past the last row
228                return false;
229            }
230        } catch (RuntimeException e) {
231            if (e.getCause() instanceof SQLException)
232                throw (SQLException)e.getCause();
233            throw e;
234        }
235
236        // Done
237        this.currentRow++;
238        return true;
239    }

241    @Override
242    public void close() throws SQLException {
243        this.closed = true;
244        for (ResultSet rs : this.resultSets)
245            rs.close();
246        this.remaining.clear();
247    }

006package org.dellroad.stuff;
007
008import java.io.IOException;
009import java.io.InputStream;
010import java.util.Properties;
011
012/**
013 * Contains library version information.
014 */
015public final class Version {
016
017    /**
018     * The version of this library.
019     */
020    public static final String DELLROAD_STUFF_VERSION;
021
022    private static final String PROPERTIES_RESOURCE = "/dellroad-stuff.properties";
023    private static final String VERSION_PROPERTY_NAME = "dellroad-stuff.version";
024
025    static {
026        Properties properties = new Properties();
027        InputStream input = Version.class.getResourceAsStream(PROPERTIES_RESOURCE);
028        if (input == null)
029            throw new RuntimeException("can't find resource " + PROPERTIES_RESOURCE);
030        try {
031            properties.load(input);
032        } catch (IOException e) {
033            throw new RuntimeException("unexpected exception", e);
034        } finally {
035            try {
036                input.close();
037            } catch (IOException e) {
038                // ignore
039            }
040        }
041        DELLROAD_STUFF_VERSION = properties.getProperty(VERSION_PROPERTY_NAME, "?");
042    }
043
044    private Version() {
045    }
046}

006package org.dellroad.stuff.dao;
007
008import java.util.List;
009
010import javax.persistence.FlushModeType;
011
012/**
013 * Data Access Object (DAO) generic interface.
014 */
015public interface DAO<T> {
016
017// Meta-data methods
018
019    /**
020     * Get the type of persistent object handled by this instance.
021     *
022     * @return associated object type
023     */
024    Class<T> getType();
025
026// Access methods
027
028    /**
029     * Get an instance by ID. This assumes object IDs are long values.
030     *
031     * @param id unique ID for instance
032     * @return instance with the given {@code id}, or null if not found
033     */
034    T getById(long id);
035
036    /**
037     * Get all instances.
038     *
039     * @return list of all instances of type {@code T}
040     */
041    List<T> getAll();
042
043    /**
044     * Get a reference to an instance by ID. This assumes object IDs are long values.
045     *
046     * <p>
047     * Note if the instance does not exist, then an exception may be thrown either here or later upon first access.
048     *
049     * @param id unique ID for instance
050     * @return reference to instance with the given {@code id}
051     */
052    T getReference(long id);
053
054// Lifecycle methods
055
056    /**
057     * Save a newly created instance.
058     *
059     * @param obj object to save
060     */
061    void save(T obj);
062
063    /**
064     * Delete the given instance from the persistent store.
065     *
066     * @param obj object to delete
067     */
068    void delete(T obj);
069
070    /**
071     * Merge the given object into the current session.
072     *
073     * @param obj object to merge
074     * @return merged persistent instance
075     */
076    T merge(T obj);
077
078    /**
079     * Refresh the given object from the database.
080     *
081     * @param obj object to refresh
082     */
083    void refresh(T obj);
084
085    /**
086     * Evict an object from the session cache.
087     *
088     * @param obj object to evict
089     */
090    void detach(T obj);
091
092// Session methods
093
094    /**
095     * Flush outstanding changes to the persistent store.
096     */
097    void flush();
098
099    /**
100     * Set flush mode.
101     *
102     * @param flushMode desired flush mode
103     */
104    void setFlushMode(FlushModeType flushMode);
105
106    /**
107     * Clear the session cache.
108     */
109    void clear();
110
111    /**
112     * Determine if the current transaction is read-only.
113     *
114     * @return true if the current transaction is read-only
115     * @throws IllegalStateException if no transaction is associated with the current thread
116     */
117    boolean isReadOnly();
118
119    /**
120     * Determine if the current session (i.e., {@code EntityManager}) contains the given instance.
121     *
122     * @param obj instance to inquire about
123     * @return true if the current transaction contains {@code obj}
124     */
125    boolean contains(T obj);
126}

006package org.dellroad.stuff.dao;
007
008import javax.persistence.Query;
009
010/**
011 * Helper class for JPA bulk updates.
012 */
013public abstract class UpdateCallback extends QueryCallback<Integer> {
014
015    /**
016     * Execute the query. This method invokes {@link Query#executeUpdate} and returns the result.
017     */
018    @Override
019    protected final Integer executeQuery(Query query) {
020        return query.executeUpdate();
021    }
022}

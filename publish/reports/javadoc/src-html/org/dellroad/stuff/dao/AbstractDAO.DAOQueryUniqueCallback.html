006package org.dellroad.stuff.dao;
007
008import java.util.List;
009
010import javax.persistence.EntityManager;
011import javax.persistence.FlushModeType;
012import javax.persistence.NoResultException;
013import javax.persistence.PersistenceContext;
014import javax.persistence.TypedQuery;
015import javax.persistence.criteria.CriteriaBuilder;
016import javax.persistence.criteria.CriteriaQuery;
017
018import org.dellroad.stuff.spring.AbstractBean;
019import org.springframework.dao.EmptyResultDataAccessException;
020import org.springframework.transaction.support.TransactionSynchronizationManager;
021
022/**
023 * Support superclass for JPA DAO implementations.
024 *
025 * @param <T> persistent instance type
026 * @see org.springframework.dao.annotation.PersistenceExceptionTranslationPostProcessor
027 */
028public abstract class AbstractDAO<T> extends AbstractBean implements DAO<T> {
029
030    /**
031     * Persistent instance type.
032     */
033    protected final Class<T> type;
034
035    /**
036     * The configured {@link EntityManager}.
037     */
038    private EntityManager entityManager;
039
040    /**
041     * Constructor.
042     *
043     * @param type persistent instance type
044     */
045    protected AbstractDAO(Class<T> type) {
046        if (type == null)
047            throw new IllegalArgumentException("null type");
048        this.type = type;
049    }
050
051// Configuration and lifecycle methods
052
053    /**
054     * Get the configured {@link EntityManager}.
055     *
056     * @return this DAO's {@link EntityManager}
057     */
058    protected EntityManager getEntityManager() {
059        return this.entityManager;
060    }
061
062    /**
063     * Set the configured {@link EntityManager}.
064     *
065     * <p>
066     * This method has a {@link PersistenceContext @PersistenceContext} annotation and normally
067     * would be invoked automatically by Spring.
068     *
069     * @param entityManager this DAO's {@link EntityManager}
070     */
071    @PersistenceContext
072    public void setEntityManager(EntityManager entityManager) {
073        this.entityManager = entityManager;
074    }
075
076    @Override
077    public void afterPropertiesSet() throws Exception {
078        super.afterPropertiesSet();
079        if (this.getEntityManager() == null)
080            throw new IllegalArgumentException("no entityManager configured");
081    }
082
083// Meta-data methods
084
085    @Override
086    public Class<T> getType() {
087        return this.type;
088    }
089
090// Access methods
091
092    @Override
093    public T getById(long id) {
094        return this.getEntityManager().find(this.type, id);
095    }
096
097    @Override
098    public List<T> getAll() {
099        return this.getBy(new DAOCriteriaListCallback() {
100            @Override
101            protected void configureQuery(CriteriaQuery<T> criteriaQuery, CriteriaBuilder criteriaBuilder) {
102                criteriaQuery.from(this.type);
103            }
104        });
105    }
106
107    @Override
108    public T getReference(long id) {
109        return this.getEntityManager().getReference(this.type, id);
110    }
111
112    /**
113     * Find instances using a query string and query parameters.
114     *
115     * @param queryString JPQL query string
116     * @param params query string parameters
117     * @return zero or more matching objects
118     */
119    protected List<T> find(final String queryString, final Object... params) {
120        return this.getBy(new DAOQueryListCallback() {
121            @Override
122            protected TypedQuery<T> buildQuery(EntityManager em) {
123                return AbstractDAO.this.buildQuery(em, queryString, params);
124            }
125        });
126    }
127
128    /**
129     * Find a unique instance using a query string and query parameters.
130     *
131     * @param queryString JPQL query string
132     * @param params query string parameters
133     * @return unique instance found, or null if none was found
134     */
135    protected T findUnique(final String queryString, final Object... params) {
136        return this.getBy(new DAOQueryUniqueCallback() {
137            @Override
138            protected TypedQuery<T> buildQuery(EntityManager em) {
139                return AbstractDAO.this.buildQuery(em, queryString, params);
140            }
141        });
142    }
143
144    /**
145     * Search using a {@link QueryCallback}.
146     *
147     * @param <R> return type
148     * @param callback query callback
149     * @return result of query
150     */
151    protected <R> R getBy(QueryCallback<R> callback) {
152        return callback.query(this.getEntityManager());
153    }
154
155    /**
156     * Perform a bulk update.
157     *
158     * @param callback query callback
159     * @return number of rows affected
160     */
161    protected int bulkUpdate(UpdateCallback callback) {
162        return callback.query(this.getEntityManager());
163    }
164
165// Lifecycle methods
166
167    @Override
168    public void save(T obj) {
169        this.getEntityManager().persist(obj);
170    }
171
172    @Override
173    public void delete(T obj) {
174        this.getEntityManager().remove(obj);
175    }
176
177    @Override
178    public T merge(T obj) {
179        return this.getEntityManager().merge(obj);
180    }
181
182    @Override
183    public void refresh(T obj) {
184        this.getEntityManager().refresh(obj);
185    }
186
187    @Override
188    public void detach(Object obj) {
189        this.getEntityManager().detach(obj);
190    }
191
192// Session methods
193
194    @Override
195    public void flush() {
196        this.getEntityManager().flush();
197    }
198
199    @Override
200    public void setFlushMode(FlushModeType flushMode) {
201        this.getEntityManager().setFlushMode(flushMode);
202    }
203
204    @Override
205    public void clear() {
206        this.getEntityManager().clear();
207    }
208
209    @Override
210    public boolean isReadOnly() {
211        return TransactionSynchronizationManager.isCurrentTransactionReadOnly();
212    }
213
214    @Override
215    public boolean contains(T obj) {
216        return this.getEntityManager().contains(obj);
217    }
218
219// Type and cast methods
220
221    /**
222     * Cast the given object to this instance's persistent instance type.
223     *
224     * @param obj object to cast
225     * @return {@code obj} cast to type {@code T}
226     */
227    protected T cast(Object obj) {
228        return this.type.cast(obj);
229    }
230
231    /**
232     * Cast the given list to a list of this instance's persistent instance type.
233     * Does not actually inspect the contents of the list.
234     *
235     * @param list list of objects
236     * @return {@code list} cast to a list of {@code T}
237     */
238    @SuppressWarnings("unchecked")
239    protected List<T> castList(List<?> list) {
240        return (List<T>)list;
241    }
242
243// Helper methods
244
245    private TypedQuery<T> buildQuery(EntityManager em, String queryString, Object[] params) {
246        TypedQuery<T> query = em.createQuery(queryString, this.type);
247        if (params != null) {
248            for (int i = 0; i < params.length; i++)
249                query.setParameter(i + 1, params[i]);
250        }
251        return query;
252    }
253
254// Helper classes
255
256    /**
257     * Convenience subclass of {@link QueryCallback} for use by DAO subclasses when returning lists of persistent instances.
258     */
259    protected abstract class DAOQueryListCallback extends TypedQueryCallback<T, List<T>> {
260
261        @Override
262        protected final List<T> executeQuery(TypedQuery<T> query) {
263            return query.getResultList();
264        }
265    }
266
267    /**
268     * Convenience subclass of {@link QueryCallback} for use by DAO subclasses when returning a single persistent instance.
269     *
270     * <p>
271     * Returns null if instance is not found.
272     */
273    protected abstract class DAOQueryUniqueCallback extends TypedQueryCallback<T, T> {
274
275        @Override
276        protected final T executeQuery(TypedQuery<T> query) {
277            try {
278                return query.getSingleResult();
279            } catch (NoResultException e) {
280                return null;
281            } catch (EmptyResultDataAccessException e) {
282                return null;
283            }
284        }
285    }
286
287    /**
288     * Convenience subclass of {@link CriteriaCallback} for use by DAO subclasses when returning lists of persistent instances.
289     */
290    protected abstract class DAOCriteriaListCallback extends CriteriaCallback<T, List<T>> {
291
292        protected DAOCriteriaListCallback() {
293            super(AbstractDAO.this.type);
294        }
295
296        @Override
297        protected final List<T> executeQuery(TypedQuery<T> query) {
298            return query.getResultList();
299        }
300    }
301
302    /**
303     * Convenience subclass of {@link CriteriaCallback} for use by DAO subclasses when returning a single persistent instance.
304     *
305     * <p>
306     * Returns null if instance is not found.
307     */
308    protected abstract class DAOCriteriaUniqueCallback extends CriteriaCallback<T, T> {
309
310        protected DAOCriteriaUniqueCallback() {
311            super(AbstractDAO.this.type);
312        }
313
314        @Override
315        protected final T executeQuery(TypedQuery<T> query) {
316            try {
317                return query.getSingleResult();
318            } catch (NoResultException e) {
319                return null;
320            } catch (EmptyResultDataAccessException e) {
321                return null;
322            }
323        }
324    }
325}

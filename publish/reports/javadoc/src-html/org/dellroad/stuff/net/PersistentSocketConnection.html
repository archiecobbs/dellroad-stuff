006package org.dellroad.stuff.net;
007
008import java.io.IOException;
009import java.net.Socket;
010import java.net.SocketAddress;
011
012/**
013 * Maintains a persistent {@link Socket} connection to a remote server, automatically reconnecting when the connection fails.
014 *
015 * <p>
016 * Subclasses need only override {@link #handleConnection handleConnection()}.
017 */
018public abstract class PersistentSocketConnection extends PersistentConnection<Socket> {
019
020    private SocketAddress serverAddress;
021
022// Properties
023
024    /**
025     * Get the address to which this instance should connect.
026     *
027     * @return remote server address
028     */
029    public SocketAddress getServerAddress() {
030        return this.serverAddress;
031    }
032
033    /**
034     * Configure the address to which this instance should connect.
035     *
036     * <p>
037     * Required property.
038     *
039     * @param serverAddress remote server address
040     */
041    public void setServerAddress(SocketAddress serverAddress) {
042        this.serverAddress = serverAddress;
043    }
044
045// Lifecycle
046
047    @Override
048    public void start() {
049        if (this.serverAddress == null)
050            throw new IllegalArgumentException("no serverAddress configured");
051        super.start();
052    }
053
054    /**
055     * Create a new server connection.
056     *
057     * <p>
058     * The implementation in {@link PersistentSocketConnection} creates a socket via {@link #createSocket}
059     * and connects the socket to the configured server.
060     *
061     * @return connected socket
062     * @throws IOException if the connection attempt fails
063     * @throws InterruptedException if the current thread is interrupted
064     */
065    @Override
066    protected Socket createConnection() throws InterruptedException, IOException {
067        final Socket socket = this.createSocket();
068        socket.connect(this.serverAddress);
069        return socket;
070    }
071
072    /**
073     * Perform cleanup after a server connection ends.
074     *
075     * <p>
076     * The implementation in {@link PersistentSocketConnection} ensures the socket is closed.
077     *
078     * @param socket socket returned from {@link #createConnection} when this connection was created
079     */
080    @Override
081    protected void cleanupConnection(Socket socket) {
082        try {
083            socket.close();
084        } catch (IOException e) {
085            // ignore
086        }
087    }
088
089    /**
090     * Create the {@link Socket} to use for a new connection. The socket should <b>not</b> be connected yet.
091     *
092     * <p>
093     * The implementation in {@link PersistentSocketConnection} creates a normal {@link Socket} and sets the keep-alive flag
094     * to ensure connection failures are detected promptly. Subclasses may override to configure the socket differently.
095     *
096     * @return new socket
097     * @throws IOException if socket creation fails
098     */
099    protected Socket createSocket() throws IOException {
100        final Socket socket = new Socket();
101        socket.setKeepAlive(true);
102        return socket;
103    }
104}

006package org.dellroad.stuff.net;
007
008import java.io.IOException;
009import java.nio.channels.SelectionKey;
010import java.nio.channels.SocketChannel;
011
012/**
013 * A TCP connection to be used with a {@link ChannelNetwork}.
014 *
015 * <b>Locking</b>
016 *
017 * <p>
018 * All access to this class must be with the associated {@link TCPNetwork} instance locked.
019 */
020public class TCPConnection extends ChannelConnection {
021
022// Constructors
023
024    protected TCPConnection(TCPNetwork network, String peer, SocketChannel channel) throws IOException {
025        super(network, peer, channel);
026    }
027
028    /**
029     * Get the associated {@link SocketChannel}.
030     *
031     * @return the socket channel for this connection
032     */
033    public SocketChannel getSocketChannel() {
034        return (SocketChannel)this.getInputChannel();
035    }
036
037// ChannelConnection
038
039    @Override
040    protected void updateSelection() {
041        if (this.getSocketChannel().isConnectionPending()) {
042            this.network.selectFor(this.inputSelectionKey, SelectionKey.OP_CONNECT, true);
043            this.network.selectFor(this.inputSelectionKey, SelectionKey.OP_READ, true);
044            this.network.selectFor(this.outputSelectionKey, SelectionKey.OP_WRITE, false);
045            return;
046        }
047        this.network.selectFor(this.inputSelectionKey, SelectionKey.OP_CONNECT, false);
048        super.updateSelection();
049    }
050
051// IOHandler
052
053    @Override
054    public void serviceIO(SelectionKey key) throws IOException {
055        assert this.network.isServiceThread();
056        assert Thread.holdsLock(this.network);
057        if (key.isConnectable())
058            this.handleConnectable();
059        super.serviceIO(key);
060    }
061
062// I/O Ready Conditions
063
064    // Handle connection succeeded
065    private void handleConnectable() throws IOException {
066
067        // Leave connecting state
068        this.network.selectFor(this.inputSelectionKey, SelectionKey.OP_CONNECT, false);
069        if (!this.getSocketChannel().finishConnect())                    // this should never occur
070            throw new IOException("connection failed");
071        if (this.log.isDebugEnabled())
072            this.log.debug(this + ": connection succeeded");
073
074        // Notify us when readable/writeable
075        this.updateSelection();
076
077        // Restart idle timer
078        this.restartIdleTimer();
079
080        // Notify client we are open for business
081        this.network.handleOutputQueueEmpty(this);
082    }
083
084// Housekeeping
085
086    @Override
087    protected void performHousekeeping() throws IOException {
088        assert Thread.holdsLock(this.network);
089        assert this.network.isServiceThread();
090        if (this.getSocketChannel().isConnectionPending()) {
091            if (this.getIdleTime() >= ((TCPNetwork)this.network).getConnectTimeout())
092                throw new IOException("connection unsuccessful after " + this.getIdleTime() + "ms");
093            return;
094        }
095        super.performHousekeeping();
096    }
097}

006package org.dellroad.stuff.net;
007
008import java.io.EOFException;
009import java.io.IOException;
010import java.nio.ByteBuffer;
011import java.nio.channels.GatheringByteChannel;
012import java.nio.channels.ReadableByteChannel;
013import java.nio.channels.SelectableChannel;
014import java.nio.channels.SelectionKey;
015import java.util.ArrayDeque;
016
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020/**
021 * Support superclass for {@link ChannelNetwork} connections.
022 *
023 * <b>Locking</b>
024 *
025 * <p>
026 * All access to this class must be with the associated {@link ChannelNetwork} instance locked.
027 */
028public abstract class ChannelConnection implements SelectorSupport.IOHandler {
029
030    /**
031     * Minimum buffer size to use a direct buffer.
032     */
033    private static final int MIN_DIRECT_BUFFER_SIZE = 128;
034
035    protected final Logger log = LoggerFactory.getLogger(this.getClass());
036    protected final ChannelNetwork network;
037    protected final String peer;
038    protected final SelectableChannel inputChannel;
039    protected final SelectableChannel outputChannel;
040    protected final SelectionKey inputSelectionKey;
041    protected final SelectionKey outputSelectionKey;        // same as inputSelectionKey if inputChannel == outputChannel
042
043    private final ArrayDeque<ByteBuffer> output = new ArrayDeque<>();
044
045    private ByteBuffer inbuf;
046    private long queueSize;                                 // invariant: always equals the total number of bytes in 'output'
047    private long lastActiveTime;
048    private boolean readingLength;                          // indicates 'inbuf' is reading the message length (4 bytes)
049    private boolean closed;
050
051// Constructors
052
053    protected ChannelConnection(ChannelNetwork network, String peer, SelectableChannel channel) throws IOException {
054        this(network, peer, channel, channel);
055    }
056
057    protected ChannelConnection(ChannelNetwork network, String peer,
058      SelectableChannel inputChannel, SelectableChannel outputChannel) throws IOException {
059
060        // Sanity check
061        if (network == null)
062            throw new IllegalArgumentException("null network");
063        if (peer == null)
064            throw new IllegalArgumentException("null peer");
065        if (!(inputChannel instanceof ReadableByteChannel))
066            throw new IllegalArgumentException("inputChannel must be a ReadableByteChannel");
067        if (!(outputChannel instanceof GatheringByteChannel))
068            throw new IllegalArgumentException("inputChannel must be a GatheringByteChannel");
069
070        // Initialize
071        this.network = network;
072        this.peer = peer;
073        this.inputChannel = inputChannel;
074        this.outputChannel = outputChannel;
075        this.restartIdleTimer();
076
077        // Set up selection
078        this.inputSelectionKey = this.network.createSelectionKey(this.inputChannel, this);
079        this.outputSelectionKey = this.outputChannel != this.inputChannel ?
080          this.network.createSelectionKey(this.outputChannel, this) : this.inputSelectionKey;
081        this.updateSelection();
082
083        // Initialize input state
084        this.inbuf = ByteBuffer.allocate(4);
085        this.readingLength = true;
086    }
087
088    /**
089     * Get remote peer's identity.
090     *
091     * @return remote peer for this connection
092     */
093    public String getPeer() {
094        return this.peer;
095    }
096
097    /**
098     * Get the associated input channel.
099     *
100     * @return input channel for this connection; could be same as the {@linkplain #getOutputChannel output channel}
101     */
102    public SelectableChannel getInputChannel() {
103        return this.inputChannel;
104    }
105
106    /**
107     * Get the associated output channel.
108     *
109     * @return input channel for this connection; could be same as the {@linkplain #getInputChannel input channel}
110     */
111    public SelectableChannel getOutputChannel() {
112        return this.outputChannel;
113    }
114
115    /**
116     * Get time in milliseconds since last activity.
117     *
118     * @return idle time in milliseconds
119     */
120    public long getIdleTime() {
121        return (System.nanoTime() - this.lastActiveTime) / 1000000L;
122    }
123
124    /**
125     * Enqueue an outgoing message on this connection.
126     *
127     * @param buf outgoing data
128     * @return true if message was enqueued, false if output buffer was full
129     */
130    public boolean output(ByteBuffer buf) {
131
132        // Sanity check
133        assert Thread.holdsLock(this.network);
134        if (buf == null)
135            throw new IllegalArgumentException("null buf");
136
137        // Avoid anyone else mucking with my buffer position, etc.
138        buf = buf.asReadOnlyBuffer();
139
140        // Check output queue capacity
141        final int length = buf.remaining();
142        final int increment = length + 4;
143        if (this.queueSize + increment > this.network.getMaxOutputQueueSize())
144            return false;
145
146        // Add to queue
147        this.output.add((ByteBuffer)ByteBuffer.allocate(4).putInt(length).flip());
148        this.output.add(buf);
149        this.queueSize += increment;
150
151        // Ensure we are notified when output is writable
152        this.updateSelection();
153
154        // Done
155        this.restartIdleTimer();
156        return true;
157    }
158
159// Object
160
161    @Override
162    public String toString() {
163        return this.getClass().getSimpleName() + "[peer=" + this.peer + ",closed=" + this.closed + "]";
164    }
165
166// IOHandler
167
168    @Override
169    public void serviceIO(SelectionKey key) throws IOException {
170        assert this.network.isServiceThread();
171        assert Thread.holdsLock(this.network);
172        if (key.isReadable())
173            this.handleReadable();
174        if (key.isWritable())
175            this.handleWritable();
176    }
177
178    @Override
179    public void close(Throwable cause) {
180        assert Thread.holdsLock(this.network);
181        if (this.closed)
182            return;
183        this.closed = true;
184        if (this.log.isDebugEnabled())
185            this.log.debug("closing " + this + (cause != null ? " due to " + cause : ""));
186        try {
187            this.inputChannel.close();
188        } catch (IOException e) {
189            // ignore
190        }
191        try {
192            this.outputChannel.close();
193        } catch (IOException e) {
194            // ignore
195        }
196        this.network.handleConnectionClosed(this);
197    }
198
199// Subclass Methods
200
201    /**
202     * Update selected keys.
203     *
204     * <p>
205     * The implementation in {@link ChannelConnection} selects for read always and write
206     * if the output queue is non-empty.
207     */
208    protected void updateSelection() {
209        this.network.selectFor(this.inputSelectionKey, SelectionKey.OP_READ, true);
210        this.network.selectFor(this.outputSelectionKey, SelectionKey.OP_WRITE, !this.output.isEmpty());
211    }
212
213    /**
214     * Restart the idle timer.
215     */
216    protected void restartIdleTimer() {
217        this.lastActiveTime = System.nanoTime();
218    }
219
220// I/O Ready Conditions
221
222    private void handleReadable() throws IOException {
223        while (true) {
224
225            // Update timestamp
226            this.restartIdleTimer();
227
228            // Read bytes
229            final long len = ((ReadableByteChannel)this.inputChannel).read(this.inbuf);
230            if (len == -1)
231                throw new EOFException("connection closed");
232
233            // Is the message (or length header) still incomplete?
234            if (this.inbuf.hasRemaining())
235                break;
236
237            // Set up for reading
238            this.inbuf.flip();
239
240            // Completed length header?
241            if (this.readingLength) {
242
243                // Get and validate length
244                assert this.inbuf.remaining() == 4;
245                final int length = this.inbuf.getInt();
246                if (length < 0 || length > this.network.getMaxMessageSize())
247                    throw new IOException("rec'd message with bogus length " + length);
248
249                // Set up for reading the actual message
250                this.inbuf = length >= MIN_DIRECT_BUFFER_SIZE ? ByteBuffer.allocateDirect(length) : ByteBuffer.allocate(length);
251                this.readingLength = false;
252                continue;
253            }
254
255            // Deliver the completed message
256            this.network.handleMessage(this, this.inbuf);
257
258            // Set up for reading next length header
259            this.inbuf = ByteBuffer.allocate(4);
260            this.readingLength = true;
261        }
262
263        // Done
264        this.restartIdleTimer();
265    }
266
267    private void handleWritable() throws IOException {
268
269        // Write more data, if present
270        boolean queueBecameEmpty = false;
271        if (!this.output.isEmpty()) {
272
273            // Write data
274            final long written = ((GatheringByteChannel)this.outputChannel).write(
275              this.output.toArray(new ByteBuffer[this.output.size()]));
276            this.queueSize -= written;
277
278            // Clear away empty buffers
279            while (!this.output.isEmpty() && !this.output.peekFirst().hasRemaining())
280                this.output.removeFirst();
281
282            // Set flag if queue became empty
283            queueBecameEmpty = this.output.isEmpty();
284        }
285
286        // If queue became empty, ensure we are no longer notified when output is writable
287        if (queueBecameEmpty)
288            this.updateSelection();
289
290        // Update timestamp
291        this.restartIdleTimer();
292
293        // Notify client if queue became empty
294        if (queueBecameEmpty)
295            this.network.handleOutputQueueEmpty(this);
296    }
297
298// Housekeeping
299
300    /**
301     * Perform housekeeping.
302     *
303     * The implementation in {@link ChannelConnection} checks the max idle time.
304     *
305     * @throws IOException if an I/O error occurs (this connection will be closed)
306     */
307    protected void performHousekeeping() throws IOException {
308        assert Thread.holdsLock(this.network);
309        assert this.network.isServiceThread();
310        if (this.getIdleTime() >= this.network.getMaxIdleTime())
311            throw new IOException("connection idle timeout after " + this.getIdleTime() + "ms");
312    }
313}

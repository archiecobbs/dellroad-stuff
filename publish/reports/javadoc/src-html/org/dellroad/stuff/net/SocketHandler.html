006package org.dellroad.stuff.net;
007
008import java.io.IOException;
009import java.net.Socket;
010
011/**
012 * Implemented by objects that handle individual connections accepted by a {@link SocketAcceptor}.
013 */
014public interface SocketHandler {
015
016    /**
017     * Handle the connection.
018     *
019     * @param socket connection socket
020     * @throws IOException if needed
021     */
022    void handleConnection(Socket socket) throws IOException;
023
024    /**
025     * Receive notification that the server is shutting down.
026     * This method should ensure that the thread currently executing {@link #handleConnection handleConnection()}
027     * returns as soon as possible.
028     *
029     * <p>
030     * After this method returns, the {@code socket} will be closed (if not already). So one way to implement this method
031     * is to simply do nothing, which will trigger an {@link IOException} on the next access from within
032     * {@link #handleConnection handleConnection()}.
033     *
034     * <p>
035     * Alternately, set some flag that {@link #handleConnection handleConnection()} detects each time around its processing loop,
036     * or use {@link Thread#interrupt}, etc.
037     *
038     * <p>
039     * In any case, it is important that {@link #handleConnection handleConnection()} thread does eventually return,
040     * otherwise the thread invoking {@link SocketAcceptor#stop SocketAcceptor.stop()} will hang.
041     *
042     * <p>
043     * Note: it may be that {@link #handleConnection handleConnection()} has already returned when this method is invoked,
044     * in which case this method should do nothing.
045     *
046     * <p>
047     *
048     * @param thread the thread invoking {@link #handleConnection handleConnection()}
049     * @param socket connection socket
050     */
051    void stop(Thread thread, Socket socket);
052}

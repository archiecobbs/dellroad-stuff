006package org.dellroad.stuff.net;
007
008import java.io.IOException;
009import java.nio.ByteBuffer;
010
011/**
012 * Abstraction layer representing "the network", over which we communicate with other nodes.
013 *
014 * <p>
015 * In general, messages may be delayed, dropped or delivered out of order. Specific implementations
016 * may provide more strict guarantees.
017 *
018 * <p>
019 * Remote peers are identified by {@link String}s; the interpretation of these {@link String}s is up to the implementation.
020 *
021 * <p>
022 * Notifications are delivered to the specified {@link Handler}.
023 */
024public interface Network {
025
026    /**
027     * Start this instance.
028     *
029     * @param handler handler for notifications
030     * @throws IllegalStateException if already started
031     * @throws IOException if an error occurs
032     */
033    void start(Handler handler) throws IOException;
034
035    /**
036     * Stop this instance.
037     *
038     * <p>
039     * Does nothing if already stopped.
040     */
041    void stop();
042
043    /**
044     * Send (or enqueue for sending) a message to a remote peer.
045     *
046     * <p>
047     * If this method returns true, then {@link Handler#outputQueueEmpty Handler.outputQueueEmpty()}
048     * is guaranteed to be invoked with parameter {@code peer} at some later point.
049     *
050     * @param peer message destination
051     * @param msg message to send
052     * @return true if message was succesfully enqueued for output; false if message failed to be delivered due to local reasons,
053     *  such as failure to initiate a new connection or output queue overflow
054     * @throws IllegalArgumentException if {@code peer} cannot be interpreted
055     * @throws IllegalArgumentException if {@code peer} or {@code msg} is null
056     */
057    boolean send(String peer, ByteBuffer msg);
058
059// Handler
060
061    /**
062     * Callback interface used by {@link Network} implementations.
063     */
064    public interface Handler {
065
066        /**
067         * Handle an incoming message from a remote peer.
068         *
069         * <p>
070         * The {@code msg} buffer is read-only; its contents are not guaranteed to be valid after this method returns.
071         *
072         * @param peer message source
073         * @param msg message received
074         */
075        void handle(String peer, ByteBuffer msg);
076
077        /**
078         * Receive notification that a remote peer's output queue has transitioned from a non empty state to an empty state
079         *
080         * @param peer remote peer
081         */
082        void outputQueueEmpty(String peer);
083    }
084}

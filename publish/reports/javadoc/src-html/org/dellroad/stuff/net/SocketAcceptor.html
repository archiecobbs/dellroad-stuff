006package org.dellroad.stuff.net;
007
008import java.io.IOException;
009import java.net.InetAddress;
010import java.net.ServerSocket;
011import java.net.Socket;
012import java.util.HashSet;
013
014import org.dellroad.stuff.spring.AbstractBean;
015
016/**
017 * Spring bean that listens for connections on a TCP socket and spawns a child thread to handle each new connection.
018 * Subclasses must implement {@link #getSocketHandler}. Only the {@link #getPort port} property is required to be set.
019 */
020public abstract class SocketAcceptor extends AbstractBean {
021
022    /**
023     * Default maximum incoming connection queue length ({@value}).
024     *
025     * @see #setBacklog
026     */
027    public static final int DEFAULT_BACKLOG = 50;
028
029    private static final long NOTIFICATION_INTERVAL = 5 * 1000 * 1000 * 1000L;      // 5 seconds
030
031    private final HashSet<SocketInfo> connections = new HashSet<SocketInfo>();
032    private InetAddress address;
033    private int backlog = DEFAULT_BACKLOG;
034    private int port;
035    private int maxConnections;
036    private ServerSocket serverSocket;
037    private Thread serverThread;
038    private boolean started;
039
040    /**
041     * Verifies configuration and invokes {@link #start}.
042     */
043    @Override
044    public void afterPropertiesSet() throws Exception {
045        super.afterPropertiesSet();
046        if (this.port == 0)
047            throw new IllegalArgumentException("TCP port not set");
048        if (this.port < 1 || this.port > 65535)
049            throw new IllegalArgumentException("invalid TCP port " + this.port);
050        if (this.maxConnections < 0)
051            throw new IllegalArgumentException("invalid maxConnections " + this.maxConnections);
052        this.start();
053    }
054
055    /**
056     * Invokes {@link #stop}.
057     */
058    @Override
059    public void destroy() throws Exception {
060        this.stop();
061        super.destroy();
062    }
063
064    /**
065     * Get address to listen on.
066     *
067     * @return address to listen on, or null for any
068     */
069    public InetAddress getInetAddress() {
070        return this.address;
071    }
072    public void setInetAddress(InetAddress address) {
073        this.address = address;
074    }
075
076    /**
077     * Get maximum connect backlog.
078     *
079     * @return maximum backlogged connections
080     */
081    public int getBacklog() {
082        return this.backlog;
083    }
084    public void setBacklog(int backlog) {
085        this.backlog = backlog;
086    }
087
088    /**
089     * Get maximum number of concurrent connections.
090     *
091     * @return max conncurrent connections, or zero for unlimited
092     */
093    public int getMaxConnections() {
094        return this.maxConnections;
095    }
096    public void setMaxConnections(int maxConnections) {
097        this.maxConnections = maxConnections;
098    }
099
100    /**
101     * Get TCP port to listen on.
102     *
103     * @return TCP listening port
104     */
105    public int getPort() {
106        return this.port;
107    }
108    public void setPort(int port) {
109        this.port = port;
110    }
111
112    /**
113     * Start accepting incoming connections. Does nothing if already started.
114     *
115     * @throws IOException if an I/O error occurs setting up the server socket
116     */
117    public void start() throws IOException {
118        synchronized (this) {
119
120            // Already started?
121            if (this.started)
122                return;
123
124            // Create server thread
125            String addr = this.address != null ? "" + this.address : "*";
126            String threadName = this.getClass().getSimpleName() + "[" + addr + ":" + this.port + "]";
127            this.serverThread = new Thread(threadName) {
128                @Override
129                public void run() {
130                    SocketAcceptor.this.run();
131                }
132            };
133
134            // Create socket
135            this.serverSocket = this.createServerSocket();
136            if (this.serverSocket == null)
137                throw new IOException("createServerSocket() returned a null socket");
138        }
139
140        // Start server thread
141        this.serverThread.start();
142        this.started = true;
143    }
144
145    private void run() {
146        try {
147            while (true) {
148
149                // Block while we've reached our connection limit
150                synchronized (this) {
151                    boolean logged = false;
152                    while (this.serverSocket != null && this.maxConnections > 0 && this.connections.size() >= this.maxConnections) {
153                        if (!logged) {
154                            this.log.warn(Thread.currentThread().getName() + " has reached connection limit of "
155                              + this.maxConnections + ", temporarily refusing new connections");
156                            logged = true;
157                        }
158                        try {
159                            this.wait();
160                        } catch (InterruptedException e) {
161                            Thread.currentThread().interrupt();
162                        }
163                    }
164                    if (logged)
165                        this.log.info(Thread.currentThread().getName() + " is accepting new connections again");
166                }
167
168                // Have we been stopped?
169                ServerSocket serverSocketCopy;
170                synchronized (this) {
171                    serverSocketCopy = this.serverSocket;
172                }
173                if (serverSocketCopy == null)
174                    break;
175
176                // Accept a new connection
177                final Socket socket = serverSocketCopy.accept();
178                final String socketDesc = socket.getInetAddress().getHostAddress() + ":" + socket.getPort();
179                this.log.info("accepted new TCP connection from " + socketDesc);
180
181                // Get a handler for it
182                final SocketHandler handler = this.getSocketHandler(socket);
183                if (handler == null) {
184                    this.log.info("null handler returned by getSocketHandler, closing connection from " + socketDesc);
185                    try {
186                        socket.close();
187                    } catch (IOException e) {
188                        // ignore
189                    }
190                    continue;
191                }
192                final SocketInfo socketInfo = new SocketInfo(socket, handler);
193
194                // Create a thread that will handle the connection.
195                Thread handlerThread = new Thread() {
196                    public void run() {
197                        try {
198                            handler.handleConnection(socket);
199                        } catch (Throwable t) {
200                            SocketAcceptor.this.log.error("error handling connection in " + Thread.currentThread().getName(), t);
201                        } finally {
202                            synchronized (SocketAcceptor.this) {
203                                SocketAcceptor.this.connections.remove(socketInfo);
204                                SocketAcceptor.this.notifyAll();
205                            }
206                            try {
207                                socket.close();
208                            } catch (IOException e) {
209                                // ignore
210                            }
211                        }
212                    }
213                };
214                handlerThread.setName(handler.getClass().getSimpleName() + "[" + socketDesc + "]");
215                socketInfo.setThread(handlerThread);
216
217                // Start handler thread and update active connections
218                synchronized (SocketAcceptor.this) {
219
220                    // Need to check again for stopped-ness because we released the lock
221                    if (this.serverSocket == null) {
222                        this.log.info("discarding connection from " + socketDesc + " due to shutdown");
223                        try {
224                            socket.close();
225                        } catch (IOException e) {
226                            // ignore
227                        }
228                        break;
229                    }
230
231                    // Start handler
232                    handlerThread.start();
233                    this.connections.add(socketInfo);
234                }
235            }
236        } catch (IOException e) {
237            boolean expected;
238            synchronized (this) {
239                expected = this.serverSocket == null;
240            }
241            if (!expected)
242                this.log.error("exception in acceptor thread " + Thread.currentThread().getName(), e);
243        } catch (Throwable t) {
244            this.log.error("exception in acceptor thread " + Thread.currentThread().getName(), t);
245        } finally {
246            this.log.info("acceptor thread " + Thread.currentThread().getName() + " exiting");
247            this.closeServerSocket();
248            synchronized (this) {
249                this.serverThread = null;
250                this.notifyAll();
251            }
252        }
253    }
254
255    /**
256     * Stop accepting connections. Does nothing if already stopped.
257     *
258     * <p>
259     * Any currently active connections are stopped via {@link SocketHandler#stop SocketHandler.stop()},
260     * and this method waits until all such connections have returned from {@link SocketHandler#handleConnection
261     * SocketHandler.handleConnection()} before returning.
262     */
263    public synchronized void stop() {
264
265        // Already shut down?
266        if (!this.started)
267            return;
268
269        // Stop acceptor thread
270        if (this.serverThread != null)
271            this.log.info("stopping acceptor thread");
272        this.closeServerSocket();
273        while (this.serverThread != null) {
274            try {
275                this.wait();
276            } catch (InterruptedException e) {
277                Thread.currentThread().interrupt();
278            }
279        }
280
281        // Notify all active connections
282        for (SocketInfo socketInfo : this.connections) {
283
284            // Notify handler
285            try {
286                socketInfo.getHandler().stop(socketInfo.getThread(), socketInfo.getSocket());
287            } catch (Throwable t) {
288                this.log.error("error stopping " + socketInfo.getHandler(), t);
289            }
290
291            // Close socket
292            try {
293                socketInfo.getSocket().close();
294            } catch (IOException e) {
295                // ignore
296            }
297        }
298
299        // Wait for all active connections to complete
300        long lastTime = System.nanoTime();
301        boolean logged = false;
302        while (!this.connections.isEmpty()) {
303            long nextTime = System.nanoTime();
304            if (!logged || nextTime - lastTime > NOTIFICATION_INTERVAL) {
305                this.log.info("waiting for " + this.connections.size() + " active connection(s) to stop...");
306                logged = true;
307            }
308            try {
309                this.wait();
310            } catch (InterruptedException e) {
311                Thread.currentThread().interrupt();
312            }
313        }
314        if (logged)
315            this.log.info("all active connection(s) have stopped");
316
317        // Done
318        this.started = false;
319    }
320
321    /**
322     * Shut down socket.
323     *
324     * @return true if socket was shut down by this invocation, false if it was already shut down
325     */
326    private synchronized void closeServerSocket() {
327        if (this.serverSocket == null)
328            return;
329        try {
330            this.serverSocket.close();
331        } catch (IOException e) {
332            // ignore
333        } finally {
334            this.serverSocket = null;
335        }
336    }
337
338    /**
339     * Create the server's socket via which connections will be accepted.
340     *
341     * <p>
342     * The implementation in {@link SocketAcceptor} creates the socket and sets the "reuse address" flag.
343     * Subclasses may override.
344     *
345     * @return server socket
346     * @throws IOException if an I/O error occurs
347     */
348    protected ServerSocket createServerSocket() throws IOException {
349        ServerSocket socket = new ServerSocket(this.port, this.backlog, this.address);
350        socket.setReuseAddress(true);
351        return socket;
352    }
353
354    /**
355     * Get the {@link SocketHandler} that will handle a new connection using the given socket.
356     *
357     * @param socket connection socket
358     * @return new handler, or <code>null</code> to disconnect the socket immediately
359     * @throws IOException if an I/O error occurs
360     */
361    protected abstract SocketHandler getSocketHandler(Socket socket) throws IOException;
362
363    // Information about one active connection
364    private static class SocketInfo {
365
366        private final Socket socket;
367        private final SocketHandler handler;
368        private Thread thread;
369
370        SocketInfo(Socket socket, SocketHandler handler) {
371            this.socket = socket;
372            this.handler = handler;
373        }
374
375        public Socket getSocket() {
376            return this.socket;
377        }
378
379        public SocketHandler getHandler() {
380            return this.handler;
381        }
382
383        public Thread getThread() {
384            return this.thread;
385        }
386        public void setThread(Thread thread) {
387            this.thread = thread;
388        }
389    }

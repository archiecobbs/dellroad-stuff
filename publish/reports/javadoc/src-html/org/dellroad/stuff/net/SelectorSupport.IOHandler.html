021/**
022 * Support superclass for classes that perform activity based on {@link SelectableChannel} asynchronous I/O notifications.
023 *
024 * <p>
025 * Instances must be {@link #start}ed before use. While running, an internal service thread continuously monitors for
026 * ready I/O operations, notifying the corresponding {@link IOHandler} when the I/O is ready (or channel closed),
027 * and performs periodic housekeeping.
028 *
029 * <p>
030 * Subclasses use {@link #createSelectionKey createSelectionKey()} setup monitoring for a {@link SelectableChannel}, and
031 * {@link #selectFor selectFor()} to adjust the I/O operations being monitored.
032 * All channels are automatically {@linkplain SelectableChannel#configureBlocking configured} for non-blocking mode.
033 *
034 * <p><b>Housekeeping</b>
035 *
036 * <p>
037 * Whether or not there is I/O activity, instances perform regular periodic housekeeping via
038 * {@link #serviceHousekeeping serviceHousekeeping()}. The maximum time between housekeeping checks is configurable
039 * via {@link #setHousekeepingInterval setHousekeepingInterval()}; periodic checks can also be disabled by setting
040 * an interval of zero.
041 *
042 * <p>
043 * Whether or not periodic housekeeping checks are enabled, an immediate housekeeping check can be forced at any
044 * time by invoking {@link #wakeup}.
045 *
046 * <p><b>Shutdown Cleanup</b>
047 *
048 * <p>
049 * On shutdown, subclasses are given an opportunity to perform final cleanup via {@link #serviceCleanup serviceCleanup()}.
050 *
051 * <p><b>Concurrency</b>
052 *
053 * <p>
054 * This class guarantees that the current instance will be locked and the current thread will be the service thread when:
055 * <ul>
056 *  <li>{@link IOHandler} notifications are delivered</li>
057 *  <li>{@link #serviceHousekeeping serviceHousekeeping()} is invoked</li>
058 *  <li>{@link #serviceCleanup serviceCleanup()} is invoked</li>
059 * </ul>
060 *
061 * <p>
062 * Because these callbacks are only ever invoked from the service thread, subclasses can be written without concern for
063 * re-entrancy issues, and subclass methods can use instance synchronization to avoid race conditions from I/O events.
064 */
065public abstract class SelectorSupport {
066
067    /**
068     * Default housekeeping interval ({@value #DEFAULT_HOUSEKEEPING_INTERVAL}ms).
069     *
070     * @see #setHousekeepingInterval setHousekeepingInterval()
071     */
072    public static final int DEFAULT_HOUSEKEEPING_INTERVAL = 1000;
073
074    protected final Logger log = LoggerFactory.getLogger(this.getClass());
075
076    private volatile Selector selector;
077    private ServiceThread serviceThread;
078    private volatile int housekeepingInterval = DEFAULT_HOUSEKEEPING_INTERVAL;

080// Configuration

082    /**
083     * Set the housekeeping interval.
084     *
085     * @param housekeepingInterval housekeeping interval in milliseconds, or zero for none
086     * @throws IllegalArgumentException if {@code housekeepingInterval} is negative
087     */
088    public void setHousekeepingInterval(int housekeepingInterval) {
089        if (housekeepingInterval < 0)
090            throw new IllegalArgumentException("housekeepingInterval < 0");
091        this.housekeepingInterval = housekeepingInterval;
092    }

094// Lifecycle
095
096    /**
097     * Start this instance.
098     *
099     * <p>
100     * Does nothing if already started.
101     *
102     * @throws IOException if a {@link Selector} cannot be created
103     */
104    public synchronized void start() throws IOException {
105        if (this.selector != null)
106            return;
107        boolean successful = false;
108        try {
109            this.selector = Selector.open();
110            this.serviceThread = new ServiceThread();
111            this.serviceThread.start();
112            successful = true;
113        } finally {
114            if (!successful)
115                this.stop();
116        }
117    }
118
119    /**
120     * Stop this instance.
121     *
122     * <p>
123     * Does nothing if already stopped.
124     */
125    public synchronized void stop() {
126
127        // Already stopped?
128        if (this.selector == null)
129            return;
130        assert this.serviceThread != null;
131
132        // Stop service thread
133        if (this.log.isDebugEnabled())
134            this.log.debug("stopping " + this);
135        if (this.selector != null) {
136            try {
137                this.selector.close();
138            } catch (Exception e) {
139                // ignore
140            }
141            this.selector = null;
142        }
143        this.serviceThread.interrupt();
144
145        // Wait for service thread to exit
146        if (this.isServiceThread())
147            return;
148        final Thread currentServiceThread = this.serviceThread;
149        String failure = null;
150        try {
151            if (!TimedWait.wait(this, 1000L, new Predicate() {
152                @Override
153                public boolean test() {
154                    return SelectorSupport.this.serviceThread != currentServiceThread;
155                }
156            }))
157                failure = "timed out";
158        } catch (InterruptedException e) {
159            failure = "interrupted";
160            Thread.currentThread().interrupt();
161        }
162        if (failure != null) {
163            this.log.warn(failure + " waiting for service thread " + currentServiceThread
164              + " while stopping " + this + ", giving up");
165        }
166    }

168// Subclass methods

170    /**
171     * Create a new {@link SelectionKey} by registering the given channel on the given {@link Selector}
172     * and associating the specified {@link IOHandler} to handle I/O ready conditions.
173     *
174     * <p>
175     * <b>Note:</b> the {@code channel} will be {@linkplain SelectableChannel#configureBlocking configured} for non-blocking mode.
176     *
177     * <p>
178     * Initially, no I/O operations will be selected. Use {@link #selectFor selectFor()} to add/remove them.
179     *
180     * <p>
181     * There is no way to explicitly unregister {@code handler} from {@code channel}, although it can be
182     * {@linkplain #selectFor selected} for zero I/O operations.
183     * Handlers are implicitly unregistered either when {@code channel} is closed or this instance is {@link #stop}ed.
184     *
185     * <p>
186     * Does nothing if this instance is not {@link #start}ed.
187     *
188     * @param channel I/O channel
189     * @param handler I/O handler
190     * @return new selection key
191     * @throws IllegalArgumentException if either parameter is null
192     * @throws java.nio.channels.ClosedChannelException if {@code channel} is closed
193     * @throws IOException if {@code channel} cannot be configured for non-blocking  mode
194     * @throws IllegalStateException if this instances is not {@link #start}ed or is shutting down
195     */
196    protected synchronized SelectionKey createSelectionKey(SelectableChannel channel, IOHandler handler) throws IOException {
197
198        // Sanity check
199        if (channel == null)
200            throw new IllegalArgumentException("null channel");
201        if (handler == null)
202            throw new IllegalArgumentException("null handler");
203        if (this.selector == null)
204            throw new IllegalArgumentException("not started");
205
206        // Wakeup service thread: we need it to release the selector's lock (held by select()) so we don't block in register()
207        this.wakeup();
208
209        // Configure channel for non-blocking mode
210        channel.configureBlocking(false);
211
212        // Register channel with our selector
213        return channel.register(this.selector, 0, handler);
214    }
215
216    /**
217     * Enable or disable listening for the specified I/O operation(s).
218     *
219     * <p>
220     * The given {@code selectionKey} must have been acquired from {@link #createSelectionKey}.
221     *
222     * <p>
223     * The change will take effect immediately.
224     *
225     * @param selectionKey selection key
226     * @param ops I/O operations to enable or disable
227     * @param enabled true to enable, false to disable
228     * @throws IllegalArgumentException if {@code selectionKey} was not created by {@link #createSelectionKey createSelectionKey()}
229     * @throws IllegalArgumentException if {@code selectionKey} is null
230     * @throws IllegalArgumentException if {@code ops} contains an invalid operation
231     */
232    protected synchronized void selectFor(SelectionKey selectionKey, int ops, boolean enabled) {
233
234        // Sanity check
235        if (selectionKey == null)
236            throw new IllegalArgumentException("null selectionKey");
237        if (!(selectionKey.attachment() instanceof IOHandler))
238            throw new IllegalArgumentException("unrecognized selectionKey");
239
240        // Wakeup service thread: we will need it to restart its select() operation with updated slection keys
241        this.wakeup();
242
243        // Add/remove ops
244        final int currentOps = selectionKey.interestOps();
245        selectionKey.interestOps(enabled ? currentOps | ops : currentOps & ~ops);
246    }
247
248    /**
249     * Wakeup the service thread.
250     *
251     * <p>
252     * This results in an immediate invocation of {@link #serviceHousekeeping serviceHousekeeping()} (from the service thread).
253     *
254     * <p>
255     * Does nothing if this instance is not {@link #start}ed.
256     *
257     * <p>
258     * This method does not acquire the lock on this instance, so it can be invoked at any time from any context.
259     */
260    protected void wakeup() {
261        assert Thread.holdsLock(this);
262        final Selector currentSelector = this.selector;
263        if (currentSelector != null) {
264            if (this.log.isTraceEnabled())
265                this.log.trace("wakeup service thread");
266            currentSelector.wakeup();
267        }
268    }
269
270    /**
271     * Perform housekeeping.
272     *
273     * <p>
274     * This method is invoked from the internal service thread; this instance will be locked at that time.
275     *
276     * <p>
277     * This method is invoked after every I/O service (while still holding this instance's lock), and periodically
278     * at least every {@linkplain #setHousekeepingInterval housekeeping interval}, if enabled.
279     * If this method is invoked, it is guaranteed that this instance is not being {@link #stop}'ed.
280     *
281     * <p>
282     * Any unchecked exceptions thrown by this method are logged but otherwise ignored. If a fatal error occurs,
283     * {@link #stop} may be invoked to initiate a graceful shutdown.
284     *
285     * <p>
286     * Use {@link #wakeup} to trigger an immediate invocation of this method.
287     *
288     * <p>
289     * The implementation in {@link SelectorSupport} does nothing.
290     */
291    protected void serviceHousekeeping() {
292    }
293
294    /**
295     * Perform shutdown cleanups.
296     *
297     * <p>
298     * This method is invoked at shutdown from the internal service thread; this instance will be locked at that time.
299     *
300     * <p>
301     * Note: ay invocation from this method to {@link #createSelectionKey createSelectionKey()} will result in
302     * an {@link IllegalStateException}.
303     *
304     * <p>
305     * The implementation in {@link SelectorSupport} does nothing.
306     */
307    protected void serviceCleanup() {
308    }

312    /**
313     * Get a debug description of the given keys.
314     *
315     * @param keys selection keys
316     * @return debug description
317     */
318    protected static String dbg(Iterable<? extends SelectionKey> keys) {
319        final ArrayList<String> strings = new ArrayList<>();
320        for (SelectionKey key : keys)
321            strings.add(SelectorSupport.dbg(key));
322        return strings.toString();
323    }
324
325    /**
326     * Get a debug description of the given key.
327     *
328     * @param key selection key
329     * @return debug description
330     */
331    protected static String dbg(SelectionKey key) {
332        try {
333            return "Key[interest=" + SelectorSupport.dbgOps(key.interestOps()) + ",ready="
334              + SelectorSupport.dbgOps(key.readyOps()) + ",obj=" + key.attachment() + "]";
335        } catch (java.nio.channels.CancelledKeyException e) {
336            return "Key[canceled]";
337        }
338    }
339
340    /**
341     * Get a debug description of the given I/O ops.
342     *
343     * @param ops I/O operation bits
344     * @return debug description
345     */
346    protected static String dbgOps(int ops) {
347        final StringBuilder buf = new StringBuilder(4);
348        if ((ops & SelectionKey.OP_ACCEPT) != 0)
349            buf.append("A");
350        if ((ops & SelectionKey.OP_CONNECT) != 0)
351            buf.append("C");
352        if ((ops & SelectionKey.OP_READ) != 0)
353            buf.append("R");
354        if ((ops & SelectionKey.OP_WRITE) != 0)
355            buf.append("W");
356        return buf.toString();
357    }

359// Service loop
360
361    private void service() throws IOException {
362        assert this.isServiceThread();
363    serviceLoop:
364        while (true) {
365
366            // Check if we're still open
367            final Selector currentSelector = this.selector;
368            if (currentSelector == null)
369                break;
370
371            // Wait for I/O readiness, timeout, or shutdown
372            try {
373                if (this.log.isTraceEnabled())
374                    this.log.trace("[SVC THREAD]: sleeping: keys=" + SelectorSupport.dbg(currentSelector.keys()));
375                currentSelector.select(this.housekeepingInterval);
376            } catch (ClosedSelectorException e) {               // close() was invoked
377                break;
378            }
379            if (Thread.interrupted() || this.selector == null)
380                break;
381
382            // Figure out what has happened
383            synchronized (this) {
384
385                // Check if we're still open
386                if (this.selector == null)
387                    break;
388
389                // Handle any ready I/O
390                if (this.log.isTraceEnabled())
391                    this.log.trace("[SVC THREAD]: awake: selectedKeys=" + SelectorSupport.dbg(currentSelector.selectedKeys()));
392                for (Iterator<SelectionKey> i = this.selector.selectedKeys().iterator(); i.hasNext(); ) {
393                    final SelectionKey key = i.next();
394                    i.remove();
395                    final IOHandler handler = (IOHandler)key.attachment();
396                    if (this.log.isTraceEnabled())
397                        this.log.trace("[SVC THREAD]: I/O ready: key=" + SelectorSupport.dbg(key) + " handler=" + handler);
398                    try {
399                        handler.serviceIO(key);
400                    } catch (IOException e) {
401                        if (this.log.isDebugEnabled())
402                            this.log.debug("I/O error from " + handler, e);
403                        handler.close(e);
404                    } catch (Throwable t) {
405                        this.log.error("service error from " + handler, t);
406                        handler.close(t);
407                    }
408                    if (this.selector == null)                              // stop() must have been invoked from handler
409                        break serviceLoop;
410                }
411
412                // Perform housekeeping
413                try {
414                    this.serviceHousekeeping();
415                } catch (Throwable t) {
416                    this.log.error("exception during housekeeping", t);
417                }
418            }
419        }
420
421        // Done
422        synchronized (this) {
423            try {
424                this.serviceCleanup();
425            } catch (Throwable t) {
426                this.log.error("exception during cleanup", t);
427            }
428            if (this.serviceThread == Thread.currentThread())
429                this.serviceThread = null;
430            this.notifyAll();
431        }
432    }
433
434// Service thread
435
436    private class ServiceThread extends Thread {
437
438        public ServiceThread() {
439            super("Service Thread for " + SelectorSupport.this);
440        }
441
442        @Override
443        public void run() {
444            try {
445                SelectorSupport.this.service();
446            } catch (ThreadDeath t) {
447                throw t;
448            } catch (Throwable t) {
449                SelectorSupport.this.log.error("unexpected error in service thread", t);
450            }
451            if (SelectorSupport.this.log.isDebugEnabled())
452                SelectorSupport.this.log.debug(this + " exiting");
453        }
454    }
455
456    /**
457     * Determine whether the current thread is this instance's service thread.
458     *
459     * @return true if the current thread is this instance's service thread
460     */
461    protected boolean isServiceThread() {
462        return Thread.currentThread().equals(this.serviceThread);
463    }

465// IOHandler
466
467    /**
468     * Callback interface used by {@link SelectorSupport}.
469     */
470    public interface IOHandler {
471
472        /**
473         * Handle ready I/O.
474         *
475         * @param key selection key
476         * @throws IOException if an I/O error occurs; this will result in {@link #close close()} being invoked
477         */
478        void serviceIO(SelectionKey key) throws IOException;
479
480        /**
481         * Invoked when an exception is thrown by {@link #serviceIO serviceIO()}.
482         *
483         * <p>
484         * This usually indicates the channel has been (or should be) closed. In any case, it allows for this
485         * handler to perform any required cleanup.
486         *
487         * <p>
488         * Typically this method will close the associated channel, which implicitly unregisters the associated
489         * {@link SelectionKey}s. This also causes the {@link #serviceIO serviceIO()} methods of other handlers
490         * waiting on the same channel to be invoked due to the ready I/O, where they will then likely throw
491         * {@link java.nio.channels.ClosedChannelException}, causing another invocation of this method.
492         * Therefore, this method should be idempotent.
493         *
494         * @param cause the error that occurred
495         */
496        void close(Throwable cause);
497    }

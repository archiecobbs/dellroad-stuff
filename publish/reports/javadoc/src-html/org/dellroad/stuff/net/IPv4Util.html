006package org.dellroad.stuff.net;
007
008import java.net.Inet4Address;
009import java.net.InetAddress;
010import java.net.UnknownHostException;
011import java.util.regex.Matcher;
012import java.util.regex.Pattern;
013
014/**
015 * Utility methods for working with IPv4 addresses and netmasks.
016 */
017public final class IPv4Util {
018
019    public static final int MIN_VALID_NETMASK_WIDTH = 8;
020    public static final int MAX_VALID_NETMASK_WIDTH = 30;
021
022    private IPv4Util() {
023    }
024
025    /**
026     * Convert from 32-bit integer representation.
027     */
028    public static Inet4Address toAddress(int address) {
029        return IPv4Util.toAddress(new byte[] {
030            (byte)((address >> 24) & 0xff),
031            (byte)((address >> 16) & 0xff),
032            (byte)((address >> 8) & 0xff),
033            (byte)(address & 0xff),
034        });
035    }
036
037    /**
038     * Convert to 32-bit integer representation.
039     */
040    public static int toInt32(Inet4Address address) {
041        byte[] bytes = address.getAddress();
042        return ((bytes[0] & 0xff) << 24)
043             | ((bytes[1] & 0xff) << 16)
044             | ((bytes[2] & 0xff) << 8)
045             | (bytes[3] & 0xff);
046    }
047
048    /**
049     * Convert to an unsigned 32-bit integer representation within a 64-bit long.
050     */
051    public static long toUInt32(Inet4Address address) {
052        return toInt32(address) & 0x00000000ffffffffL;
053    }
054
055    /**
056     * Convert raw bytes to an {@link Inet4Address}.
057     *
058     * @throws IllegalArgumentException if array does not have length four
059     */
060    public static Inet4Address toAddress(byte[] bytes) {
061        if (bytes.length != 4)
062            throw new IllegalArgumentException("array has length " + bytes.length + " != 4");
063        try {
064            return (Inet4Address)InetAddress.getByAddress(toString(bytes), bytes);
065        } catch (UnknownHostException e) {
066            throw new RuntimeException("unexpected exception", e);
067        }
068    }
069
070    /**
071     * Convert an IPv4 address to string representation.
072     *
073     * @throws IllegalArgumentException if array does not have length four
074     */
075    public static String toString(Inet4Address address) {
076        return IPv4Util.toString(address.getAddress());
077    }
078
079    /**
080     * Convert a network address (IPv4 base address and netmask) to string representation.
081     *
082     * @throws IllegalArgumentException if array does not have length four
083     */
084    public static String toString(Inet4Address address, Inet4Address netmask) {
085        return IPv4Util.toString(address) + "/" + IPv4Util.getWidth(netmask);
086    }
087
088    private static String toString(byte[] bytes) {
089        assert bytes.length == 4;
090        return (bytes[0] & 0xff) + "." + (bytes[1] & 0xff) + "." + (bytes[2] & 0xff) + "." + (bytes[3] & 0xff);
091    }
092
093    /**
094     * Convert from string representation.
095     *
096     * @throws IllegalArgumentException if string is not an IP address
097     */
098    public static Inet4Address fromString(String string) {
099        String bytepat = "([0-9]|[1-9][0-9]|1[0-9][0-9]|2[0-4][0-9]|25[0-5])";
100        Pattern pattern = Pattern.compile(bytepat + "\\." + bytepat + "\\." + bytepat + "\\." + bytepat);
101        Matcher matcher = pattern.matcher(string);
102        if (!matcher.matches())
103            throw new IllegalArgumentException("invalid IP address: " + string);
104        return toAddress(new byte[] {
105          (byte)Integer.parseInt(matcher.group(1)),
106          (byte)Integer.parseInt(matcher.group(2)),
107          (byte)Integer.parseInt(matcher.group(3)),
108          (byte)Integer.parseInt(matcher.group(4))
109        });
110    }
111
112    /**
113     * Get netmask width.
114     *
115     * @return a value between zero and 32 (inclusive)
116     * @throws IllegalArgumentException if the value is not a valid CIDR netmask
117     */
118    public static int getWidth(Inet4Address netmask) {
119        int mask = toInt32(netmask);
120        if (mask == 0)
121            return 0;
122        for (int shift = 0; shift < 32; shift++) {
123            if (mask == (~0 << shift))
124                return 32 - shift;
125        }
126        throw new IllegalArgumentException("invalid netmask " + netmask);
127    }
128
129    /**
130     * Determine if the given address is a valid CIDR netmask.
131     * The netmask must have a width between 8 and 30 (inclusive).
132     */
133    public static boolean isValidNetmask(Inet4Address netmask) {
134        try {
135            int width = getWidth(netmask);
136            return width >= MIN_VALID_NETMASK_WIDTH && width <= MAX_VALID_NETMASK_WIDTH;
137        } catch (IllegalArgumentException e) {
138            return false;
139        }
140    }
141
142    /**
143     * Get netmask with the given width.
144     *
145     * @throws IllegalArgumentException if width is less than zero or greater than 32
146     */
147    public static Inet4Address getNetmaskForWidth(int width) {
148        if (width < 0 || width > 32)
149            throw new IllegalArgumentException("invalid netmask width " + width);
150        return toAddress(~0 << (32 - width));
151    }
152
153    /**
154     * Get the base address (all host bits zero) of the given IP network.
155     *
156     * @param address any IP address on the network
157     * @param netmask the netmask of the network
158     * @throws IllegalArgumentException if {@code netmask} is not a valid netmask
159     */
160    public static Inet4Address getBaseAddress(Inet4Address address, Inet4Address netmask) {
161        int addr = toInt32(address);
162        int width = getWidth(netmask);
163        addr &= ~0 << (32 - width);
164        return toAddress(addr);
165    }
166
167    /**
168     * Get the broadcast address (all host bits one) of the given IP network.
169     *
170     * @param address any IP address on the network
171     * @param netmask the netmask of the network
172     * @throws IllegalArgumentException if {@code netmask} is not a valid netmask
173     */
174    public static Inet4Address getBroadcastAddress(Inet4Address address, Inet4Address netmask) {
175        int addr = toInt32(address);
176        int width = getWidth(netmask);
177        addr |= ~(~0 << (32 - width));
178        return toAddress(addr);
179    }
180
181    /**
182     * Determine if the given address lives on the given network.
183     *
184     * @param address IP address in question
185     * @param network network address
186     * @param netmask network netmask
187     * @throws IllegalArgumentException if {@code netmask} is not a valid netmask
188     */
189    public static boolean isOnNetwork(Inet4Address address, Inet4Address network, Inet4Address netmask) {
190        int addr1 = toInt32(address);
191        int addr2 = toInt32(network);
192        int width = getWidth(netmask);
193        int mask = ~0 << (32 - width);
194        return (addr1 & mask) == (addr2 & mask);
195    }
196
197    /**
198     * Determine if the given address is a valid host address on the given network.
199     * The address must lie on the given network and not equal either the base or broadcast addresses.
200     *
201     * @param address IP address in question
202     * @param network network address
203     * @param netmask network netmask
204     * @throws IllegalArgumentException if {@code netmask} is not a valid netmask
205     */
206    public static boolean isValidHostOnNetwork(Inet4Address address, Inet4Address network, Inet4Address netmask) {
207        return isOnNetwork(address, network, netmask)
208         && !address.equals(getBaseAddress(network, netmask))
209         && !address.equals(getBroadcastAddress(network, netmask));
210    }
211}

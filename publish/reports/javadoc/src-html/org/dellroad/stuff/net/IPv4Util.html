006package org.dellroad.stuff.net;
007
008import java.net.Inet4Address;
009import java.net.InetAddress;
010import java.net.UnknownHostException;
011import java.util.regex.Matcher;
012import java.util.regex.Pattern;
013
014/**
015 * Utility methods for working with IPv4 addresses and netmasks.
016 */
017public final class IPv4Util {
018
019    public static final int MIN_VALID_NETMASK_WIDTH = 8;
020    public static final int MAX_VALID_NETMASK_WIDTH = 30;
021
022    private IPv4Util() {
023    }
024
025    /**
026     * Convert from 32-bit integer representation.
027     *
028     * @param address IP address as a 32-bit integer
029     * @return the corresponding IP address
030     */
031    public static Inet4Address toAddress(int address) {
032        return IPv4Util.toAddress(new byte[] {
033            (byte)((address >> 24) & 0xff),
034            (byte)((address >> 16) & 0xff),
035            (byte)((address >> 8) & 0xff),
036            (byte)(address & 0xff),
037        });
038    }
039
040    /**
041     * Convert to 32-bit integer representation.
042     *
043     * @param address IP address
044     * @return the corresponding 32-bit integer
045     */
046    public static int toInt32(Inet4Address address) {
047        byte[] bytes = address.getAddress();
048        return ((bytes[0] & 0xff) << 24)
049             | ((bytes[1] & 0xff) << 16)
050             | ((bytes[2] & 0xff) << 8)
051             | (bytes[3] & 0xff);
052    }
053
054    /**
055     * Convert to an unsigned 32-bit integer representation within a 64-bit long.
056     *
057     * @param address IP address
058     * @return the corresponding unsigned 32-bit integer
059     */
060    public static long toUInt32(Inet4Address address) {
061        return toInt32(address) & 0x00000000ffffffffL;
062    }
063
064    /**
065     * Convert raw bytes to an {@link Inet4Address}.
066     *
067     * @param bytes IP address bytes (big endian)
068     * @return corresponding IP address
069     * @throws IllegalArgumentException if array does not have length four
070     */
071    public static Inet4Address toAddress(byte[] bytes) {
072        if (bytes.length != 4)
073            throw new IllegalArgumentException("array has length " + bytes.length + " != 4");
074        try {
075            return (Inet4Address)InetAddress.getByAddress(toString(bytes), bytes);
076        } catch (UnknownHostException e) {
077            throw new RuntimeException("unexpected exception", e);
078        }
079    }
080
081    /**
082     * Convert an IPv4 address to string representation.
083     *
084     * @param address IP address
085     * @return {@code address} in {@link String} form
086     * @throws IllegalArgumentException if array does not have length four
087     */
088    public static String toString(Inet4Address address) {
089        return IPv4Util.toString(address.getAddress());
090    }
091
092    /**
093     * Convert a network address (IPv4 base address and netmask) to string representation.
094     * Example: {@code 192.168.1.39/24}
095     *
096     * @param address IP address
097     * @param netmask netmask for network
098     * @return address plus netmask width
099     * @throws IllegalArgumentException if array does not have length four
100     */
101    public static String toString(Inet4Address address, Inet4Address netmask) {
102        return IPv4Util.toString(address) + "/" + IPv4Util.getWidth(netmask);
103    }
104
105    private static String toString(byte[] bytes) {
106        assert bytes.length == 4;
107        return (bytes[0] & 0xff) + "." + (bytes[1] & 0xff) + "." + (bytes[2] & 0xff) + "." + (bytes[3] & 0xff);
108    }
109
110    /**
111     * Convert from string representation.
112     *
113     * @param string IP address in string form
114     * @return parsed IP address
115     * @throws IllegalArgumentException if string is not an IP address
116     */
117    public static Inet4Address fromString(String string) {
118        String bytepat = "([0-9]|[1-9][0-9]|1[0-9][0-9]|2[0-4][0-9]|25[0-5])";
119        Pattern pattern = Pattern.compile(bytepat + "\\." + bytepat + "\\." + bytepat + "\\." + bytepat);
120        Matcher matcher = pattern.matcher(string);
121        if (!matcher.matches())
122            throw new IllegalArgumentException("invalid IP address: " + string);
123        return toAddress(new byte[] {
124          (byte)Integer.parseInt(matcher.group(1)),
125          (byte)Integer.parseInt(matcher.group(2)),
126          (byte)Integer.parseInt(matcher.group(3)),
127          (byte)Integer.parseInt(matcher.group(4))
128        });
129    }
130
131    /**
132     * Get netmask width.
133     *
134     * @param netmask netmask
135     * @return netmask width in bits, a value between zero and 32 (inclusive)
136     * @throws IllegalArgumentException if the value is not a valid CIDR netmask
137     */
138    public static int getWidth(Inet4Address netmask) {
139        int mask = toInt32(netmask);
140        if (mask == 0)
141            return 0;
142        for (int shift = 0; shift < 32; shift++) {
143            if (mask == (~0 << shift))
144                return 32 - shift;
145        }
146        throw new IllegalArgumentException("invalid netmask " + netmask);
147    }
148
149    /**
150     * Determine if the given address is a valid CIDR netmask.
151     * The netmask must have a width between 8 and 30 (inclusive).
152     *
153     * @param netmask netmask
154     * @return true if {@code netmask} is a valid netmask
155     */
156    public static boolean isValidNetmask(Inet4Address netmask) {
157        try {
158            int width = getWidth(netmask);
159            return width >= MIN_VALID_NETMASK_WIDTH && width <= MAX_VALID_NETMASK_WIDTH;
160        } catch (IllegalArgumentException e) {
161            return false;
162        }
163    }
164
165    /**
166     * Get netmask with the given width.
167     *
168     * @param width netmask width in bits
169     * @return netmask with the given width
170     * @throws IllegalArgumentException if width is less than zero or greater than 32
171     */
172    public static Inet4Address getNetmaskForWidth(int width) {
173        if (width < 0 || width > 32)
174            throw new IllegalArgumentException("invalid netmask width " + width);
175        return toAddress(~0 << (32 - width));
176    }
177
178    /**
179     * Get the base address (all host bits zero) of the given IP network.
180     *
181     * @param address any IP address on the network
182     * @param netmask the netmask of the network
183     * @return base address for the specified network
184     * @throws IllegalArgumentException if {@code netmask} is not a valid netmask
185     */
186    public static Inet4Address getBaseAddress(Inet4Address address, Inet4Address netmask) {
187        int addr = toInt32(address);
188        int width = getWidth(netmask);
189        addr &= ~0 << (32 - width);
190        return toAddress(addr);
191    }
192
193    /**
194     * Get the broadcast address (all host bits one) of the given IP network.
195     *
196     * @param address any IP address on the network
197     * @param netmask the netmask of the network
198     * @return broadcast address for the specified network
199     * @throws IllegalArgumentException if {@code netmask} is not a valid netmask
200     */
201    public static Inet4Address getBroadcastAddress(Inet4Address address, Inet4Address netmask) {
202        int addr = toInt32(address);
203        int width = getWidth(netmask);
204        addr |= ~(~0 << (32 - width));
205        return toAddress(addr);
206    }
207
208    /**
209     * Determine if the given address lives on the given network.
210     *
211     * @param address IP address in question
212     * @param network network address
213     * @param netmask network netmask
214     * @return true if {@code address} is on the specified network
215     * @throws IllegalArgumentException if {@code netmask} is not a valid netmask
216     */
217    public static boolean isOnNetwork(Inet4Address address, Inet4Address network, Inet4Address netmask) {
218        int addr1 = toInt32(address);
219        int addr2 = toInt32(network);
220        int width = getWidth(netmask);
221        int mask = ~0 << (32 - width);
222        return (addr1 & mask) == (addr2 & mask);
223    }
224
225    /**
226     * Determine if the given address is a valid host address on the given network.
227     * The address must lie on the given network and not equal either the base or broadcast addresses.
228     *
229     * @param address IP address in question
230     * @param network network address
231     * @param netmask network netmask
232     * @return true if {@code address} is valid on the specified network
233     * @throws IllegalArgumentException if {@code netmask} is not a valid netmask
234     */
235    public static boolean isValidHostOnNetwork(Inet4Address address, Inet4Address network, Inet4Address netmask) {
236        return isOnNetwork(address, network, netmask)
237         && !address.equals(getBaseAddress(network, netmask))
238         && !address.equals(getBroadcastAddress(network, netmask));
239    }
240}

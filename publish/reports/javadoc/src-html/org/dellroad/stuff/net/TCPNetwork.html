016/**
017 * {@link Network} implementation based on TCP connections.
018 *
019 * <p>
020 * Remote peers have the {@link String} form <code>IP-Address[:port]</code>. If the port is omitted,
021 * the default port provided to the constructor is assumed.
022 */
023public class TCPNetwork extends ChannelNetwork implements Network {
024
025    /**
026     * Default connect timeout for outgoing connections ({@value #DEFAULT_CONNECT_TIMEOUT} milliseconds).
027     *
028     * @see #getConnectTimeout
029     */
030    public static final long DEFAULT_CONNECT_TIMEOUT = 20 * 1000L;              // 20 sec
031
032    private final int defaultPort;
033
034    private InetSocketAddress listenAddress;
035    private long connectTimeout = DEFAULT_CONNECT_TIMEOUT;
036
037    private ServerSocketChannel serverSocketChannel;
038    private SelectionKey selectionKey;
039
040// Constructors
041
042    /**
043     * Constructor.
044     *
045     * @param defaultPort default TCP port when no port is explicitly proviced
046     * @throws IllegalArgumentException if {@code port} is invalid
047     */
048    public TCPNetwork(int defaultPort) {
049        if (defaultPort <= 0 || defaultPort > 65535)
050            throw new IllegalArgumentException("invalid default port " + defaultPort);
051        this.defaultPort = defaultPort;
052        this.listenAddress = new InetSocketAddress(this.defaultPort);
053    }

097// Lifecycle
098
099    @Override
100    public synchronized void start(Handler handler) throws IOException {
101        super.start(handler);
102        boolean successful = false;
103        try {
104            if (this.serverSocketChannel != null)
105                return;
106            this.serverSocketChannel = ServerSocketChannel.open();
107            this.configureServerSocketChannel(this.serverSocketChannel);
108            this.serverSocketChannel.bind(this.listenAddress);
109            this.selectionKey = this.createSelectionKey(this.serverSocketChannel, new IOHandler() {
110                @Override
111                public void serviceIO(SelectionKey key) throws IOException {
112                    if (key.isAcceptable())
113                        TCPNetwork.this.handleAccept();
114                }
115                @Override
116                public void close(Throwable cause) {
117                    TCPNetwork.this.log.error("stopping " + this + " due to exception", cause);
118                    TCPNetwork.this.stop();
119                }
120            });
121            this.selectForAccept(true);
122            successful = true;
123        } finally {
124            if (!successful)
125                this.stop();
126        }
127    }
128
129    @Override
130    public void stop() {
131        super.stop();
132        synchronized (this) {
133            if (this.serverSocketChannel != null) {
134                try {
135                    this.serverSocketChannel.close();
136                } catch (Exception e) {
137                    // ignore
138                }
139                this.serverSocketChannel = null;
140            }
141            this.selectionKey = null;
142        }
143    }

145// Utility methods
146
147    /**
148     * Parse out the address part of an address that has an optional colon plus TCP port number suffix.
149     *
150     * @param address address of the form {@code ipaddr} or {@code ipaddr:port}
151     * @return the IP address part
152     */
153    public static String parseAddressPart(String address) {
154        return (String)TCPNetwork.parseAddress(address, 0)[0];
155    }
156
157    /**
158     * Parse out the port part of an address that has an optional colon plus TCP port number suffix.
159     *
160     * @param address address of the form {@code ipaddr} or {@code ipaddr:port}
161     * @param defaultPort default port if none specified in {@code address}
162     * @return the port part, or {@code defaultPort} if there is no explicit port
163     */
164    public static int parsePortPart(String address, int defaultPort) {
165        return (Integer)TCPNetwork.parseAddress(address, defaultPort)[1];
166    }
167
168    private static Object[] parseAddress(String string, int defaultPort) {
169        final int colon = string.lastIndexOf(':');
170        if (colon == -1)
171            return new Object[] { string, defaultPort };
172        try {
173            final int port = Integer.parseInt(string.substring(colon + 1), 10);
174            if (port < 1 || port > 65535)
175                return new Object[] { string, defaultPort };
176            return new Object[] { string.substring(0, colon), port };
177        } catch (Exception e) {
178            return new Object[] { string, defaultPort };
179        }
180    }

214// I/O Ready Conditions
215
216    // Invoked when we have a new incoming connection
217    private void handleAccept() throws IOException {
218
219        // Sanity check
220        assert this.isServiceThread();
221
222        // Check connection size limit
223        if (this.connectionMap.size() >= this.getMaxConnections()) {
224            this.log.warn("too many network connections (" + this.connectionMap.size() + " >= "
225              + this.getMaxConnections() + "), not accepting any more (for now)");
226            this.selectForAccept(false);
227            return;
228        }
229
230        // Accept connection
231        SocketChannel socketChannel = this.serverSocketChannel.accept();
232        if (socketChannel == null)
233            return;
234        this.log.info("accepted incoming connection from " + socketChannel.getRemoteAddress());
235        socketChannel
236          .setOption(StandardSocketOptions.SO_KEEPALIVE, true)
237          .setOption(StandardSocketOptions.TCP_NODELAY, true);
238        this.configureSocketChannel(socketChannel);
239
240        // Create peer
241        final InetSocketAddress remote = (InetSocketAddress)socketChannel.socket().getRemoteSocketAddress();
242        final String peer = remote.getHostString() + (remote.getPort() != this.defaultPort ? ":" + remote.getPort() : "");
243
244        // Are we already connected to this peer? If so (deterministically) choose which connection wins
245        TCPConnection connection = (TCPConnection)this.connectionMap.get(peer);
246        if (connection != null) {
247
248            // Compare the socket addresses of the initiator side of each connection
249            final SocketAddress oldAddr = (InetSocketAddress)connection.getSocketChannel().socket().getLocalSocketAddress();
250            final SocketAddress newAddr = (InetSocketAddress)socketChannel.getRemoteAddress();
251            final String oldDesc = oldAddr.toString().replaceAll("^[^/]*/", "");            // strip off hostname part, if any
252            final String newDesc = newAddr.toString().replaceAll("^[^/]*/", "");            // strip off hostname part, if any
253            final int diff = newDesc.compareTo(oldDesc);
254            this.log.info("connection mid-air collision: old: " + oldDesc + ", new: " + newDesc
255              + ", winner: " + (diff < 0 ? "new" : diff > 0 ? "old" : "neither (?)"));
256
257            // Close the loser(s)
258            if (diff >= 0) {
259                this.log.info("rejecting incoming connection from " + socketChannel.getRemoteAddress() + " as duplicate");
260                socketChannel.close();
261                socketChannel = null;
262            }
263            if (diff <= 0) {
264                this.log.info("closing existing duplicate connection to " + socketChannel.getRemoteAddress());
265                this.connectionMap.remove(peer);
266                connection.close(new IOException("duplicate connection"));
267                connection = null;
268            }
269        }
270
271        // Create connection from new socket if needed
272        if (connection == null && socketChannel != null) {
273            connection = new TCPConnection(this, peer, socketChannel);
274            this.connectionMap.put(peer, connection);
275            this.handleOutputQueueEmpty(connection);
276        }
277    }
278
279    // Enable/disable incoming connections
280    private void selectForAccept(boolean enabled) throws IOException {
281        if (this.selectionKey == null)
282            return;
283        if (enabled && (this.selectionKey.interestOps() & SelectionKey.OP_ACCEPT) == 0) {
284            this.selectFor(this.selectionKey, SelectionKey.OP_ACCEPT, true);
285            if (this.log.isDebugEnabled())
286                this.log.debug(this + " started listening for incoming connections");
287        } else if (!enabled && (this.selectionKey.interestOps() & SelectionKey.OP_ACCEPT) != 0) {
288            this.selectFor(this.selectionKey, SelectionKey.OP_ACCEPT, false);
289            if (this.log.isDebugEnabled())
290                this.log.debug(this + " stopped listening for incoming connections");
291        }
292    }

294// ChannelNetwork

296    /**
297     * Normalize the given peer.
298     *
299     * <p>
300     * The implementation in {@link TCPNetwork} converts {@code peer} to lower case and appends the
301     * (default) port if not specified in the string.
302     *
303     * @throws IllegalArgumentException {@code inheritDoc}
304     */
305    @Override
306    protected String normalizePeerName(String peer) {
307        if (peer == null)
308            throw new IllegalArgumentException("null peer");
309        final int colon = peer.lastIndexOf(':');
310        if (colon == -1)
311            peer += ":" + this.defaultPort;
312        return peer;
313    }
314
315    @Override
316    protected synchronized TCPConnection createConnection(String peer) throws IOException {
317
318        // Create new socket
319        final SocketChannel socketChannel = SocketChannel.open()
320          .setOption(StandardSocketOptions.SO_KEEPALIVE, true)
321          .setOption(StandardSocketOptions.TCP_NODELAY, true);
322        this.configureSocketChannel(socketChannel);
323        socketChannel.configureBlocking(false);
324
325        // Resolve peer name into a socket address
326        if (this.log.isDebugEnabled())
327            this.log.debug(this + " looking up peer address `" + peer + "'");
328        InetSocketAddress socketAddress = null;
329        try {
330            socketAddress = new InetSocketAddress(
331              TCPNetwork.parseAddressPart(peer), TCPNetwork.parsePortPart(peer, this.defaultPort));
332        } catch (IllegalArgumentException e) {
333            if (this.log.isTraceEnabled())
334                this.log.trace(this + " peer address `" + peer + "' is invalid", e);
335        }
336        if (socketAddress == null || socketAddress.isUnresolved())
337            throw new IOException("invalid or unresolvable peer address `" + peer + "'");
338
339        // Initiate connection to peer
340        if (this.log.isDebugEnabled()) {
341            this.log.debug(this + ": resolved peer address `" + peer + "' to " + socketAddress.getAddress()
342              + "; now initiating connection");
343        }
344        socketChannel.connect(socketAddress);
345
346        // Create new connection
347        return new TCPConnection(this, peer, socketChannel);
348    }

350// SelectorSupport

352    @Override
353    protected void serviceHousekeeping() {
354        super.serviceHousekeeping();
355        try {
356            this.selectForAccept(this.connectionMap.size() < this.getMaxConnections());
357        } catch (IOException e) {
358            throw new RuntimeException("unexpected exception", e);
359        }
360    }

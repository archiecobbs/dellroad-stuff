025/**
026 * {@link Network} implementation based on TCP sockets.
027 *
028 * <p>
029 * Remote peers have the {@link String} form <code>IP-Address[:port]</code>. If the port is omitted,
030 * the default port provided to the constructor is assumed.
031 * </p>
032 */

198    @Override
199    public synchronized void start(Handler handler) throws IOException {
200        super.start();
201        boolean successful = false;
202        try {
203            if (this.handler != null)
204                return;
205            if (this.log.isDebugEnabled())
206                this.log.debug("starting " + this + " listening on " + this.listenAddress);
207            this.serverSocketChannel = ServerSocketChannel.open();
208            this.configureServerSocketChannel(this.serverSocketChannel);
209            this.serverSocketChannel.bind(this.listenAddress);
210            this.selectionKey = this.createSelectionKey(this.serverSocketChannel, new IOHandler() {
211                @Override
212                public void serviceIO(SelectionKey key) throws IOException {
213                    if (key.isAcceptable())
214                        TCPNetwork.this.handleAccept();
215                }
216                @Override
217                public void close(Throwable cause) {
218                    TCPNetwork.this.log.error("stopping " + this + " due to exception", cause);
219                    TCPNetwork.this.stop();
220                }
221            });
222            this.selectForAccept(true);
223            this.executor = Executors.newSingleThreadExecutor();
224            this.handler = handler;
225            successful = true;
226        } finally {
227            if (!successful)
228                this.stop();
229        }
230    }
231
232    @Override
233    public void stop() {
234        super.stop();
235        synchronized (this) {
236            if (this.handler == null)
237                return;
238            if (this.log.isDebugEnabled())
239                this.log.debug("stopping " + this);
240            if (this.serverSocketChannel != null) {
241                try {
242                    this.serverSocketChannel.close();
243                } catch (Exception e) {
244                    // ignore
245                }
246                this.serverSocketChannel = null;
247            }
248            if (this.executor != null) {
249                this.executor.shutdownNow();
250                try {
251                    this.executor.awaitTermination(1000, TimeUnit.MILLISECONDS);
252                } catch (InterruptedException e) {
253                    Thread.currentThread().interrupt();
254                }
255                this.executor = null;
256            }
257            this.selectionKey = null;
258            this.handler = null;
259        }
260    }

264    @Override
265    public synchronized boolean send(String peer, ByteBuffer msg) {
266
267        // Sanity check
268        if (peer == null)
269            throw new IllegalArgumentException("null peer");
270
271        // Get/create connection
272        TCPConnection connection = this.connectionMap.get(peer);
273        if (connection == null) {
274
275            // Create connection
276            try {
277                connection = this.createConnection(peer);
278            } catch (IOException e) {
279                this.log.info(this + " unable to send message to `" + peer + "': " + e.getMessage());
280                return false;
281            }
282
283            // Record connection
284            this.connectionMap.put(peer, connection);
285        }
286
287        // Send message
288        return connection.output(msg);
289    }

293    /**
294     * Parse out the address part of an address that has an optional colon plus TCP port number suffix.
295     *
296     * @param address address of the form {@code ipaddr} or {@code ipaddr:port}
297     * @return the IP address part
298     */
299    public static String parseAddressPart(String address) {
300        return (String)TCPNetwork.parseAddress(address, 0)[0];
301    }
302
303    /**
304     * Parse out the port part of an address that has an optional colon plus TCP port number suffix.
305     *
306     * @param address address of the form {@code ipaddr} or {@code ipaddr:port}
307     * @param defaultPort default port if none specified in {@code address}
308     * @return the port part, or {@code defaultPort} if there is no explicit port
309     */
310    public static int parsePortPart(String address, int defaultPort) {
311        return (Integer)TCPNetwork.parseAddress(address, defaultPort)[1];
312    }
313
314    private static Object[] parseAddress(String string, int defaultPort) {
315        final int colon = string.lastIndexOf(':');
316        if (colon == -1)
317            return new Object[] { string, defaultPort };
318        try {
319            final int port = Integer.parseInt(string.substring(colon + 1), 10);
320            if (port < 1 || port > 65535)
321                return new Object[] { string, defaultPort };
322            return new Object[] { string.substring(0, colon), port };
323        } catch (Exception e) {
324            return new Object[] { string, defaultPort };
325        }
326    }

362// I/O Ready Conditions

364    // Invoked when we have a new incoming connection
365    private void handleAccept() throws IOException {
366
367        // Sanity check
368        assert this.isServiceThread();
369
370        // Check connection size limit
371        if (this.connectionMap.size() >= this.maxConnections) {
372            this.log.warn("too many network connections (" + this.connectionMap.size() + " >= "
373              + this.maxConnections + "), not accepting any more (for now)");
374            this.selectForAccept(false);
375            return;
376        }
377
378        // Accept connection
379        SocketChannel socketChannel = this.serverSocketChannel.accept();
380        if (socketChannel == null)
381            return;
382        this.log.info("accepted incoming connection from " + socketChannel.getRemoteAddress());
383        socketChannel
384          .setOption(StandardSocketOptions.SO_KEEPALIVE, true)
385          .setOption(StandardSocketOptions.TCP_NODELAY, true);
386        this.configureSocketChannel(socketChannel);
387
388        // Create peer
389        final InetSocketAddress remote = (InetSocketAddress)socketChannel.socket().getRemoteSocketAddress();
390        final String peer = remote.getHostString() + (remote.getPort() != this.defaultPort ? ":" + remote.getPort() : "");
391
392        // Are we already connected to this peer? If so (deterministically) choose which connection wins
393        TCPConnection connection = this.connectionMap.get(peer);
394        if (connection != null) {
395
396            // Compare the socket addresses of the initiator side of each connection
397            final SocketAddress oldAddr = (InetSocketAddress)connection.getSocketChannel().socket().getLocalSocketAddress();
398            final SocketAddress newAddr = (InetSocketAddress)socketChannel.getRemoteAddress();
399            final String oldDesc = oldAddr.toString().replaceAll("^[^/]*/", "");            // strip off hostname part, if any
400            final String newDesc = newAddr.toString().replaceAll("^[^/]*/", "");            // strip off hostname part, if any
401            final int diff = newDesc.compareTo(oldDesc);
402            this.log.info("connection mid-air collision: old: " + oldDesc + ", new: " + newDesc
403              + ", winner: " + (diff < 0 ? "new" : diff > 0 ? "old" : "neither (?)"));
404
405            // Close the loser(s)
406            if (diff >= 0) {
407                this.log.info("rejecting incoming connection from " + socketChannel.getRemoteAddress() + " as duplicate");
408                socketChannel.close();
409                socketChannel = null;
410            }
411            if (diff <= 0) {
412                this.log.info("closing existing duplicate connection to " + socketChannel.getRemoteAddress());
413                this.connectionMap.remove(peer);
414                connection.close(new IOException("duplicate connection"));
415                connection = null;
416            }
417        }
418
419        // Create connection from new socket if needed
420        if (connection == null && socketChannel != null) {
421            connection = new TCPConnection(this, peer, socketChannel);
422            this.connectionMap.put(peer, connection);
423            this.handleOutputQueueEmpty(connection);
424        }
425    }

427    // Enable/disable incoming connections
428    private void selectForAccept(boolean enabled) throws IOException {
429        if (this.selectionKey == null)
430            return;
431        if (enabled && (this.selectionKey.interestOps() & SelectionKey.OP_ACCEPT) == 0) {
432            this.selectFor(this.selectionKey, SelectionKey.OP_ACCEPT, true);
433            if (this.log.isDebugEnabled())
434                this.log.debug(this + " started listening for incoming connections");
435        } else if (!enabled && (this.selectionKey.interestOps() & SelectionKey.OP_ACCEPT) != 0) {
436            this.selectFor(this.selectionKey, SelectionKey.OP_ACCEPT, false);
437            if (this.log.isDebugEnabled())
438                this.log.debug(this + " stopped listening for incoming connections");
439        }
440    }

444    // Invoked when a message arrives on a connection
445    void handleMessage(final TCPConnection connection, final ByteBuffer msg) {
446        assert Thread.holdsLock(this);
447        assert this.isServiceThread();
448        this.executor.submit(new Runnable() {
449            @Override
450            public void run() {
451                try {
452                    TCPNetwork.this.handler.handle(connection.getPeer(), msg);
453                } catch (Throwable t) {
454                    TCPNetwork.this.log.error("exception in callback", t);
455                }
456            }
457        });
458    }

460    // Invoked a connection's output queue goes empty
461    void handleOutputQueueEmpty(final TCPConnection connection) {
462        assert Thread.holdsLock(this);
463        assert this.isServiceThread();
464        this.executor.submit(new Runnable() {
465            @Override
466            public void run() {
467                try {
468                    TCPNetwork.this.handler.outputQueueEmpty(connection.getPeer());
469                } catch (Throwable t) {
470                    TCPNetwork.this.log.error("exception in callback", t);
471                }
472            }
473        });
474    }

476    // Invoked when a connection closes
477    void handleConnectionClosed(TCPConnection connection) {
478        assert Thread.holdsLock(this);
479        assert this.isServiceThread();
480        if (this.log.isDebugEnabled())
481            this.log.debug(this + " handling closed connection " + connection);
482        this.connectionMap.remove(connection.getPeer());
483        this.handleOutputQueueEmpty(connection);
484        this.wakeup();
485    }

489    // Create a new connection to the specified peer
490    private synchronized TCPConnection createConnection(String peer) throws IOException {
491
492        // Create new one
493        final SocketChannel socketChannel = SocketChannel.open()
494          .setOption(StandardSocketOptions.SO_KEEPALIVE, true)
495          .setOption(StandardSocketOptions.TCP_NODELAY, true);
496        this.configureSocketChannel(socketChannel);
497        socketChannel.configureBlocking(false);
498        if (this.log.isDebugEnabled())
499            this.log.debug(this + " looking up peer address `" + peer + "'");
500
501        // Resolve peer name into a socket address
502        InetSocketAddress socketAddress = null;
503        try {
504            socketAddress = new InetSocketAddress(
505              TCPNetwork.parseAddressPart(peer), TCPNetwork.parsePortPart(peer, this.defaultPort));
506        } catch (IllegalArgumentException e) {
507            if (this.log.isTraceEnabled())
508                this.log.trace(this + " peer address `" + peer + "' is invalid", e);
509        }
510        if (socketAddress == null || socketAddress.isUnresolved())
511            throw new IOException("invalid or unresolvable peer address `" + peer + "'");
512
513        // Initiate connection to peer
514        if (this.log.isDebugEnabled()) {
515            this.log.debug(this + ": resolved peer address `" + peer + "' to " + socketAddress.getAddress()
516              + "; now initiating connection");
517        }
518        socketChannel.connect(socketAddress);
519
520        // Create new connection
521        return new TCPConnection(this, peer, socketChannel);
522    }

526    @Override
527    protected void serviceHousekeeping() {
528
529        // Perform connection housekeeping
530        for (TCPConnection connection : new ArrayList<TCPConnection>(this.connectionMap.values())) {
531            try {
532                connection.performHousekeeping();
533            } catch (IOException e) {
534                if (this.log.isDebugEnabled())
535                    this.log.debug("I/O error from " + connection, e);
536                connection.close(e);
537            } catch (Throwable t) {
538                this.log.error("error performing housekeeping for " + connection, t);
539                connection.close(t);
540            }
541        }
542
543        // Perform my own housekeeping
544        try {
545            this.selectForAccept(this.connectionMap.size() < this.maxConnections);
546        } catch (IOException e) {
547            throw new RuntimeException("unexpected exception", e);
548        }
549    }
550
551    @Override
552    protected void serviceCleanup() {
553        for (TCPConnection connection : new ArrayList<TCPConnection>(this.connectionMap.values()))
554            connection.close(null);
555    }

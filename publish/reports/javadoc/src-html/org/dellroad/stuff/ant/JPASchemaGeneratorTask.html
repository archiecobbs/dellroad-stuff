006package org.dellroad.stuff.ant;
007
008import java.io.File;
009import java.net.MalformedURLException;
010import java.net.URL;
011import java.sql.Connection;
012import java.sql.DriverManager;
013import java.sql.SQLException;
014import java.util.HashMap;
015
016import javax.persistence.Persistence;
017import javax.persistence.PersistenceException;
018
019import org.apache.tools.ant.AntClassLoader;
020import org.apache.tools.ant.BuildException;
021import org.apache.tools.ant.Project;
022import org.apache.tools.ant.Task;
023import org.apache.tools.ant.types.Path;
024
025/**
026 * Ant task for JPA 2.1 schema generation.
027 *
028 * @see <a href="https://blogs.oracle.com/arungupta/entry/jpa_2_1_schema_generation">JPA 2.1 Schema Generation (TOTD #187)</a>
029 */
030public class JPASchemaGeneratorTask extends Task {
031
032    private boolean create;
033    private boolean drop;
034    private boolean schemas;
035    private String unit;
036    private String driver;
037    private String url;
038    private String username;
039    private String password;
040    private String databaseProductName;
041    private int databaseMajorVersion;
042    private int databaseMinorVersion;
043    private File file;
044    private Path classPath;
045
046    public void setCreate(boolean create) {
047        this.create = create;
048    }
049
050    public void setDrop(boolean drop) {
051        this.drop = drop;
052    }
053
054    public void setSchemas(boolean schemas) {
055        this.schemas = schemas;
056    }
057
058    public void setUnit(String unit) {
059        this.unit = unit;
060    }
061
062    public void setDriver(String driver) {
063        this.driver = driver;
064    }
065
066    public void setURL(String url) {
067        this.url = url;
068    }
069
070    public void setUsername(String username) {
071        this.username = username;
072    }
073
074    public void setPassword(String password) {
075        this.password = password;
076    }
077
078    public void setDatabaseProductName(String databaseProductName) {
079        this.databaseProductName = databaseProductName;
080    }
081
082    public void setDatabaseMajorVersion(int databaseMajorVersion) {
083        this.databaseMajorVersion = databaseMajorVersion;
084    }
085
086    public void setDatabaseMinorVersion(int databaseMinorVersion) {
087        this.databaseMinorVersion = databaseMinorVersion;
088    }
089
090    public void setFile(File file) {
091        this.file = file;
092    }
093
094    public Path createClasspath() {
095        this.classPath = new Path(this.getProject());
096        return this.classPath;
097    }
098
099    public void setClasspath(Path classPath) {
100        this.classPath = classPath;
101    }
102
103    /**
104     * @throws BuildException
105     */
106    @Override
107    public void execute() {
108
109        // Sanity check
110        if (this.unit == null)
111            throw new BuildException("`unit' attribute is required specifying persistence unit name");
112        if (this.file == null)
113            throw new BuildException("`file' attribute is required specifying output file");
114
115        // Create directory containing file
116        if (this.file.getParent() != null && !this.file.getParentFile().exists() && !this.file.getParentFile().mkdirs())
117            throw new BuildException("error creating directory `" + this.file.getParentFile() + "'");
118
119        // Get URL of output file
120        final URL fileURL;
121        try {
122            fileURL = this.file.toURI().toURL();
123        } catch (MalformedURLException e) {
124            throw new BuildException("unexpected exception: " + e, e);
125        }
126
127        // Set up properties
128        final HashMap<String, Object> map = new HashMap<>();
129        map.put("javax.persistence.schema-generation.scripts.create-target", fileURL.toString());
130        map.put("javax.persistence.schema-generation.scripts.drop-target", fileURL.toString());
131        map.put("javax.persistence.schema-generation.scripts.create-source", "metadata");
132        map.put("javax.persistence.schema-generation.scripts.drop-source", "metadata");
133        map.put("javax.persistence.schema-generation.database.action", "none");
134        map.put("javax.persistence.schema-generation.scripts.action",
135          this.create && this.drop ? "drop-and-create" : this.create ? "create" : "drop");
136        map.put("javax.persistence.schema-generation.create-database-schemas", this.schemas);
137
138        // Set up mysterious classloader stuff
139        final AntClassLoader loader = this.getProject().createClassLoader(this.classPath);
140        final ClassLoader currentLoader = this.getClass().getClassLoader();
141        if (currentLoader != null)
142            loader.setParent(currentLoader);
143        loader.setThreadContextLoader();
144        try {
145
146            // Use connection to get database info, or use explicitly provided info
147            Connection con = null;
148            if (this.url != null && this.url.length() > 0) {
149                if (this.driver != null && this.driver.length() > 0) {
150                    try {
151                        Class.forName(this.driver);
152                    } catch (Exception e) {
153                        throw new BuildException("can't load database driver class `" + this.driver + "': " + e, e);
154                    }
155                }
156                try {
157                    con = DriverManager.getConnection(this.url, this.username, this.password);
158                } catch (Exception e) {
159                    throw new BuildException("can't database connection to " + this.url, e);
160                }
161                map.put("javax.persistence.schema-generation.connection", con);
162            } else {
163                if (databaseProductName == null) {
164                    throw new BuildException("must specify database type using \"databaseProductName\", \"databaseMajorVersion\","
165                      + " and \"databaseMinorVersion\" attributes, or using \"driver\", \"url\", \"username\", and \"password\""
166                      + " attributes");
167                }
168                map.put("javax.persistence.database-product-name", this.databaseProductName);
169                map.put("javax.persistence.database-major-version", this.databaseMajorVersion);
170                map.put("javax.persistence.database-minor-version", this.databaseMinorVersion);
171            }
172
173            // Debug
174            this.log("Properties for persistence unit `" + this.unit + "' JPA schema generation: " + map, Project.MSG_VERBOSE);
175
176            // Generate schema
177            try {
178                Persistence.generateSchema(this.unit, map);
179            } catch (PersistenceException e) {
180                throw new BuildException("schema generation for persistence unit `" + this.unit + "' failed: " + e, e);
181            } finally {
182                if (con != null) {
183                    try {
184                        con.close();
185                    } catch (SQLException e) {
186                        // ignore
187                    }
188                }
189            }
190        } finally {
191            loader.resetThreadContextLoader();
192            loader.cleanup();
193        }
194    }
195}

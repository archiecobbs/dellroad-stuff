006package org.dellroad.stuff.pobj;
007
008import java.util.ArrayList;
009import java.util.List;
010
011import javax.xml.stream.XMLStreamException;
012import javax.xml.stream.XMLStreamReader;
013
014import org.dellroad.stuff.xml.AnnotatedXMLStreamReader;
015
016/**
017 * {@link AnnotatedXMLStreamReader} that reads the nested schema update list. Used by {@link PersistentObjectSchemaUpdater}.
018 */
019public class UpdatesXMLStreamReader extends AnnotatedXMLStreamReader {
020
021    private ArrayList<String> updates;
022
023    public UpdatesXMLStreamReader(XMLStreamReader inner) {
024        super(inner);
025    }
026
027    /**
028     * Get the updates gleaned from the scan.
029     *
030     * @return list of updates, or null if the document didn't contain an updates element
031     */
032    public List<String> getUpdates() {
033        return this.updates;
034    }
035
036    @Override
037    protected boolean readAnnotationElement(XMLStreamReader reader) throws XMLStreamException {
038
039        // Check element name
040        if (reader.getEventType() != START_ELEMENT || !reader.getName().equals(PersistentObjectSchemaUpdater.UPDATES_ELEMENT_NAME))
041            return false;
042
043        // Read updates
044        this.updates = new ArrayList<String>();
045        while (true) {
046
047            // Ignore leading whitespace, comments, and PI's
048            int eventType = reader.next();
049            while (reader.isWhiteSpace() || eventType == COMMENT || eventType == PROCESSING_INSTRUCTION)
050                eventType = reader.next();
051
052            // Done with updates?
053            if (reader.isEndElement())
054                return true;
055
056            // Read <update>text</update>
057            reader.require(START_ELEMENT, PersistentObjectSchemaUpdater.UPDATE_ELEMENT_NAME.getNamespaceURI(),
058              PersistentObjectSchemaUpdater.UPDATE_ELEMENT_NAME.getLocalPart());
059            this.updates.add(reader.getElementText());
060        }
061    }
062}

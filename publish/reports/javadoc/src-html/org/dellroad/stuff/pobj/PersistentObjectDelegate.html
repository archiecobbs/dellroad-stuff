006package org.dellroad.stuff.pobj;
007
008import java.io.IOException;
009import java.util.Set;
010
011import javax.validation.ConstraintViolation;
012import javax.xml.transform.Result;
013import javax.xml.transform.Source;
014
015/**
016 * Delegate interface required for {@link PersistentObject}s.
017 * Instances provide methods for converting to/from XML, validation, etc.
018 *
019 * @param <T> type of the root persistent object
020 * @see PersistentObject
021 */
022public interface PersistentObjectDelegate<T> {
023
024    /**
025     * Serialize a root object graph into XML.
026     *
027     * <p>
028     * Note that this method effectively defines what is contained in the object graph
029     * rooted at {@code obj}.
030     *
031     * <p>
032     * This method must not modify {@code obj} or any other object in its object graph.
033     *
034     * @param obj object to serialize; must not be modified
035     * @param result XML destination
036     * @throws PersistentObjectException if an error occurs
037     */
038    void serialize(T obj, Result result) throws IOException;
039
040    /**
041     * Deserialize a root object graph from XML.
042     *
043     * <p>
044     * This method also presents an opportunity to (re)construct any indexes used for optimizing queries
045     * into the root object graph.
046     *
047     * @param source XML source
048     * @return deserialized object
049     * @throws PersistentObjectException if an error occurs
050     */
051    T deserialize(Source source) throws IOException;
052
053    /**
054     * Make a deep copy of the given object.
055     *
056     * <p>
057     * For correct behavior, this behavior of this method should be equivalent to a
058     * {@linkplain #serialize serialization} followed by a {@linkplain #deserialize deserialization}.
059     *
060     * <p>
061     * This method must not modify {@code original} or any other object in its object graph.
062     *
063     * @throws IllegalArgumentException if {@code original} is null
064     * @throws PersistentObjectException if an error occurs
065     */
066    T copy(T original);
067
068    /**
069     * Attempt to determine whether two object graphs are identical.
070     *
071     * <p>
072     * This optional method is an optimization to detect invocations to {@link PersistentObject#setRoot PersistentObject.setRoot()}
073     * where the new object graph and the old object graph are identical. In such cases, no change is applied,
074     * the version number does not increase, and no notifications are sent.
075     *
076     * <p>
077     * It is always safe and correct for this method to return false. If it returns true, then it must be the case
078     * that the two object graphs are identical.
079     *
080     * <p>
081     * This method must not modify {@code oldRoot} or {@code newRoot} or any other object in their object respective graphs.
082     *
083     * @param root1 root of first object graph
084     * @param root2 root of second object graph
085     * @throws IllegalArgumentException if {@code oldRoot} or {@code newRoot} is null
086     * @throws PersistentObjectException if an error occurs
087     */
088    boolean isSameGraph(T root1, T root2);
089
090    /**
091     * Validate the given object.
092     *
093     * <p>
094     * This method must not modify {@code obj} or any other object in its object graph.
095     *
096     * @throws IllegalArgumentException if {@code obj} is null
097     * @return set of zero or more constraint violations
098     */
099    Set<ConstraintViolation<T>> validate(T obj);
100
101    /**
102     * Handle an exception thrown during a delayed write-back attempt. {@link ThreadDeath} exceptions are not
103     * passed to this method, but all others are.
104     *
105     * @param pobj the instance that encountered the exception
106     * @param t the exception thrown
107     */
108    void handleWritebackException(PersistentObject<T> pobj, Throwable t);
109
110    /**
111     * Get the default value for the root object graph. This method is invoked at startup when the persistent file does not exist.
112     * If this method returns null, then an {@linkplain PersistentObject#isAllowEmptyStart empty start} occurs unless the
113     * {@link PersistentObject} object is configured to disallow them, in which case an exception is thrown.
114     *
115     * @return root object initial value, or null if there is no default value
116     */
117    T getDefaultValue();
118}

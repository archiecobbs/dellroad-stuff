006package org.dellroad.stuff.pobj;
007
008import java.util.List;
009
010import javax.xml.namespace.QName;
011import javax.xml.stream.XMLStreamException;
012import javax.xml.stream.XMLStreamWriter;
013
014import org.dellroad.stuff.xml.AnnotatedXMLStreamWriter;
015
016/**
017 * {@link AnnotatedXMLStreamWriter} that inserts the schema update list into the document
018 * using an {@link PersistentObjectSchemaUpdater#UPDATES_ELEMENT_NAME} annotation element.
019 * Used by {@link PersistentObjectSchemaUpdater}.
020 */
021public class UpdatesXMLStreamWriter extends AnnotatedXMLStreamWriter {
022
023    private final List<String> updates;
024
025    /**
026     * Constructor.
027     *
028     * @param inner nested output
029     * @param updates list of updates to add
030     * @throws IllegalArgumentException if {@code updates} is null
031     */
032    public UpdatesXMLStreamWriter(XMLStreamWriter inner, List<String> updates) {
033        super(inner);
034        if (updates == null)
035            throw new IllegalArgumentException("null updates");
036        this.updates = updates;
037    }
038
039    @Override
040    protected void addAnnotationElement(XMLStreamWriter writer) throws XMLStreamException {
041        final QName updatesTag = PersistentObjectSchemaUpdater.UPDATES_ELEMENT_NAME;
042        final QName updateTag = PersistentObjectSchemaUpdater.UPDATE_ELEMENT_NAME;
043        if (this.updates.isEmpty()) {
044            writer.writeEmptyElement(updatesTag.getPrefix(), updatesTag.getLocalPart(), updatesTag.getNamespaceURI());
045            writer.writeNamespace(updatesTag.getPrefix(), updatesTag.getNamespaceURI());
046        } else {
047            writer.writeStartElement(updatesTag.getPrefix(), updatesTag.getLocalPart(), updatesTag.getNamespaceURI());
048            writer.writeNamespace(updatesTag.getPrefix(), updatesTag.getNamespaceURI());
049            final String space = this.getTrailingSpace();
050            final String space2 = space.length() > 0 ? space + "    " : "";
051            for (String updateName : this.updates) {
052                writer.writeCharacters(space2);
053                writer.writeStartElement(updateTag.getPrefix(), updateTag.getLocalPart(), updateTag.getNamespaceURI());
054                writer.writeCharacters(updateName);
055                writer.writeEndElement();
056            }
057            writer.writeCharacters(space);
058            writer.writeEndElement();
059        }
060    }
061}

006package org.dellroad.stuff.pobj;
007
008import javax.xml.transform.TransformerFactory;
009
010import org.springframework.beans.factory.config.BeanPostProcessor;
011import org.springframework.context.ResourceLoaderAware;
012import org.springframework.core.io.DefaultResourceLoader;
013import org.springframework.core.io.ResourceLoader;
014
015/**
016 * Spring {@link BeanPostProcessor} that looks for {@link SpringXSLPersistentObjectSchemaUpdate} beans that don't have
017 * an explicit {@linkplain SpringXSLPersistentObjectSchemaUpdate#setTransform transform resource configured}, and configures
018 * them using a resource location based on the bean name, by simply adding a configured prefix and suffix.
019 *
020 * <p>
021 * In addition, this class will optionally configure the {@code transformerFactory} property of all of the
022 * {@link SpringXSLPersistentObjectSchemaUpdate} beans it finds that don't already have one explicitly configured.
023 *
024 * @see SpringXSLPersistentObjectSchemaUpdate
025 */
026public class SpringXSLUpdateTransformConfigurer implements BeanPostProcessor, ResourceLoaderAware {
027
028    /**
029     * Default location prefix: <code>{@value}</code>.
030     */
031    public static final String DEFAULT_LOCATION_PREFIX = "/";
032
033    /**
034     * Default location suffix: <code>{@value}</code>.
035     */
036    public static final String DEFAULT_LOCATION_SUFFIX = ".xsl";
037
038    private ResourceLoader resourceLoader = new DefaultResourceLoader();
039    private String prefix = DEFAULT_LOCATION_PREFIX;
040    private String suffix = DEFAULT_LOCATION_SUFFIX;
041    private TransformerFactory transformerFactory;
042
043    /**
044     * Set the location prefix.
045     */
046    public void setPrefix(String prefix) {
047        if (prefix == null)
048            prefix = "";
049        this.prefix = prefix;
050    }
051
052    /**
053     * Set the location suffix.
054     */
055    public void setSuffix(String suffix) {
056        if (suffix == null)
057            suffix = "";
058        this.suffix = suffix;
059    }
060
061    /**
062     * Set the {@link TransformerFactory} to configure all updates with.
063     *
064     * <p>
065     * This property is optional.
066     */
067    public void setTransformerFactory(TransformerFactory transformerFactory) {
068        this.transformerFactory = transformerFactory;
069    }
070
071    @Override
072    public void setResourceLoader(ResourceLoader resourceLoader) {
073        this.resourceLoader = resourceLoader;
074    }
075
076    @Override
077    public Object postProcessBeforeInitialization(Object bean, String beanName) {
078        if (bean instanceof SpringXSLPersistentObjectSchemaUpdate) {
079            SpringXSLPersistentObjectSchemaUpdate<?> update = (SpringXSLPersistentObjectSchemaUpdate<?>)bean;
080            if (update.getTransform() == null)
081                update.setTransform(this.resourceLoader.getResource(this.getImpliedTransformResourceLocation(beanName)));
082            if (update.getTransformerFactory() == null && this.transformerFactory != null)
083                update.setTransformerFactory(this.transformerFactory);
084        }
085        return bean;
086    }
087
088    @Override
089    public Object postProcessAfterInitialization(Object bean, String beanName) {
090        return bean;
091    }
092
093    /**
094     * Derive the implied transform resource location for the update with the given bean name.
095     *
096     * <p>
097     * The implementation in {@link SpringXSLUpdateTransformConfigurer} simply prepends the configured
098     * prefix and appends the configured suffix to {@code beanName}.
099     */
100    protected String getImpliedTransformResourceLocation(String beanName) {
101        return this.prefix + beanName + this.suffix;
102    }
103}

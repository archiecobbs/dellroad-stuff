027/**
028 * {@link PersistentObject} implementation of Spring's
029 * {@link org.springframework.transaction.PlatformTransactionManager PlatformTransactionManager} interface,
030 * allowing methods annotated with Spring's {@link org.springframework.transaction.annotation.Transactional &#64;Transactional}
031 * annotation to perform transactions on {@link PersistentObject}s.
032 *
033 * <p>
034 * During a transaction, the transaction's root object is available via the method {@link #getRoot}, and may be replaced entirely
035 * via {@link #setRoot setRoot()}. When the transaction completes, the transaction's root object will be automatically
036 * written back to the {@link PersistentObject} via {@link PersistentObject#setRoot(Object, long) PersistentObject.setRoot()}
037 * (unless the transaction was read-only). During commit, the {@link PersistentObject} version number is verified, and
038 * if another update has occurred since the transaction was opened, a {@link PersistentObjectVersionException} is thrown
039 * (consider using {@link org.dellroad.stuff.spring.RetryTransaction &#64;RetryTransaction} for automatic retry in this case).
040 *
041 * <p>
042 * Transactions are associated with threads; the {@link PersistentObjectTransactionManager} associated with the current thread's
043 * transaction is available via {@link #getCurrent()} (or {@link #getCurrent(String)} if there is more than one).
044 *
045 * <p>
046 * The {@code persistentObject} property is required. The {@code readOnlySharedRoot} property is optional and configures whether,
047 * during read-only transactions only, {@link #getRoot} returns the {@linkplain PersistentObject#getSharedRoot shared root} object.
048 * In this mode, root object graph copies are avoided entirely for read-only transactions, but the application code must be
049 * careful to not modify the object graph returned by {@link #getRoot} at any time, either during or after the transaction.
050 *
051 * <p>
052 * This class also provides an {@link XAResource} for use with JTA/XA tranasactions. Once elisted in a JTA transaction,
053 * the usage described above using {@link #getRoot} and {@link #setRoot setRoot()} is identical.
054 *
055 * @param <T> type of the root object
056 * @see PersistentObject
057 * @see PersistentObject#getSharedRoot
058 */
059@SuppressWarnings("serial")
060public class PersistentObjectTransactionManager<T> extends AbstractPlatformTransactionManager
061  implements BeanNameAware, ResourceTransactionManager, InitializingBean {
062

160    /**
161     * Get the instance associated with an open transaction in the current thread and having the given bean name.
162     *
163     * @param <T> root object type
164     * @param beanName bean name assigned to the desired instance
165     * @return current thread's transaction manager with the given name
166     * @throws IllegalArgumentException if {@code name} is null
167     * @throws IllegalStateException if the current thread is not running within a transaction managed by
168     *  a {@link PersistentObjectTransactionManager} assigned the given name
169     */
170    @SuppressWarnings("unchecked")
171    public static <T> PersistentObjectTransactionManager<T> getCurrent(String beanName) {
172        if (beanName == null)
173            throw new IllegalArgumentException("null beanName");
174        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
175          = PersistentObjectTransactionManager.getManagerMap(false);
176        final PersistentObjectTransactionManager<?> manager = managerMap != null ? managerMap.get(beanName) : null;
177        if (manager == null) {
178            throw new IllegalStateException("no PersistentObjectTransactionManager named `" + beanName
179              + "' has an open transaction in the current thread");
180        }
181        if (manager.logger.isTraceEnabled())
182            manager.logger.trace("PersistentObjectTransactionManager.getCurrent(\"" + beanName + "\"): returning " + manager);
183        return (PersistentObjectTransactionManager<T>)manager;
184    }
185
186    /**
187     * Get the (unique) instance associated with an open transaction in the current thread.
188     *
189     * <p>
190     * This is a convenience method for the common case where there is only one instance associated with the current thread.
191     *
192     * @param <T> root object type
193     * @return current thread's transaction manager
194     * @throws IllegalStateException if the current thread is not running within a
195     *  {@link PersistentObjectTransactionManager} transaction
196     * @throws IllegalStateException there is more than one {@link PersistentObjectTransactionManager} transaction
197     *  open in the current thread
198     */
199    @SuppressWarnings("unchecked")
200    public static <T> PersistentObjectTransactionManager<T> getCurrent() {
201        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
202          = PersistentObjectTransactionManager.getManagerMap(false);
203        if (managerMap == null || managerMap.isEmpty()) {
204            throw new IllegalStateException("there are no PersistentObjectTransactionManager transactions open"
205              + " in the current thread");
206        }
207        if (managerMap.size() > 1) {
208            throw new IllegalStateException("there are multiple PersistentObjectTransactionManager transactions open"
209              + " in the current thread: " + managerMap.keySet() + "; invoke getCurrent() with explicitly specified bean name");
210        }
211        final PersistentObjectTransactionManager<T> manager
212          = (PersistentObjectTransactionManager<T>)managerMap.values().iterator().next();
213        if (manager.logger.isTraceEnabled())
214            manager.logger.trace("PersistentObjectTransactionManager.getCurrent(): returning " + manager);
215        return manager;
216    }
217
218    private static HashMap<String, PersistentObjectTransactionManager<?>> getManagerMap(boolean create) {
219        HashMap<String, PersistentObjectTransactionManager<?>> managerMap = MANAGER_MAP.get();
220        if (managerMap == null && create) {
221            managerMap = new HashMap<String, PersistentObjectTransactionManager<?>>();
222            MANAGER_MAP.set(managerMap);
223        }
224        return managerMap;
225    }

229    /**
230     * Determine whether the transaction associated with the current thread is a read-only transaction.

232     * @return true if the transaction associated with the current thread is a read-only transaction

255    /**

261    public T getRoot() {
262        final TxInfo<T> info = this.currentInfo.get();

268    /**
269     * Change the root object graph to be committed in the transaction associated with the current thread.
270     * Subsequent calls to {@link #getRoot getRoot} will return the new object.
271     *
272     * <p>
273     * This method may be invoked during read-only transactions (as before, the root will not actually be committed).
274     *
275     * @param root new root object
276     * @throws IllegalStateException if there is no transaction associated with the current thread
277     * @throws IllegalArgumentException if {@code root} is null
278     */
279    public void setRoot(T root) {
280        if (root == null)
281            throw new IllegalArgumentException("null root");
282        final TxInfo<T> info = this.currentInfo.get();
283        if (info == null)
284            throw new IllegalStateException("there is no transaction associated with the current thread");
285        info.setSnapshot(this.persistentObject.new Snapshot(root, info.getSnapshot().getVersion()));
286    }

311    @Override
312    @SuppressWarnings("unchecked")
313    protected void doBegin(Object txObj, TransactionDefinition txDef) {
314
315        // Logging
316        if (this.logger.isTraceEnabled())
317            this.logger.trace("PersistentObjectTransactionManager.doBegin(): txObj=" + txObj + " txDef=" + txDef);
318
319        // Sanity check
320        if (this.currentInfo.get() != null)
321            throw new TransactionUsageException("there is already a transaction associated with the current thread");
322        if (this.isExistingTransaction(txObj))
323            throw new TransactionUsageException("there is already a transaction associated with the given transaction object");
324        final TxWrapper<T> tx = (TxWrapper<T>)txObj;
325
326        // Create transaction
327        final PersistentObject<T>.Snapshot snapshot;
328        try {
329            snapshot = txDef.isReadOnly() && this.readOnlySharedRoot ?
330              this.persistentObject.getSharedRootSnapshot() : this.persistentObject.getRootSnapshot();
331        } catch (PersistentObjectException e) {
332            throw new CannotCreateTransactionException("error creating new JSimpleDB transaction", e);
333        }
334
335        // Associate transaction with this thread for getCurrent()
336        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
337          = PersistentObjectTransactionManager.getManagerMap(true);
338        if (managerMap.containsKey(this.beanName)) {
339            throw new IllegalStateException("A PersistentObjectTransactionManager named `" + this.beanName
340              + "' already has an open transaction in the current thread; all bean names must be distinct");
341        }
342        managerMap.put(this.beanName, this);
343
344        // Associate new transaction with the current thread
345        final TxInfo<T> info = new TxInfo<T>(snapshot, txDef.isReadOnly());
346        this.currentInfo.set(info);
347
348        // Set transaction info into the Spring transaction object
349        tx.setInfo(info);
350    }

352    /**
353     * Suspend the current transaction.
354     */
355    @Override
356    @SuppressWarnings("unchecked")
357    protected Object doSuspend(Object txObj) {
358
359        // Logging
360        if (this.logger.isTraceEnabled())
361            this.logger.trace("PersistentObjectTransactionManager.doSuspend(): txObj=" + txObj);
362
363        // Sanity check
364        final TxWrapper<T> tx = (TxWrapper<T>)txObj;
365        final TxInfo<T> info = tx.getInfo();
366        if (info == null)
367            throw new TransactionUsageException("no PersistentObject transaction exists in the given transaction object");
368        if (info != this.currentInfo.get())
369            throw new TransactionUsageException("the provided transaction object contains the wrong PersistentObject transaction");
370
371        // Suspend it
372        if (this.logger.isTraceEnabled())
373            this.logger.trace("PersistentObjectTransactionManager.doSuspend(): suspending " + info.getSnapshot());
374        this.currentInfo.remove();
375        tx.setInfo(null);
376        PersistentObjectTransactionManager.getManagerMap(true).remove(this.beanName);
377
378        // Done
379        return info;
380    }

382    /**
383     * Resume a previously suspended transaction.
384     */
385    @Override
386    @SuppressWarnings("unchecked")
387    protected void doResume(Object txObj, Object suspendedResources) {
388
389        // Logging
390        if (this.logger.isTraceEnabled())
391            this.logger.trace("PersistentObjectTransactionManager.doResume(): txObj=" + txObj + " res=" + suspendedResources);
392
393        // Sanity check
394        if (this.currentInfo.get() != null)
395            throw new TransactionUsageException("there is already a transaction associated with the current thread");
396        if (this.isExistingTransaction(txObj))
397            throw new TransactionUsageException("there is already a transaction associated with the given transaction object");
398
399        // Get previously saved info
400        final TxWrapper<T> tx = (TxWrapper<T>)txObj;
401        final TxInfo<T> info = (TxInfo<T>)suspendedResources;
402
403        // Resume transaction
404        if (this.logger.isTraceEnabled())
405            this.logger.trace("PersistentObjectTransactionManager.doSuspend(): resuming " + info.getSnapshot());
406        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
407          = PersistentObjectTransactionManager.getManagerMap(true);
408        if (managerMap.containsKey(this.beanName)) {
409            throw new IllegalStateException("A PersistentObjectTransactionManager named `" + this.beanName
410              + "' already has an open transaction in the current thread; all bean names must be distinct");
411        }
412        managerMap.put(this.beanName, this);
413        this.currentInfo.set(info);
414        tx.setInfo(info);
415    }

417    @Override
418    @SuppressWarnings("unchecked")
419    protected void prepareForCommit(DefaultTransactionStatus status) {
420        final TxInfo<T> info = ((TxWrapper<T>)status.getTransaction()).getInfo();
421        if (this.logger.isTraceEnabled())
422            this.logger.trace("PersistentObjectTransactionManager.prepareForCommit(): status=" + status + " info=" + info);
423        if (info == null)
424            throw new NoTransactionException("no current transaction exists");
425    }
426
427    @Override
428    @SuppressWarnings("unchecked")
429    protected void doCommit(DefaultTransactionStatus status) {
430
431        // Logging
432        if (this.logger.isTraceEnabled())
433            this.logger.trace("PersistentObjectTransactionManager.doCommit(): status=" + status);
434
435        // Get transaction
436        final TxWrapper<T> tx = (TxWrapper<T>)status.getTransaction();
437        final TxInfo<T> info = tx.getInfo();
438        if (info == null)
439            throw new NoTransactionException("no current transaction exists");
440        final PersistentObject<T>.Snapshot snapshot = info.getSnapshot();
441
442        // Commit
443        try {
444            if (!info.isReadOnly()) {
445                if (this.logger.isTraceEnabled())
446                    this.logger.trace("PersistentObjectTransactionManager.doCommit(): committing " + info);
447                this.persistentObject.setRoot(snapshot.getRoot(), snapshot.getVersion());
448            } else
449                this.logger.trace("PersistentObjectTransactionManager.doCommit(): not committing read-only " + info);
450        } catch (PersistentObjectVersionException e) {
451            throw new OptimisticLockingFailureException(null, e);
452        } catch (PersistentObjectValidationException e) {
453            throw new DataIntegrityViolationException(null, e);
454        } catch (PersistentObjectException e) {
455            throw new TransactionSystemException("error committing transaction", e);
456        }
457    }

461    protected void doRollback(DefaultTransactionStatus status) {
462
463        // Logging
464        if (this.logger.isTraceEnabled())
465            this.logger.trace("PersistentObjectTransactionManager.doRollback(): status=" + status);
466
467        // Get transaction
468        final TxWrapper<T> tx = (TxWrapper<T>)status.getTransaction();
469        final TxInfo<T> info = tx.getInfo();
470        if (info == null)
471            throw new NoTransactionException("no current transaction exists");
472        final PersistentObject<T>.Snapshot snapshot = info.getSnapshot();
473
474        // Rollback
475        if (this.logger.isTraceEnabled())
476            this.logger.trace("PersistentObjectTransactionManager.doRollback(): rolling back " + info);
477        // no action required to rollback a PersistentObject transaction
478    }

500    @Override
501    @SuppressWarnings("unchecked")
502    protected void doCleanupAfterCompletion(Object txObj) {
503
504        // Logging
505        if (this.logger.isTraceEnabled())
506            this.logger.trace("PersistentObjectTransactionManager.doCleanupAfterCompletion(): txObj=" + txObj);
507
508        // Get transaction
509        final TxWrapper<T> tx = (TxWrapper<T>)txObj;
510
511        // Clean up
512        tx.setInfo(null);
513        this.currentInfo.remove();
514        final HashMap<String, PersistentObjectTransactionManager<?>> managerMap
515          = PersistentObjectTransactionManager.getManagerMap(false);
516        if (managerMap != null) {
517            managerMap.remove(this.beanName);
518            if (managerMap.isEmpty())
519                MANAGER_MAP.remove();
520        }
521    }

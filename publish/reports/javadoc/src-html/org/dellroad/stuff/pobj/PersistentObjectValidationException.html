006package org.dellroad.stuff.pobj;
007
008import java.util.Set;
009
010import javax.validation.ConstraintViolation;
011
012/**
013 * Runtime exception thrown during {@link PersistentObject} operations.
014 */
015@SuppressWarnings("serial")
016public class PersistentObjectValidationException extends PersistentObjectException {
017
018    private final Set<ConstraintViolation<?>> violations;
019
020    /**
021     * Constructor.
022     *
023     * @param violations set of violations
024     * @throws IllegalArgumentException if {@code violations} is null
025     */
026    @SuppressWarnings("unchecked")
027    public <T> PersistentObjectValidationException(Set<ConstraintViolation<T>> violations) {
028        super(PersistentObjectValidationException.generateMessage(violations));
029        this.violations = (Set<ConstraintViolation<?>>)(Object)violations;
030    }
031
032    /**
033     * Get the set of constraint violations.
034     */
035    public Set<ConstraintViolation<?>> getViolations() {
036        return this.violations;
037    }
038
039    private static <T> String generateMessage(Set<ConstraintViolation<T>> violations) {
040        if (violations == null)
041            throw new IllegalArgumentException("null violations");
042        StringBuilder buf = new StringBuilder("object failed to validate with " + violations.size() + " violation(s):\n");
043        for (ConstraintViolation<T> violation : violations) {
044            buf.append("    [")
045              .append(violation.getPropertyPath())
046              .append("]: ")
047              .append(violation.getMessage())
048              .append('\n');
049        }
050        return buf.toString();
051    }
052}

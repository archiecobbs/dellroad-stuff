006package org.dellroad.stuff.pobj;
007
008import java.io.IOException;
009import java.io.InputStream;
010import java.util.Properties;
011
012import javax.xml.transform.OutputKeys;
013import javax.xml.transform.Transformer;
014import javax.xml.transform.TransformerException;
015import javax.xml.transform.TransformerFactory;
016import javax.xml.transform.stream.StreamSource;
017
018import org.dellroad.stuff.xml.TransformErrorListener;
019import org.slf4j.LoggerFactory;
020import org.springframework.core.io.Resource;
021
022/**
023 * {@link SpringPersistentObjectSchemaUpdate} that applies a configured XSL transform to the XML form of the persistent object.
024 *
025 * <p>
026 * The {@link #setTransform transform} property is required.
027 *
028 * <p>
029 * See {@link SpringPersistentObjectSchemaUpdater} for a Spring XML configuration example.
030 *
031 * @param <T> type of the persistent object
032 */
033public class SpringXSLPersistentObjectSchemaUpdate<T> extends SpringPersistentObjectSchemaUpdate<T> {
034
035    private Resource transform;
036    private Properties parameters;
037    private TransformerFactory transformerFactory;
038
039    @Override
040    public void afterPropertiesSet() throws Exception {
041        super.afterPropertiesSet();
042        if (this.transform == null)
043            throw new Exception("no transform configured");
044    }
045
046    /**
047     * Configure the XSLT transform as a resource.
048     *
049     * @param transform transform for this schema update
050     */
051    public void setTransform(Resource transform) {
052        this.transform = transform;
053    }
054
055    public Resource getTransform() {
056        return this.transform;
057    }
058
059    /**
060     * Configure XSLT parameters. This is an optional property.
061     *
062     * @param parameters transform parameters
063     */
064    public void setParameters(Properties parameters) {
065        this.parameters = parameters;
066    }
067
068    /**
069     * Get the configured {@link TransformerFactory} that will be used to create the {@link Transformer}
070     * that will be used to actually apply the {@linkplain #setTransform configured XSL transform}.
071     *
072     * <p>
073     * This property is optional; if not specified, {@link TransformerFactory#newInstance} is used.
074     *
075     * @return custom factory for XSL transformers, or null if none configured
076     */
077    public TransformerFactory getTransformerFactory() {
078        return this.transformerFactory;
079    }
080
081    /**
082     * Set the {@link TransformerFactory} to use.
083     *
084     * <p>
085     * This property is optional; if not specified, {@link TransformerFactory#newInstance} is used.
086     *
087     * @param transformerFactory custom factory for XSL transformers, or null for none
088     * @see #getTransformerFactory
089     */
090    public void setTransformerFactory(TransformerFactory transformerFactory) {
091        this.transformerFactory = transformerFactory;
092    }
093
094    /**
095     * Apply this update to the given transaction.
096     */
097    @Override
098    public void apply(PersistentFileTransaction transaction) {
099
100        // Get transform source
101        InputStream input;
102        try {
103            input = this.transform.getInputStream();
104        } catch (IOException e) {
105            throw new RuntimeException(e);
106        }
107        try {
108
109            // Setup transformer
110            TransformerFactory tf = this.transformerFactory;
111            if (tf == null)
112                tf = TransformerFactory.newInstance();
113            Transformer transformer = tf.newTransformer(new StreamSource(input, this.transform.getURI().toString()));
114            transformer.setErrorListener(new TransformErrorListener(LoggerFactory.getLogger(this.getClass()), true));
115            transformer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
116            if (this.parameters != null) {
117                for (String name : this.parameters.stringPropertyNames())
118                    transformer.setParameter(name, this.parameters.getProperty(name));
119            }
120
121            // Do the transform
122            transaction.transform(transformer);
123
124        } catch (IOException e) {
125            throw new RuntimeException(e);
126        } catch (TransformerException e) {
127            throw new RuntimeException(e);
128        } catch (RuntimeException e) {
129            if (e.getCause() instanceof TransformerException && e.getCause().getCause() instanceof RuntimeException)
130                e = (RuntimeException)e.getCause().getCause();
131            throw e;
132        } finally {
133            try {
134                input.close();
135            } catch (IOException e) {
136                // ignore
137            }
138        }
139    }
140}

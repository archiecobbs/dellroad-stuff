006package org.dellroad.stuff.pobj;
007
008/**
009 * Listener interface for changes to a {@link PersistentObject}.
010 *
011 * <p>
012 * Listener notifications are delivered in a separate thread from the one that caused the root object to change.
013 *
014 * @param <T> type of the root persistent object
015 */
016public interface PersistentObjectListener<T> {
017
018    /**
019     * Handle notification of an updated root object.
020     */
021    void handleEvent(PersistentObjectEvent<T> event);
022}

022/**
023 * Represents a {@code git(1)} repository and provides a few simple methods that wrap the command-line client
024 * and provide the functionality needed by {@link Synchronizer}.
025 * Only normal (i.e., non-bare) repositories are supported.
026 *
027 * <p>
028 * This class assumes the repository already has all branches and remotes already configured, and that no
029 * other external process is accessing the {@code git(1)} repository without our knowledge.
030 *
031 * <p>
032 * Instances are thread safe and guarantees exclusive access to the working directory
033 * to each {@link Accessor} while it is in use.
034 *
035 * <p>
036 * Requires git version at least 1.7.4.
037 *
038 * @see <a href="http://git-scm.com/">Git web site</a>
039 */
040public class GitRepository extends AbstractBean {
041
042    /**
043     * Accepted git branch name pattern.
044     */
045    public static final Pattern BRANCH_NAME_PATTERN = Pattern.compile("[\\w][-:\\w]*");
046
047    /**
048     * Accepted git remote name pattern.
049     */
050    public static final Pattern REMOTE_NAME_PATTERN = Pattern.compile("[\\w][-:\\w]*");
051
052    /**
053     * SHA1 pattern.
054     */
055    public static final Pattern SHA1_PATTERN = Pattern.compile("[0-9a-f]{40}");
056
057    private final File dir;
058
059// Constructors
060
061    /**
062     * Constructor.
063     *
064     * @param dir repository filesystem location (i.e., the directory containing {@code .git} as a subdirectory)
065     * @throws IllegalArgumentException if {@code dir} is null
066     */
067    public GitRepository(File dir) {
068        if (dir == null)
069            throw new IllegalArgumentException("null dir");
070        this.dir = dir;
071    }
072
073// Lifecycle
074
075    @Override
076    public void afterPropertiesSet() throws Exception {
077        super.afterPropertiesSet();
078        final File gitDir = new File(this.dir, ".git");
079        if (!gitDir.exists() || !gitDir.isDirectory())
080            throw new Exception("`" + this.dir + "' is not a git repository");
081    }

083// Methods
084
085    /**
086     * Get the git working directory associated with this instance.
087     *
088     * @return repository working directory location
089     */
090    public File getDirectory() {
091        return this.dir;
092    }

094    /**
095     * Fetch data from all configured remotes to update our local copy of their information.
096     *
097     * @throws GitException if an error occurs
098     */
099    public synchronized void fetch() {
100        new GitCommand(this.dir, "fetch", "--all").run();
101    }
102
103    /**
104     * Fetch data from the named remotes to update our local copy of their information.
105     *
106     * @param remotes list of remotes
107     * @throws GitException if an error occurs
108     * @throws IllegalArgumentException if {@code remotes} or any element is null
109     * @throws IllegalArgumentException if an remote does not match {@link #REMOTE_NAME_PATTERN}
110     */
111    public synchronized void fetch(List<String> remotes) {
112
113        // Sanity check
114        if (remotes == null)
115            throw new IllegalArgumentException("null remotes");
116        for (String remote : remotes) {
117            if (remote == null)
118                throw new IllegalArgumentException("null remote");
119            if (!REMOTE_NAME_PATTERN.matcher(remote).matches())
120                throw new IllegalArgumentException("illegal remote name `" + remote + "'");
121        }
122
123        // Fetch
124        final ArrayList<String> params = new ArrayList<String>(remotes.size() + 2);
125        params.add("fetch");
126        params.add("--multiple");
127        params.addAll(remotes);
128        new GitCommand(this.dir, params).run();
129    }
130
131    /**
132     * Get the author date associated with the given commit.
133     *
134     * @param tree commit reference
135     * @return author date
136     * @throws GitException if git fails
137     * @throws IllegalArgumentException if {@code tree} is null
138     */
139    public synchronized Date getAuthorDate(String tree) {
140        try {
141            return new Date(Long.parseLong(new GitCommand(this.dir,
142              "log", "-1", "--format=format:%at", tree).runAndGetOutput(), 10) * 1000L);
143        } catch (NumberFormatException e) {
144            throw new GitException("error parsing git output", e);
145        }
146    }
147
148    /**
149     * Access the head of the specified branch or the named commit.
150     * Any changes made by the {@link Accessor} are discarded when this method completes.
151     *
152     * @param branch name of the branch or commit we want to access
153     * @param accessor callback interface to access the working directory contents
154     * @return commit name (SHA-1 hash) corresponding to what tree was accessed
155     * @throws GitException if {@code branch} is not a local branch or a commit ID
156     * @throws IllegalArgumentException if {@code branch} is null
157     * @throws IllegalArgumentException if {@code branch} does not match {@link #BRANCH_NAME_PATTERN}
158     * @throws IllegalArgumentException if {@code branch} is null
159     * @throws IllegalArgumentException if {@code accessor} is null
160     */
161    public synchronized String access(String branch, Accessor accessor) {
162
163        // Sanity check
164        if (branch == null)
165            throw new IllegalArgumentException("null branch");
166        if (!BRANCH_NAME_PATTERN.matcher(branch).matches())
167            throw new IllegalArgumentException("illegal branch name `" + branch + "'");
168        if (accessor == null)
169            throw new IllegalArgumentException("null accessor");
170
171        // Reset
172        this.reset();
173
174        // Checkout the commit
175        new GitCommand(this.dir, "checkout", "--force", branch).run();
176
177        // Get the commit ID that we have checked out
178        final String commit = this.getCurrentCommit();
179
180        // Grant access
181        try {
182            accessor.accessWorkingCopy(this.dir);
183        } finally {
184            this.reset();
185        }
186
187        // Done
188        return commit;
189    }
190
191    /**
192     * Commit changes onto the specified branch. The branch must already exist.
193     *
194     * <p>
195     * If after the {@code accessor} runs nothing has changed, then no commit is performed
196     * and the previous commit ID is returned.
197     *
198     * @param branch name of the branch we want to commit onto
199     * @param accessor callback interface to update the working directory contents
200     * @param message commit message
201     * @return commit name (SHA-1 hash), or previous existing commit if no changes were made
202     * @throws GitException if {@code branch} is not an existing local branch
203     * @throws IllegalArgumentException if {@code branch} is null
204     * @throws IllegalArgumentException if {@code branch} does not match {@link #BRANCH_NAME_PATTERN}
205     * @throws IllegalArgumentException if {@code accessor} is null
206     * @throws IllegalArgumentException if {@code message} is null
207     */
208    public synchronized String commit(String branch, Accessor accessor, String message) {
209
210        // Sanity check
211        if (branch == null)
212            throw new IllegalArgumentException("null branch");
213        if (!BRANCH_NAME_PATTERN.matcher(branch).matches())
214            throw new IllegalArgumentException("illegal branch name `" + branch + "'");
215        if (accessor == null)
216            throw new IllegalArgumentException("null accessor");
217        if (message == null)
218            throw new IllegalArgumentException("null message");
219
220        // Get ref file and verify branch is really a local branch
221        final File refFile = this.getRepoFile("refs/heads/" + branch);
222        if (!refFile.exists())
223            throw new GitException("branch `" + branch + "' does not exist or is not a local branch");
224
225        // Reset
226        this.reset();
227
228        // Perform commit steps
229        try {
230
231            // Check out branch
232            new GitCommand(this.dir, "checkout", "--force", branch).run();
233
234            // Apply changes
235            accessor.accessWorkingCopy(this.dir);
236
237            // Attempt commit only if something has changed
238            if (new GitCommand(this.dir, "status", "--porcelain").runAndGetOutput().length() > 0) {
239
240                // Stage them into the index
241                new GitCommand(this.dir, "add", "--all").run();
242
243                // Commit them
244                new GitCommand(this.dir, "commit", "--message", message).run();
245            }
246        } finally {
247            this.reset();
248        }
249
250        // Done
251        return this.getCurrentCommit();
252    }

254    /**
255     * Ensure the specified local branch exists. If it does not, it will be created with an initial empty commit.
256     *
257     * @param branch name of the branch we want to ensure exists
258     * @param message commit message (only used if branch does not yet exist)
259     * @throws IllegalArgumentException if {@code branch} is null
260     * @throws IllegalArgumentException if {@code branch} does not match {@link #BRANCH_NAME_PATTERN}
261     * @throws IllegalArgumentException if {@code message} is null
262     */
263    public synchronized void ensureBranch(String branch, String message) {
264
265        // Sanity check
266        if (branch == null)
267            throw new IllegalArgumentException("null branch");
268        if (!BRANCH_NAME_PATTERN.matcher(branch).matches())
269            throw new IllegalArgumentException("illegal branch name `" + branch + "'");
270        if (message == null)
271            throw new IllegalArgumentException("null message");
272
273        // Does branch already exist?
274        final File refFile = this.getRepoFile("refs/heads/" + branch);
275        if (refFile.exists())
276            return;
277
278        // Create a commit with no parents and an empty tree
279        this.log.info("creating new local branch `" + branch + "' starting with an empty commit in directory `" + this.dir + "'");
280        new GitCommand(this.dir, "read-tree", "--empty").run();
281        final String tree = new GitCommand(this.dir, "write-tree").runAndGetOutput();
282        if (!SHA1_PATTERN.matcher(tree).matches())
283            throw new GitException("can't interpret output from `git write-tree': " + tree);
284        final String commit = new GitCommand(this.dir, "commit-tree", "-m", message, tree).runAndGetOutput();
285
286        // Create a branch starting there
287        new GitCommand(this.dir, "branch", branch, commit).run();
288    }

290    /**
291     * Compare two committed trees for equality.
292     *
293     * @param tree1 name of the first tree (branch name, commit, etc.)
294     * @param tree2 name of the second tree (branch name, commit, etc.)
295     * @return true if the trees are equal, otherwise false
296     * @throws GitException if {@code tree1} or {@code tree2} is not a valid branch or commit reference
297     * @throws IllegalArgumentException if {@code tree1} or {@code tree2} is null
298     */
299    public synchronized boolean equalTrees(String tree1, String tree2) {
300        final GitCommand diff = new GitCommand(this.dir, "diff", "--quiet", tree1, tree2);
301        final int exitValue = diff.run(true);
302        switch (exitValue) {
303        case 0:
304            return true;
305        case 1:
306            return false;
307        default:
308            throw new GitException("command `" + diff + "' in directory `" + this.dir + "' failed with exit value " + exitValue);
309        }
310    }

312    /**
313     * Merge and commit.
314     *
315     * <p>
316     * Trivial merges, i.e., when {@code other} is already an ancestor of {@code branch}, do not result in a new commit
317     * and the {@code accessor} is not run.
318     *
319     * @param branch name of the branch we want to commit onto
320     * @param other name of the other commit or branch we want to merge into {@code branch}
321     * @param strategy merge strategy
322     * @param accessor callback interface to access the working directory contents after the merge; may be null
323     * @param message commit message
324     * @return commit name (SHA-1 hash)
325     * @throws GitException if {@code branch} is not a local branch
326     * @throws GitException if {@code other} is not a valid branch or commit reference
327     * @throws GitMergeConflictException if the merge fails with conflicts
328     * @throws IllegalArgumentException if {@code branch} or {@code other} is null
329     * @throws IllegalArgumentException if {@code branch} does not match {@link #BRANCH_NAME_PATTERN}
330     * @throws IllegalArgumentException if {@code message} is null
331     */
332    public synchronized String merge(String branch, String other, MergeStrategy strategy, Accessor accessor, String message) {
333
334        // Sanity check
335        if (branch == null)
336            throw new IllegalArgumentException("null branch");
337        if (!BRANCH_NAME_PATTERN.matcher(branch).matches())
338            throw new IllegalArgumentException("illegal branch name `" + branch + "'");
339        if (other == null)
340            throw new IllegalArgumentException("null other");
341        if (strategy == null)
342            throw new IllegalArgumentException("null strategy");
343        if (message == null)
344            throw new IllegalArgumentException("null message");
345
346        // Get ref file and verify branch is really a local branch
347        final File refFile = this.getRepoFile("refs/heads/" + branch);
348        if (!refFile.exists())
349            throw new GitException("branch `" + branch + "' does not exist or is not a local branch");
350
351        // Reset
352        this.reset();
353
354        // Perform merge steps
355        try {
356
357            // Check out branch
358            new GitCommand(this.dir, "checkout", "--force", branch).run();
359
360            // Perform the merge
361            strategy.merge(this.dir, other);
362
363            // Detect and ignore trivial merge
364            if (new GitCommand(this.dir, "status", "--porcelain").runAndGetOutput().length() > 0) {
365
366                // Grant access
367                if (accessor != null)
368                    accessor.accessWorkingCopy(this.dir);
369
370                // Commit changes
371                new GitCommand(this.dir, "commit", "--message", message).run();
372            }
373        } finally {
374
375            // Clean up
376            this.reset();
377        }
378
379        // Done
380        return this.getCurrentCommit();
381    }

383    /**
384     * Read file possibly containing a reference and follow it until we get a commit ID.
385     *
386     * @param filename filename relative to {@code .git}, e.g., {@code "HEAD"} or {@code "refs/heads/foobar"}
387     * @return SHA-1 commit ID
388     * @throws GitException if we can't figure it out
389     */
390    public String followReference(String filename) {
391        for (int i = 0; i < 10; i++) {
392
393            // Read file
394            final File file = this.getRepoFile(filename);
395            String line;
396            try {
397                line = this.readFirstLine(file);
398            } catch (IOException e) {
399                throw new GitException("error reading `" + file + "'", e);
400            }
401
402            // Is it a SHA-1?
403            if (SHA1_PATTERN.matcher(line).matches())
404                return line;
405
406            // Is it a reference?
407            final Matcher matcher = Pattern.compile("ref: (.*)$").matcher(line);
408            if (matcher.matches()) {
409                filename = matcher.group(1);
410                continue;
411            }
412
413            // Dunno
414            throw new GitException("can't interpret contents of `" + file + "': " + line);
415        }
416        throw new GitException("too many levels of git references");
417    }
418
419    /**
420     * Clean out and reset working directory.
421     */
422    protected void reset() {
423        new GitCommand(this.dir, "clean", "-xdf").run();                      // clean out working directory
424        new GitCommand(this.dir, "reset", "--hard").run();                    // discard uncommitted changes
425    }
426
427    /**
428     * Get the commit ID of what we have currently checked out (i.e., our HEAD).
429     *
430     * @return HEAD commit ID
431     * @throws GitException if we can't figure it out
432     */
433    protected String getCurrentCommit() {
434        return this.followReference("HEAD");
435    }

437    /**
438     * Read the first line of a file (trimmed) as a {@link String}.
439     *
440     * @param file file to read
441     * @return first line of {@code file}
442     * @throws IOException if an I/O error occurs
443     */
444    protected String readFirstLine(File file) throws IOException {
445        final FileInputStream inputStream = new FileInputStream(file);
446        try {
447            final LineNumberReader reader = new LineNumberReader(new InputStreamReader(inputStream, Charset.forName("UTF-8")));
448            final String firstLine = reader.readLine();
449            if (firstLine == null)
450                throw new GitException("read empty content from `" + file + "'");
451            return firstLine.trim();
452        } finally {
453            try {
454                inputStream.close();
455            } catch (IOException e) {
456                // ignore
457            }
458        }
459    }

463     *
464     * @param path relative path, e.g., {@code HEAD} or {@code refs/heads/foo}
465     * @return corresponding file
466     */
467    public File getRepoFile(String path) {
468        return new File(new File(this.dir, ".git"), path);
469    }

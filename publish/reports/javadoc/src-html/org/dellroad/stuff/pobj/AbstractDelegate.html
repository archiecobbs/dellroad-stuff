006package org.dellroad.stuff.pobj;
007
008import java.io.IOException;
009import java.io.StringReader;
010import java.io.StringWriter;
011import java.util.Set;
012
013import javax.validation.ConstraintViolation;
014import javax.xml.transform.stream.StreamResult;
015import javax.xml.transform.stream.StreamSource;
016
017import org.dellroad.stuff.validation.ValidationContext;
018import org.slf4j.Logger;
019import org.slf4j.LoggerFactory;
020
021/**
022 * Support superclass for {@link PersistentObjectDelegate} classes, with implementations of all methods
023 * other than {@link #serialize serialize()} and {@link #deserialize deserialize()}.
024 *
025 * @param <T> type of the root persistent object
026 * @see PersistentObject
027 */
028public abstract class AbstractDelegate<T> implements PersistentObjectDelegate<T> {
029
030    private static final int BUFFER_SIZE = 16 * 1024 - 32;
031
032    protected final Logger log = LoggerFactory.getLogger(this.getClass());
033
034    /**
035     * Make a deep copy of the given object.
036     *
037     * <p>
038     * The implementation in {@link AbstractDelegate} does this by {@linkplain #serialize serializing}
039     * and then {@linkplain #deserialize deserializing} the object graph.
040     * Subclasses are encouraged to provide a more efficient implementation, for example, by implementing
041     * {@link org.dellroad.stuff.java.GraphCloneable}.
042     *
043     * @throws IllegalArgumentException if {@code original} is null
044     * @throws PersistentObjectException if an error occurs
045     */
046    public T copy(T original) {
047        if (original == null)
048            throw new IllegalArgumentException("null original");
049        StringWriter buffer = new StringWriter(BUFFER_SIZE);
050        try {
051            this.serialize(original, new StreamResult(buffer));
052        } catch (IOException e) {
053            throw new PersistentObjectException("exception during serialize()");
054        }
055        StreamSource source = new StreamSource(new StringReader(buffer.toString()));
056        T copy;
057        try {
058            copy = this.deserialize(source);
059        } catch (IOException e) {
060            throw new PersistentObjectException("exception during deserialize()");
061        }
062        if (copy == null)
063            throw new PersistentObjectException("null object returned by deserialize()");
064        return copy;
065    }
066
067    /**
068     * Compare two object graphs.
069     *
070     * <p>
071     * The implementation in {@link AbstractDelegate} only returns true if {@code root1 == root2}.
072     * This is a very conservative implementation. If your root object graph correctly implements
073     * {@link Object#equals equals()}, then {@code root1.equals(root2)} would be a more appropriate test.
074     *
075     * @param root1 first object graph (never null)
076     * @param root2 second object graph (never null)
077     */
078    @Override
079    public boolean isSameGraph(T root1, T root2) {
080        return root1 == root2;
081    }
082
083    /**
084     * Validate the given instance.
085     *
086     * <p>
087     * The implementation in {@link AbstractDelegate} performs validation using {@link ValidationContext#validate()}.
088     *
089     * @throws IllegalArgumentException if {@code obj} is null
090     * @return set of zero or more constraint violations
091     */
092    @Override
093    public Set<ConstraintViolation<T>> validate(T obj) {
094        return new ValidationContext<T>(obj).validate();
095    }
096
097    /**
098     * Handle an exception thrown during a delayed write-back attempt. {@link ThreadDeath} exceptions are not
099     * passed to this method, but all others are.
100     *
101     * <p>
102     * The implementation in {@link AbstractDelegate} simply logs an error to {@link #log}.
103     *
104     * @param pobj the instance that encountered the exception
105     * @param t the exception thrown
106     */
107    @Override
108    public void handleWritebackException(PersistentObject<T> pobj, Throwable t) {
109        this.log.error(pobj + ": error during write-back", t);
110    }
111
112    /**
113     * Get the default value for the root object graph. This method is invoked at startup when the persistent file does not exist.
114     * If this method returns null, then an {@linkplain PersistentObject#isAllowEmptyStart empty start} occurs unless the
115     * {@link PersistentObject} object is configured to disallow them, in which case an exception is thrown.
116     *
117     * <p>
118     * The implementation in {@link AbstractDelegate} returns null
119     *
120     * @return root object initial value, or null if there is no default value
121     */
122    @Override
123    public T getDefaultValue() {
124        return null;
125    }
126}

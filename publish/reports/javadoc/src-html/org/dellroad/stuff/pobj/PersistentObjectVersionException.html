006package org.dellroad.stuff.pobj;
007
008/**
009 * Optimistic locking exception thrown by {@link PersistentObject#setRoot PersistentObject.setRoot()}
010 * when the expected version number does not agree.
011 */
012@SuppressWarnings("serial")
013public class PersistentObjectVersionException extends PersistentObjectException {
014
015    private final long actualVersion;
016    private final long expectedVersion;
017
018    public PersistentObjectVersionException(long actualVersion, long expectedVersion) {
019        super("expected version " + expectedVersion + " but actual version was " + actualVersion);
020        this.actualVersion = actualVersion;
021        this.expectedVersion = expectedVersion;
022    }
023
024    /**
025     * Get the actual, unexpected version number.
026     *
027     * @return actual version number
028     */
029    public long getActualVersion() {
030        return this.actualVersion;
031    }
032
033    /**
034     * Get the version number that was expected.
035     *
036     * @return expected version number
037     */
038    public long getExpectedVersion() {
039        return this.expectedVersion;
040    }
041}

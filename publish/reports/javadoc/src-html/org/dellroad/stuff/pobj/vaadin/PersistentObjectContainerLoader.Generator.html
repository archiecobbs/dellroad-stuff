006package org.dellroad.stuff.pobj.vaadin;
007
008import com.vaadin.server.VaadinSession;
009
010import org.dellroad.stuff.pobj.PersistentObject;
011import org.dellroad.stuff.vaadin7.AbstractSimpleContainer;
012
013/**
014 * Facilitates populating a {@link AbstractSimpleContainer} with data from {@link PersistentObject}
015 * and automatically updating it when that {@link PersistentObject} changes.
016 *
017 * <p>
018 * The constructor requires a {@link Generator} to generate, from the {@link PersistentObject}
019 * root object, the Java objects that populate the {@link AbstractSimpleContainer}.
020 * </p>
021 *
022 * <p>
023 * <b>Note:</b> Instances of this class start out in a non-listening state. You must invoke {@link #connect connect()}
024 * to start listening to the {@link PersistentObject} and (re)load the container contents.
025 * You should invoke {@link #disconnect} whenever the container is no longer used to avoid a memory leak.
026 * Typically, {@link #connect connect()} and {@link #disconnect} are invoked from the
027 * {@link com.vaadin.ui.Component#attach attach()} and {@link com.vaadin.ui.Component#detach detach()} methods (respectively)
028 * of the GUI widget that is using the container.
029 * </p>
030 *
031 * @param <T> type of the persistent object
032 * @param <K> the type of the Java objects that back each {@link com.vaadin.data.Item} in the container
033 */
034@SuppressWarnings("serial")
035public class PersistentObjectContainerLoader<T, K> {
036
037    private final AbstractSimpleContainer<?, K> container;
038    private final Generator<T, K> generator;
039
040    private VaadinPersistentObjectListener<T> listener;
041
042    /**
043     * Constructor.
044     *
045     * @param container container to populate
046     * @param generator object that generates the Java objects that back the {@link com.vaadin.data.Item}s in the container
047     * @throws IllegalArgumentException if {@code container} is null
048     * @throws IllegalArgumentException if {@code generator} is null
049     */
050    public PersistentObjectContainerLoader(AbstractSimpleContainer<?, K> container, Generator<T, K> generator) {
051        if (container == null)
052            throw new IllegalArgumentException("null container");
053        if (generator == null)
054            throw new IllegalArgumentException("null generator");
055        this.container = container;
056        this.generator = generator;
057    }
058
059    /**
060     * Register this container as a listener on the persistent object and load the container.
061     * This method may only be invoked while the {@link VaadinSession} is locked by the current thread.
062     * If this method is invoked twice with no intervening {@link #disconnect}, this instance will
063     * automatically disconnect from the old {@link PersistentObject} before connecting to the new one.
064     *
065     * @throws IllegalArgumentException if {@code persistentObject} is null
066     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
067     */
068    public void connect(PersistentObject<T> persistentObject) {
069
070        // Sanity check
071        this.assertSession();
072        if (persistentObject == null)
073            throw new IllegalArgumentException("null persistentObject");
074
075        // Connect
076        boolean needReload = false;
077        T root = null;
078        synchronized (this) {
079            if (this.listener != null)
080                this.listener.unregister();
081            this.listener = new VaadinPersistentObjectListener<T>(persistentObject) {
082                @Override
083                protected void handlePersistentObjectChange(T oldRoot, T newRoot, long version) {
084                    PersistentObjectContainerLoader.this.reload(newRoot);
085                }
086            };
087            this.listener.register();
088            root = persistentObject.getSharedRoot();
089            needReload = true;
090        }
091
092        // Load container if there is data
093        if (needReload)
094            this.reload(root);
095    }
096
097    /**
098     * Shutdown this container. This unregisters the container from the {@link PersistentObject} it was previously
099     * {@linkplain #connect connected} to. This method may be invoked from any thread and is idempotent.
100     * The associated container is not notified about any change.
101     *
102     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
103     */
104    public synchronized void disconnect() {
105        this.assertSession();
106        if (this.listener != null) {
107            this.listener.unregister();
108            this.listener = null;
109        }
110    }
111
112    /**
113     * Reload the container associated with this instance using the given {@link PersistentObject} root.
114     *
115     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
116     */
117    protected void reload(T root) {
118        this.assertSession();
119        this.container.load(this.generator.generateContainerObjects(root));
120    }
121
122    /**
123     * Verify there is a {@link VaadinSession} associated with the current thread.
124     *
125     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
126     */
127    protected void assertSession() {
128        if (VaadinSession.getCurrent() == null)
129             throw new IllegalStateException("there is no VaadinSession associated with the current thread");
130    }
131
132// Generator
133
134    /**
135     * Interface used by {@link PersistentObjectContainerLoader} to derive {@link AbstractSimpleContainer} backing
136     * objects from a {@link PersistentObject} root.
137     *
138     * @param <T> type of the persistent object
139     * @param <K> the type of the Java objects that back each {@link com.vaadin.data.Item} in the {@link AbstractSimpleContainer}
140     */
141    public interface Generator<T, K> {
142
143        /**
144         * Generate the Java objects that will back the {@link AbstractSimpleContainer} associated with
145         * a {@link PersistentObjectContainerLoader} from the given {@link PersistentObject} root.
146         *
147         * <p>
148         * This method should not modify {@code root}.
149         * </p>
150         *
151         * <p>
152         * If the associated {@link PersistentObject} allows empty starts or stops, then {@code root} might be null.
153         * </p>
154         *
155         * @param root updated persistent object root, or null in the case of an empty start or stop
156         * @return container backing objects
157         */
158        Iterable<? extends K> generateContainerObjects(T root);
159    }
160}

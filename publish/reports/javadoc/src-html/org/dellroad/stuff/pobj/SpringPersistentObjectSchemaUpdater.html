006package org.dellroad.stuff.pobj;
007
008import java.io.BufferedInputStream;
009import java.io.IOException;
010import java.io.InputStream;
011import java.util.Collection;
012import java.util.Comparator;
013
014import javax.xml.transform.stream.StreamSource;
015
016import org.dellroad.stuff.schema.SchemaUpdate;
017import org.dellroad.stuff.spring.BeanNameComparator;
018import org.springframework.beans.factory.BeanFactory;
019import org.springframework.beans.factory.BeanFactoryAware;
020import org.springframework.beans.factory.InitializingBean;
021import org.springframework.beans.factory.ListableBeanFactory;
022import org.springframework.core.io.Resource;
023
024/**
025 * {@link PersistentObjectSchemaUpdater} optimized for use with Spring:
026 * <ul>
027 * <li>{@link #getOrderingTieBreaker} is overridden to break ties by ordering updates in the same order
028 *  as they are defined in the bean factory.</li>
029 * <li>This class implements {@link InitializingBean} and verifies all required properties are set.</li>
030 * <li>If no updates are {@linkplain #setUpdates explicitly configured}, then all {@link SpringPersistentObjectSchemaUpdate}s
031 *  found in the containing bean factory are automatically configured; this requires that all of the schema updates
032 *  are defined in the same {@link ListableBeanFactory}.</li>
033 * <li>The default value may be configured as an XML resource</li>
034 * </ul>
035 *
036 * <p>
037 * An example of how this class can be combined with custom XML to define an updater and all its updates:
038 * <blockquote><pre>
039 *  &lt;beans xmlns="http://www.springframework.org/schema/beans"
040 *    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
041 *    xmlns:p="http://www.springframework.org/schema/p"
042 *    xsi:schemaLocation="
043 *      http://www.springframework.org/schema/beans
044 *        http://www.springframework.org/schema/beans/spring-beans-3.0.xsd"&gt;
045 *
046 *      &lt;!-- Normal nested persistent object delegate. You supply the XML (un)marshaller (not shown). --&gt;
047 *      <b>&lt;bean id="normalDelegate" class="org.dellroad.stuff.pobj.SpringDelegate"
048 *        p:marshaller-ref="marshaller" p:unmarshaller-ref="unmarshaller"/&gt;</b>
049 *
050 *      &lt;!-- Schema updating persistent object delegate; the updates below will be auto-detected. --&gt;
051 *      <b>&lt;bean id="updatingDelegate" class="org.dellroad.stuff.pobj.SpringPersistentObjectSchemaUpdater"
052 *        p:marshaller-ref="marshaller" p:unmarshaller-ref="unmarshaller" p:defaultXML="classpath:default.xml"&gt;
053 *          &lt;constructor-arg ref="normalDelegate"/&gt;
054 *      &lt;/bean&gt;</b>
055 *
056 *      &lt;!-- Persistent object, configured to use our schema updating delegate --&gt;
057 *      <b>&lt;bean id="persistentObject" class="org.dellroad.stuff.pobj.PersistentObject"
058 *        init-method="start" destroy-method="stop" p:file="/var/lib/pobj.xml" p:allowEmptyStart="true"
059 *        p:numBackups="3" p:delegate-ref="updatingDelegate"/&gt;</b>
060 *
061 *      &lt;!-- Define a default location for schema update XSL files --&gt;
062 *      <b>&lt;bean class="org.dellroad.stuff.pobj.SpringXSLUpdateTransformConfigurer"
063 *        p:prefix="classpath:updates/" p:suffix=".xsl"/&gt;</b>
064 *
065 *      &lt;!-- Schema update #1 with an explicitly configured XSL resource --&gt;
066 *      <b>&lt;bean id="update1" class="org.dellroad.stuff.pobj.SpringXSLPersistentObjectSchemaUpdate"
067 *        transform="file:///usr/share/updates/update1.xsl"/&gt;</b>
068 *
069 *      &lt;!-- Schema update #2: implicitly uses "classpath:updates/update2.xsl" --&gt;
070 *      <b>&lt;bean id="update2" class="org.dellroad.stuff.pobj.SpringXSLPersistentObjectSchemaUpdate"/&gt;</b>
071 *
072 *      &lt;!-- Schema update #3: requires that update #1 be applied first --&gt;
073 *      <b>&lt;bean id="update3" depends-on="update1"
074 *        class="org.dellroad.stuff.pobj.SpringXSLPersistentObjectSchemaUpdate"/&gt;</b>
075 *
076 *      &lt;!-- Add more schema updates over time as needed and everything just works... --&gt;
077 *
078 *  &lt;/beans&gt;
079 * </pre></blockquote>
080 *
081 * @param <T> type of the root persistent object
082 */
083public class SpringPersistentObjectSchemaUpdater<T> extends PersistentObjectSchemaUpdater<T>
084  implements BeanFactoryAware, InitializingBean {
085
086    private ListableBeanFactory beanFactory;
087    private Resource defaultXML;
088
089    /**
090     * Constructor.
091     *
092     * @param delegate delegate that will be wrapped by this instance
093     * @throws IllegalArgumentException if {@code delegate} is null
094     * @see PersistentObjectSchemaUpdater#PersistentObjectSchemaUpdater
095     */
096    public SpringPersistentObjectSchemaUpdater(PersistentObjectDelegate<T> delegate) {
097        super(delegate);
098    }
099
100    /**
101     * Set the resource containing the default value, encoded as XML, to be used on an uninitialized persistent object.
102     * This will override whatever default value is returned by the nested delegate.
103     *
104     * @param resource default database content
105     */
106    public void setDefaultXML(Resource resource) {
107        this.defaultXML = resource;
108    }
109
110    /**
111     * Get the default value for the persistent object when no persistent file is found.
112     *
113     * <p>
114     * The implementation in {@link SpringPersistentObjectSchemaUpdater} parses and returns the
115     * {@linkplain #setDefaultXML default value resource}, if any; otherwise, the delegate provided
116     * to the constructor is queried for a default value.
117     */
118    @Override
119    public T getDefaultValue() {
120
121        // If no XML configured, fall back to nested delegate
122        if (this.defaultXML == null)
123            return this.delegate.getDefaultValue();
124
125        // Use configured XML
126        try {
127            this.log.info("loading default content from " + this.defaultXML.getURI());
128            InputStream input = this.defaultXML.getInputStream();
129            try {
130                return this.delegate.deserialize(
131                  new StreamSource(new BufferedInputStream(input), this.defaultXML.getURI().toString()));
132            } finally {
133                try {
134                    input.close();
135                } catch (IOException e) {
136                    // ignore
137                }
138            }
139        } catch (RuntimeException e) {
140            throw e;
141        } catch (Exception e) {
142            throw new PersistentObjectException(e);
143        }
144    }
145
146    @Override
147    public void setBeanFactory(BeanFactory beanFactory) {
148        if (this.beanFactory == null && beanFactory instanceof ListableBeanFactory)
149            this.beanFactory = (ListableBeanFactory)beanFactory;
150    }
151
152    @Override
153    @SuppressWarnings("unchecked")
154    public void afterPropertiesSet() throws Exception {
155        if (this.getUpdates() == null) {
156            if (this.beanFactory == null) {
157                throw new IllegalArgumentException("no updates explicitly configured and the containing BeanFactory"
158                  + " is not a ListableBeanFactory: " + this.beanFactory);
159            }
160            this.setUpdates((Collection<SpringPersistentObjectSchemaUpdate<T>>)(Object)this.beanFactory.getBeansOfType(
161              SpringPersistentObjectSchemaUpdate.class).values());
162        }
163    }
164
165    /**
166     * Get the preferred ordering of two updates that do not have any predecessor constraints
167     * (including implied indirect constraints) between them.
168     *
169     * <p>
170     * In the case no schema updates are explicitly configured, the {@link Comparator} returned by the
171     * implementation in {@link SpringPersistentObjectSchemaUpdater} sorts updates in the same order that they appear
172     * in the containing {@link ListableBeanFactory}. Otherwise, the
173     * {@linkplain org.dellroad.stuff.schema.AbstractSchemaUpdater#getOrderingTieBreaker superclass method} is used.
174     */
175    @Override
176    protected Comparator<SchemaUpdate<PersistentFileTransaction>> getOrderingTieBreaker() {
177        if (this.beanFactory == null)
178            return super.getOrderingTieBreaker();
179        final BeanNameComparator beanNameComparator = new BeanNameComparator(this.beanFactory);
180        return new Comparator<SchemaUpdate<PersistentFileTransaction>>() {
181            @Override
182            public int compare(SchemaUpdate<PersistentFileTransaction> update1, SchemaUpdate<PersistentFileTransaction> update2) {
183                return beanNameComparator.compare(update1.getName(), update2.getName());
184            }
185        };
186    }
187}

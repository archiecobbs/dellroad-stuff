006package org.dellroad.stuff.pobj.distrib;
007
008import java.io.File;
009import java.util.List;
010
011/**
012 * Available merge strategies for a {@link GitRepository} merge operation.
013 */
014public enum MergeStrategy {
015
016    /**
017     * Use our version; discard other version.
018     */
019    OURS {
020        @Override
021        void addOptions(List<String> args) {
022            args.add("--strategy=ours");
023        }
024    },
025
026    /**
027     * Use other version; discard our version.
028     */
029    THEIRS {
030
031        // Note, this would be easy with `--stragegy=theirs' except for this jerk: http://marc.info/?l=git&m=121637513604413&w=2
032        // Links:
033        //  http://stackoverflow.com/a/4969679/263801
034        //  http://stackoverflow.com/a/4912267/263801
035        @Override
036        public void merge(File dir, String other) {
037
038            // Sanity check
039            if (other == null)
040                throw new IllegalArgumentException("null other");
041
042            // Simulate --stragegy=theirs
043            new GitCommand(dir, "merge", "--no-commit", "--no-ff", "--strategy=ours", "--", other).run();
044            new GitCommand(dir, "read-tree", "--reset", other).run();
045        }
046
047        @Override
048        void addOptions(List<String> args) {
049            throw new UnsupportedOperationException();
050        }
051    },
052
053    /**
054     * Use the traditional `resolve' merge algorithm.
055     */
056    RESOLVE {
057        @Override
058        void addOptions(List<String> args) {
059            args.add("--strategy=resolve");
060        }
061    },
062
063    /**
064     * Use the `recursive' merge algorithm.
065     */
066    RECURSIVE {
067        @Override
068        void addOptions(List<String> args) {
069            args.add("--strategy=recursive");
070        }
071    },
072
073    /**
074     * Use the `recursive' merge algorithm, resolving conflicts in favor of our branch.
075     */
076    RECURSIVE_OURS {
077        @Override
078        void addOptions(List<String> args) {
079            args.add("--strategy=recursive");
080            args.add("--strategy-option=ours");
081        }
082    },
083
084    /**
085     * Use the `recursive' merge algorithm, resolving conflicts in favor of the other branch.
086     */
087    RECURSIVE_THEIRS {
088        @Override
089        void addOptions(List<String> args) {
090            args.add("--strategy=recursive");
091            args.add("--strategy-option=theirs");
092        }
093    },
094
095    /**
096     * Use the `recursive' merge algorithm with the `patience' option.
097     */
098    RECURSIVE_PATIENCE {
099        @Override
100        void addOptions(List<String> args) {
101            args.add("--strategy=recursive");
102            args.add("--strategy-option=patience");
103        }
104    },
105
106    /**
107     * Use the `recursive' merge algorithm with the `patience' option, resolving conflicts in favor of our branch.
108     */
109    RECURSIVE_PATIENCE_OURS {
110        @Override
111        void addOptions(List<String> args) {
112            args.add("--strategy=recursive");
113            args.add("--strategy-option=patience");
114            args.add("--strategy-option=ours");
115        }
116    },
117
118    /**
119     * Use the `recursive' merge algorithm with the `patience' option, resolving conflicts in favor of the other branch.
120     */
121    RECURSIVE_PATIENCE_THEIRS {
122        @Override
123        void addOptions(List<String> args) {
124            args.add("--strategy=recursive");
125            args.add("--strategy-option=patience");
126            args.add("--strategy-option=theirs");
127        }
128    };
129
130    /**
131     * Peform a merge using this strategy, without committing.
132     * This assumes the checkout directory is prepared appropriately for the merge with the target branch checked out.
133     *
134     * @param dir checkout directory
135     * @param other other branch to merge with
136     * @throws IllegalArgumentException if {@code dir} or {@code other} is null
137     * @throws GitMergeConflictException if a merge conflict occurs
138     * @throws GitException if an error other than a merge conflict occurs
139     */
140    public void merge(File dir, String other) {
141
142        // Run merge command
143        final GitCommand merge = this.buildMergeCommand(dir, other);
144        final int exitValue = merge.run(true);
145
146        // Any conflict or other error?
147        if (exitValue == 0)
148            return;
149
150        // Check for errors other than conflict
151        final File mergeHeadFile = new File(new File(dir, ".git"), "MERGE_HEAD");
152        if (!mergeHeadFile.exists())
153            throw new GitException("git merge failed: " + merge.getStandardError().trim());
154
155        // Abort the merge
156        new GitCommand(dir, "merge", "--abort").run();
157
158        // Bail out
159        throw new GitMergeConflictException("merge failed with conflict(s)");
160    }
161
162    /**
163     * Build the merge command. Delegates to {@link #addOptions} for strategy-specific merge command options.
164     *
165     * @param dir checkout directory
166     * @param other other branch to merge with
167     * @throws IllegalArgumentException if {@code dir} or {@code other} is null
168     */
169    private GitCommand buildMergeCommand(File dir, String other) {
170
171        // Sanity check
172        if (other == null)
173            throw new IllegalArgumentException("null other");
174
175        // Build command
176        final GitCommand merge = new GitCommand(dir, "merge", "--no-commit", "--no-ff");
177        this.addOptions(merge.getArgs());
178        merge.getArgs().add("--");
179        merge.getArgs().add(other);
180        return merge;
181    }
182
183    /**
184     * Add the appropriate {@code git merge} command line flags.
185     */
186    abstract void addOptions(List<String> args);
187}

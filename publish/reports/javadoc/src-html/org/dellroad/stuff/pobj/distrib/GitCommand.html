006package org.dellroad.stuff.pobj.distrib;
007
008import java.io.File;
009import java.io.IOException;
010import java.nio.charset.Charset;
011import java.util.ArrayList;
012import java.util.Arrays;
013import java.util.List;
014
015import org.dellroad.stuff.java.ProcessRunner;
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019/**
020 * A simple wrapper around the {@code git(1)} command.
021 */
022public class GitCommand {
023
024    private final Logger log = LoggerFactory.getLogger(this.getClass());
025    private final File dir;
026    private final List<String> args = new ArrayList<String>();
027
028    private ProcessRunner runner;
029
030    /**
031     * Constructor.
032     *
033     * @param dir repository filesystem location (i.e., the directory containing {@code .git} as a subdirectory)
034     * @param params the command line arguments (not including the initial {@code "git"})
035     */
036    public GitCommand(File dir, String... params) {
037        this(dir, Arrays.asList(params));
038    }
039
040    /**
041     * Constructor.
042     *
043     * @param dir repository filesystem location (i.e., the directory containing {@code .git} as a subdirectory)
044     * @param params the command line arguments (not including the initial {@code "git"})
045     * @throws IllegalArgumentException if {@code dir} is null
046     */
047    public GitCommand(File dir, List<String> params) {
048        if (dir == null)
049            throw new IllegalArgumentException("null dir");
050        if (params == null)
051            throw new IllegalArgumentException("null params");
052        this.dir = dir;
053        this.args.add("git");
054        this.args.addAll(params);
055    }
056
057    /**
058     * Get command line arguments.
059     */
060    public List<String> getArgs() {
061        return this.args;
062    }
063
064    /**
065     * Run command. Equivalent to {@code run(false)}.
066     */
067    public int run() {
068        return this.run(false);
069    }
070
071    /**
072     * Run command.
073     *
074     * @param errorOk true if a non-zero exit value may be expected
075     * @return {@code git(1)} exit value; if {@code errorOk} is false this will always be zero
076     * @throws IllegalStateException if this command has already been run
077     * @throws RuntimeException if the current thread is interrupted
078     * @throws GitException if any {@code GIT_*} environment variables are set
079     * @throws GitException if {@code errorOk} is false and {@code git(1}} exits with a non-zero value
080     */
081    public int run(boolean errorOk) {
082
083        // Sanity check
084        if (this.runner != null)
085            throw new IllegalStateException("command already executed");
086
087        // Sanity check environment
088        final ArrayList<String> vars = new ArrayList<String>();
089        for (String var : System.getenv().keySet()) {
090            if (var.startsWith("GIT_"))
091                vars.add(var);
092        }
093        if (!vars.isEmpty())
094            throw new GitException("need to unset GIT_* environment variables first: " + vars);
095
096        // Start process
097        Process process;
098        try {
099            process = Runtime.getRuntime().exec(args.toArray(new String[args.size()]), null, this.dir);
100        } catch (IOException e) {
101            this.log.debug("command `" + this + "' in directory `" + this.dir + "' failed: ", e);
102            throw new GitException("error invoking git(1)", e);
103        }
104        this.runner = new ProcessRunner(process);
105
106        // Let it finish
107        int exitValue;
108        try {
109            exitValue = this.runner.run();
110        } catch (InterruptedException e) {
111            throw new RuntimeException("unexpected exception", e);      // XXX
112        }
113
114        // Log command output
115        final StringBuilder buf = new StringBuilder();
116        final String[] stdout = this.getStandardOutput().trim().split("\n");
117        if (!(stdout.length == 1 && stdout[0].length() == 0)) {
118            for (String line : stdout)
119                buf.append("\n  [stdout] ").append(line);
120        }
121        final String[] stderr = this.getStandardError().trim().split("\n");
122        if (!(stderr.length == 1 && stderr[0].length() == 0)) {
123            for (String line : stderr)
124                buf.append("\n  [stderr] ").append(line);
125        }
126
127        // Check exit value
128        if (exitValue != 0) {
129            final String msg = "command `" + this + "' in directory `" + this.dir + "' failed with exit value " + exitValue;
130            if (errorOk)
131                this.log.debug(msg);
132            else {
133                this.log.error(msg);
134                throw new GitException(msg);
135            }
136        } else
137            this.log.debug("successfully executed command `" + this + "' in directory `" + this.dir + "'" + buf);
138
139        // Done
140        return exitValue;
141    }
142
143    /**
144     * Run command and return standard output, interpreted as a UTF-8 string, with leading and trailing whitespace trimmed.
145     *
146     * @throws IllegalStateException if this command has already been run
147     * @throws RuntimeException if the current thread is interrupted
148     * @throws GitException if any {@code GIT_*} environment variables are set
149     * @throws GitException if {@code git(1}} exits with a non-zero value
150     */
151    public String runAndGetOutput() {
152        this.run();
153        return this.getStandardOutput().trim();
154    }
155
156    /**
157     * Get the standard output from {@code git(1)}.
158     *
159     * @throws IllegalStateException if command has not yet completed
160     * @return command standard output interpreted as UTF-8
161     */
162    public String getStandardOutput() {
163        if (this.runner == null)
164            throw new IllegalStateException("command has not yet executed");
165        return new String(this.runner.getStandardOutput(), Charset.forName("UTF-8"));
166    }
167
168    /**
169     * Get the standard error from {@code git(1)}.
170     *
171     * @throws IllegalStateException if command has not yet completed
172     * @return command standard error interpreted as UTF-8
173     */
174    public String getStandardError() {
175        if (this.runner == null)
176            throw new IllegalStateException("command has not yet executed");
177        return new String(this.runner.getStandardError(), Charset.forName("UTF-8"));
178    }
179
180    @Override
181    public String toString() {
182        final StringBuilder buf = new StringBuilder();
183        for (String arg : this.args) {
184            if (buf.length() > 0)
185                buf.append(' ');
186            buf.append(arg);
187        }
188        return buf.toString();
189    }
190}

006package org.dellroad.stuff.pobj;
007
008import java.util.EventObject;
009
010/**
011 * Notification event emitted by a {@link PersistentObject} to listeners whenever there is an update to the root object.
012 *
013 * @param <T> type of the root persistent object
014 */
015@SuppressWarnings("serial")
016public class PersistentObjectEvent<T> extends EventObject {
017
018    private final long version;
019    private final T oldRoot;
020    private final T newRoot;
021
022    /**
023     * Constructor.
024     *
025     * @param persistentObject source of this event
026     * @param version the new persistent object version (i.e., the version of {@code newRoot})
027     * @param oldRoot previous root object; null if exiting from an empty start period
028     * @param newRoot updated root object; null if entering an empty stop period
029     */
030    public PersistentObjectEvent(PersistentObject<T> persistentObject, long version, T oldRoot, T newRoot) {
031        super(persistentObject);
032        this.version = version;
033        this.oldRoot = oldRoot;
034        this.newRoot = newRoot;
035    }
036
037    /**
038     * Get the {@link PersistentObject} that originated this event.
039     */
040    @SuppressWarnings("unchecked")
041    public PersistentObject<T> getSource() {
042        return (PersistentObject<T>)super.getSource();
043    }
044
045    /**
046     * Get the version that this event is associated with. This will be the version of the {@linkplain #getNewRoot new root}.
047     *
048     * <p>
049     * The {@link PersistentObject} class always delivers notifications in order, so this
050     * number should always increase over time.
051     *
052     * @return new root version number
053     */
054    public long getVersion() {
055        return this.version;
056    }
057
058    /**
059     * Get the old root prior to the update.
060     *
061     * <p>
062     * The caller must not modify the returned object, as it is shared among all listeners.
063     *
064     * @return the old root object; will be null if an empty start period has just ended
065     */
066    public T getOldRoot() {
067        return this.oldRoot;
068    }
069
070    /**
071     * Get the new root after the update.
072     *
073     * <p>
074     * The caller must not modify the returned object, as it is shared among all listeners.
075     *
076     * @return the new root object; will be null if an empty stop period has just started
077     */
078    public T getNewRoot() {
079        return this.newRoot;
080    }
081}

006package org.dellroad.stuff.string;
007
008import java.math.BigInteger;
009import java.util.Comparator;
010
011/**
012 * Comparator for version number parts.
013 *
014 * <p>
015 * This comparator puts numerical parts before non-numerical parts.
016 * Then, numerical parts are compared numerically, and non-numerical parts are compared lexically.
017 *
018 * @see VersionComparator
019 */
020public class VersionPartComparator implements Comparator<String> {
021
022    @Override
023    public int compare(String part1, String part2) {
024        BigInteger num1 = part1.matches("[0-9]+") ? new BigInteger(part1, 10) : null;
025        BigInteger num2 = part2.matches("[0-9]+") ? new BigInteger(part2, 10) : null;
026        if (num1 != null && num2 != null)
027            return num1.compareTo(num2);
028        else if (num1 != null)
029            return 1;
030        else if (num2 != null)
031            return -1;
032        else
033            return part1.compareTo(part2);
034    }
035}

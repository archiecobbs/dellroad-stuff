006package org.dellroad.stuff.string;
007
008import java.util.regex.Matcher;
009import java.util.regex.Pattern;
010
011/**
012 * Utility class supporting parsing of strings.
013 *
014 * <p>
015 * Instances of this class are not thread safe.
016 * </p>
017 */
018public class ParseContext implements Cloneable {
019
020    private static final int MAX_REJECT_QUOTE = 15;
021
022    private final String input;
023
024    private int index;
025
026    /**
027     * Constructor.
028     *
029     * @param input the input string to parse
030     * @throws IllegalArgumentException if {@code input} is null
031     */
032    public ParseContext(String input) {
033        if (input == null)
034            throw new IllegalArgumentException("null input");
035        this.input = input;
036    }
037
038    /**
039     * Get the original input string as passed to the constructor.
040     *
041     * @return original input string
042     */
043    public String getOriginalInput() {
044        return this.input;
045    }
046
047    /**
048     * Get the current input.
049     *
050     * @return substring of the original input string starting at the current parse position
051     */
052    public String getInput() {
053        return this.input.substring(this.index);
054    }
055
056    /**
057     * Get the current index into the original input string.
058     *
059     * @return current parse position
060     * @see #setIndex
061     */
062    public int getIndex() {
063        return this.index;
064    }
065
066    /**
067     * Set the current index into the original input string.
068     *
069     * @param index new parse position
070     * @throws IllegalArgumentException if {@code index} is greater than the original string length
071     * @see #getIndex
072     */
073    public void setIndex(int index) {
074        this.index = index;
075    }
076
077    /**
078     * Reset this instance. This instance will return to the state it was in
079     * immediately after construction.
080     *
081     * <p>
082     * This method just invokes:
083     * <blockquote>
084     *  <code>setIndex(0)</code>
085     * </blockquote>
086     */
087    public void reset() {
088        this.setIndex(0);
089    }
090
091    /**
092     * Match the current input against the given regular expression and advance past it.
093     *
094     * @param regex regular expression to match against the current input
095     * @throws IllegalArgumentException if the current input does not match
096     */
097    public Matcher matchPrefix(String regex) {
098        return this.matchPrefix(Pattern.compile(regex));
099    }
100
101    /**
102     * Match the current input against the given regular expression and advance past it.
103     *
104     * @param regex regular expression to match against the current input
105     * @throws IllegalArgumentException if the current input does not match
106     */
107    public Matcher matchPrefix(Pattern regex) {
108        final Matcher matcher = this.tryPattern(regex);
109        if (matcher == null)
110            throw buildException("expected input matching pattern `" + regex + "'");
111        return matcher;
112    }
113
114    /**
115     * Determine if the current input starts with the given literal prefix.
116     * If so, advance past it. If not, do not advance.
117     *
118     * @param prefix literal string to try to match against the current input
119     * @return whether the current input matched {@code prefix}
120     */
121    public boolean tryLiteral(String prefix) {
122        final boolean match = this.input.startsWith(prefix, this.index);
123        if (match)
124            this.index += prefix.length();
125        return match;
126    }
127
128    /**
129     * Determine if the current input starts with the given regular expression.
130     * If so, advance past it and return a successful {@link Matcher}; otherwise, return null.
131     *
132     * @param pattern regular expression to match against input
133     * @return match if successful, null if not
134     */
135    public Matcher tryPattern(String pattern) {
136        return this.tryPattern(Pattern.compile(pattern));
137    }
138
139    /**
140     * Determine if the current input starts with the given regular expression.
141     * If so, advance past it and return a successful {@link Matcher}; otherwise, return null.
142     *
143     * @param regex regular expression to match against input
144     * @return match if successful, null if not
145     */
146    public Matcher tryPattern(Pattern regex) {
147        final Matcher matcher = regex.matcher(this.getInput());
148        if (!matcher.lookingAt())
149            return null;
150        this.index += matcher.end();
151        return matcher;
152    }
153
154    /**
155     * Determine if we are at the end of the input.
156     */
157    public boolean isEOF() {
158        return this.index >= this.input.length();
159    }
160
161    /**
162     * Read and advance past the next character.
163     *
164     * @return the next character of input
165     * @throws IllegalArgumentException if there are no more characters
166     */
167    public char read() {
168        char ch = this.peek();
169        this.index++;
170        return ch;
171    }
172
173    /**
174     * Read, but do not advance past, the next character.
175     *
176     * @return the next character of input
177     * @throws IllegalArgumentException if there are no more characters
178     */
179    public char peek() {
180        try {
181            return this.input.charAt(this.index);
182        } catch (StringIndexOutOfBoundsException e) {
183            throw this.buildException("truncated input");
184        }
185    }
186
187    /**
188     * Push back the previously read character.
189     *
190     * @throws IllegalStateException if the beginning of the original string has been reached
191     */
192    public void unread() {
193        if (this.index == 0)
194            throw new IllegalStateException();
195        this.index--;
196    }
197
198    /**
199     * Read and advance past the next character, which must match {@code ch}.
200     *
201     * @param ch the expected next character of input
202     * @throws IllegalArgumentException if there are no more characters or the
203     *  next character read is not {@code ch}
204     */
205    public void expect(char ch) {
206        if (this.read() != ch) {
207            this.unread();
208            throw buildException("expected `" + ch + "'");
209        }
210    }
211
212    /**
213     * Skip leading whitespace, if any.
214     *
215     * @see Character#isWhitespace
216     */
217    public void skipWhitespace() {
218        while (this.index < this.input.length() && Character.isWhitespace(this.input.charAt(this.index)))
219            this.index++;
220    }
221
222    /**
223     * Clone this instance.
224     */
225    public ParseContext clone() {
226        try {
227            return (ParseContext)super.clone();
228        } catch (CloneNotSupportedException e) {
229            throw new RuntimeException(e);
230        }
231    }
232
233    /**
234     * Create a generic exception for rejecting the current input.
235     */
236    public IllegalArgumentException buildException() {
237        return this.buildException(null);
238    }
239
240    /**
241     * Create an exception for rejecting the current input.
242     *
243     * @param message problem description, or {@code null} for none
244     */
245    public IllegalArgumentException buildException(String message) {
246        String text = "parse error ";
247        String bogus = getInput();
248        if (bogus.length() == 0)
249            text += "at end of input";
250        else {
251            if (bogus.length() > MAX_REJECT_QUOTE)
252                bogus = bogus.substring(0, MAX_REJECT_QUOTE - 3) + "...";
253            text += "starting with `" + bogus + "'";
254        }
255        if (message != null)
256            text += ": " + message;
257        return new IllegalArgumentException(text);
258    }
259
260    @Override
261    public boolean equals(Object obj) {
262        if (obj == null || obj.getClass() != this.getClass())
263            return false;
264        final ParseContext that = (ParseContext)obj;
265        return this.input.equals(that.input) && this.index == that.index;
266    }
267
268    @Override
269    public int hashCode() {
270        return this.input.hashCode() ^ this.index;
271    }
272}

006package org.dellroad.stuff.string;
007
008import java.util.regex.Matcher;
009import java.util.regex.Pattern;
010
011/**
012 * Encodes/decodes Java strings, escaping control and XML-invalid characters.
013 */
014public final class StringEncoder {
015
016    /**
017     * A regular expression that matches exactly the set of valid {@link #encode encode()}'d strings.
018     */
019    public static final Pattern ENCODE_PATTERN
020      = Pattern.compile("([\\t\\r\\n\\x20-\\x5b\\x5d-\\ud7ff]|[\\ue000-\\ufffd]"
021        + "|\\\\([\\\\bftrn]|u[\\p{XDigit}]{4}))*");
022
023    /**
024     * A regular expression that matches exactly the set of valid {@link #enquote enquote()}'d strings.
025     */
026    public static final Pattern ENQUOTE_PATTERN
027      = Pattern.compile("\"([\\t\\r\\n\\x20\\x21\\x23-\\x5b\\x5d-\\ud7ff]|[\\ue000-\\ufffd]"
028        + "|\\\\([\\\\bftrn\"]|u[\\p{XDigit}]{4}))*\"");
029
030    private static final char[] HEXDIGITS = "0123456789abcdef".toCharArray();
031
032    private StringEncoder() {
033    }
034
035    /**
036     * Encode a string, escaping control and XML-invalid characters.
037     * Whether tab, newline, and carriage return are escaped is optional;
038     * these are the three control characters that are valid inside XML documents.
039     *
040     * <p>
041     * Characters are escaped using <code>&#92;uNNNN</code> notation like Java unicode characters,
042     * e.g., <code>0x001f</code> would appear in the encoded string as <code>&#92;u001f</code>.
043     * Normal Java backslash escapes are used for tab, newline, carriage return, backspace, and formfeed.
044     * Backslash characters are themselves encoded with a double backslash.
045     *
046     * @param value         string to encode (possibly null)
047     * @param escapeTABNLCR escape tab, newline, and carriage return characters as well
048     * @return the encoded version of {@code value}, or {@code null} if {@code value} was {@code null}
049     * @see #decode
050     * @see #isValidXMLChar
051     */
052    public static String encode(String value, boolean escapeTABNLCR) {
053        if (value == null)
054            return value;
055        StringBuilder buf = new StringBuilder(value.length() + 4);
056        final int limit = value.length();
057        for (int i = 0; i < limit; i++) {
058            final char ch = value.charAt(i);
059
060            // Handle special escapes
061            switch (ch) {
062            case '\\':
063                buf.append('\\').append('\\');
064                continue;
065            case '\b':
066                buf.append('\\').append('b');
067                continue;
068            case '\f':
069                buf.append('\\').append('f');
070                continue;
071            case '\t':
072                if (escapeTABNLCR) {
073                    buf.append('\\').append('t');
074                    continue;
075                }
076                break;
077            case '\n':
078                if (escapeTABNLCR) {
079                    buf.append('\\').append('n');
080                    continue;
081                }
082                break;
083            case '\r':
084                if (escapeTABNLCR) {
085                    buf.append('\\').append('r');
086                    continue;
087                }
088                break;
089            default:
090                break;
091            }
092
093            // If character is an otherwise valid XML character, pass it through unchanged
094            if (isValidXMLChar(ch)) {
095                buf.append(ch);
096                continue;
097            }
098
099            // Escape it using 4 digit hex
100            buf.append('\\');
101            buf.append('u');
102            for (int shift = 12; shift >= 0; shift -= 4)
103                buf.append(HEXDIGITS[(ch >> shift) & 0x0f]);
104        }
105        return buf.toString();
106    }
107
108    /**
109     * Decode a string encoded by {@link #encode}.
110     *
111     * <p>
112     * The parsing is strict; any ill-formed backslash escape sequence (i.e., not of the form
113     * <code>&#92;uNNNN</code>, <code>\b</code>, <code>\t</code>, <code>\n</code>, <code>\f</code>, <code>\r</code>
114     * or <code>\\</code>) will cause an exception to be thrown.
115     *
116     * @param text string to decode (possibly null)
117     * @return the decoded version of {@code text}, or {@code null} if {@code text} was {@code null}
118     * @throws IllegalArgumentException if {@code text} contains an invalid escape sequence
119     * @see #encode
120     */
121    public static String decode(String text) {
122        if (text == null)
123            return null;
124        StringBuilder buf = new StringBuilder(text.length());
125        final int limit = text.length();
126        for (int i = 0; i < limit; i++) {
127            char ch = text.charAt(i);
128
129            // Handle unescaped characters
130            if (ch != '\\') {
131                if (!isValidXMLChar(ch))
132                    throw new IllegalArgumentException(String.format("illegal character 0x%04x in encoded string", ch));
133                buf.append(ch);
134                continue;
135            }
136
137            // Get next char
138            if (++i >= limit)
139                throw new IllegalArgumentException("illegal trailing '\\' in encoded string");
140            ch = text.charAt(i);
141
142            // Check for special escapes
143            switch (ch) {
144            case '\\':
145                buf.append('\\');
146                continue;
147            case 'b':
148                buf.append('\b');
149                continue;
150            case 't':
151                buf.append('\t');
152                continue;
153            case 'n':
154                buf.append('\n');
155                continue;
156            case 'f':
157                buf.append('\f');
158                continue;
159            case 'r':
160                buf.append('\r');
161                continue;
162            default:
163                break;
164            }
165
166            // Must be unicode escape
167            if (ch != 'u')
168                throw new IllegalArgumentException("illegal escape sequence '\\" + ch + "' in encoded string");
169
170            // Decode hex value
171            if (i + 4 >= limit)
172                throw new IllegalArgumentException("illegal truncated '\\u' escape sequence in encoded string");
173            int value = 0;
174            for (int j = 0; j < 4; j++) {
175                int nibble = Character.digit(text.charAt(++i), 16);
176                if (nibble == -1) {
177                    throw new IllegalArgumentException(
178                      "illegal escape sequence '" + text.substring(i - j - 2, i - j + 4) + "' in encoded string");
179                }
180                assert nibble >= 0 && nibble <= 0xf;
181                value = (value << 4) | nibble;
182            }
183
184            // Append decodec character
185            buf.append((char)value);
186        }
187        return buf.toString();
188    }

190    /**
191     * Enquote a string. Functions like {@link #encode encode(string, true)} but in addition the resulting
192     * string is surrounded by double quotes and double quotes in the string are backslash-escaped.
193     *
194     * <p>
195     * Note: the strings returned by this method are not suitable for decoding by {@link #decode}.
196     * Use {@link #dequote} instead.
197     *
198     * @param string string to enquote
199     * @return enquoted string, or the string {@code null} (not quoted) if {@code string} is null
200     * @throws IllegalArgumentException if {@code string} is null
201     */
202    public static String enquote(String string) {
203        if (string == null)
204            return "null";
205        return '"' + StringEncoder.encode(string, true).replaceAll(Pattern.quote("\""), Matcher.quoteReplacement("\\\"")) + '"';
206    }
207
208    /**
209     * Enquote bytes, treating them as an ASCII string.
210     *
211     * @param data ascii character buffer
212     * @param off starting offset in {@code data}
213     * @param len number of characters in {@code data}
214     * @return enquoted string
215     * @throws IllegalArgumentException if {@code data} is null
216     * @throws IndexOutOfBoundsException if {@code off} and/or {@code len} is invalid
217     * @see #enquote(String)
218     */
219    public static String enquote(byte[] data, int off, int len) {
220        if (data == null)
221            throw new IllegalArgumentException("null data");
222        if (off < 0 || len < 0 || off + len < 0 || off + len > data.length)
223            throw new IndexOutOfBoundsException("invalid off/len");
224        char[] chars = new char[len];
225        for (int i = 0; i < len; i++)
226            chars[i] = (char)(data[off + i] & 0xff);
227        return enquote(new String(chars));
228    }
229
230    /**
231     * Dequote a string previously enquoted by {@link #enquote}.
232     *
233     * @param quotedString a string returned by {@link #enquote}
234     * @return original unquoted string
235     * @throws IllegalArgumentException if {@code quotedString} has an invalid format (i.e., it could not have
236     *  ever been returned by {@link #enquote})
237     */
238    public static String dequote(String quotedString) {
239        int len = quotedString.length();
240        if (len < 2 || quotedString.charAt(0) != '"' || quotedString.charAt(len - 1) != '"')
241            throw new IllegalArgumentException("invalid quoted string: not surrounded by quote characters");
242        quotedString = quotedString.substring(1, len - 1);
243        if (quotedString.matches("(?s)^(\"|.*[^\\\\]\").*$"))
244            throw new IllegalArgumentException("invalid quoted string: unescaped nested quote character");
245        quotedString = quotedString.replaceAll(Pattern.quote("\\\""), Matcher.quoteReplacement("\""));
246        return decode(quotedString);
247    }
248
249    /**
250     * Determine the length of a string previously enquoted by {@link #enquote} when it appears
251     * as the prefix of a longer string. This method assumes that the prefix is a valid quoted
252     * string; use {@link #dequote} to verify.
253     *
254     * @param string a string containing a prefix returned by {@link #enquote}
255     * @return length of {@code string} when enquoted
256     * @throws IllegalArgumentException if a starting or terminating quote character is not found
257     */
258    public static int enquotedLength(String string) {
259        int len = string.length();
260        if (len == 0 || string.charAt(0) != '"')
261            throw new IllegalArgumentException("invalid quoted string prefix: string does not begin with a quote character");
262        for (int i = 1; i < len; i++) {
263            if (string.charAt(i) == '"' && string.charAt(i - 1) != '\\')
264                return i + 1;
265        }
266        throw new IllegalArgumentException("invalid quoted string prefix: no terminating quote character found");
267    }
268
269    /**
270     * Determine if the given character is a valid XML character according to the XML 1.0 specification.
271     *
272     * <p>
273     * Valid characters are tab, newline, carriage return, and characters in the ranges
274     * <code>&#92;u0020 - &#92;ud7ff</code> and <code>&#92;ue000 - &#92;fffdf</code> (inclusive).
275     *
276     * @param ch character to test
277     * @return true if {@code ch} is valid within XML
278     * @see <a href="http://www.w3.org/TR/REC-xml/#charsets">The XML 1.0 Specification</a>
279     */
280    public static boolean isValidXMLChar(char ch) {
281        return (ch >= '\u0020' && ch <= '\ud7ff') || ch == '\n' || ch == '\r' || ch == '\t' || (ch >= '\ue000' && ch <= '\ufffd');
282    }

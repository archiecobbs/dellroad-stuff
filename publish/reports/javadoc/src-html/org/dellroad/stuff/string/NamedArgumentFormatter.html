006package org.dellroad.stuff.string;
007
008import java.util.ArrayList;
009import java.util.Collections;
010import java.util.HashMap;
011import java.util.Map;
012import java.util.Set;
013import java.util.regex.Matcher;
014import java.util.regex.Pattern;
015
016/**
017 * Interpreter for printf-style formatting strings that identifies arguments using names rather than indicies.
018 *
019 * <p>
020 * Works just like {@link java.util.Formatter} but arguments are specified using names instead of numbers.
021 * A mapping from argument name to argument value must be provided during the format operation.
022 *
023 * @since 1.0.64
024 */
025public class NamedArgumentFormatter {
026
027    private static final Pattern FORMAT_PARAM_PATTERN = Pattern.compile("%(\\w+)\\$");
028
029    /**
030     * The list of argument names in the order they are used. This is the inverse mapping of {@link #fieldMap}.
031     */
032    protected final ArrayList<String> fieldList = new ArrayList<String>();
033
034    /**
035     * Mapping from argument name to argument list index. This is the inverse mapping of {@link #fieldList}.
036     */
037    protected final HashMap<String, Integer> fieldMap = new HashMap<String, Integer>();
038
039    /**
040     * The original format string provided to the constructor.
041     */
042    protected final String originalFormat;
043
044    /**
045     * The modified format string containing indexes instead of argument names.
046     */
047    protected final String indexedFormat;
048
049    /**
050     * Constructor.
051     *
052     * @param format format string containing argument names instead of indicies
053     */
054    public NamedArgumentFormatter(final String format) {
055        Matcher matcher = FORMAT_PARAM_PATTERN.matcher(format);
056        StringBuilder buf = new StringBuilder();
057        for (int offset = 0; true; offset = matcher.end(1)) {
058
059            // Find next conversion
060            if (!matcher.find(offset)) {
061                buf.append(format.substring(offset));
062                break;
063            }
064
065            // Extract argument name
066            String fieldName = matcher.group(1);
067
068            // If we haven't seen this argument name yet, assign it the next parameter index
069            Integer fieldIndex = this.fieldMap.get(fieldName);
070            if (fieldIndex == null) {
071                fieldIndex = this.fieldList.size();
072                this.fieldMap.put(fieldName, fieldIndex);
073                this.fieldList.add(fieldName);
074            }
075
076            // Replace field name in format string with parameter index
077            buf.append(format.substring(offset, matcher.start(1)));
078            buf.append(fieldIndex + 1);
079        }
080        this.originalFormat = format;
081        this.indexedFormat = buf.toString();
082    }
083
084    /**
085     * Format the string using the given arguments.
086     *
087     * @param argMap mapping from argument name to argument value
088     * @return formatted string
089     * @throws java.util.IllegalFormatException if the format provided to the constructor contained illegal syntax
090     * @throws java.util.IllegalFormatException if an argument value is incompatible or missing (and null would be invalid)
091     */
092    public String format(Map<String, Object> argMap) {
093
094        // Put values into format parameter array
095        ArrayList<Object> parameterList = new ArrayList<Object>(this.fieldList.size());
096        for (String fieldName : this.fieldList)
097            parameterList.add(argMap.get(fieldName));
098
099        // Format string
100        return String.format(this.indexedFormat, parameterList.toArray());
101    }
102
103    /**
104     * Get the original format string provided to the constructor.
105     *
106     * @return original format string
107     */
108    public String getFormat() {
109        return this.originalFormat;
110    }
111
112    /**
113     * Get the argument names found in the configured format string.
114     *
115     * @return argument names as a unmodifiable set
116     */
117    public Set<String> getArgumentNames() {
118        return Collections.unmodifiableSet(this.fieldMap.keySet());
119    }
120}

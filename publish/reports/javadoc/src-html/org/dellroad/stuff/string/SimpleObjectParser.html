008import java.beans.BeanInfo;
009import java.beans.IndexedPropertyDescriptor;
010import java.beans.IntrospectionException;
011import java.beans.Introspector;
012import java.beans.PropertyDescriptor;
013import java.lang.reflect.Method;
014import java.util.Collections;
015import java.util.HashMap;
016import java.util.Map;
017import java.util.regex.Matcher;
018import java.util.regex.Pattern;
019import java.util.regex.PatternSyntaxException;
020
021import org.dellroad.stuff.java.Primitive;

023/**
024 * Parses strings using regular expressions into new instances of some class by parsing substrings.
025 * Primitive and String values are handled automatically. Other property types can be handled by
026 * overriding {@link #setProperty}.
027 *
028 * <a name="namedgroup"/>
029 * <p>
030 * This class supports parsing using a <i>named group regular expression</i>, which is pattern string
031 * using normal {@link Pattern} regular expression syntax with one additional grouping construct of the
032 * form <code>({property}...)</code>, allowing the Java bean property name to be specified inside the
033 * curly braces at the start of a grouped subexpression.
034 *
035 * <p>
036 * Instances of this class are immutable and thread-safe.
037 */
038public class SimpleObjectParser<T> {
039
040    private final Class<T> targetClass;
041    private final HashMap<String, PropertyDescriptor> propertyMap = new HashMap<String, PropertyDescriptor>();
042
043    /**
044     * Constructor.
045     *
046     * @param targetClass type of target object we will be parsing
047     */
048    public SimpleObjectParser(Class<T> targetClass) {
049        this.targetClass = targetClass;
050        this.buildPropertyMap();
051    }
052
053    /**
054     * Get the target class.
055     */
056    public Class<T> getTargetClass() {
057        return this.targetClass;
058    }
059
060    /**
061     * Same as {@link #parse(Object, String, String, boolean)} but this method creates the target instance using
062     * the target type's default constructor.
063     *
064     * @throws RuntimeException if a new target instance cannot be created using the default constructor
065     * @since 1.0.85
066     */
067    public T parse(String text, String regex, boolean allowSubstringMatch) {
068        T target;
069        try {
070            target = this.targetClass.newInstance();
071        } catch (Exception e) {
072            throw new RuntimeException("can't create instance of " + this.targetClass + " using default constructor", e);
073        }
074        return this.parse(target, text, regex, allowSubstringMatch);
075    }
076
077    /**
078     * Parse the given text using the provided <i>named group regular expression</i>.
079     *
080     * <p>
081     * This method assumes the following about {@code regex}:
082     * <ul>
083     * <li>All instances of an opening parenthesis not preceded by a backslash are actual grouped sub-expressions</li>
084     * <li>In particular, all instances of substrings like <code>({foo}</code> are actual named group sub-expressions</li>
085     * </ul>
086     *
087     * @param target              target instance
088     * @param text                string to parse
089     * @param regex               named group regular expression containing object property names
090     * @param allowSubstringMatch if false, entire text must match, otherwise only a (the first) substring need match
091     * @return parsed object or null if parse fails
092     * @throws PatternSyntaxException if the regular expression with the named group property names removed is invalid
093     * @throws PatternSyntaxException if this method cannot successfully parse the regular expression
094     * @throws IllegalArgumentException if a named group specfies a property that is not a parseable
095     *                                  property of this instance's target class
096     * @since 1.0.95
097     */
098    public T parse(T target, String text, String regex, boolean allowSubstringMatch) {
099
100        // Scan regular expression for named sub-groups and parse them out
101        HashMap<Integer, String> patternMap = new HashMap<Integer, String>();
102        StringBuilder buf = new StringBuilder(regex.length());
103        Pattern namedGroup = Pattern.compile("\\(\\{(\\p{javaJavaIdentifierStart}\\p{javaJavaIdentifierPart}*)\\}");
104        Matcher matcher = namedGroup.matcher(regex);
105        int pos = 0;
106        int groupCount = 0;
107        while (true) {
108            int match = matcher.find(pos) ? matcher.start() : regex.length();
109            String chunk = regex.substring(pos, match);
110            for (int i = 0; i < chunk.length(); i++) {
111                if (chunk.charAt(i) == '('
112                  && (i == 0 || chunk.charAt(i - 1) != '\\')
113                  && (i == chunk.length() - 1 || chunk.charAt(i + 1) != '?'))
114                    groupCount++;
115            }
116            buf.append(chunk);
117            if (match == regex.length())
118                break;
119            buf.append('(');
120            patternMap.put(++groupCount, matcher.group(1));
121            pos = matcher.end();
122        }
123
124        // Sanity check our parse attempt
125        Pattern pattern = Pattern.compile(buf.toString());
126        int numGroups = pattern.matcher("").groupCount();
127        if (numGroups != groupCount) {
128            throw new PatternSyntaxException("the given regular expression is not supported (counted "
129              + groupCount + " != " + numGroups + " groups)", regex, 0);
130        }
131
132        // Proceed
133        return this.parse(target, text, pattern, patternMap, allowSubstringMatch);
134    }
135
136    /**
137     * Same as {@link #parse(Object, String, Pattern, Map, boolean)} but this method creates the target instance using
138     * the target type's default constructor.
139     *
140     * @throws RuntimeException if a new target instance cannot be created using the default constructor
141     * @since 1.0.85
142     */
143    public T parse(String text, Pattern pattern, Map<Integer, String> patternMap, boolean allowSubstringMatch) {
144        T target;
145        try {
146            target = this.targetClass.newInstance();
147        } catch (Exception e) {
148            throw new RuntimeException("can't create instance of " + this.targetClass + " using default constructor", e);
149        }
150        return this.parse(target, text, pattern, patternMap, allowSubstringMatch);
151    }
152
153    /**
154     * Parse the given text using the provided pattern and mapping from pattern sub-group to Java bean property name.
155     *
156     * @param target              target instance
157     * @param text                string to parse
158     * @param pattern             pattern with substring matching groups that match object properties
159     * @param patternMap          mapping from pattern substring group index to object property name
160     * @param allowSubstringMatch if false, entire text must match, otherwise only a (the first) substring need match
161     * @return parsed object or null if parse fails
162     * @throws IllegalArgumentException if the map contains a property that is not a parseable
163     *                                  property of this instance's target class
164     * @throws IllegalArgumentException if a subgroup index key in patternMap is out of bounds
165     * @since 1.0.95
166     */
167    public T parse(T target, String text, Pattern pattern, Map<Integer, String> patternMap, boolean allowSubstringMatch) {
168
169        // Compose given map with target class' property map
170        HashMap<Integer, PropertyDescriptor> subgroupMap = new HashMap<Integer, PropertyDescriptor>();
171        for (Map.Entry<Integer, String> entry : patternMap.entrySet()) {
172            String propName = entry.getValue();
173            PropertyDescriptor property = this.propertyMap.get(propName);
174            if (property == null)
175                throw new IllegalArgumentException("parseable property \"" + propName + "\" not found in " + this.targetClass);
176            subgroupMap.put(entry.getKey(), property);
177        }
178
179        // Attempt to match the string
180        Matcher matcher = pattern.matcher(text);
181        boolean matches = allowSubstringMatch ? matcher.find() : matcher.matches();
182        if (!matches)
183            return null;
184
185        // Set fields based on matching substrings
186        for (Map.Entry<Integer, PropertyDescriptor> entry : subgroupMap.entrySet()) {
187
188            // Get substring
189            String substring;
190            try {
191                substring = matcher.group(entry.getKey());
192            } catch (IndexOutOfBoundsException e) {
193                throw new IllegalArgumentException(
194                  "regex subgroup " + entry.getKey() + " does not exist in pattern `" + pattern + "'");
195            }
196
197            // If substring was not matched, don't set property
198            if (substring == null)
199                continue;
200
201            // Set property from substring
202            this.setProperty(target, entry.getValue(), substring);
203        }
204
205        // Post-process
206        this.postProcess(target);
207
208        // Done
209        return target;
210    }
211
212    /**
213     * Get the mapping from property name to setter method.
214     */
215    public Map<String, PropertyDescriptor> getPropertyMap() {
216        return Collections.unmodifiableMap(this.propertyMap);
217    }
218
219    /**
220     * Set a property value.
221     * <p/>
222     * <p>
223     * The implementation in {@link SimpleObjectParser} simply invokes {@link #setSimpleProperty}.
224     * Other property types can be handled by overriding this method.
225     * </p>
226     *
227     * @param obj       newly created instance
228     * @param property  descriptor for the property being set
229     * @param substring matched substring
230     * @throws IllegalArgumentException if substring cannot be successfully parsed
231     * @throws IllegalArgumentException if an exception is thrown attempting to set the property
232     */
233    public void setProperty(T obj, PropertyDescriptor property, String substring) {
234        this.setSimpleProperty(obj, property, substring);
235    }
236
237    /**
238     * Set a primitive or string property value.
239     * <p/>
240     * <p>
241     * The implementation in {@link SimpleObjectParser} handles primitives using the corresponding
242     * {@code valueOf} method; String values are handled by setting the value directly.
243     * </p>
244     *
245     * @throws IllegalArgumentException if property is not a primitive or String property.
246     * @throws IllegalArgumentException if substring cannot be successfully parsed (if primitive)
247     * @throws IllegalArgumentException if an exception is thrown attempting to set the property
248     */
249    public void setSimpleProperty(T obj, PropertyDescriptor property, String substring) {
250
251        // Parse substring
252        Object value;
253        if (property.getPropertyType() == String.class)
254            value = substring;
255        else {
256            Primitive<?> prim = Primitive.get(property.getPropertyType());
257            if (prim == null) {
258                throw new IllegalArgumentException(
259                  "property `" + property.getName() + "' of " + this.targetClass + " is not a primitive or String");
260            }
261            value = prim.parseValue(substring);
262        }
263
264        // Set value
265        try {
266            property.getWriteMethod().invoke(obj, value);
267        } catch (Exception e) {
268            throw new IllegalArgumentException("can't set property `" + property.getName() + "' of " + this.targetClass, e);
269        }
270    }
271
272    /**
273     * Post-process newly created instances. The instance's properties will have already been
274     * set by a successful parse.
275     * <p/>
276     * <p>
277     * The implementation in {@link SimpleObjectParser} does nothing. Subclasses may override if needed.
278     * </p>
279     */
280    protected void postProcess(T obj) {
281    }
282
283    private void buildPropertyMap() {
284
285        // Introspect target class
286        BeanInfo beanInfo;
287        try {
288            beanInfo = Introspector.getBeanInfo(this.targetClass);
289        } catch (IntrospectionException e) {
290            throw new RuntimeException(e);
291        }
292
293        // Build map from property name -> setter method
294        for (PropertyDescriptor property : beanInfo.getPropertyDescriptors()) {
295            if (property instanceof IndexedPropertyDescriptor)
296                continue;
297            Method setter = property.getWriteMethod();
298            if (setter == null)
299                continue;
300            this.propertyMap.put(property.getName(), property);
301        }
302    }
303}

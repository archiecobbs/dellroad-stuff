006package org.dellroad.stuff.string;
007
008/**
009 * Encodes {@code byte[]} arrays to and from hexadecimal strings.
010 */
011public final class ByteArrayEncoder {
012
013    private ByteArrayEncoder() {
014    }
015
016    /**
017     * Encode a {@code byte[]} array as a {@link String}.
018     * Equivalent to:
019     * <blockquote>
020     * <code>encode(array, 0, array.length)</code>
021     * </blockquote>
022     *
023     * @param array byte array, or {@code null}
024     * @return hexadecimal string (or {@code null} if {@code array} was {@code null})
025     */
026    public static String encode(byte[] array) {
027        return encode(array, 0, array.length);
028    }
029
030    /**
031     * Encode a {@code byte[]} array as a {@link String}.
032     *
033     * @param array byte array, or {@code null}
034     * @param off   offset into the array
035     * @param len   number of bytes to convert
036     * @return hexadecimal string (or {@code null} if {@code array} was {@code null})
037     * @throws IndexOutOfBoundsException if array bounds are exceeded
038     * @throws IllegalArgumentException  if {@code len} is greater than {@code Integer.MAX_VALUE / 2}
039     */
040    public static String encode(byte[] array, int off, int len) {
041
042        // Check for null
043        if (array == null)
044            return null;
045
046        // Check bounds
047        if (off < 0 || len < 0 || off + len < 0 || off + len > array.length)
048            throw new IndexOutOfBoundsException("array bounds exceeded");
049
050        // Encode bytes
051        if (len > Integer.MAX_VALUE / 2)
052            throw new IllegalArgumentException("array is too long");
053        char[] buf = new char[len * 2];
054        for (int i = 0; i < len; i++) {
055            int value = array[off + i] & 0xff;
056            buf[i * 2] = Character.forDigit(value >> 4, 16);
057            buf[i * 2 + 1] = Character.forDigit(value & 0xf, 16);
058        }
059
060        // Done
061        return new String(buf);
062    }
063
064    /**
065     * Decode a {@link String} back into a {@code byte[]} array.
066     * Any extra whitespace in the string is ignored.
067     *
068     * @param text string previously encoded by {@link #encode}, or {@code null}
069     * @return original {@code byte[]} array (or {@code null} if {@code text} was {@code null})
070     * @throws IllegalArgumentException if any invalid non-whitespace characters are seen, or the number of hex digits is odd
071     */
072    public static byte[] decode(String text) {
073
074        // Check for null
075        if (text == null)
076            return null;
077
078        // Allocate array
079        byte[] array = new byte[text.length() / 2];
080
081        // Parse bytes
082        int len = 0;
083        boolean flipflop = false;
084        int prevNibble = 0;
085        final int limit = text.length();
086        for (int pos = 0; pos < limit; pos++) {
087            char ch = text.charAt(pos);
088            if (Character.isWhitespace(ch))
089                continue;
090            int nibble = Character.digit(ch, 16);
091            if (nibble == -1)
092                throw new IllegalArgumentException("invalid character '" + ch + "' in byte array");
093            if (flipflop)
094                array[len++] = (byte)((prevNibble << 4) | nibble);
095            else
096                prevNibble = nibble;
097
098            // bitwise inversion of boolean
099            flipflop ^= true;
100        }
101        if (flipflop)
102            throw new IllegalArgumentException("byte array has an odd number of digits");
103
104        // Account for any squeezed-out whitespace
105        if (len < array.length) {
106            byte[] temp = new byte[len];
107            System.arraycopy(array, 0, temp, 0, len);
108            array = temp;
109        }
110
111        // Done
112        return array;
113    }
114}

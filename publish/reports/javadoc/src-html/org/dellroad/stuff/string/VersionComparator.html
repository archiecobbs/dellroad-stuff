006package org.dellroad.stuff.string;
007
008import java.util.ArrayList;
009import java.util.Comparator;
010import java.util.List;
011import java.util.regex.Matcher;
012import java.util.regex.Pattern;
013
014/**
015 * Comparator for version numbers.
016 *
017 * <p>
018 * Version numbers are broken up into <b>parts</b>, where a <b>part</b> is a contiguous sequence
019 * of one or more digits, or a contiguous sequence of one or more non-digits, and where the period character
020 * ('.') serves as a part separator. Then, parts are compared pair-wise until there is a difference, as
021 * determined by the {@link Comparator} provided to the constructor, or one version string runs out of parts,
022 * in which case it is less than the other.
023 * </p>
024 */
025public class VersionComparator implements Comparator<String> {
026
027    private static final Pattern PART_PATTERN = Pattern.compile("[0-9]+|[^0-9.]+|\\.");
028
029    private final Comparator<String> partComparator;
030
031    /**
032     * Primary constructor.
033     *
034     * @param partComparator compares individual parts
035     * @throws IllegalArgumentException if {@code partComparator} is null
036     */
037    public VersionComparator(Comparator<String> partComparator) {
038        if (partComparator == null)
039            throw new IllegalArgumentException("null partComparator");
040        this.partComparator = partComparator;
041    }
042
043    /**
044     * Convenience constructor.
045     *
046     * <p>
047     * Equivalent to <code>VersionComparator(new VersionPartComparator())</code>.
048     */
049    public VersionComparator() {
050        this(new VersionPartComparator());
051    }
052
053    @Override
054    public int compare(String v1, String v2) {
055
056        // Chop up version into parts, where a part is a contiguous all-digit or all-non-digit sequence
057        List<List<String>> partsList = new ArrayList<List<String>>(2);
058        partsList.add(this.separateIntoParts(v1));
059        partsList.add(this.separateIntoParts(v2));
060
061        // Compare parts pair-wise
062        for (int i = 0; true; i++) {
063            final boolean hasPart1 = i < partsList.get(0).size();
064            final boolean hasPart2 = i < partsList.get(1).size();
065            if (!hasPart1 && !hasPart2)
066                return 0;
067            else if (hasPart1 && !hasPart2)
068                return 1;
069            else if (!hasPart1 && hasPart2)
070                return -1;
071            final String part1 = partsList.get(0).get(i);
072            final String part2 = partsList.get(1).get(i);
073            int diff = this.partComparator.compare(part1, part2);
074            if (diff != 0)
075                return diff;
076        }
077    }
078
079    /**
080     * Split a version string into parts.
081     *
082     * <p>
083     * The implementation in {@link VersionComparator} defines a <b>part</b> as a contiguous sequence
084     * of one or more digits, or a contiguous sequence of one or more non-digits, and where the period character
085     * ('.') serves as a part separator.
086     * </p>
087     *
088     * <p>
089     * Subclasses may choose to override this method to define parts differently.
090     * </p>
091     *
092     * @param version the original version string
093     * @return the parts of the version string
094     */
095    protected List<String> separateIntoParts(String version) {
096        final ArrayList<String> partsList = new ArrayList<String>();
097        int pos = 0;
098        for (Matcher matcher = PART_PATTERN.matcher(version); matcher.find(pos); pos = matcher.end()) {
099            final String part = matcher.group();
100            if (!part.equals("."))
101                partsList.add(part);
102            else if (matcher.end() == version.length() || version.charAt(matcher.end()) == '.')
103                partsList.add("");
104        }
105        return partsList;
106    }
107}

006package org.dellroad.stuff.validation;
007
008import java.util.Collection;
009import java.util.Map;
010import java.util.Set;
011
012import javax.validation.ConstraintValidatorContext;
013
014/**
015 * Validator for the {@link Unique @Unique} constraint.
016 *
017 * @see Unique
018 */
019public class UniqueValidator extends AbstractValidator<Unique, Object> {
020
021    private String domain;
022    private Uniquifier<Object> uniquifier;
023
024    @Override
025    @SuppressWarnings("unchecked")
026    public void initialize(Unique annotation) {
027        super.initialize(annotation);
028        this.domain = annotation.domain();
029        Class<? extends Uniquifier<?>> uniquifierClass = annotation.uniquifier();
030        try {
031            this.uniquifier = (Uniquifier<Object>)uniquifierClass.newInstance();
032        } catch (Exception e) {
033            throw new RuntimeException("can't create an instance of " + uniquifierClass + ": " + e.getMessage(), e);
034        }
035    }
036
037    @Override
038    public boolean isValid(final Object propertyValue, ConstraintValidatorContext context) {
039
040        // Ignore null values
041        if (propertyValue == null)
042            return true;
043
044        // Recurse on collections and maps
045        if (propertyValue instanceof Collection)
046            return this.isCollectionValid((Collection<?>)propertyValue, context);
047        if (propertyValue instanceof Map)
048            return this.isCollectionValid(((Map<?, ?>)propertyValue).values(), context);
049
050        // If we are only validating a sub-tree then skip this check
051        ValidationContext<?> validationContext;
052        try {
053            validationContext = ValidationContext.getCurrentContext();
054        } catch (ClassCastException e) {
055            return true;
056        }
057
058        // Uniquify value
059        Object uniqueValue;
060        try {
061            uniqueValue = this.uniquifier.getUniqued(propertyValue);
062        } catch (ClassCastException e) {
063            throw new RuntimeException("uniquifier " + this.annotation.uniquifier()
064              + " does not support uniquifying values of type " + propertyValue.getClass());
065        }
066
067        // Ignore null values
068        if (uniqueValue == null)
069            return true;
070
071        // Get the domain
072        final Set<Object> uniqueDomain = validationContext.getUniqueDomain(this.domain);
073
074        // Verify value is unique XXX: this assumes isValid() is invoked only once per property
075        if (!uniqueDomain.add(uniqueValue)) {
076            this.setViolation(context, "non-unique value `" + uniqueValue + "' in uniqueness domain `" + this.domain + "'");
077            return false;
078        }
079
080        // Done
081        return true;
082    }
083}

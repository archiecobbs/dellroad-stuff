006package org.dellroad.stuff.validation;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Retention;
011import java.lang.annotation.RetentionPolicy;
012import java.lang.annotation.Target;
013
014import javax.validation.Constraint;
015import javax.validation.Payload;
016import javax.validation.ReportAsSingleViolation;
017
018/**
019 * Works like the standard {@link javax.validation.constraints.Pattern @Pattern} but applies to any
020 * type of object, converting to {@link String} as necessary via {@link Object#toString}, and recursing
021 * on collection types.
022 */
023@Documented
024@Constraint(validatedBy = PatternValidator.class)
025@Target({ ElementType.METHOD, ElementType.FIELD, ElementType.ANNOTATION_TYPE })
026@Retention(RetentionPolicy.RUNTIME)
027@ReportAsSingleViolation
028public @interface Pattern {
029
030    String message() default "Does not match the pattern \"{regexp}\"";
031
032    Class<?>[] groups() default {};
033
034    Class<? extends Payload>[] payload() default {};
035
036    /**
037     * Regular expression that must be matched.
038     */
039    String regexp();
040
041    /**
042     * Regular expression flags.
043     */
044    javax.validation.constraints.Pattern.Flag[] flags() default {};
045}

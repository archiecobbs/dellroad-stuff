006package org.dellroad.stuff.validation;
007
008import java.util.Collection;
009import java.util.regex.PatternSyntaxException;
010
011import javax.validation.ConstraintDeclarationException;
012import javax.validation.ConstraintValidatorContext;
013
014/**
015 * Validator for the @{@link Pattern} constraint.
016 *
017 * @see Pattern
018 */
019public class PatternValidator extends AbstractValidator<Pattern, Object> {
020
021    private java.util.regex.Pattern pattern;
022
023    @Override
024    public void initialize(Pattern pattern) {
025        super.initialize(pattern);
026
027        // Get flags
028        int flags = 0;
029        for (javax.validation.constraints.Pattern.Flag flag : this.annotation.flags())
030            flags |= flag.getValue();
031
032        // Compile regular expression
033        try {
034            this.pattern = java.util.regex.Pattern.compile(this.annotation.regexp(), flags);
035        } catch (PatternSyntaxException e) {
036            throw new ConstraintDeclarationException("Invalid regular expression `" + annotation.regexp() + "': " + e.getMessage());
037        }
038    }
039
040    @Override
041    public boolean isValid(Object value, ConstraintValidatorContext context) {
042
043        // Handle null
044        if (value == null)
045            return true;
046
047        // Handle collection
048        if (value instanceof Collection)
049            return this.isCollectionValid((Collection<?>)value, context);
050
051        // Check pattern
052        return this.pattern.matcher(value.toString()).matches();
053    }
054}

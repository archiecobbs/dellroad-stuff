006package org.dellroad.stuff.validation;
007
008/**
009 * Converts values into some other object that has the desired uniqueness.
010 *
011 * @param <T> the type of object being validated
012 * @see Unique @Unique
013 */
014public interface Uniquifier<T> {
015
016    /**
017     * Get an object representing the unique value of the given instance.
018     * In other words, if and only two values are "the same", this method should return object(s) that
019     * match (according to {@link Object#equals equals()} and {@link Object#hashCode}).
020     *
021     * <p>
022     * This method may return a value of {@code null} to indicate that {@code value} does not in fact
023     * need to be unique.
024     * </p>
025     *
026     * @param value the value to be uniquified
027     * @return unique representative, or {@code null} to except {@code value} from the uniqueness requirement
028     * @throws ClassCastException if {@code value} is not of a supported type
029     */
030    Object getUniqued(T value);
031}

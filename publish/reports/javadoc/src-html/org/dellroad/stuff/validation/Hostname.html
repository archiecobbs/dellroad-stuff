006package org.dellroad.stuff.validation;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Retention;
011import java.lang.annotation.RetentionPolicy;
012import java.lang.annotation.Target;
013
014import javax.validation.Constraint;
015import javax.validation.Payload;
016import javax.validation.ReportAsSingleViolation;
017import javax.validation.constraints.Pattern;
018
019/**
020 * Validation constraint requiring a {@link String} to be a valid DNS hostname (or hostname component).
021 */
022@Documented
023@Constraint(validatedBy = {})
024@Target({ ElementType.METHOD, ElementType.FIELD, ElementType.ANNOTATION_TYPE })
025@Retention(RetentionPolicy.RUNTIME)
026@Pattern(regexp = "^(([\\p{Alnum}]([-\\p{Alnum}]*[\\p{Alnum}])?)\\.)*([\\p{Alpha}]([-\\p{Alnum}]*[\\p{Alnum}])?)$")
027@ReportAsSingleViolation
028public @interface Hostname {
029
030    String message() default "Invalid hostname or hostname component";
031
032    Class<?>[] groups() default {};
033
034    Class<? extends Payload>[] payload() default {};
035}

006package org.dellroad.stuff.validation;
007
008import java.util.HashMap;
009import java.util.HashSet;
010import java.util.Set;
011import java.util.concurrent.Callable;
012
013import javax.validation.ConstraintViolation;
014import javax.validation.Validation;
015import javax.validation.Validator;
016
017import org.dellroad.stuff.java.ThreadLocalHolder;
018
019/**
020 * Provides additional context for {@link javax.validation.ConstraintValidator} implementations.
021 *
022 * <p>
023 * {@link ValidationContext} gives {@link javax.validation.ConstraintValidator} implementations access to the root object
024 * being validated. This breaks the usual principle of locality for validation (i.e., that validation of a specific bean
025 * proceeds unaware of that bean's parents) but it can make custom validators more convenient to implement.
026 * Subclasses are encouraged to provide additional application-specific information.
027 *
028 * <p>
029 * Validation must be performed via {@link #validate(Validator) validate()} for this class to work.
030 *
031 * @param <T> type of the root object being validated
032 */
033public class ValidationContext<T> {
034
035    private static final ThreadLocalHolder<ValidationContext<?>> CURRENT = new ThreadLocalHolder<ValidationContext<?>>();
036
037    private final HashMap<String, Set<Object>> uniqueDomainMap = new HashMap<String, Set<Object>>();
038    private final T root;
039
040    /**
041     * Construct a new validation context configured to validate the given root object.
042     *
043     * @param root root object to be validated
044     * @throws IllegalArgumentException if {@code root} is null
045     */
046    public ValidationContext(T root) {
047        if (root == null)
048            throw new IllegalArgumentException("null root");
049        this.root = root;
050    }
051
052    /**
053     * Get the root object associated with this instance.
054     */
055    public final T getRoot() {
056        return this.root;
057    }
058
059    /**
060     * Validate this instance's root object. This is a convenience method, equivalent to:
061     *  <blockquote>
062     *  <code>{@link #validate(Validator) validate}(Validation.buildDefaultValidatorFactory().getValidator())</code>
063     *  <blockquote>
064     *
065     * @throws IllegalStateException if this method is invoked re-entrantly
066     */
067    public Set<ConstraintViolation<T>> validate() {
068        return this.validate(Validation.buildDefaultValidatorFactory().getValidator());
069    }
070
071    /**
072     * Validate this instance's root object using the given {@link Validator}, making the context
073     * available to the current thread during the validation process via {@link #getCurrentContext}.
074     *
075     * @throws IllegalStateException if this method is invoked re-entrantly
076     */
077    public Set<ConstraintViolation<T>> validate(final Validator validator) {
078
079        // Sanity check
080        if (ValidationContext.CURRENT.get() != null)
081            throw new IllegalStateException("re-entrant invocation is not allowed");
082
083        // Validate
084        try {
085            return ValidationContext.CURRENT.invoke(this, new Callable<Set<ConstraintViolation<T>>>() {
086                @Override
087                public Set<ConstraintViolation<T>> call() {
088                    return validator.validate(ValidationContext.this.root);
089                }
090            });
091        } catch (RuntimeException e) {
092            throw e;
093        } catch (Exception e) {
094            throw new RuntimeException(e);
095        }
096    }
097
098    /**
099     * Get the {@link ValidationContext} associated with the current thread.
100     * This method is only valid during invocations of {@link #validate(Validator) validate()}.
101     *
102     * @return current {@link ValidationContext}
103     * @throws IllegalStateException if {@link #validate(Validator) validate()} is not currently executing
104     */
105    public static ValidationContext<?> getCurrentContext() {
106        ValidationContext<?> context = ValidationContext.CURRENT.get();
107        if (context == null)
108            throw new IllegalStateException("current thread is not executing validate()");
109        return context;
110    }
111
112    /**
113     * Get the {@link ValidationContext} associated with the current thread, cast to the desired type.
114     * This method is only valid during invocations of {@link #validate(Validator) validate()}.
115     *
116     * @param type required type
117     * @return current {@link ValidationContext}
118     * @throws IllegalStateException if {@link #validate(Validator) validate()} is not currently executing
119     * @throws ClassCastException if the current {@link ValidationContext} is not of type {@code type}
120     * @throws NullPointerException if {@code type} is null
121     */
122    public static <V extends ValidationContext<?>> V getCurrentContext(Class<V> type) {
123        return type.cast(ValidationContext.getCurrentContext());
124    }
125
126    /**
127     * Convenience method to get the root object being validated by the current thread.
128     * This method is only valid during invocations of {@link #validate(Validator) validate()}.
129     *
130     * @return current validation root object
131     * @throws IllegalStateException if {@link #validate(Validator) validate()} is not currently executing
132     */
133    public static Object getCurrentRoot() {
134        return ValidationContext.getCurrentContext().getRoot();
135    }
136
137    /**
138     * Convenience method to get the root object being validated by the current thread, cast to the desired type.
139     * This method is only valid during invocations of {@link #validate(Validator) validate()}.
140     *
141     * @return current validation root object
142     * @throws IllegalStateException if {@link #validate(Validator) validate()} is not currently executing
143     * @throws ClassCastException if the current validation root is not of type {@code type}
144     * @throws NullPointerException if {@code type} is null
145     */
146    public static <T> T getCurrentRoot(Class<T> type) {
147        return type.cast(ValidationContext.getCurrentRoot());
148    }
149
150    /**
151     * Get the uniqueness domain with the given name. Used to validate {@link Unique @Unique} constraints.
152     *
153     * @param domain name of the uniqueness domain
154     * @return set containing the unique values that have already been seen during this validation check
155     */
156    public Set<Object> getUniqueDomain(String domain) {
157        Set<Object> uniqueDomain = this.uniqueDomainMap.get(domain);
158        if (uniqueDomain == null) {
159            uniqueDomain = new HashSet<Object>();
160            this.uniqueDomainMap.put(domain, uniqueDomain);
161        }
162        return uniqueDomain;
163    }
164}

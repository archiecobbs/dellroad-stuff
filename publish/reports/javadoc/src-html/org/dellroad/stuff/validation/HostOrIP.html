006package org.dellroad.stuff.validation;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Retention;
011import java.lang.annotation.RetentionPolicy;
012import java.lang.annotation.Target;
013
014import javax.validation.Constraint;
015import javax.validation.Payload;
016import javax.validation.ReportAsSingleViolation;
017import javax.validation.constraints.Pattern;
018
019/**
020 * Validation constraint requiring a {@link String} to be either a valid fully-qualified DNS hostname
021 * or an IPv4 address.
022 */
023@Documented
024@Constraint(validatedBy = {})
025@Target({ ElementType.METHOD, ElementType.FIELD, ElementType.ANNOTATION_TYPE })
026@Retention(RetentionPolicy.RUNTIME)
027@Pattern(regexp =
028    "^((([\\p{Alnum}]([-\\p{Alnum}]*[\\p{Alnum}])?)\\.)+([\\p{Alpha}]([-\\p{Alnum}]*[\\p{Alnum}])?))"
029  + "|("
030    + "([0-9]|[1-9][0-9]|1[0-9][0-9]|2[0-4][0-9]|25[0-5])" + "\\."
031    + "([0-9]|[1-9][0-9]|1[0-9][0-9]|2[0-4][0-9]|25[0-5])" + "\\."
032    + "([0-9]|[1-9][0-9]|1[0-9][0-9]|2[0-4][0-9]|25[0-5])" + "\\."
033    + "([0-9]|[1-9][0-9]|1[0-9][0-9]|2[0-4][0-9]|25[0-5])"
034  + ")$")
035@ReportAsSingleViolation
036public @interface HostOrIP {
037
038    String message() default "Not a valid hostname or IPv4 address";
039
040    Class<?>[] groups() default {};
041
042    Class<? extends Payload>[] payload() default {};
043}

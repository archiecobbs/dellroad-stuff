006package org.dellroad.stuff.validation;
007
008import java.lang.annotation.Annotation;
009import java.util.Collection;
010
011import javax.validation.ConstraintValidator;
012import javax.validation.ConstraintValidatorContext;
013
014/**
015 * Support superclass for validators.
016 */
017public abstract class AbstractValidator<C extends Annotation, T> implements ConstraintValidator<C, T> {
018
019    /**
020     * The constraint being checked by this instance.
021     */
022    protected C annotation;
023
024    @Override
025    public void initialize(C annotation) {
026        this.annotation = annotation;
027    }
028
029    /**
030     * Convenience method to add a constraint violation described by {@code message} and disable the default violation.
031     */
032    protected void setViolation(ConstraintValidatorContext context, String message) {
033        context.disableDefaultConstraintViolation();
034        context.buildConstraintViolationWithTemplate(message).addConstraintViolation();
035    }
036
037    /**
038     * Apply this constraint to all values in a collection. This is a convenience method for validators
039     * that want to work with both simple properties and collection properties.
040     */
041    protected boolean isCollectionValid(Collection<? extends T> collection, ConstraintValidatorContext context) {
042        boolean result = true;
043        for (T value : collection) {
044            if (!this.isValid(value, context))
045                result = false;
046        }
047        return result;
048    }
049}

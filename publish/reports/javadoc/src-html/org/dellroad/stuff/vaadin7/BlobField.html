006package org.dellroad.stuff.vaadin7;
007
008import com.vaadin.data.Property;
009import com.vaadin.ui.Alignment;
010import com.vaadin.ui.Component;
011import com.vaadin.ui.CustomField;
012import com.vaadin.ui.HorizontalLayout;
013import com.vaadin.ui.Label;
014import com.vaadin.ui.Notification;
015import com.vaadin.ui.ProgressBar;
016import com.vaadin.ui.Upload;
017
018import java.io.ByteArrayOutputStream;
019import java.io.OutputStream;
020
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;
023
024/**
025 * A Vaadin {@link com.vaadin.ui.Field} for editing {@link Property}s with {@code byte[]} array values
026 * (i.e., "blobs"). The blob is updated using a file upload.
027 */
028@SuppressWarnings("serial")
029public class BlobField extends CustomField<byte[]> implements Upload.Receiver,
030  Upload.StartedListener, Upload.ProgressListener, Upload.SucceededListener, Upload.FailedListener {
031
032    protected final Logger log = LoggerFactory.getLogger(this.getClass());
033
034    private final Label descriptionLabel = new Label();
035    private final HorizontalLayout layout = new HorizontalLayout();
036    private final ByteArrayOutputStream buffer = new ByteArrayOutputStream();
037    private final ProgressBar progressBar = new ProgressBar();
038    private final Upload upload = new Upload();
039
040    /**
041     * Constructor.
042     */
043    public BlobField() {
044        this.addValueChangeListener(new Property.ValueChangeListener() {
045            @Override
046            public void valueChange(Property.ValueChangeEvent event) {
047                BlobField.this.updateDisplay();
048            }
049        });
050        this.addReadOnlyStatusChangeListener(new Property.ReadOnlyStatusChangeListener() {
051            @Override
052            public void readOnlyStatusChange(Property.ReadOnlyStatusChangeEvent event) {
053                BlobField.this.updateDisplay();
054            }
055        });
056    }
057
058    @Override
059    public void attach() {
060        super.attach();
061        this.updateDisplay();
062    }
063
064    // Cancel any upload in progress if we are detached
065    @Override
066    public void detach() {
067        super.detach();
068        this.upload.interruptUpload();
069    }
070
071    /**
072     * Get the {@link Upload} associated with this instance.
073     */
074    public Upload getUpload() {
075        return this.upload;
076    }
077
078    /**
079     * Get the description {@link Label} associated with this instance.
080     */
081    public Label getDescriptionLabel() {
082        return this.descriptionLabel;
083    }
084
085// CustomField
086
087    @Override
088    public Class<byte[]> getType() {
089        return byte[].class;
090    }
091
092    @Override
093    protected Component initContent() {
094
095        // Initialize layout
096        this.layout.setMargin(false);
097        this.layout.setSpacing(true);
098
099        // Add description
100        this.descriptionLabel.setSizeUndefined();
101        this.layout.addComponent(this.descriptionLabel);
102        this.layout.setComponentAlignment(this.descriptionLabel, Alignment.MIDDLE_LEFT);
103        this.layout.addComponent(new Label("\u00a0\u00a0"));
104
105        // Add upload
106        this.upload.setReceiver(this);
107        this.upload.addStartedListener(this);
108        this.upload.addProgressListener(this);
109        this.upload.addSucceededListener(this);
110        this.upload.addFailedListener(this);
111        this.upload.setImmediate(true);
112        this.layout.addComponent(this.upload);
113
114        // Add progress bar
115        this.progressBar.setIndeterminate(false);
116        this.progressBar.setVisible(false);
117        this.layout.addComponent(this.progressBar);
118        this.layout.setComponentAlignment(this.progressBar, Alignment.MIDDLE_LEFT);
119        this.layout.setExpandRatio(this.progressBar, 1.0f);
120
121        // Done
122        return layout;
123    }
124
125    @Override
126    protected void setInternalValue(byte[] value) {
127        super.setInternalValue(value);
128        this.updateDisplay();
129    }
130
131// Upload.Receiver
132
133    @Override
134    public OutputStream receiveUpload(String filename, String mimeType) {
135        return this.buffer;
136    }
137
138// Upload.StartedListener
139
140    @Override
141    public void uploadStarted(Upload.StartedEvent event) {
142        this.log.info("started upload of file `" + event.getFilename() + "'");
143        this.progressBar.setValue(0.0f);
144        this.progressBar.setVisible(true);
145        this.updateDisplay();
146    }
147
148// Upload.ProgressListener
149
150    @Override
151    public void updateProgress(long readBytes, long contentLength) {
152        float fraction = (float)readBytes / (float)contentLength;
153        fraction = Math.max(fraction, 0.0f);
154        fraction = Math.min(fraction, 1.0f);
155        fraction = (int)(fraction * 64.0f) / 64.0f;                 // quantize fraction to avoid zillions of PUSH updates
156        this.progressBar.setValue(fraction);
157    }
158
159// Upload.SucceededListener
160
161    @Override
162    public void uploadSucceeded(Upload.SucceededEvent event) {
163        final byte[] data = this.buffer.toByteArray();
164        this.log.info("finished upload of file `" + event.getFilename() + "' (" + data.length + " bytes)");
165        this.buffer.reset();
166        this.setValue(data);
167        this.progressBar.setVisible(false);
168        this.updateDisplay();
169    }
170
171// Upload.FailedListener
172
173    @Override
174    public void uploadFailed(Upload.FailedEvent event) {
175        this.log.info("failed upload of file `" + event.getFilename() + "': " + event.getReason());
176        this.progressBar.setVisible(false);
177        this.updateDisplay();
178        if (this.getUI() != null) {
179            final Notification notification = new Notification("Upload failed",
180              "" + event.getReason(), Notification.Type.ERROR_MESSAGE);
181            notification.setDelayMsec(3000);
182            notification.show(this.getUI().getPage());
183        }
184    }
185
186// Internal Methods
187
188    /**
189     * Update the description label to reflect the current property value and read-only status.
190     */
191    protected void updateDisplay() {
192        final byte[] value = this.getValue();
193        this.descriptionLabel.setValue(value != null ? value.length + " bytes" : "Null");
194        this.upload.setEnabled(!this.isReadOnly());
195    }
196}

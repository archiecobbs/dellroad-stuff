006package org.dellroad.stuff.vaadin7;
007
008/**
009 * A {@link QueryList} that always throws {@link InvalidQueryListException}.
010 */
011public class AlwaysInvalidQueryList<T> implements QueryList<T> {
012
013    private final long size;
014
015    /**
016     * Constructor.
017     *
018     * @param size size of this list to report via {@link #size}
019     * @throws IllegalArgumentException if {@code size} is negative
020     */
021    public AlwaysInvalidQueryList(long size) {
022        if (size < 0)
023            throw new IllegalArgumentException("size < 0");
024        this.size = size;
025    }
026
027    public long size() {
028        return this.size;
029    }
030
031    public T get(long index) throws InvalidQueryListException {
032        throw new InvalidQueryListException();
033    }
034}

006package org.dellroad.stuff.vaadin7;
007
008import com.vaadin.server.SessionDestroyEvent;
009import com.vaadin.server.SessionDestroyListener;
010import com.vaadin.server.VaadinSession;
011
012import org.slf4j.LoggerFactory;
013
014/**
015 * Support superclass customized for use by listeners that are part of a Vaadin application when listening
016 * to non-Vaadin ("external") event sources.
017 *
018 * <p>
019 * Listeners that are part of a Vaadin application should use this superclass if they are going to be registered
020 * with non-Vaadin event sources, and use the methods {@link #register()} and {@link #unregister()} to control listener
021 * registration. Subclasses implement {@link #register(Object) register(S)} and {@link #unregister(Object) register(S)}
022 * to perform the actual listener registration/unregister operations.
023 *
024 * <p>
025 * Use of this class will ensure two things:
026 * <ul>
027 *  <li>Events are delivered {@linkplain VaadinUtil#invoke in the proper Vaadin application context}; and</li>
028 *  <li>The listener is automatically unregistered from the external event source when the Vaadin application is closed,
029 *      avoiding a memory leak</li>
030 * </ul>
031 *
032 * <p>
033 * <b>Important:</b> subclass listener methods must use {@link #handleEvent handleEvent()} when handling events.
034 * This will ensure proper locking to avoid race conditions.
035 *
036 * <p>
037 * Note: when listening to event sources that are scoped to specific Vaadin application instances and already originate events
038 * within the proper Vaadin application context (i.e., event sources that are not external to the Vaadin application),
039 * then the use of this superclass is not necessary (however, it also doesn't hurt to use it anyway).
040 *
041 * @param <S> The type of the event source
042 * @see VaadinUtil#invoke
043 * @see VaadinApplicationScope
044 * @see VaadinApplicationListener
045 * @see SpringVaadinSessionListener
046 */
047public abstract class VaadinExternalListener<S> {
048
049    private final S eventSource;
050    private final VaadinSession session;
051    private final CloseListener closeListener = new CloseListener();
052
053    private volatile boolean asynchronous;
054
055    /**
056     * Convenience constructor. Equivalent to:
057     * <blockquote>
058     *  {@link #VaadinExternalListener(Object, VaadinSession)
059     *      VaadinExternalListener(eventSource, VaadinUtil.getCurrentSession())}
060     * </blockquote>
061     *
062     * @param eventSource the event source on which this listener will register
063     * @throws IllegalArgumentException if {@code eventSource} is null
064     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
065     */
066    protected VaadinExternalListener(S eventSource) {
067        this(eventSource, VaadinUtil.getCurrentSession());
068    }
069
070    /**
071     * Convenience constructor. Equivalent to:
072     * <blockquote>
073     *  {@link #VaadinExternalListener(Object, VaadinSession) VaadinExternalListener(eventSource, application.getSession())}
074     * </blockquote>
075     *
076     * @param eventSource the event source on which this listener will register
077     * @param application the {@link VaadinApplication} that this listener is part of
078     * @throws IllegalArgumentException if {@code eventSource} is null
079     * @throws NullPointerException if {@code application} is null
080     */
081    protected VaadinExternalListener(S eventSource, VaadinApplication application) {
082        this(eventSource, application.getSession());
083    }
084
085    /**
086     * Primary constructor.
087     *
088     * @param eventSource the event source on which this listener will register when {@link #register} is invoked
089     * @param session the associated Vaadin application's session
090     * @throws IllegalArgumentException if either parameter is null
091     */
092    protected VaadinExternalListener(S eventSource, VaadinSession session) {
093        if (eventSource == null)
094            throw new IllegalArgumentException("null eventSource");
095        if (session == null)
096            throw new IllegalArgumentException("null session");
097        this.eventSource = eventSource;
098        this.session = session;
099    }
100
101    /**
102     * Determine whether this instance is configured for asynchronous notification. Default false.
103     *
104     * @return true if this instance will notify asynchronously
105     */
106    public boolean isAsynchronous() {
107        return this.asynchronous;
108    }
109
110    /**
111     * Set whether to notify asynchronously. If set, {@link VaadinUtil#invokeLater VaadinUtil.invokeLater()} will
112     * be used for notifications, so that these occur on a different thread from the original notifying thread.
113     *
114     * @param asynchronous true to notify asynchronously, false for synchronous notifications
115     * @see #handleEvent(Runnable) handleEvent()
116     */
117    public void setAsynchronous(boolean asynchronous) {
118        this.asynchronous = asynchronous;
119    }
120
121    /**
122     * Register as a listener on configured event source.
123     *
124     * <p>
125     * This also registers a {@link SessionDestroyListener} on the {@linkplain #getSession configured Vaadin application},
126     * so that when the application closes we can unregister this instance from the event source to avoid a memory leak.
127     */
128    public void register() {
129        VaadinUtil.addSessionDestroyListener(this.session, this.closeListener);
130        this.register(this.eventSource);
131    }
132
133    /**
134     * Un-register as a listener on configured event source.
135     *
136     * <p>
137     * This also unregisters the {@link SessionDestroyListener} registered by {@link #register}.
138     */
139    public void unregister() {
140        VaadinUtil.removeSessionDestroyListener(this.session, this.closeListener);
141        this.unregister(this.eventSource);
142    }
143
144    /**
145     * Get the {@link VaadinSession} (aka Vaadin application) with which this instance is associated.
146     *
147     * @return associated session
148     */
149    public final VaadinSession getSession() {
150        return this.session;
151    }
152
153    /**
154     * Get the event source with which this instance is (or was) registered as a listener.
155     *
156     * @return associated event source
157     */
158    public final S getEventSource() {
159        return this.eventSource;
160    }
161
162    /**
163     * Execute the given listener action using the {@link VaadinSession} with which this instance is associated.
164     *
165     * <p>
166     * Subclass listener methods should handle events by invoking this method to ensure proper locking to avoid race conditions.
167     *
168     * <p>
169     * This method delegates to {@link VaadinUtil#invoke VaadinUtil.invoke()}, or
170     * {@link VaadinUtil#invokeLater VaadinUtil.invokeLater()} if this instance is configured to be
171     * {@linkplain #setAsynchronous asynchronous}.
172     *
173     * @param action action to perform
174     * @see VaadinUtil#invoke
175     */
176    protected void handleEvent(final Runnable action) {
177        if (this.asynchronous) {
178            VaadinUtil.invokeLater(this.getSession(), new Runnable() {
179                @Override
180                public void run() {
181                    try {
182                        action.run();
183                    } catch (RuntimeException e) {
184                        LoggerFactory.getLogger(VaadinExternalListener.this.getClass()).error(
185                          "exception in asynchrnous listener", e);
186                        throw e;
187                    }
188                }
189            });
190        } else
191            VaadinUtil.invoke(this.getSession(), action);
192    }
193
194    /**
195     * Register as a listener on the given event source.
196     *
197     * <p>
198     * Subclass must implement this to perform the actual listener registration.
199     *
200     * @param eventSource event source, never null; will be same as provided to the constructor
201     */
202    protected abstract void register(S eventSource);
203
204    /**
205     * Register as a listener from the given event source.
206     *
207     * <p>
208     * Subclass must implement this to perform the actual listener registration.
209     *
210     * @param eventSource event source, never null; will be same as provided to the constructor
211     */
212    protected abstract void unregister(S eventSource);
213
214// Application close listener
215
216    @SuppressWarnings("serial")
217    private final class CloseListener implements SessionDestroyListener {
218
219        @Override
220        public void sessionDestroy(SessionDestroyEvent event) {
221            VaadinExternalListener.this.unregister();
222        }
223    }
224}

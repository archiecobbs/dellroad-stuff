006package org.dellroad.stuff.vaadin7;
007
008import java.util.Collection;
009import java.util.HashSet;
010import java.util.Set;
011
012/**
013 * Simple read-only {@link com.vaadin.data.Item} implementation backed by a Java object.
014 *
015 * <p>
016 * Item {@link com.vaadin.data.Property}s are defined by {@link PropertyDef}s and retrieved by a {@link PropertyExtractor}.
017 * </p>
018 *
019 * <p>
020 * Although the item properties are read-only from "above", they can be made mutable from "below" by modifying the
021 * backing object and invoking {@link #fireValueChange fireValueChange()}.
022 * </p>
023 *
024 * @param <T> the type of the underlying Java object
025 * @see AbstractQueryContainer
026 * @see AbstractSimpleContainer
027 */
028@SuppressWarnings("serial")
029public class SimpleItem<T> implements BackedItem<T> {
030
031    protected final T object;
032    protected final PropertyExtractor<? super T> propertyExtractor;
033
034    private final Object[] propertyList;                    // elements are either PropertyDef's or Property's
035
036    /**
037     * Constructor.
038     *
039     * @param object underlying Java object
040     * @param propertyDefs property definitions
041     * @param propertyExtractor extracts the property value from {@code object}
042     * @throws IllegalArgumentException if any parameter is null
043     */
044    public SimpleItem(T object, Collection<? extends PropertyDef<?>> propertyDefs, PropertyExtractor<? super T> propertyExtractor) {
045        if (object == null)
046            throw new IllegalArgumentException("null object");
047        if (propertyExtractor == null)
048            throw new IllegalArgumentException("null propertyExtractor");
049        if (propertyDefs == null)
050            throw new IllegalArgumentException("null propertyDefs");
051        this.object = object;
052        this.propertyExtractor = propertyExtractor;
053        this.propertyList = propertyDefs.toArray();
054    }
055
056    @Override
057    public T getObject() {
058        return this.object;
059    }
060
061    @Override
062    @SuppressWarnings("unchecked")
063    public Property<?> getItemProperty(Object id) {
064        for (int i = 0; i < this.propertyList.length; i++) {
065            final Object value = this.propertyList[i];
066            if (value instanceof PropertyDef) {
067                final PropertyDef<?> propertyDef = (PropertyDef<?>)value;
068                if (!propertyDef.getName().equals(id))
069                    continue;
070                final Property<?> property = this.createProperty(propertyDef);
071                this.propertyList[i] = property;
072                return property;
073            } else if (value instanceof Property) {
074                final Property<?> property = (Property<?>)value;
075                if (!property.getPropertyDef().getName().equals(id))
076                    continue;
077                return property;
078            } else
079                throw new RuntimeException("internal error: found " + value);
080        }
081        return null;
082    }
083
084    @Override
085    @SuppressWarnings("unchecked")
086    public Set<String> getItemPropertyIds() {
087        final HashSet<String> names = new HashSet<>(this.propertyList.length);
088        for (int i = 0; i < this.propertyList.length; i++) {
089            final Object value = this.propertyList[i];
090            if (value instanceof PropertyDef) {
091                final PropertyDef<?> propertyDef = (PropertyDef<?>)value;
092                names.add(propertyDef.getName());
093            } else if (value instanceof Property) {
094                final Property<?> property = (Property<?>)value;
095                names.add(property.getPropertyDef().getName());
096            } else
097                throw new RuntimeException("internal error: found " + value);
098        }
099        return names;
100    }
101
102    /**
103     * @throws UnsupportedOperationException always
104     */
105    @Override
106    @SuppressWarnings("rawtypes")
107    public boolean addItemProperty(Object id, com.vaadin.data.Property/*<?>*/ property) {
108        throw new UnsupportedOperationException();
109    }
110
111    /**
112     * @throws UnsupportedOperationException always
113     */
114    @Override
115    public boolean removeItemProperty(Object id) {
116        throw new UnsupportedOperationException();
117    }
118
119    /**
120     * Issue {@link com.vaadin.data.Property.ValueChangeEvent}s to all
121     * {@link com.vaadin.data.Property.ValueChangeListener}s registered on the specified property of this item.
122     *
123     * <p>
124     * Does nothing if {@code propertyName} is not one of this item's properties.
125     * </p>
126     *
127     * @param propertyName property ID
128     */
129    @SuppressWarnings("unchecked")
130    public void fireValueChange(String propertyName) {
131        for (int i = 0; i < this.propertyList.length; i++) {
132            final Object value = this.propertyList[i];
133            if (value instanceof Property) {
134                final Property<?> property = (Property<?>)value;
135                if (property.getPropertyDef().getName().equals(propertyName)) {
136                    property.fireValueChange();
137                    return;
138                }
139            }
140        }
141    }
142
143    /**
144     * Issue {@link com.vaadin.data.Property.ValueChangeEvent}s to all
145     * {@link com.vaadin.data.Property.ValueChangeListener}s registered on any property of this item.
146     */
147    @SuppressWarnings("unchecked")
148    public void fireValueChange() {
149        for (int i = 0; i < this.propertyList.length; i++) {
150            final Object value = this.propertyList[i];
151            if (value instanceof Property) {
152                final Property<?> property = (Property<?>)value;
153                property.fireValueChange();
154            }
155        }
156    }
157
158    /**
159     * Create a {@link Property} to be used for the given property definition.
160     * This method will be invoked at most once for any property; the returned value is cached.
161     *
162     * <p>
163     * The implementation in {@link SimpleItem} returns {@code new Property<V>(propertyDef)}.
164     * </p>
165     *
166     * @param propertyDef property definition
167     * @throws IllegalArgumentException if {@code propertyDef} is null
168     */
169    protected <V> Property<V> createProperty(PropertyDef<V> propertyDef) {
170        return new Property<V>(propertyDef);
171    }
172
173// Property
174
175    /**
176     * {@link Property} implementation used by {@link SimpleItem}.
177     */
178    public class Property<V> extends ReadOnlyProperty<V> implements BackedProperty<T, V> {
179
180        private final PropertyDef<V> propertyDef;
181
182        public Property(PropertyDef<V> propertyDef) {
183            if (propertyDef == null)
184                throw new IllegalArgumentException("null propertyDef");
185            this.propertyDef = propertyDef;
186        }
187
188        public PropertyDef<V> getPropertyDef() {
189            return this.propertyDef;
190        }
191
192        @Override
193        public T getObject() {
194            return SimpleItem.this.object;
195        }
196
197        @Override
198        public Class<V> getType() {
199            return this.propertyDef.getType();
200        }
201
202        @Override
203        public V getValue() {
204            return SimpleItem.this.propertyExtractor.getPropertyValue(SimpleItem.this.object, this.propertyDef);
205        }
206    }
207}

006package org.dellroad.stuff.vaadin7;
007
008import com.vaadin.data.Property;
009
010/**
011 * Extension of the {@link Property} interface for implementations that are backed by an underlying Java object.
012 *
013 * @param <T> the type of the underlying Java object
014 * @param <V> the type of the property
015 */
016@SuppressWarnings("serial")
017public interface BackedProperty<T, V> extends Property<V> {
018
019    /**
020     * Retrieve the underlying Java object.
021     *
022     * @return underlying Java object, never null
023     */
024    T getObject();
025}

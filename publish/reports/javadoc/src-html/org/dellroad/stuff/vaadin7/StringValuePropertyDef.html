006package org.dellroad.stuff.vaadin7;
007
008/**
009 * A {@link PropertyDef} representing the {@link String} value of an object using {@link Object#toString Object#toString()}.
010 *
011 * <p>
012 * Instances also serve as a {@link SortingPropertyExtractor} that can actually extract the property from any object.
013 * </p>
014 */
015public class StringValuePropertyDef extends PropertyDef<String> implements SortingPropertyExtractor<Object> {
016
017    /**
018     * Constructor.
019     *
020     * @param name property name; also serves as the property ID
021     */
022    public StringValuePropertyDef(String name) {
023        super(name, String.class, null);
024    }
025
026// PropertyExtractor
027
028    @Override
029    @SuppressWarnings("unchecked")
030    public <V> V getPropertyValue(Object obj, PropertyDef<V> propertyDef) {
031        if (!(propertyDef instanceof StringValuePropertyDef))
032            throw new IllegalArgumentException("unknown property " + propertyDef);
033        return (V)obj.toString();
034    }
035
036// SortingPropertyExtractor
037
038    @Override
039    public boolean canSort(PropertyDef<?> propertyDef) {
040        return propertyDef instanceof StringValuePropertyDef;
041    }
042
043    @Override
044    public int sort(PropertyDef<?> propertyDef, Object obj1, Object obj2) {
045        if (!(propertyDef instanceof StringValuePropertyDef))
046            throw new UnsupportedOperationException("unknown property " + propertyDef);
047        return obj1.toString().compareTo(obj2.toString());
048    }
049}

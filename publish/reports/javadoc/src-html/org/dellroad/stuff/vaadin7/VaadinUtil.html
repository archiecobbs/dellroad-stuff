006package org.dellroad.stuff.vaadin7;
007
008import com.vaadin.server.SessionDestroyEvent;
009import com.vaadin.server.SessionDestroyListener;
010import com.vaadin.server.VaadinRequest;
011import com.vaadin.server.VaadinService;
012import com.vaadin.server.VaadinSession;
013
014import java.util.concurrent.Future;
015
016/**
017 * Miscellaneous utility methods.
018 */
019public final class VaadinUtil {
020
021    private VaadinUtil() {
022    }
023
024    /**
025     * Verify that we are running in the context of the given session and holding the session's lock.
026     * This method can be used by any code that manipulates Vaadin state to assert that the proper Vaadin
027     * locking has been performed.
028     *
029     * @param session session we are supposed to be running with
030     * @throws IllegalArgumentException if {@code session} is null
031     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
032     * @throws IllegalStateException if the {@link VaadinSession} associated with the current thread is not {@code session}
033     * @throws IllegalStateException if the {@link VaadinSession} associated with the current thread is not locked
034     * @throws IllegalStateException if the {@link VaadinSession} associated with the current thread is locked by another thread
035     */
036    public static void assertSession(VaadinSession session) {
037        if (session == null)
038            throw new IllegalArgumentException("null session");
039        final VaadinSession currentSession = VaadinSession.getCurrent();
040        if (currentSession == null)
041            throw new IllegalStateException("there is no VaadinSession associated with the current thread");
042        if (currentSession != session) {
043            throw new IllegalStateException("the VaadinSession associated with the current thread " + currentSession
044              + " is not the same session as the given one " + session);
045        }
046        if (!session.hasLock()) {
047            throw new IllegalStateException("the VaadinSession associated with the current thread " + currentSession
048              + " is not locked by this thread");
049        }
050    }
051
052    /**
053     * Get the {@link VaadinSession} associated with the current thread.
054     * This is just a wrapper around {@link VaadinSession#getCurrent} that throws an exception instead
055     * of returning null when there is no session associated with the current thread.
056     *
057     * @return current {@link VaadinSession}, never null
058     *
059     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
060     */
061    public static VaadinSession getCurrentSession() {
062        VaadinSession session = VaadinSession.getCurrent();
063        if (session == null) {
064            throw new IllegalStateException("there is no VaadinSession associated with the current thread;"
065              + " are we executing within a Vaadin HTTP request or VaadinUtil.invoke()?");
066        }
067        return session;
068    }
069
070    /**
071     * Get the {@link VaadinRequest} associated with the current thread.
072     * This is just a wrapper around {@link VaadinService#getCurrentRequest} that throws an exception instead
073     * of returning null when there is no request associated with the current thread.
074     *
075     * @return current {@link VaadinRequest}, never null
076     *
077     * @throws IllegalStateException if there is no {@link VaadinRequest} associated with the current thread
078     */
079    public static VaadinRequest getCurrentRequest() {
080        VaadinRequest request = VaadinService.getCurrentRequest();
081        if (request == null) {
082            throw new IllegalStateException("there is no VaadinRequest associated with the current thread;"
083              + " are we executing within a Vaadin HTTP request?");
084        }
085        return request;
086    }
087
088    /**
089     * Peform some action while holding the given {@link VaadinSession}'s lock.
090     *
091     * <p>
092     * This method now just invokes {@link VaadinSession#accessSynchronously}, a method which didn't exist in earlier
093     * versions of Vaadin.
094     *
095     * <p>
096     * All back-end threads that interact with Vaadin components must use this method (or {@link #invokeLater invokeLater()})
097     * to avoid race conditions. Since session locks are re-entrant, it will not cause problems if this method is also
098     * used by a "front-end" (i.e., Vaadin HTTP request) thread.
099     *
100     * <p>
101     * Note: when executing within a Vaadin HTTP request, the current thread's {@link VaadinSession} is available
102     * via {@link VaadinSession#getCurrent}; consider also using {@link VaadinApplication#invoke} instead of this method.
103     *
104     * <p>
105     * <b>Warning:</b> background threads should be careful when invoking this method to ensure they
106     * are not already holding an application-specific lock that a separate HTTP request thread could
107     * attempt to acquire during its normal processing: because the HTTP request thread will probably
108     * already be holding the session lock when it attempts to acquire the application-specific lock,
109     * this creates the potential for a lock-ordering reversal deadlock.
110     *
111     * @throws IllegalArgumentException if either parameter is null
112     * @see VaadinApplication#invoke
113     */
114    public static void invoke(VaadinSession session, Runnable action) {
115        if (session == null)
116            throw new IllegalArgumentException("null session");
117        if (action == null)
118            throw new IllegalArgumentException("null action");
119        session.accessSynchronously(action);
120    }
121
122    /**
123     * Peform some action while holding the given {@link VaadinSession}'s lock, but do so asynchronously.
124     *
125     * <p>
126     * Here the term "asynchronously" means:
127     * <ul>
128     *  <li>If any thread holds the session lock (including the current thread), this method will return
129     *      immediately and the action will be performed later, when the session is eventually unlocked.</li>
130     *  <li>If no thread holds the session lock, the session will be locked and the action performed
131     *      synchronously by the current thread.</li>
132     * </ul>
133     *
134     * <p>
135     * This method now just invokes {@link VaadinSession#access}, a method which didn't exist in earlier
136     * versions of Vaadin.
137     *
138     * <p>
139     * Note: when executing within a Vaadin HTTP request, the current thread's {@link VaadinSession} is available
140     * via {@link VaadinSession#getCurrent}; consider also using {@link VaadinApplication#invokeLater} instead of this method.
141     *
142     * @return a corresponding {@link Future}
143     * @throws IllegalArgumentException if either parameter is null
144     * @see #invoke
145     * @see VaadinApplication#invokeLater
146     */
147    public static Future<Void> invokeLater(VaadinSession session, Runnable action) {
148        if (session == null)
149            throw new IllegalArgumentException("null session");
150        if (action == null)
151            throw new IllegalArgumentException("null action");
152        return session.access(action);
153    }
154
155    /**
156     * Register for a notification when the {@link VaadinSession} is closed, without creating a memory leak.
157     * This method is intended to be used by listeners that are themselves part of a Vaadin application.
158     *
159     * <p>
160     * Explanation: the {@link VaadinSession} class does not provide a listener API directly; instead, you must
161     * use the {@link com.vaadin.server.VaadinService} class. However, registering as a listener on the
162     * {@link com.vaadin.server.VaadinService} when you are part of a Vaadin application sets you up for a memory leak
163     * if you forget to unregister yourself when the notification arrives, because the {@link com.vaadin.server.VaadinService}
164     * lifetime is longer than the {@link VaadinSession} lifetime. This method handles that de-registration for
165     * you automatically.
166     *
167     * @throws IllegalArgumentException if either parameter is null
168     * @see VaadinApplication#addSessionDestroyListener
169     */
170    public static void addSessionDestroyListener(VaadinSession session, SessionDestroyListener listener) {
171        session.getService().addSessionDestroyListener(new LeakAvoidingDestroyListener(session, listener));
172    }
173
174    /**
175     * Remove a listener added via {@link #addSessionDestroyListener addSessionDestroyListener()}.
176     *
177     * @throws IllegalArgumentException if either parameter is null
178     * @see VaadinApplication#removeSessionDestroyListener
179     */
180    public static void removeSessionDestroyListener(VaadinSession session, SessionDestroyListener listener) {
181        session.getService().removeSessionDestroyListener(new LeakAvoidingDestroyListener(session, listener));
182    }
183
184// LeakAvoidingDestroyListener
185
186    @SuppressWarnings("serial")
187    private static class LeakAvoidingDestroyListener implements SessionDestroyListener {
188
189        private final VaadinSession session;
190        private final SessionDestroyListener listener;
191
192        public LeakAvoidingDestroyListener(VaadinSession session, SessionDestroyListener listener) {
193            if (session == null)
194                throw new IllegalArgumentException("null session");
195            if (listener == null)
196                throw new IllegalArgumentException("null listener");
197            this.session = session;
198            this.listener = listener;
199        }
200
201        @Override
202        public void sessionDestroy(SessionDestroyEvent event) {
203            final VaadinSession closedSession = event.getSession();
204            if (closedSession == this.session) {
205                this.session.getService().removeSessionDestroyListener(this);       // remove myself as listener to avoid mem leak
206                this.listener.sessionDestroy(event);
207            }
208        }
209
210        @Override
211        public boolean equals(Object obj) {
212            if (obj == null || obj.getClass() != this.getClass())
213                return false;
214            LeakAvoidingDestroyListener that = (LeakAvoidingDestroyListener)obj;
215            return this.session == that.session && this.listener.equals(that.listener);
216        }
217
218        @Override
219        public int hashCode() {
220            return this.session.hashCode() ^ this.listener.hashCode();
221        }
222    }
223}

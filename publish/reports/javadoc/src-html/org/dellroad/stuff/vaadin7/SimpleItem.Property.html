006package org.dellroad.stuff.vaadin7;
007
008import java.util.Collection;
009import java.util.HashSet;
010import java.util.Set;
011
012/**
013 * Simple read-only {@link com.vaadin.data.Item} implementation backed by a Java object.
014 *
015 * <p>
016 * Item {@link com.vaadin.data.Property}s are defined by {@link PropertyDef}s and retrieved by a {@link PropertyExtractor}.
017 *
018 * <p>
019 * Although the item properties are read-only from "above", they can be made mutable from "below" by modifying the
020 * backing object and invoking {@link #fireValueChange fireValueChange()}.
021 *
022 * @param <T> the type of the underlying Java object
023 * @see AbstractQueryContainer
024 * @see AbstractSimpleContainer
025 */
026@SuppressWarnings("serial")
027public class SimpleItem<T> implements BackedItem<T> {
028
029    protected final T object;
030    protected final PropertyExtractor<? super T> propertyExtractor;
031
032    private final Object[] propertyList;                    // elements are either PropertyDef's or Property's
033
034    /**
035     * Constructor.
036     *
037     * @param object underlying Java object
038     * @param propertyDefs property definitions
039     * @param propertyExtractor extracts the property value from {@code object}
040     * @throws IllegalArgumentException if any parameter is null
041     */
042    public SimpleItem(T object, Collection<? extends PropertyDef<?>> propertyDefs, PropertyExtractor<? super T> propertyExtractor) {
043        if (object == null)
044            throw new IllegalArgumentException("null object");
045        if (propertyExtractor == null)
046            throw new IllegalArgumentException("null propertyExtractor");
047        if (propertyDefs == null)
048            throw new IllegalArgumentException("null propertyDefs");
049        this.object = object;
050        this.propertyExtractor = propertyExtractor;
051        this.propertyList = propertyDefs.toArray();
052    }
053
054    @Override
055    public T getObject() {
056        return this.object;
057    }
058
059    @Override
060    @SuppressWarnings("unchecked")
061    public Property<?> getItemProperty(Object id) {
062        for (int i = 0; i < this.propertyList.length; i++) {
063            final Object value = this.propertyList[i];
064            if (value instanceof PropertyDef) {
065                final PropertyDef<?> propertyDef = (PropertyDef<?>)value;
066                if (!propertyDef.getName().equals(id))
067                    continue;
068                final Property<?> property = this.createProperty(propertyDef);
069                this.propertyList[i] = property;
070                return property;
071            } else if (value instanceof Property) {
072                final Property<?> property = (Property<?>)value;
073                if (!property.getPropertyDef().getName().equals(id))
074                    continue;
075                return property;
076            } else
077                throw new RuntimeException("internal error: found " + value);
078        }
079        return null;
080    }
081
082    @Override
083    @SuppressWarnings("unchecked")
084    public Set<String> getItemPropertyIds() {
085        final HashSet<String> names = new HashSet<>(this.propertyList.length);
086        for (int i = 0; i < this.propertyList.length; i++) {
087            final Object value = this.propertyList[i];
088            if (value instanceof PropertyDef) {
089                final PropertyDef<?> propertyDef = (PropertyDef<?>)value;
090                names.add(propertyDef.getName());
091            } else if (value instanceof Property) {
092                final Property<?> property = (Property<?>)value;
093                names.add(property.getPropertyDef().getName());
094            } else
095                throw new RuntimeException("internal error: found " + value);
096        }
097        return names;
098    }
099
100    /**
101     * @throws UnsupportedOperationException always
102     */
103    @Override
104    @SuppressWarnings("rawtypes")
105    public boolean addItemProperty(Object id, com.vaadin.data.Property/*<?>*/ property) {
106        throw new UnsupportedOperationException();
107    }
108
109    /**
110     * @throws UnsupportedOperationException always
111     */
112    @Override
113    public boolean removeItemProperty(Object id) {
114        throw new UnsupportedOperationException();
115    }
116
117    /**
118     * Issue {@link com.vaadin.data.Property.ValueChangeEvent}s to all
119     * {@link com.vaadin.data.Property.ValueChangeListener}s registered on the specified property of this item.
120     *
121     * <p>
122     * Does nothing if {@code propertyName} is not one of this item's properties.
123     *
124     * @param propertyName property ID
125     */
126    @SuppressWarnings("unchecked")
127    public void fireValueChange(String propertyName) {
128        for (int i = 0; i < this.propertyList.length; i++) {
129            final Object value = this.propertyList[i];
130            if (value instanceof Property) {
131                final Property<?> property = (Property<?>)value;
132                if (property.getPropertyDef().getName().equals(propertyName)) {
133                    property.fireValueChange();
134                    return;
135                }
136            }
137        }
138    }
139
140    /**
141     * Issue {@link com.vaadin.data.Property.ValueChangeEvent}s to all
142     * {@link com.vaadin.data.Property.ValueChangeListener}s registered on any property of this item.
143     */
144    @SuppressWarnings("unchecked")
145    public void fireValueChange() {
146        for (int i = 0; i < this.propertyList.length; i++) {
147            final Object value = this.propertyList[i];
148            if (value instanceof Property) {
149                final Property<?> property = (Property<?>)value;
150                property.fireValueChange();
151            }
152        }
153    }
154
155    /**
156     * Create a {@link Property} to be used for the given property definition.
157     * This method will be invoked at most once for any property; the returned value is cached.
158     *
159     * <p>
160     * The implementation in {@link SimpleItem} returns {@code new Property<V>(propertyDef)}.
161     *
162     * @param propertyDef property definition
163     * @throws IllegalArgumentException if {@code propertyDef} is null
164     */
165    protected <V> Property<V> createProperty(PropertyDef<V> propertyDef) {
166        return new Property<V>(propertyDef);
167    }
168
169// Property
170
171    /**
172     * {@link Property} implementation used by {@link SimpleItem}.
173     */
174    public class Property<V> extends ReadOnlyProperty<V> implements BackedProperty<T, V> {
175
176        private final PropertyDef<V> propertyDef;
177
178        public Property(PropertyDef<V> propertyDef) {
179            if (propertyDef == null)
180                throw new IllegalArgumentException("null propertyDef");
181            this.propertyDef = propertyDef;
182        }
183
184        public PropertyDef<V> getPropertyDef() {
185            return this.propertyDef;
186        }
187
188        @Override
189        public T getObject() {
190            return SimpleItem.this.object;
191        }
192
193        @Override
194        public Class<V> getType() {
195            return this.propertyDef.getType();
196        }
197
198        @Override
199        public V getValue() {
200            return SimpleItem.this.propertyExtractor.getPropertyValue(SimpleItem.this.object, this.propertyDef);
201        }
202    }
203}

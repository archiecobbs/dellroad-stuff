006package org.dellroad.stuff.vaadin7;
007
008import java.util.Collection;
009
010/**
011 * A {@link SimpleKeyedContainer} where the item IDs are the underlying container objects themselves.
012 *
013 * <p>
014 * Restriction: instances can't contain two objects that are {@link Object#equals equal()} to each other.
015 *
016 * @param <T> the type of the Java objects that back each {@link com.vaadin.data.Item} in the container
017 */
018@SuppressWarnings("serial")
019public class SelfKeyedContainer<T> extends SimpleKeyedContainer<T, T> {
020
021    /**
022     * Constructor.
023     *
024     * <p>
025     * After using this constructor, subsequent invocations of {@link #setPropertyExtractor setPropertyExtractor()}
026     * and {@link #setProperties setProperties()} are required to define the properties of this container
027     * and how to extract them.
028     */
029    public SelfKeyedContainer() {
030    }
031
032    /**
033     * Constructor.
034     *
035     * <p>
036     * After using this constructor, a subsequent invocation of {@link #setProperties setProperties()} is required
037     * to define the properties of this container.
038     *
039     * @param propertyExtractor used to extract properties from the underlying Java objects;
040     *  may be null but then container is not usable until one is configured via
041     * {@link #setPropertyExtractor setPropertyExtractor()}
042     */
043    public SelfKeyedContainer(PropertyExtractor<? super T> propertyExtractor) {
044        this(propertyExtractor, null);
045    }
046
047    /**
048     * Constructor.
049     *
050     * <p>
051     * After using this constructor, a subsequent invocation of {@link #setPropertyExtractor setPropertyExtractor()} is required
052     * to define how to extract the properties of this container; alternately, subclasses can override
053     * {@link #getPropertyValue getPropertyValue()}.
054     *
055     * @param propertyDefs container property definitions; null is treated like the empty set
056     */
057    protected SelfKeyedContainer(Collection<? extends PropertyDef<?>> propertyDefs) {
058        this(null, propertyDefs);
059    }
060
061    /**
062     * Constructor.
063     *
064     * @param propertyExtractor used to extract properties from the underlying Java objects;
065     *  may be null but then container is not usable until one is configured via
066     * {@link #setPropertyExtractor setPropertyExtractor()}
067     * @param propertyDefs container property definitions; null is treated like the empty set
068     */
069    public SelfKeyedContainer(PropertyExtractor<? super T> propertyExtractor, Collection<? extends PropertyDef<?>> propertyDefs) {
070        super(propertyExtractor, propertyDefs);
071    }
072
073    /**
074     * Constructor.
075     *
076     * <p>
077     * Properties will be determined by the {@link ProvidesProperty &#64;ProvidesProperty} and
078     * {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods in the given class.
079     *
080     * @param type class to introspect for annotated methods
081     * @throws IllegalArgumentException if {@code type} is null
082     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}
083     *  or {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods for the same property
084     * @throws IllegalArgumentException if a {@link ProvidesProperty &#64;ProvidesProperty}-annotated method with no
085     *  {@linkplain ProvidesProperty#value property name specified} has a name which cannot be interpreted as a bean
086     *  property "getter" method
087     * @see ProvidesProperty
088     * @see ProvidesPropertySort
089     * @see ProvidesPropertyScanner
090     */
091    protected SelfKeyedContainer(Class<? super T> type) {
092        super(type);
093    }
094
095    /**
096     * Get the key to be used as item ID for the given object.
097     *
098     * <p>
099     * The implementation in {@link SelfKeyedContainer} always returns {@code obj}.
100     *
101     * @param obj underlying container object
102     * @return key for object
103     */
104    @Override
105    protected T getKeyFor(T obj) {
106        return obj;
107    }
108}

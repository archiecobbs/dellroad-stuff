008/**
009 * {@link PropertyExtractor} that also sorts properties, given two instances of the target Java type.
010 *
011 * @param <T> target object type for extraction
012 */
013public interface SortingPropertyExtractor<T> extends PropertyExtractor<T> {
014
015    /**
016     * Determine if the given property can be sorted by this instance.
017     * If this method returns false, then the property's sorting resorts to the default behavior,
018     * i.e., depending on whether the property value implements {@link Comparable}.
019     *
020     * @param propertyDef definition of property
021     * @return true if the property defined by {@code propertyDef} can be sorted by this instance
022     * @throws NullPointerException if {@code propertyDef} is null
023     */
024    boolean canSort(PropertyDef<?> propertyDef);
025
026    /**
027     * Sort two values based on the given property.
028     *
029     * @param obj1 first Java object
030     * @param obj2 second Java object
031     * @param propertyDef definition of property to sort on
032     * @return result of comparing {@code obj1}'s property value to {@code obj2}'s property value
033     * @throws NullPointerException if any parameter is null
034     * @throws UnsupportedOperationException if the property defined by {@code propertyDef} cannot be sorted by this instance
035     */
036    int sort(PropertyDef<?> propertyDef, T obj1, T obj2);
037}

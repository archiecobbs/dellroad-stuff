006package org.dellroad.stuff.vaadin7;
007
008import com.vaadin.server.SessionDestroyListener;
009import com.vaadin.server.VaadinSession;
010
011import java.io.IOException;
012import java.io.ObjectInputStream;
013import java.io.Serializable;
014import java.util.concurrent.Future;
015
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019/**
020 * A globally accessible "Vaadin application" singleton.
021 *
022 * <p>
023 * At most one instance is associated with each {@link VaadinSession} (attempting to create more than one will result in
024 * an exception). This singleton instance is then always accessible from any Vaadin thread via {@link #get()}.
025 * </p>
026 *
027 * <p>
028 * Although this class does not require Spring, the singleton instance can be declared as a bean in the Spring XML
029 * application context created by a {@link SpringVaadinSessionListener}, which allows other beans and widgets
030 * in the Vaadin application context to autowire it and have access to the methods provided here:
031 * <blockquote><pre>
032 *  &lt;bean class="org.dellroad.stuff.vaadin7.VaadinApplication"/&gt;
033 * </pre></blockquote>
034 * If your {@link VaadinApplication} instance allocates any resources on startup, be sure to configure it so
035 * that it cleans them up on application context shutdown, for example, by specifying {@code destroy-method="..."}.
036 * </p>
037 *
038 * <p>
039 * If you are in a clustered environment and are subclassing this class, {@code scope="session"} is recommended;
040 * without it, there will be one instance per server+session rather than one unique instance per session. Of course,
041 * whether that matters depends on the semantics of the subclass fields. In any case, be aware that no instance will
042 * exist in session until the application context is refreshed.
043 * </p>
044 *
045 * <p>
046 * Note: using {@code scope="session"} requires adding a {@code <listener>} clause registering Spring's
047 * {@link org.springframework.web.context.request.RequestContextListener} in your {@code web.xml}.
048 * </p>
049 *
050 * <p>
051 * If this class is subclassed, additional application-specific fields and methods can be supplied to the entire
052 * application via the same mechanism. Subclasses can override {@link #init} for any application-specific initialization.
053 * </p>
054 *
055 * <p>
056 * Note: the singleton instance of this class is stored in the Vaadin session and is {@link Serializable}. Therefore,
057 * in a clustered environment, subclasses must be prepared to handle (de)serialization. Also, avoid declaring fields
058 * that reference this class in non-session-scope beans; instead, use {@link #get VaadinApplication.get()}.
059 * </p>
060 *
061 * @see SpringVaadinServlet
062 * @see SpringVaadinSessionListener
063 * @see com.vaadin.server.VaadinService
064 */
065public class VaadinApplication implements Serializable {
066
067    private static final long serialVersionUID = -4202507689739224846L;
068    private static final Class<VaadinApplication> ATTRIBUTE_KEY = VaadinApplication.class;
069
070    protected final Logger log = LoggerFactory.getLogger(this.getClass());
071
072    private transient VaadinSession session;
073
074    /**
075     * Convenience constructor. Equivalent to:
076     * <blockquote></code>
077     *  {@link #VaadinApplication(VaadinSession) VaadinApplication}({@link VaadinUtil#getCurrentSession})
078     * </code></blockquote>
079     *
080     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
081     * @throws IllegalStateException if there is already a {@link VaadinApplication} instance associated with the current session
082     */
083    public VaadinApplication() {
084        this(VaadinUtil.getCurrentSession());
085    }
086
087    /**
088     * Primary Constructor.
089     *
090     * @param session the session with which this instance should be associated
091     * @throws IllegalArgumentException if {@code session} is null
092     * @throws IllegalStateException if there is already a {@link VaadinApplication} instance associated with {@code session}
093     */
094    public VaadinApplication(VaadinSession session) {
095
096        // Get session
097        if (session == null)
098            throw new IllegalArgumentException("null session");
099        this.session = session;
100
101        // Check for already-existing instance
102        VaadinApplication vaadinApplication = this.session.getAttribute(ATTRIBUTE_KEY);
103        if (vaadinApplication != null) {
104            throw new IllegalStateException("there is already a VaadinApplication associated with VaadinSession "
105              + this.session + ": " + vaadinApplication + "; did you accidentally declare more than one instance of"
106              + " VaadinApplication in the Vaadin Spring XML application context?");
107        }
108
109        // Set session attribute
110        VaadinApplication.setAttribute(this.session, ATTRIBUTE_KEY, this);
111
112        // Delegate to subclass for further initialization
113        this.init();
114    }
115
116    /**
117     * Perform any further initialization at construction time.
118     *
119     * <p>
120     * The implementation in {@link VaadinApplication} does nothing. Subclasses may override as desired.
121     * </p>
122     */
123    protected void init() {
124    }
125
126    /**
127     * Get the {@link VaadinSession} associated with this instance.
128     *
129     * @return associated VaadinSession, never null
130     */
131    public VaadinSession getSession() {
132        return this.session;
133    }
134
135    /**
136     * Close the {@link VaadinSession} associated with this instance.
137     * After invoking this method, the caller would normally ensure that no further references to this
138     * instance remain so that it and the associated {@link VaadinSession} can be freed.
139     *
140     * <p>
141     * The implementation in {@link VaadinApplication} just delegates to {@link com.vaadin.server.VaadinSession#close}.
142     */
143    public void close() {
144        this.session.close();
145    }
146
147    /**
148     * Get the singleton {@link VaadinApplication} instance associated with the current {@link VaadinSession}.
149     *
150     * <p>
151     * This is a convenience method, equivalent to:
152     * <blockquote><code>
153     *  {@link #get(Class) VaadinApplication.get}({@link VaadinApplication VaadinApplication.class})
154     * </code></blockquote>
155     *
156     * @return singleton instance for the current Vaadin application, never null
157     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
158     * @throws IllegalStateException if there is no {@link VaadinApplication} instance associated with the current session
159     */
160    public static VaadinApplication get() {
161        return VaadinApplication.get(VaadinApplication.class);
162    }
163
164    /**
165     * Get the singleton instance of the specified class associated with the current {@link VaadinSession}.
166     *
167     * <p>
168     * This is a convenience method, equivalent to:
169     * <blockquote><code>
170     *  {@link #get(Class) VaadinApplication.get}({@link VaadinUtil#getCurrentSession}, clazz)
171     * </code></blockquote>
172     *
173     * <p>
174     * Useful for subclasses of {@link VaadinApplication} that want to provide their own zero-parameter {@code get()} methods.
175     *
176     * @param clazz singleton instance type
177     * @return singleton instance of {@code clazz} in the session, never null
178     * @throws IllegalArgumentException if {@code clazz} is null
179     * @throws IllegalStateException if there is no {@link VaadinSession} associated with the current thread
180     * @throws IllegalStateException if there is no singleton of type {@code clazz} associated with the current session
181     */
182    public static <T extends VaadinApplication> T get(Class<T> clazz) {
183        return VaadinApplication.get(VaadinUtil.getCurrentSession(), clazz);
184    }
185
186    /**
187     * Get the singleton instance of the specified class associated with the given session.
188     *
189     * @param session Vaadin session
190     * @param clazz singleton instance type
191     * @return singleton instance of {@code clazz} in the session, never null
192     * @throws IllegalArgumentException if {@code session} is null
193     * @throws IllegalArgumentException if {@code clazz} is null
194     * @throws IllegalStateException if there is no singleton of type {@code clazz} associated with the {@code session}
195     */
196    public static <T extends VaadinApplication> T get(VaadinSession session, Class<T> clazz) {
197
198        // Sanity check
199        if (session == null)
200            throw new IllegalArgumentException("null session");
201        if (clazz == null)
202            throw new IllegalArgumentException("null clazz");
203
204        // Get the application
205        VaadinApplication vaadinApplication = session.getAttribute(ATTRIBUTE_KEY);
206        if (vaadinApplication == null) {
207            throw new IllegalStateException("there is no VaadinApplication associated with the current VaadinSession"
208              + "; did you declare an instance of VaadinApplication in the Vaadin Spring XML application context?"
209              + " And if declared scope=\"session\", did you register Spring's RequestContextListener in web.xml?");
210        }
211
212        // Check type
213        if (!clazz.isInstance(vaadinApplication)) {
214            throw new IllegalStateException("there is a VaadinApplication associated with the current VaadinSession"
215              + " but it is not an instance of " + clazz + "; instead it has type " + vaadinApplication.getClass().getName());
216        }
217
218        // Done
219        return clazz.cast(vaadinApplication);
220    }
221
222    /**
223     * Peform some action while holding the lock of the {@link VaadinSession} associated with this instance.
224     *
225     * <p>
226     * This is a convenience method that in turn invokes {@link VaadinUtil#invoke VaadinUtil.invoke()} using the
227     * {@link VaadinSession} associated with this instance.
228     * </p>
229     *
230     * @param action action to perform
231     * @throws IllegalArgumentException if {@code action} is null
232     * @see VaadinUtil#invoke
233     */
234    public void invoke(Runnable action) {
235        VaadinUtil.invoke(this.session, action);
236    }
237
238    /**
239     * Peform some action asynchronously while holding the lock of the {@link VaadinSession} associated with this instance.
240     *
241     * <p>
242     * This is a convenience method that in turn invokes {@link VaadinUtil#invokeLater VaadinUtil.invokeLater()} using the
243     * {@link VaadinSession} associated with this instance.
244     * </p>
245     *
246     * @param action action to perform
247     * @throws IllegalArgumentException if {@code action} is null
248     * @see VaadinUtil#invokeLater
249     */
250    public Future<Void> invokeLater(Runnable action) {
251        return VaadinUtil.invokeLater(this.session, action);
252    }
253
254    /**
255     * Register for a notification when the {@link VaadinSession} is closed, without creating a memory leak.
256     *
257     * <p>
258     * This is a convenience method that in turn invokes
259     * {@link VaadinUtil#addSessionDestroyListener VaadinUtil.addSessionDestroyListener()} using the
260     * {@link VaadinSession} associated with this instance.
261     *
262     * @throws IllegalArgumentException if {@code listener} is null
263     * @see VaadinUtil#addSessionDestroyListener
264     */
265    public void addSessionDestroyListener(SessionDestroyListener listener) {
266        VaadinUtil.addSessionDestroyListener(this.session, listener);
267    }
268
269    /**
270     * Remove a listener added via {@link #addSessionDestroyListener addSessionDestroyListener()}.
271     *
272     * <p>
273     * This is a convenience method that in turn invokes
274     * {@link VaadinUtil#removeSessionDestroyListener VaadinUtil.removeSessionDestroyListener()} using the
275     * {@link VaadinSession} associated with this instance.
276     *
277     * @throws IllegalArgumentException if {@code listener} is null
278     * @see VaadinUtil#removeSessionDestroyListener
279     */
280    public void removeSessionDestroyListener(SessionDestroyListener listener) {
281        VaadinUtil.removeSessionDestroyListener(this.session, listener);
282    }
283
284    // This method exists solely to bind the generic type
285    private static <T> void setAttribute(VaadinSession session, Class<T> clazz, Object value) {
286        session.setAttribute(clazz, clazz.cast(value));
287    }
288
289// Serialization
290
291    private void readObject(ObjectInputStream input) throws IOException, ClassNotFoundException {
292        input.defaultReadObject();
293        this.session = VaadinUtil.getCurrentSession();
294    }
295}

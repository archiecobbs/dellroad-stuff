006package org.dellroad.stuff.vaadin7;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Inherited;
011import java.lang.annotation.Retention;
012import java.lang.annotation.RetentionPolicy;
013import java.lang.annotation.Target;
014
015import org.dellroad.stuff.java.ErrorAction;
016import org.springframework.beans.factory.annotation.Autowire;
017
018/**
019 * Indicates that the class is a candidate for configuration using the {@code VaadinConfigurableAspect} aspect.
020 *
021 * <p>
022 * Analogous to Spring's {@link org.springframework.beans.factory.annotation.Configurable @Configurable} annotation,
023 * but causes beans to be autowired into the Spring application context associated with the current
024 * {@link com.vaadin.server.VaadinSession} (aka "Vaadin application") by {@link SpringVaadinSessionListener} instead of
025 * the Spring application context associated with the servlet context.
026 *
027 * <p>
028 * For an extra safety check, consider setting {@link org.dellroad.stuff.vaadin7.VaadinConfigurable#ifSessionNotLocked}
029 * where appropriate.
030 *
031 * <p>
032 * For the this annotation to function properly, {@link VaadinConfigurable @VaadinConfigurable} classes must be woven
033 * (either at build time or runtime) using the
034 * <a href="http://www.eclipse.org/aspectj/doc/released/faq.php#compiler">AspectJ compiler</a> with the
035 * {@code VaadinConfigurableAspect} aspect (included in the <code>dellroad-stuff</code> JAR file), and the
036 * {@link SpringVaadinServlet} must be used.
037 *
038 * @see org.dellroad.stuff.vaadin7
039 * @see SpringVaadinServlet
040 * @see SpringVaadinSessionListener
041 * @see <a href="https://github.com/archiecobbs/dellroad-stuff-vaadin-spring-demo3/tree/vaadin7">Example Code on GitHub</a>
042 */
043@Retention(RetentionPolicy.RUNTIME)
044@Target(ElementType.TYPE)
045@Documented
046@Inherited
047public @interface VaadinConfigurable {
048
049    /**
050     * Configuration bean definition template name, if any.
051     */
052    String value() default "";
053
054    /**
055     * Whether and how to automatically autowire dependencies.
056     */
057    Autowire autowire() default Autowire.NO;
058
059    /**
060     * Whether to enable dependency checking.
061     */
062    boolean dependencyCheck() default false;
063
064    /**
065     * Whether to inject dependencies prior to constructor execution.
066     */
067    boolean preConstruction() default false;
068
069    /**
070     * What to do when we discover that the {@link com.vaadin.server.VaadinSession} is not
071     * {@linkplain com.vaadin.server.VaadinSession#getLockInstance locked} when the annotated bean is constructed.
072     * For beans that are (or will interact with) Vaadin widgets, containers, etc., this typically
073     * indicates a programming error. In such cases, this property configures what to do, if anything.
074     *
075     * @see com.vaadin.server.VaadinSession#getLockInstance
076     */
077    ErrorAction ifSessionNotLocked() default ErrorAction.EXCEPTION;
078}

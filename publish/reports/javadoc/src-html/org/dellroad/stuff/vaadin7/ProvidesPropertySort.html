006package org.dellroad.stuff.vaadin7;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Retention;
011import java.lang.annotation.RetentionPolicy;
012import java.lang.annotation.Target;
013
014/**
015 * Declares that a Java method provides either a {@link Comparable} value or a {@link java.util.Comparator} that
016 * should be used to determine property sort ordering in a Java class whose instances back the items in a
017 * {@link com.vaadin.data.Container}.
018 *
019 * <p>
020 * {@link ProvidesProperty &#64;ProvidesProperty} and {@link ProvidesPropertySort &#64;ProvidesPropertySort} method annotations
021 * can be used to automatically generate a list of {@link PropertyDef}s and a {@link PropertyExtractor} using a
022 * {@link ProvidesPropertyScanner}. This happens automatically when using the appropriate constructors of the various
023 * {@link com.vaadin.data.Container} classes in this package.
024 * </p>
025 *
026 * <p>
027 * This annotation is used in conjunction with the {@link ProvidesProperty &#64;ProvidesProperty}
028 * annotation when the value returned by the {@link ProvidesProperty &#64;ProvidesProperty}-annotated method is not
029 * {@link Comparable} or does not naturally sort as desired. By declaring a
030 * {@link ProvidesPropertySort &#64;ProvidesPropertySort}-annotated method for a property, any arbitrary sorting
031 * function can be supplied.
032 * </p>
033 *
034 * <p>
035 * If the annotated method returns a {@link java.util.Comparator}, then the return value will be used to sort property values;
036 * otherwise, the annotated method must return a sub-type of {@link Comparable}, in which case the returned value determines how
037 * that instance's property value sorts (with {@code null} values sorting first). If the method returns neither a
038 * {@link java.util.Comparator} nor a {@link Comparable}, an exception is thrown during scanning.
039 * </p>
040 *
041 * <p>
042 * Here is a typical situation where {@link ProvidesPropertySort &#64;ProvidesPropertySort} is needed: you have
043 * a {@link String} property containing a formatted {@link java.util.Date}, but the way the {@link java.util.Date}
044 * strings are formatted does not sort in chronological order. This will usually be the case unless your
045 * {@link java.util.Date} strings are formatted like {@code 2013-03-12}, etc.
046 * </p>
047 *
048 * <p>
049 * To address this problem, define a {@link ProvidesPropertySort &#64;ProvidesPropertySort}-annotated method that
050 * provides a properly sorting {@link Comparable} value corresponding to the associated property. For example:
051 * <blockquote><pre>
052 * // Container backing object class
053 * public class User {
054 *
055 *     private Date birthday;
056 *
057 *     <b>&#64;ProvidesPropertySort</b>             // property name "birthday" is implied by method name
058 *     public Date getBirthday() {
059 *         return this.birthday;
060 *     }
061 *     public void setBirthday(Date birthday) {
062 *         this.birthday = birthday;
063 *     }
064 *
065 *     <b>&#64;ProvidesProperty("birthday")</b>     // the actual property value is returned here
066 *     private Label birthdayProperty() {
067 *         return new Label(new SimpleDateFormat("MM/dd/yyyy").format(this.birthday));
068 *     }
069 * </pre></blockquote>
070 * </p>
071 *
072 * <p>
073 * Alternately, have the {@link ProvidesPropertySort &#64;ProvidesPropertySort}-annotated method return
074 * a {@link java.util.Comparator}, for example, if you wanted to sort {@code null} values last instead of first:
075 * <blockquote><pre>
076 *     <b>&#64;ProvidesPropertySort("birthday")</b>
077 *     private static Comparator&lt;User&gt; birthdayComparator() {
078 *         return new Comparator&lt;User&gt;(User user1, User user2) {
079 *              final Date date1 = user1.getBirthday();
080 *              final Date date2 = user2.getBirthday();
081 *              if (date1 == null &amp;&amp; date2 != null)
082 *                  return 1;
083 *              if (date1 != null &amp;&amp; date2 == null)
084 *                  return -1;
085 *              if (date1 == null &amp;&amp; date2 == null)
086 *                  return 0;
087 *              return date1.compareTo(date2);
088 *         };
089 *     }
090 * </pre></blockquote>
091 * Note that the returned {@link java.util.Comparator} compares <i>backing instances</i>, not property values,
092 * and that methods returning {@link java.util.Comparator} may be declared {@code static}. The returned
093 * {@link java.util.Comparator} may be cached by the implementation.
094 * </p>
095 *
096 * <p>
097 * Some details regarding {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotations on methods:
098 *  <ul>
099 *  <li>Only non-void methods taking zero parameters are supported; {@link ProvidesPropertySort &#64;ProvidesPropertySort}
100 *      annotations on other methods are ignored</li>
101 *  <li>Protected, package private, and private methods are supported.</li>
102 *  <li>{@link ProvidesPropertySort &#64;ProvidesPropertySort} annotations on interface methods are supported</li>
103 *  <li>If a method and the superclass or superinterface method it overrides are both annotated with
104 *      {@link ProvidesPropertySort &#64;ProvidesPropertySort}, then the overridding method's annotation takes precedence.
105 *  <li>If two methods with different names are annotated with {@link ProvidesPropertySort &#64;ProvidesPropertySort} for the same
106 *      {@linkplain #value property name}, then the declaration in the class which is a sub-type of the other
107 *      wins (if the two methods are delcared in the same class, an exception is thrown). This allows subclasses
108 *      to "override" which method supplies the sort value for a given property.</li>
109 *  </ul>
110 * </p>
111 *
112 *
113 * @see ProvidesProperty
114 * @see ProvidesPropertyScanner
115 */
116@Retention(RetentionPolicy.RUNTIME)
117@Target({ ElementType.METHOD, ElementType.FIELD })
118@Documented
119public @interface ProvidesPropertySort {
120
121    /**
122     * Get the name of the Vaadin property. If this is left unset (empty string), then the
123     * bean property name of the annotated bean property "getter" method is used.
124     */
125    String value() default "";
126}

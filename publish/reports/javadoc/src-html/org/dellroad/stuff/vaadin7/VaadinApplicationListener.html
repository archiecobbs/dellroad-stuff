006package org.dellroad.stuff.vaadin7;
007
008import com.vaadin.server.VaadinSession;
009
010import org.springframework.context.ApplicationEvent;
011import org.springframework.context.ApplicationListener;
012import org.springframework.context.event.ApplicationEventMulticaster;
013
014/**
015 * A Spring {@link org.springframework.context.ApplicationListener} support superclass customized for use by
016 * listeners that are part of a Vaadin application when listening to non-Vaadin application event sources.
017 *
018 * <p>
019 * Listeners that are part of a Vaadin application should use this superclass if they are going to be registered
020 * with non-Vaadin event sources. This will ensure that events are delivered in the proper Vaadin application context
021 * and memory leaks are avoided when the session closes.
022 * </p>
023 *
024 * @param <E> The type of the event
025 * @see VaadinExternalListener
026 * @see VaadinUtil#invoke
027 * @see VaadinApplicationScope
028 * @see SpringVaadinSessionListener
029 */
030public abstract class VaadinApplicationListener<E extends ApplicationEvent>
031  extends VaadinExternalListener<ApplicationEventMulticaster> implements ApplicationListener<E> {
032
033    /**
034     * Convenience constructor. Equivalent to:
035     * <blockquote>
036     *  {@link #VaadinApplicationListener(ApplicationEventMulticaster, VaadinSession)
037     *      VaadinApplicationListener(multicaster, VaadinUtil.getCurrentSession())}
038     * </blockquote>
039     *
040     * @param multicaster the application event multicaster on which this listener will register
041     * @throws IllegalArgumentException if {@code multicaster} is null
042     * @throws IllegalArgumentException if there is no {@link VaadinSession} associated with the current thread
043     */
044    public VaadinApplicationListener(ApplicationEventMulticaster multicaster) {
045        this(multicaster, VaadinUtil.getCurrentSession());
046    }
047
048    /**
049     * Primary constructor.
050     *
051     * @param multicaster the application event multicaster on which this listener will register
052     * @param session the associated Vaadin application
053     * @throws IllegalArgumentException if either parameter is null
054     */
055    public VaadinApplicationListener(ApplicationEventMulticaster multicaster, VaadinSession session) {
056        super(multicaster, session);
057    }
058
059    @Override
060    protected void register(ApplicationEventMulticaster multicaster) {
061        multicaster.addApplicationListener(this);
062    }
063
064    @Override
065    protected void unregister(ApplicationEventMulticaster multicaster) {
066        multicaster.removeApplicationListener(this);
067    }
068
069    @Override
070    public final void onApplicationEvent(final E event) {
071        this.handleEvent(new Runnable() {
072            @Override
073            public void run() {
074                VaadinApplicationListener.this.onApplicationEventInternal(event);
075            }
076        });
077    }
078
079    /**
080     * Handle a listener event. When this method is invoked, it will already be within the context
081     * of the {@link VaadinSession} with which this listener is associated.
082     *
083     * @see VaadinUtil#invoke
084     * @see VaadinUtil#getCurrentSession
085     */
086    protected abstract void onApplicationEventInternal(E event);
087}

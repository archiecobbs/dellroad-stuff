007
008import java.lang.reflect.Method;
009import java.util.ArrayList;
010import java.util.Collections;
011import java.util.Comparator;
012import java.util.HashMap;
013import java.util.List;
014import java.util.Map;
015import java.util.Set;
016
017import org.dellroad.stuff.java.MethodAnnotationScanner;
018import org.dellroad.stuff.java.Primitive;
019
020/**
021 * Scans a Java class hierarchy for {@link ProvidesProperty &#64;ProvidesProperty} and
022 * {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated getter methods and creates
023 * a corresponding set of {@link PropertyDef} property definitions and a {@link SortingPropertyExtractor}
024 * that will extract the properties from instances of the given class and sort them accordingly.
025 *
026 * @param <T> Java class to be introspected
027 * @see ProvidesProperty &#64;ProvidesProperty
028 * @see ProvidesPropertySort &#64;ProvidesPropertySort
029 */
030public class ProvidesPropertyScanner<T> {
031
032    private final ArrayList<PropertyDef<?>> propertyList = new ArrayList<>();
033    private final SortingPropertyExtractor<T> propertyExtractor;
034
035    /**
036     * Constructor.
037     *
038     * @param type Java class to be introspected
039     * @throws IllegalArgumentException if {@code type} is null
040     * @throws IllegalArgumentException if an annotated method with no {@linkplain ProvidesProperty#value property name specified}
041     *  has a name which cannot be interpreted as a bean property "getter" method
042     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}-annotated
043     *  fields or methods with the same {@linkplain ProvidesProperty#value property name}
044     */
045    public ProvidesPropertyScanner(Class<T> type) {
046
047        // Sanity check
048        if (type == null)
049            throw new IllegalArgumentException("null type");
050
051        // Scan for @ProvidesProperty and @ProvidesPropertySort annotations
052        final Set<MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo> providesPropertyMethods
053          = new MethodAnnotationScanner<T, ProvidesProperty>(type, ProvidesProperty.class).findAnnotatedMethods();
054        final Set<MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo> providesPropertySortMethods
055          = new MethodAnnotationScanner<T, ProvidesPropertySort>(type, ProvidesPropertySort.class).findAnnotatedMethods();
056
057        // Check for duplicate @ProvidesProperty names
058        final HashMap<String, MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo> providesPropertyNameMap = new HashMap<>();
059        for (MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo : providesPropertyMethods) {
060            final String propertyName = this.getPropertyName(methodInfo);
061
062            // Check for name conflict
063            final MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo previousInfo = providesPropertyNameMap.get(propertyName);
064            if (previousInfo == null) {
065                providesPropertyNameMap.put(propertyName, methodInfo);
066                continue;
067            }
068
069            // If there is a name conflict, the sub-type method declaration wins
070            switch (this.compareDeclaringClass(previousInfo.getMethod(), methodInfo.getMethod())) {
071            case 0:
072                throw new IllegalArgumentException("duplicate @" + ProvidesProperty.class.getSimpleName()
073                  + " declaration for property `" + propertyName + "' on method " + previousInfo.getMethod()
074                  + " and " + methodInfo.getMethod() + " declared in the same class");
075            case 1:
076                providesPropertyNameMap.put(propertyName, methodInfo);
077                break;
078            default:
079                break;
080            }
081        }
082
083        // Check for duplicate @ProvidesPropertySort names, etc.
084        final HashMap<String, MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo> providesPropertySortNameMap
085          = new HashMap<>();
086        for (MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo methodInfo : providesPropertySortMethods) {
087            final String propertyName = this.getSortPropertyName(methodInfo);
088
089            // Verify the method's return type (or wrapper type if primitive) implements Comparable or Comparator
090            Class<?> methodType = methodInfo.getMethod().getReturnType();
091            if (methodType.isPrimitive())
092                methodType = Primitive.get(methodType).getWrapperType();
093            if (!Comparable.class.isAssignableFrom(methodType) && !Comparator.class.isAssignableFrom(methodType)) {
094                throw new IllegalArgumentException("invalid @" + ProvidesPropertySort.class.getSimpleName()
095                  + " declaration for property `" + propertyName + "': method " + methodInfo.getMethod()
096                  + " return type " + methodType.getName() + " implements neither " + Comparable.class.getName()
097                  + " nor " + Comparator.class.getName());
098            }
099
100            // Check for name conflict
101            final MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo previousInfo
102              = providesPropertySortNameMap.get(propertyName);
103            if (previousInfo == null) {
104                providesPropertySortNameMap.put(propertyName, methodInfo);
105                continue;
106            }
107
108            // If there is a name conflict, the sub-type method declaration wins
109            switch (this.compareDeclaringClass(previousInfo.getMethod(), methodInfo.getMethod())) {
110            case 0:
111                throw new IllegalArgumentException("duplicate @" + ProvidesPropertySort.class.getSimpleName()
112                  + " declaration for property `" + propertyName + "' on method " + previousInfo.getMethod()
113                  + " and " + methodInfo.getMethod() + " declared in the same class");
114            case 1:
115                providesPropertySortNameMap.put(propertyName, methodInfo);
116                break;
117            default:
118                break;
119            }
120        }
121
122        // Build PropertyDef list
123        for (Map.Entry<String, MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo> e : providesPropertyNameMap.entrySet()) {
124            final String propertyName = e.getKey();
125            final MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo = e.getValue();
126
127            // Get property type
128            Class<?> propertyType = methodInfo.getMethod().getReturnType();
129
130            // Get property default value
131            Object defaultValue = null;
132            if (propertyType.isPrimitive()) {
133                final Primitive<?> primitiveType = Primitive.get(propertyType);
134                defaultValue = primitiveType.getDefaultValue();
135                propertyType = primitiveType.getWrapperType();
136            }
137
138            // Get sort property (if any)
139            final MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo sortMethodInfo
140              = providesPropertySortNameMap.get(propertyName);
141
142            // Create property definition
143            this.propertyList.add(this.createAnnotationPropertyDef(
144              propertyName, propertyType, defaultValue, methodInfo, sortMethodInfo));
145        }
146
147        // Build PropertyExtractor
148        this.propertyExtractor = new SortingPropertyExtractor<T>() {
149
150            @Override
151            @SuppressWarnings("unchecked")
152            public <V> V getPropertyValue(T obj, PropertyDef<V> propertyDef) {
153                if (!(propertyDef instanceof AnnotationPropertyDef))
154                    throw new IllegalArgumentException("unknown property " + propertyDef);
155                final AnnotationPropertyDef<V> annotationPropertyDef = (AnnotationPropertyDef<V>)propertyDef;
156                return propertyDef.getType().cast(annotationPropertyDef.getMethodInfo().invoke(obj));
157            }
158
159            @Override
160            @SuppressWarnings("unchecked")
161            public boolean canSort(PropertyDef<?> propertyDef) {
162                if (!(propertyDef instanceof AnnotationPropertyDef))
163                    return false;
164                final AnnotationPropertyDef<?> annotationPropertyDef = (AnnotationPropertyDef)propertyDef;
165                return annotationPropertyDef.getSortMethodInfo() != null;
166            }
167
168            @Override
169            @SuppressWarnings("unchecked")
170            public int sort(PropertyDef<?> propertyDef, T obj1, T obj2) {
171                if (!(propertyDef instanceof AnnotationPropertyDef))
172                    throw new UnsupportedOperationException("unknown property " + propertyDef);
173                final AnnotationPropertyDef<?> annotationPropertyDef = (AnnotationPropertyDef<?>)propertyDef;
174                return annotationPropertyDef.getComparator().compare(obj1, obj2);
175            }
176        };
177    }
178
179    /**
180     * Get the list of {@link PropertyDef}s generated from the annotated methods.
181     *
182     * <p>
183     * All of the properties in the returned list can be extracted from instances of this reader's configured
184     * class by the {@link PropertyExtractor} returned by {@link #getPropertyExtractor}.
185     *
186     * @return unmodifiable list of properties
187     * @see #getPropertyExtractor
188     */
189    public List<PropertyDef<?>> getPropertyDefs() {
190        return Collections.unmodifiableList(this.propertyList);
191    }
192
193    /**
194     * Get the {@link PropertyExtractor} that extracts Vaadin {@link com.vaadin.data.Property} values
195     * from instances of the annotated class when given one of the {@link PropertyDef}s returned by
196     * {@link #getPropertyDefs}.
197     *
198     * @see #getPropertyDefs
199     */
200    public SortingPropertyExtractor<T> getPropertyExtractor() {
201        return this.propertyExtractor;
202    }
203
204    private String getPropertyName(MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo) {
205        return methodInfo.getAnnotation().value().length() > 0 ?
206          methodInfo.getAnnotation().value() : methodInfo.getMethodPropertyName();
207    }
208
209    private String getSortPropertyName(MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo methodInfo) {
210        return methodInfo.getAnnotation().value().length() > 0 ?
211          methodInfo.getAnnotation().value() : methodInfo.getMethodPropertyName();
212    }
213
214    // This method exists solely to bind the generic type
215    private <V> AnnotationPropertyDef<V> createAnnotationPropertyDef(String propertyName, Class<V> propertyType,
216      Object defaultValue, MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo,
217      MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo sortMethodInfo) {
218        return new AnnotationPropertyDef<V>(propertyName, propertyType,
219          propertyType.cast(defaultValue), methodInfo, sortMethodInfo);
220    }
221
222    // Compare two methods to determine which one has the declaring class that is a sub-type of the other's
223    private int compareDeclaringClass(Method method1, Method method2) {
224        final Class<?> class1 = method1.getDeclaringClass();
225        final Class<?> class2 = method2.getDeclaringClass();
226        if (class1 == class2)
227            return 0;
228        if (class1.isAssignableFrom(class2))
229            return 1;
230        if (class2.isAssignableFrom(class1))
231            return -1;
232        throw new RuntimeException("internal error: incomparable classes " + class1.getName() + " and " + class2.getName());
233    }
234
235    private class AnnotationPropertyDef<V> extends PropertyDef<V> {
236
237        private final MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo;
238        private final MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo sortMethodInfo;
239
240        private Comparator<T> comparator;
241
242        AnnotationPropertyDef(String name, Class<V> type, V defaultValue,
243          MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo methodInfo,
244          MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo sortMethodInfo) {
245            super(name, type, defaultValue);
246            this.methodInfo = methodInfo;
247            this.sortMethodInfo = sortMethodInfo;
248        }
249
250        public MethodAnnotationScanner<T, ProvidesProperty>.MethodInfo getMethodInfo() {
251            return this.methodInfo;
252        }
253
254        public MethodAnnotationScanner<T, ProvidesPropertySort>.MethodInfo getSortMethodInfo() {
255            return this.sortMethodInfo;
256        }
257
258        public Comparator<T> getComparator() {
259
260            // Sanity check
261            if (this.sortMethodInfo == null)
262                throw new UnsupportedOperationException("can't sort property " + this);
263
264            // Already created?
265            if (this.comparator != null)
266                return this.comparator;
267
268            // Handle Comparator case
269            if (Comparator.class.isAssignableFrom(this.sortMethodInfo.getMethod().getReturnType())) {
270                this.comparator = new Comparator<T>() {
271
272                    private Comparator<T> comparator;
273
274                    @Override
275                    @SuppressWarnings("unchecked")
276                    public int compare(T obj1, T obj2) {
277                        if (this.comparator == null)
278                            this.comparator = (Comparator<T>)AnnotationPropertyDef.this.sortMethodInfo.invoke(obj1);
279                        return this.comparator.compare(obj1, obj2);
280                    }
281                };
282                return this.comparator;
283            }
284
285            // Handle Comparable case
286            this.comparator = new Comparator<T>() {
287                @Override
288                @SuppressWarnings("unchecked")
289                public int compare(T obj1, T obj2) {
290                    final Comparable<Object> value1 = (Comparable<Object>)AnnotationPropertyDef.this.sortMethodInfo.invoke(obj1);
291                    final Comparable<Object> value2 = (Comparable<Object>)AnnotationPropertyDef.this.sortMethodInfo.invoke(obj2);
292                    if (value1 == null && value2 != null)
293                        return -1;
294                    if (value1 != null && value2 == null)
295                        return 1;
296                    if (value1 == null && value2 == null)
297                        return 0;
298                    return value1.compareTo(value2);
299                }
300            };
301            return this.comparator;
302        }
303    }

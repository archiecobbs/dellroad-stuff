006package org.dellroad.stuff.vaadin7;
007
008import com.vaadin.server.DeploymentConfiguration;
009import com.vaadin.server.ServiceException;
010import com.vaadin.server.VaadinRequest;
011import com.vaadin.server.VaadinServlet;
012import com.vaadin.server.VaadinServletService;
013import com.vaadin.server.VaadinSession;
014
015import java.util.ArrayList;
016import java.util.List;
017import java.util.Properties;
018import java.util.WeakHashMap;
019
020import javax.servlet.ServletConfig;
021import javax.servlet.ServletException;
022
023/**
024 * A {@link VaadinServlet} that associates and manages a Spring
025 * {@link org.springframework.web.context.ConfigurableWebApplicationContext} with each
026 * {@link com.vaadin.server.VaadinSession} (aka, "Vaadin application" in the old terminology).
027 *
028 * <p>
029 * The {@code vaadinContextConfigLocation} servlet parameter may be used to specify the Spring XML config
030 * file location(s). For example:
031 *
032 * <blockquote><pre>
033 * &lt;servlet&gt;
034 *     &lt;servlet-name&gt;My Vaadin App&lt;/servlet-name&gt;
035 *     &lt;servlet-class&gt;org.dellroad.stuff.vaadin7.SpringVaadinServlet&lt;/servlet-class&gt;
036 *     &lt;init-param&gt;
037 *         &lt;param-name&gt;UI&lt;/param-name&gt;
038 *         &lt;param-value&gt;com.example.MyApplicationUI&lt;/param-value&gt;
039 *     &lt;/init-param&gt;
040 *     &lt;init-param&gt;
041 *         &lt;param-name&gt;configLocation&lt;/param-name&gt;
042 *         &lt;param-value&gt;classpath:com/example/MyApplicationContext.xml&lt;/param-value&gt;
043 *     &lt;/init-param&gt;
044 * &lt;/servlet&gt;
045 * </pre></blockquote>
046 *
047 * <p>
048 * The main function of this servlet is to create and register a {@link SpringVaadinSessionListener} as a listener on the
049 * {@link com.vaadin.server.VaadinService} associated with this servlet. The {@link SpringVaadinSessionListener} in turn detects
050 * the creation and destruction of Vaadin application instances (represented by {@link com.vaadin.server.VaadinSession}
051 * instances) and does the work of managing the associated Spring application contexts.
052 *
053 * <p>
054 * Use of this servlet in place of the standard Vaadin servlet is required for the {@link VaadinConfigurable @VaadinConfigurable}
055 * annotation to work.
056 *
057 * <p>
058 * Supported URL parameters:
059 * <div style="margin-left: 20px;">
060 * <table border="1" cellpadding="3" cellspacing="0" summary="Servlet URL Parameters">
061 * <tr bgcolor="#ccffcc">
062 *  <th align="left">Parameter Name</th>
063 *  <th align="left">Required?</th>
064 *  <th align="left">Description</th>
065 * </tr>
066 * <tr>
067 * <td>{@code applicationName}</td>
068 * <td align="center">No</td>
069 * <td>
070 *  Vaadin application name. Used for logging purposes and as the name of the XML application context file
071 *  when {@code configLocation} is not specified. If this parameter is not specified, the
072 *  name of the servlet is used.
073 * </td>
074 * </tr>
075 * <tr>
076 * <td>{@code configLocation}</td>
077 * <td align="center">No</td>
078 * <td>
079 *  Location of Spring application context XML file(s). Multiple locations are separated by whitespace.
080 *  If omitted, {@code /WEB-INF/ServletName.xml} is used, where {@code ServletName} is the name of the Vaadin
081 *  application (see {@code applicationName}).
082 * </td>
083 * </tr>
084 * <tr>
085 * <td>{@code listenerClass}</td>
086 * <td align="center">No</td>
087 * <td>
088 *  Specify the name of a custom class extending {@link SpringVaadinSessionListener} and having the same constructor arguments.
089 *  If omitted, {@link SpringVaadinSessionListener} is used.
090 * </td>
091 * </tr>
092 * <tr>
093 * <td>{@code sessionTracking}</td>
094 * <td align="center">No</td>
095 * <td>
096 *  Boolean value that configures whether the {@link SpringVaadinSessionListener} should track Vaadin sessions; default
097 *  {@code false}. If set to {@code true}, then {@link #getSessions} can be used to access all active sessions.
098 *  Session tracking should not be used unless sessions are normally kept in memory; e.g., don't use session tracking
099 *  when sessions are being serialized and persisted. See also {@link VaadinSessionContainer}.
100 * </td>
101 * </tr>
102 * <tr>
103 * <td>{@code maxSessions}</td>
104 * <td align="center">No</td>
105 * <td>
106 *  Configures a limit on the number of simultaneous Vaadin sessions that may exist at one time. Going over this
107 *  limit will result in a {@link com.vaadin.server.ServiceException} being thrown. A zero or negative number
108 *  means there is no limit (this is the default). Ignored unless {@value #SESSION_TRACKING_PARAMETER} is set to {@code true}.
109 * </td>
110 * </tr>
111 * </table>
112 * </div>
113 *
114 * <p>
115 * Note: if any beans are declared {@code scope="session"} in your application context, you must register Spring's
116 * {@link org.springframework.web.context.request.RequestContextListener} as a listener in your {@code web.xml}
117 * (normally, Spring's {@link org.springframework.web.servlet.DispatcherServlet} would handle this task).
118 *
119 * @see SpringVaadinSessionListener
120 * @see VaadinConfigurable
121 * @see VaadinApplication
122 */
123@SuppressWarnings("serial")
124public class SpringVaadinServlet extends VaadinServlet {
125
126    /**
127     * Servlet initialization parameter (<code>{@value #CONFIG_LOCATION_PARAMETER}</code>) used to specify
128     * the location(s) of the Spring application context XML file(s). Multiple XML files may be separated by whitespace.
129     * This parameter is optional.
130     */
131    public static final String CONFIG_LOCATION_PARAMETER = "configLocation";
132
133    /**
134     * Servlet initialization parameter (<code>{@value #LISTENER_CLASS_PARAMETER}</code>) used to specify
135     * the name of an custom subclass of {@link SpringVaadinSessionListener}.
136     * This parameter is optional.
137     */
138    public static final String LISTENER_CLASS_PARAMETER = "listenerClass";
139
140    /**
141     * Servlet initialization parameter (<code>{@value #APPLICATION_NAME_PARAMETER}</code>) used to specify
142     * the name the application.
143     * This parameter is optional.
144     */
145    public static final String APPLICATION_NAME_PARAMETER = "applicationName";
146
147    /**
148     * Servlet initialization parameter (<code>{@value #SESSION_TRACKING_PARAMETER}</code>) that enables
149     * tracking of all Vaadin session.
150     * This parameter is optional, and defaults to <code>false</code>.
151     */
152    public static final String SESSION_TRACKING_PARAMETER = "sessionTracking";
153
154    /**
155     * Servlet initialization parameter (<code>{@value #MAX_SESSIONS_PARAMETER}</code>) that configures the
156     * maximum number of simultaneous Vaadin sessions. Requires {@link #SESSION_TRACKING_PARAMETER} to be set to {@code true}.
157     * This parameter is optional, and defaults to zero, which means no limit.
158     */
159    public static final String MAX_SESSIONS_PARAMETER = "maxSessions";
160
161    // We use weak references to avoid leaks caused by exceptions in SessionInitListeners; see http://dev.vaadin.com/ticket/12915
162    private final WeakHashMap<VaadinSession, Void> liveSessions = new WeakHashMap<>();
163
164    private String servletName;
165
166    @Override
167    public void init(ServletConfig config) throws ServletException {
168        this.servletName = config.getServletName();
169        if (this.servletName == null)
170            throw new IllegalArgumentException("null servlet name");
171        super.init(config);
172    }
173
174    @Override
175    protected void servletInitialized() throws ServletException {
176
177        // Sanity check
178        if (this.servletName == null)
179            throw new IllegalArgumentException("servlet not initialized");
180
181        // Defer to superclass
182        super.servletInitialized();
183
184        // Get params
185        final VaadinServletService servletService = this.getService();
186        final Properties params = servletService.getDeploymentConfiguration().getInitParameters();
187        final String contextLocation = params.getProperty(CONFIG_LOCATION_PARAMETER);
188        final String listenerClassName = params.getProperty(LISTENER_CLASS_PARAMETER);
189        String applicationName = params.getProperty(APPLICATION_NAME_PARAMETER);
190        if (applicationName == null)
191            applicationName = this.servletName;
192
193        // Detect listener class to use
194        Class<? extends SpringVaadinSessionListener> listenerClass = SpringVaadinSessionListener.class;
195        if (listenerClassName != null) {
196            try {
197                listenerClass = Class.forName(listenerClassName, false, Thread.currentThread().getContextClassLoader())
198                  .asSubclass(SpringVaadinSessionListener.class);
199            } catch (RuntimeException e) {
200                throw e;
201            } catch (Exception e) {
202                throw new ServletException("error finding class " + listenerClassName, e);
203            }
204        }
205
206        // Create session listener
207        SpringVaadinSessionListener sessionListener;
208        try {
209            sessionListener = listenerClass.getConstructor(String.class, String.class)
210              .newInstance(applicationName, contextLocation);
211        } catch (RuntimeException e) {
212            throw e;
213        } catch (Exception e) {
214            throw new ServletException("error instantiating " + listenerClass, e);
215        }
216
217        // Register session listener
218        servletService.addSessionInitListener(sessionListener);
219        servletService.addSessionDestroyListener(sessionListener);
220    }
221
222    @Override
223    protected VaadinServletService createServletService(DeploymentConfiguration deploymentConfiguration) throws ServiceException {
224
225        // Get session tracking parameters
226        final Properties params = deploymentConfiguration.getInitParameters();
227        final boolean sessionTracking = Boolean.valueOf(params.getProperty(SESSION_TRACKING_PARAMETER));
228        int maxSessionsParam = 0;
229        try {
230            maxSessionsParam = Integer.parseInt(params.getProperty(MAX_SESSIONS_PARAMETER));
231        } catch (Exception e) {
232            // ignore
233        }
234        final int maxSessions = maxSessionsParam;
235
236        // If not tracking sessions, do the normal thing
237        if (!sessionTracking)
238            return super.createServletService(deploymentConfiguration);
239
240        // Return a VaadinServletService that tracks sessions
241        final VaadinServletService service = new VaadinServletService(this, deploymentConfiguration) {
242
243            @Override
244            protected VaadinSession createVaadinSession(VaadinRequest request) throws ServiceException {
245                final VaadinSession session;
246                synchronized (SpringVaadinServlet.this.liveSessions) {
247                    if (maxSessions > 0 && SpringVaadinServlet.this.liveSessions.size() >= maxSessions)
248                        throw new ServiceException("The maximum number of active sessions has been reached");
249                    session = super.createVaadinSession(request);
250                    SpringVaadinServlet.this.liveSessions.put(session, null);
251                }
252                return session;
253            }
254
255            @Override
256            public void fireSessionDestroy(VaadinSession session) {
257                synchronized (SpringVaadinServlet.this.liveSessions) {
258                    SpringVaadinServlet.this.liveSessions.remove(session);
259                }
260                super.fireSessionDestroy(session);
261            }
262        };
263        service.init();
264        return service;
265    }
266
267    /**
268     * Get all live {@link VaadinSession}s associated with this instance.
269     *
270     * @return live tracked sessions, or an empty collection if session tracking is not enabled
271     * @see VaadinSessionContainer
272     */
273    public List<VaadinSession> getSessions() {
274        synchronized (this.liveSessions) {
275            return new ArrayList<VaadinSession>(this.liveSessions.keySet());
276        }
277    }
278
279    /**
280     * Get the {@link SpringVaadinServlet} that is associated with the given {@link VaadinSession}.
281     *
282     * @throws IllegalStateException if the {@link VaadinServlet} associated with {@code session} is not a
283     *  {@link SpringVaadinServlet}
284     */
285    public static SpringVaadinServlet getServlet(VaadinSession session) {
286        if (session == null)
287            throw new IllegalArgumentException("null session");
288        if (!(session.getService() instanceof VaadinServletService))
289            throw new IllegalStateException("the VaadinService associated with the session is not a VaadinServletService instance");
290        final VaadinServletService service = (VaadinServletService)session.getService();
291        if (!(service.getServlet() instanceof SpringVaadinServlet))
292            throw new IllegalStateException("the VaadinServlet associated with the session is not a SpringVaadinServlet instance");
293        return (SpringVaadinServlet)service.getServlet();
294    }
295}

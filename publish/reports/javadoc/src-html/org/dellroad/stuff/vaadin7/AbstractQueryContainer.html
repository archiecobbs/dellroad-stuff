020/**
021 * Support superclass for read-only {@link Container} implementations where each {@link Item} in the container
022 * is backed by a Java object, and the Java objects are generated via {@linkplain #query a query} that returns
023 * a {@link QueryList} containing only some portion of the total list of backing objects at any one time.
024 *
025 * <p>
026 * This {@link Container}'s {@link Property}s are defined via {@link PropertyDef}s, and a {@link PropertyExtractor}
027 * is used to actually extract the property values from each underlying object (alternately, subclasses can override
028 * {@link #getPropertyValue getPropertyValue()}). However, the easist way to configure the container {@link Property}s
029 * is to pass a {@link ProvidesProperty &#64;ProvidesProperty}-annotated Java class to the {@link #AbstractQueryContainer(Class)}
030 * constructor.
031 * </p>
032 *
033 * <p>
034 * This {@link Container}'s item ID's are simply the indexes of the corresponding objects in the overall list. The {@link QueryList}
035 * interface is designed for scalability; at minimum, it is required to provide only the size of the total list
036 * (which may only be an estimate) and one backing object at a specified index.
037 *
038 * <p>
039 * This class will invoke {@link #query} as needed to (re)generate the {@link QueryList}; the {@link QueryList} is then cached.
040 * However, if any invocation of {@link QueryList#get} throws an {@link InvalidQueryListException}, then the cached
041 * list is discarded and {@link #query} is invoked again to regenerate it. In this way, the {@link QueryList} is
042 * allowed to decide, on demand, when it is invalid or incapable of providing a specific list member. For example,
043 * when using JPA, a list may be considered invalid if the current EntityManager session has changed.
044 * </p>
045 *
046 * <p>
047 * Note that the {@link QueryList} being invalid is an orthogonal concept from the contents of the list having changed.
048 * Invalid means "this list can no longer be used" while changed means "this list contains out-of-date information".
049 * Normally, the latter implies the former (but not vice-versa). The list becoming invalid does not in itself not cause
050 * any notifications to be sent, so no new query will be performed until e.g. the user interface explicitly requests
051 * more information.
052 * </p>
053 *
054 * <p>
055 * Therefore, if the list size or content changes, first invoke {@link #invalidate} to discard the cached {@link QueryList},
056 * and then {@link #fireItemSetChange} to notify listeners; for convenience, {@link #reload} will perform these two steps for
057 * you. The new size and content will be provided by the {@link QueryList} returned by the next invocation of {@link #query}.
058 * </p>
059 *
060 * <p>
061 * The subclass may forcibly invalidate the current {@link QueryList} via {@link #invalidate}; this merely discards it
062 * and will force a new invocation of {@link #query} on the next container access. In many situations, however,
063 * the use of {@link #invalidate} is never required.
064 * </p>
065 *
066 * <p>
067 * For scalability reasons the {@link QueryList} may actually only contain a portion of the list, throwing
068 * {@link InvalidQueryListException}s when other list members are accessed.  An "index hint" parameter provided to
069 * {@link #query} indicates which member of the container is of current interest. The returned {@link QueryList}
070 * is required to provide exception-free access only to the indicated member, so in the extreme case only a single
071 * list member could be kept. In practice, normally range of members near to the index hint would be kept;
072 * see for example {@link WindowQueryList}.
073 * </p>
074 *
075 * @param <T> the type of the Java objects that back each {@link Item} in the container
076 * @see QueryList
077 * @see SimpleQueryList
078 * @see WindowQueryList
079 * @see AbstractUnsizedContainer
080 */
081@SuppressWarnings({ "serial", "deprecation" })
082public abstract class AbstractQueryContainer<T> extends AbstractContainer implements PropertyExtractor<T>,
083  Container.Ordered, Container.Indexed, Container.PropertySetChangeNotifier, Container.ItemSetChangeNotifier, Connectable {

228        if (propertyDefs == null)
229            propertyDefs = Collections.<PropertyDef<?>>emptySet();
230        this.propertyMap.clear();
231        for (PropertyDef<?> propertyDef : propertyDefs) {
232            if (this.propertyMap.put(propertyDef.getName(), propertyDef) != null)
233                throw new IllegalArgumentException("duplicate property name `" + propertyDef.getName() + "'");
234        }
235        this.fireContainerPropertySetChange();
236    }

279    /**
280     * Perform a query to (re)generate the list of Java objects backing this container.
281     *
282     * <p>
283     * The particular position in the list we are interested in is given as a hint by the {@code hint} parameter.
284     * That is, an invocation of {@link QueryList#get}{@code (hint)} is likely immediately after this method
285     * returns and if so it must complete without throwing an exception, unless {@code hint} is out of range.
286     * </p>
287     *
288     * <p>
289     * The {@code hint} can be used to implement a highly scalable query list containing external objects
290     * (such as from a database) where only a small "window" of objects is actually kept in memory at any one time.
291     * Of course, implementations are also free to ignore {@code hint}. However, the returned {@link QueryList}
292     * must at least tolerate one invocation of {@link QueryList#get get}{@code (hint)} without throwing an exception
293     * when {@code hint} is less that the {@link QueryList#size size()} of the returned {@link QueryList}.
294     * </p>
295     *
296     * @param hint index of the list element we are interested in
297     * @return list of Java objects backing this container
298     */
299    protected abstract QueryList<? extends T> query(long hint);

301    /**
302     * Discard the current cached {@link QueryList}, if any.
303     */
304    protected void invalidate() {
305        this.queryList = null;
306        this.itemMap.clear();
307    }

309    /**
310     * Get the Java backing object at the given index in the list.
311     *
312     * @return backing object, or null if {@code index} is out of range
313     */
314    protected T getJavaObject(int index) {
315        Exception exception;
316        long currentSize;
317        int attempt = 1;
318        while (true) {
319            currentSize = this.ensureList(index);
320            if (index < 0 || index >= currentSize)
321                return null;
322            try {
323                return this.queryList.get(index);
324            } catch (InvalidQueryListException e) {
325                this.invalidate();
326                if (attempt == 100) {           // avoid infinite loops
327                    exception = e;
328                    break;
329                }
330            } catch (IndexOutOfBoundsException e) {
331                exception = e;
332                break;
333            }
334            attempt++;
335        }
336
337        // The QueryList is behaving badly
338        throw new RuntimeException("query(" + index + ") returned a QueryList with size() = "
339          + currentSize + " but QueryList.get(" + index + ") failed (attempt #" + attempt + ")", exception);
340    }

342    /**
343     * Ensure we have a cached query list.
344     *
345     * @param hint index of the list element we are interested in, passed to {@link #query} if no query list is cached
346     * @return the size of the cached query list
347     */
348    protected long ensureList(int hint) {
349        if (this.queryList == null) {
350            this.queryList = this.query(hint);
351            final long oldTotalSize = this.totalSize;
352            final long newTotalSize = this.queryList.size();
353            this.totalSize = newTotalSize;
354            if (oldTotalSize != -1 && newTotalSize != oldTotalSize)
355                this.handleSizeChange();
356        }
357        return this.totalSize;
358    }

404    @Override
405    public BackedItem<T> getItem(Object itemId) {
406        if (!(itemId instanceof Integer))
407            return null;
408        int index = ((Integer)itemId).intValue();
409        T obj = this.getJavaObject(index);
410        if (obj == null)
411            return null;
412        BackedItem<T> item = this.itemMap.get(index);
413        if (item == null) {
414            item = this.createBackedItem(obj, this.propertyMap.values(), this);
415            this.itemMap.put(index, item);
416        }
417        return item;
418    }

532    @Override
533    public List<Integer> getItemIds(int startIndex, int numberOfItems) {
534        if (numberOfItems < 0)
535            throw new IllegalArgumentException("numberOfItems < 0");
536        final long size = this.ensureList(startIndex);
537        if (startIndex < 0 || startIndex > size)
538            throw new IndexOutOfBoundsException("startIndex=" + startIndex + " but size=" + size);
539        if (startIndex + numberOfItems > size)
540            numberOfItems = (int)(size - startIndex);
541        return new IntList(startIndex, numberOfItems);
542    }

671    private static class IntList extends AbstractList<Integer> {
672
673        private final int min;
674        private final int size;
675
676        public IntList(int size) {
677            this(0, size);
678        }
679
680        public IntList(int min, int size) {
681            if (size < 0)
682                throw new IllegalArgumentException("size < 0");
683            this.min = min;
684            this.size = size;
685        }
686
687        @Override
688        public int size() {
689            return this.size;
690        }
691
692        @Override
693        public Integer get(int index) {
694            if (index < 0 || index >= this.size)
695                throw new IndexOutOfBoundsException();
696            return this.min + index;
697        }
698    }

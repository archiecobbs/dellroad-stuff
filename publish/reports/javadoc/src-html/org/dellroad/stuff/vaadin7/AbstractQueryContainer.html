020/**
021 * Support superclass for read-only {@link Container} implementations where each {@link Item} in the container
022 * is backed by a Java object, and the Java objects are generated via {@linkplain #query a query} that returns
023 * a {@link QueryList} containing only some portion of the total list of backing objects at any one time.
024 *
025 * <p>
026 * This {@link Container}'s {@link Property}s are defined via {@link PropertyDef}s, and a {@link PropertyExtractor}
027 * is used to actually extract the property values from each underlying object (alternately, subclasses can override
028 * {@link #getPropertyValue getPropertyValue()}). However, the easist way to configure the container {@link Property}s
029 * is to pass a {@link ProvidesProperty &#64;ProvidesProperty}-annotated Java class to the {@link #AbstractQueryContainer(Class)}
030 * constructor.
031 *
032 * <p>
033 * This {@link Container}'s item ID's are simply the indexes of the corresponding objects in the overall list. The {@link QueryList}
034 * interface is designed for scalability; at minimum, it is required to provide only the size of the total list
035 * (which may only be an estimate) and one backing object at a specified index.
036 *
037 * <p>
038 * This class will invoke {@link #query} as needed to (re)generate the {@link QueryList}; the {@link QueryList} is then cached.
039 * However, if any invocation of {@link QueryList#get} throws an {@link InvalidQueryListException}, then the cached
040 * list is discarded and {@link #query} is invoked again to regenerate it. In this way, the {@link QueryList} is
041 * allowed to decide, on demand, when it is invalid or incapable of providing a specific list member. For example,
042 * when using JPA, a list may be considered invalid if the current EntityManager session has changed.
043 *
044 * <p>
045 * Note that the {@link QueryList} being invalid is an orthogonal concept from the contents of the list having changed.
046 * Invalid means "this list can no longer be used" while changed means "this list contains out-of-date information".
047 * Normally, the latter implies the former (but not vice-versa). The list becoming invalid does not in itself not cause
048 * any notifications to be sent, so no new query will be performed until e.g. the user interface explicitly requests
049 * more information.
050 *
051 * <p>
052 * Therefore, if the list size or content changes, first invoke {@link #invalidate} to discard the cached {@link QueryList},
053 * and then {@link #fireItemSetChange} to notify listeners; for convenience, {@link #reload} will perform these two steps for
054 * you. The new size and content will be provided by the {@link QueryList} returned by the next invocation of {@link #query}.
055 *
056 * <p>
057 * The subclass may forcibly invalidate the current {@link QueryList} via {@link #invalidate}; this merely discards it
058 * and will force a new invocation of {@link #query} on the next container access. In many situations, however,
059 * the use of {@link #invalidate} is never required.
060 *
061 * <p>
062 * For scalability reasons the {@link QueryList} may actually only contain a portion of the list, throwing
063 * {@link InvalidQueryListException}s when other list members are accessed.  An "index hint" parameter provided to
064 * {@link #query} indicates which member of the container is of current interest. The returned {@link QueryList}
065 * is required to provide exception-free access only to the indicated member, so in the extreme case only a single
066 * list member could be kept. In practice, normally range of members near to the index hint would be kept;
067 * see for example {@link WindowQueryList}.
068 *
069 * @param <T> the type of the Java objects that back each {@link Item} in the container
070 * @see QueryList
071 * @see SimpleQueryList
072 * @see WindowQueryList
073 * @see AbstractUnsizedContainer
074 */
075@SuppressWarnings({ "serial", "deprecation" })
076public abstract class AbstractQueryContainer<T> extends AbstractContainer implements PropertyExtractor<T>,
077  Container.Ordered, Container.Indexed, Container.PropertySetChangeNotifier, Container.ItemSetChangeNotifier, Connectable {

217        if (propertyDefs == null)

219        this.propertyMap.clear();
220        for (PropertyDef<?> propertyDef : propertyDefs) {

222                throw new IllegalArgumentException("duplicate property name `" + propertyDef.getName() + "'");
223        }
224        this.fireContainerPropertySetChange();
225    }

266    /**
267     * Perform a query to (re)generate the list of Java objects backing this container.
268     *
269     * <p>
270     * The particular position in the list we are interested in is given as a hint by the {@code hint} parameter.
271     * That is, an invocation of {@link QueryList#get}{@code (hint)} is likely immediately after this method
272     * returns and if so it must complete without throwing an exception, unless {@code hint} is out of range.
273     *
274     * <p>
275     * The {@code hint} can be used to implement a highly scalable query list containing external objects
276     * (such as from a database) where only a small "window" of objects is actually kept in memory at any one time.
277     * Of course, implementations are also free to ignore {@code hint}. However, the returned {@link QueryList}
278     * must at least tolerate one invocation of {@link QueryList#get get}{@code (hint)} without throwing an exception
279     * when {@code hint} is less that the {@link QueryList#size size()} of the returned {@link QueryList}.
280     *
281     * @param hint index of the list element we are interested in
282     * @return list of Java objects backing this container
283     */
284    protected abstract QueryList<? extends T> query(long hint);

289    protected void invalidate() {
290        this.queryList = null;
291        this.itemMap.clear();
292    }

294    /**
295     * Get the Java backing object at the given index in the list.
296     *
297     * @return backing object, or null if {@code index} is out of range
298     */
299    protected T getJavaObject(int index) {
300        Exception exception;
301        long currentSize;
302        int attempt = 1;
303        while (true) {
304            currentSize = this.ensureList(index);
305            if (index < 0 || index >= currentSize)
306                return null;
307            try {
308                return this.queryList.get(index);
309            } catch (InvalidQueryListException e) {
310                this.invalidate();
311                if (attempt == 100) {           // avoid infinite loops
312                    exception = e;
313                    break;
314                }
315            } catch (IndexOutOfBoundsException e) {
316                exception = e;
317                break;
318            }
319            attempt++;
320        }
321
322        // The QueryList is behaving badly
323        throw new RuntimeException("query(" + index + ") returned a QueryList with size() = "
324          + currentSize + " but QueryList.get(" + index + ") failed (attempt #" + attempt + ")", exception);
325    }

327    /**
328     * Ensure we have a cached query list.
329     *
330     * @param hint index of the list element we are interested in, passed to {@link #query} if no query list is cached
331     * @return the size of the cached query list
332     */
333    protected long ensureList(int hint) {
334        if (this.queryList == null) {
335            this.queryList = this.query(hint);
336            final long oldTotalSize = this.totalSize;
337            final long newTotalSize = this.queryList.size();
338            this.totalSize = newTotalSize;
339            if (oldTotalSize != -1 && newTotalSize != oldTotalSize)
340                this.handleSizeChange();
341        }
342        return this.totalSize;
343    }

385    @Override
386    public BackedItem<T> getItem(Object itemId) {
387        if (!(itemId instanceof Integer))
388            return null;
389        int index = ((Integer)itemId).intValue();
390        T obj = this.getJavaObject(index);
391        if (obj == null)
392            return null;
393        BackedItem<T> item = this.itemMap.get(index);
394        if (item == null) {
395            item = this.createBackedItem(obj, this.propertyMap.values(), this);
396            this.itemMap.put(index, item);
397        }
398        return item;
399    }

513    @Override
514    public List<Integer> getItemIds(int startIndex, int numberOfItems) {
515        if (numberOfItems < 0)
516            throw new IllegalArgumentException("numberOfItems < 0");
517        final long size = this.ensureList(startIndex);
518        if (startIndex < 0 || startIndex > size)
519            throw new IndexOutOfBoundsException("startIndex=" + startIndex + " but size=" + size);
520        if (startIndex + numberOfItems > size)
521            numberOfItems = (int)(size - startIndex);
522        return new IntList(startIndex, numberOfItems);
523    }

652    private static class IntList extends AbstractList<Integer> {
653
654        private final int min;
655        private final int size;
656
657        public IntList(int size) {
658            this(0, size);
659        }
660
661        public IntList(int min, int size) {
662            if (size < 0)
663                throw new IllegalArgumentException("size < 0");
664            this.min = min;
665            this.size = size;
666        }
667
668        @Override
669        public int size() {
670            return this.size;
671        }
672
673        @Override
674        public Integer get(int index) {
675            if (index < 0 || index >= this.size)
676                throw new IndexOutOfBoundsException();
677            return this.min + index;
678        }
679    }

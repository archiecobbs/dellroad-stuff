049/**
050 * Automatically builds and binds fields for a Java bean annotated with {@link FieldBuilder} annotations.
051 * The various nested {@link FieldBuilder} annotation types annotate Java bean property "getter" methods and specify
052 * how the the bean properties of that class should be edited using {@link AbstractField}s. This allows all information
053 * about how to edit a Java model class to stay contained within that class.
054 *
055 * <p>
056 * This class supports two types of annotations: first, the {@link ProvidesField &#64;ProvidesField} annotation annotates
057 * a method that knows how to build an {@link AbstractField} suitable for editing the bean property specified by
058 * its {@link ProvidesField#value value()}. So {@link ProvidesField &#64;ProvidesField} is analgous to
059 * {@link ProvidesProperty &#64;ProvidesProperty}, except that it defines an editing field rather than a container property.
060 * </p>
061 *
062 * <p>
063 * The {@link FieldBuilder.AbstractField &#64;FieldBuilder.AbstractField} hierarchy annotations are the other type of annotation.
064 * These annotations annotate a Java bean property getter method and specify how to configure an {@link AbstractField} subclass
065 * instance to edit the bean property corresponding to the getter method.
066 * {@link FieldBuilder.AbstractField &#64;FieldBuilder.AbstractField} is the top level annotation in a hierarchy of annotations
067 * that correspond to the {@link AbstractField} class hierarchy. {@link FieldBuilder.AbstractField &#64;FieldBuilder.AbstractField}
068 * corresponds to {@link AbstractField}, and its properties configure corresponding {@link AbstractField} properties.
069 * More specific annotations correspond to the various {@link AbstractField} subclasses,
070 * for example {@link ComboBox &#64;FieldBuilder.ComboBox} corresponds to {@link ComboBox}.
071 * When using more specific annotations, the "superclass" annotations configure the corresponding superclass' properties.
072 * </p>
073 *
074 * <p>
075 * A simple example shows how these annotations are used:
076 * <blockquote><pre>
077 * // Use a 10x40 TextArea to edit the "description" property
078 * <b>&#64;FieldBuilder.AbstractField(caption = "Description:")</b>
079 * <b>&#64;FieldBuilder.TextArea(columns = 40, rows = 10)</b>
080 * public String getDescription() {
081 *     return this.description;
082 * }
083 *
084 * // Use my own custom field to edit the "foobar" property
085 * <b>&#64;FieldBuilder.ProvidesField("foobar")</b>
086 * private MyCustomField createFoobarField() {
087 *     ...
088 * }
089 * </pre></blockquote>
090 * </p>
091 *
092 * <p>
093 * A {@link FieldBuilder} instance will read these annotations and build the fields automatically. For example:
094 * <blockquote><pre>
095 * // Create fields based on FieldGroup.* annotations
096 * Person person = new Person("Joe Smith", 100);
097 * BeanFieldGroup&lt;Person&gt; fieldGroup = <b>FieldBuilder.buildFieldGroup(person)</b>;
098 *
099 * // Layout the fields in a form
100 * FormLayout layout = new FormLayout();
101 * for (Field&lt;?&gt; field : fieldGroup.getFields())
102 *     layout.addComponent(field);
103 * </pre></blockquote>
104 * </p>
105 *
106 * <p>
107 * For all annotations in the {@link FieldBuilder.AbstractField &#64;FieldBuilder.AbstractField} hierarchy, leaving properties
108 * set to their default values results in the default behavior.
109 * </p>
110 *
111 * @see AbstractSelect
112 * @see AbstractTextField
113 * @see CheckBox
114 * @see ComboBox
115 * @see DateField
116 * @see EnumComboBox
117 * @see ListSelect
118 * @see PasswordField
119 * @see TextArea
120 * @see TextField
121 */

124    /**
125     * Introspect for {@link FieldBuilder} annotations on property getter methods of the
126     * {@link BeanFieldGroup}'s data source, and then build and bind the corresponding fields.
127     *
128     * @param fieldGroup field group to configure
129     * @throws IllegalArgumentException if {@code fieldGroup} is null
130     * @throws IllegalArgumentException if {@code fieldGroup} does not yet
131     *  {@linkplain BeanFieldGroup#setItemDataSource(Object) have a data source}
132     */
133    public void buildAndBind(BeanFieldGroup<?> fieldGroup) {
134
135        // Sanity check
136        if (fieldGroup == null)
137            throw new IllegalArgumentException("null beanType");
138        final BeanItem<?> beanItem = fieldGroup.getItemDataSource();
139        if (beanItem == null)
140            throw new IllegalArgumentException("fieldGroup does not yet have a data source");
141
142        // Scan bean properties to build fields
143        for (Map.Entry<String, Field<?>> entry : this.buildBeanPropertyFields(beanItem.getBean()).entrySet())
144            fieldGroup.bind(entry.getValue(), entry.getKey());
145    }

147    /**
148     * Introspect for {@link FieldBuilder} annotations on property getter methods and build
149     * a mapping from Java bean property name to a field that may be used to edit that property.
150     * Only bean properties that have {@link FieldBuilder} annotations are detected.
151     *
152     * @param bean Java bean
153     * @return mapping from bean property name to field
154     * @throws IllegalArgumentException if {@code bean} is null
155     * @throws IllegalArgumentException if invalid or conflicting annotations are encountered
156     */
157    public Map<String, Field<?>> buildBeanPropertyFields(Object bean) {
158
159        // Sanity check
160        if (bean == null)
161            throw new IllegalArgumentException("null bean");
162
163        // Look for all bean property getter methods
164        BeanInfo beanInfo;
165        try {
166            beanInfo = Introspector.getBeanInfo(bean.getClass());
167        } catch (IntrospectionException e) {
168            throw new RuntimeException("unexpected exception", e);
169        }
170        final HashMap<String, Method> getterMap = new HashMap<>();              // contains all getter methods
171        for (PropertyDescriptor propertyDescriptor : beanInfo.getPropertyDescriptors()) {
172            Method method = propertyDescriptor.getReadMethod();
173
174            // Work around Introspector stupidity where it returns overridden superclass' getter method
175            if (method != null && method.getClass() != bean.getClass()) {
176                for (Class<?> c = bean.getClass(); c != null && c != method.getClass(); c = c.getSuperclass()) {
177                    try {
178                        method = c.getDeclaredMethod(method.getName(), method.getParameterTypes());
179                    } catch (Exception e) {
180                        continue;
181                    }
182                    break;
183                }
184            }
185
186            // Add getter, if appropriate
187            if (method != null && method.getReturnType() != void.class && method.getParameterTypes().length == 0)
188                getterMap.put(propertyDescriptor.getName(), method);
189        }
190
191        // Scan getters for FieldBuilder.* annotations other than FieldBuidler.ProvidesField
192        final HashMap<String, Field<?>> map = new HashMap<>();              // contains @FieldBuilder.* fields
193        for (Map.Entry<String, Method> entry : getterMap.entrySet()) {
194            final String propertyName = entry.getKey();
195            final Method method = entry.getValue();
196
197            // Get annotations, if any
198            final List<AnnotationApplier<?, ?>> applierList = this.buildApplierList(method);
199            if (applierList.isEmpty())
200                continue;
201
202            // Build field
203            map.put(propertyName, this.buildField(applierList, "method " + method));
204        }
205
206        // Scan all methods for @FieldBuilder.ProvidesField annotations
207        final HashMap<String, Method> providerMap = new HashMap<>();            // contains @FieldBuilder.ProvidesField methods
208        this.buildProviderMap(providerMap, bean.getClass());
209
210        // Check for conflicts between @FieldBuidler.ProvidesField and other annotations and add fields to map
211        for (Map.Entry<String, Method> entry : providerMap.entrySet()) {
212            final String propertyName = entry.getKey();
213            final Method method = entry.getValue();
214
215            // Verify field is not already defined
216            if (map.containsKey(propertyName)) {
217                throw new IllegalArgumentException("conflicting annotations exist for property `" + propertyName + "': annotation @"
218                  + ProvidesField.class.getName() + " on method " + method
219                  + " cannot be combined with other @FieldBuilder.* annotation types");
220            }
221
222            // Invoke method to create field
223            Field<?> field;
224            try {
225                method.setAccessible(true);
226            } catch (Exception e) {
227                // ignore
228            }
229            try {
230                field = (Field<?>)method.invoke(bean);
231            } catch (Exception e) {
232                throw new RuntimeException("error invoking @" + ProvidesField.class.getName()
233                  + " annotation on method " + method, e);
234            }
235
236            // Save field
237            map.put(propertyName, field);
238        }
239
240        // Done
241        return map;
242    }

244    // This method exists solely to bind the generic type
245    private <T> void buildProviderMap(Map<String, Method> providerMap, Class<T> type) {
246        final MethodAnnotationScanner<T, ProvidesField> scanner
247          = new MethodAnnotationScanner<T, ProvidesField>(type, ProvidesField.class);
248        for (MethodAnnotationScanner<T, ProvidesField>.MethodInfo methodInfo : scanner.findAnnotatedMethods())
249            this.buildProviderMap(providerMap, methodInfo.getMethod().getDeclaringClass(), methodInfo.getMethod().getName());
250    }

252    // Used by buildBeanPropertyFields() to validate @FieldBuilder.ProvidesField annotations
253    private void buildProviderMap(Map<String, Method> providerMap, Class<?> type, String methodName) {
254
255        // Terminate recursion
256        if (type == null)
257            return;
258
259        // Check the method in this class
260        do {
261
262            // Get method
263            Method method;
264            try {
265                method = type.getDeclaredMethod(methodName);
266            } catch (NoSuchMethodException e) {
267                break;
268            }
269
270            // Get annotation
271            final ProvidesField providesField = method.getAnnotation(ProvidesField.class);
272            if (providesField == null)
273                break;
274
275            // Validate method return type is compatible with Field
276            if (!Field.class.isAssignableFrom(method.getReturnType())) {
277                throw new IllegalArgumentException("invalid @" + ProvidesField.class.getName() + " annotation on method " + method
278                  + ": return type " + method.getReturnType().getName() + " is not a subtype of " + Field.class.getName());
279            }
280
281            // Check for two methods declaring fields for the same property
282            final String propertyName = providesField.value();
283            final Method otherMethod = providerMap.get(propertyName);
284            if (otherMethod != null && !otherMethod.getName().equals(methodName)) {
285                throw new IllegalArgumentException("conflicting @" + ProvidesField.class.getName()
286                  + " annotations exist for property `" + propertyName + "': both method "
287                  + otherMethod + " and method " + method + " are specified");
288            }
289
290            // Save method
291            if (otherMethod == null)
292                providerMap.put(propertyName, method);
293        } while (false);
294
295        // Recurse on interfaces
296        for (Class<?> iface : type.getInterfaces())
297            this.buildProviderMap(providerMap, iface, methodName);
298
299        // Recurse on superclass
300        this.buildProviderMap(providerMap, type.getSuperclass(), methodName);
301    }

303    /**
304     * Create a {@link BeanFieldGroup} using the given instance, introspect for {@link FieldBuilder} annotations
305     * on property getter methods of the given bean's class, and build and bind the corresponding fields, and return
306     * the result.
307     *
308     * @param bean model bean annotated with {@link FieldBuilder} annotations
309     * @throws IllegalArgumentException if {@code bean} is null
310     */
311    @SuppressWarnings("unchecked")
312    public static <T> BeanFieldGroup<T> buildFieldGroup(T bean) {
313
314        // Sanity check
315        if (bean == null)
316            throw new IllegalArgumentException("null bean");
317
318        // Create field group
319        final BeanFieldGroup<T> fieldGroup = new BeanFieldGroup<>((Class<T>)bean.getClass());
320        fieldGroup.setItemDataSource(bean);
321        new FieldBuilder().buildAndBind(fieldGroup);
322        return fieldGroup;
323    }

325    /**
326     * Instantiate and configure an {@link AbstractField} according to the given scanned annotations.
327     */
328    protected com.vaadin.ui.AbstractField<?> buildField(Collection<AnnotationApplier<?, ?>> appliers, String description) {
329
330        // Get comparator that sorts by class hierarcy, narrower types first; note Collections.sort() is stable,
331        // so for any specific annotation type, that annotation on subtype appears before that annotation on supertype.
332        final Comparator<AnnotationApplier<?, ?>> comparator = new Comparator<AnnotationApplier<?, ?>>() {
333            @Override
334            public int compare(AnnotationApplier<?, ?> a1, AnnotationApplier<?, ?> a2) {
335                final Class<? extends com.vaadin.ui.AbstractField<?>> type1 = a1.getFieldType();
336                final Class<? extends com.vaadin.ui.AbstractField<?>> type2 = a2.getFieldType();
337                if (type1 == type2)
338                    return 0;
339                if (type1.isAssignableFrom(type2))
340                    return 1;
341                if (type2.isAssignableFrom(type1))
342                    return -1;
343                return 0;
344            }
345        };
346
347        // Sanity check for duplicates and conflicts
348        final ArrayList<AnnotationApplier<?, ?>> applierList = new ArrayList<>(appliers);
349        Collections.sort(applierList, comparator);
350        for (int i = 0; i < applierList.size() - 1; ) {
351            final AnnotationApplier<?, ?> a1 = applierList.get(i);
352            final AnnotationApplier<?, ?> a2 = applierList.get(i + 1);
353
354            // Let annotations on subclass override annotations on superclass
355            if (a1.getAnnotation().annotationType() == a2.getAnnotation().annotationType()) {
356                applierList.remove(i + 1);
357                continue;
358            }
359
360            // Check for conflicting annotation types (e.g., both FieldBuilder.TextField and FieldBuilder.DateField)
361            if (comparator.compare(a1, a2) == 0) {
362                throw new IllegalArgumentException("conflicting annotations of type "
363                  + a1.getAnnotation().annotationType().getName() + " and " + a2.getAnnotation().annotationType().getName()
364                  + " for " + description);
365            }
366            i++;
367        }
368
369        // Determine field type
370        Class<? extends com.vaadin.ui.AbstractField<?>> type = null;
371        AnnotationApplier<?, ?> typeApplier = null;
372        for (AnnotationApplier<?, ?> applier : applierList) {
373
374            // Pick up type() if specified
375            if (applier.getActualFieldType() == null)
376                continue;
377            if (type == null) {
378                type = applier.getActualFieldType();
379                typeApplier = applier;
380                continue;
381            }
382
383            // Verify the field type specified by a narrower annotation has compatible narrower field type
384            if (!applier.getActualFieldType().isAssignableFrom(type)) {
385                throw new IllegalArgumentException("conflicting field types specified by annotations of type "
386                  + typeApplier.getAnnotation().annotationType().getName() + " (type() = " + type.getName() + ") and "
387                  + applier.getAnnotation().annotationType().getName() + " (type() = " + applier.getActualFieldType().getName()
388                  + ") for " + description);
389            }
390        }
391        if (type == null)
392            throw new IllegalArgumentException("cannot determine field type; no type() specified for " + description);
393
394        // Instantiate field
395        final com.vaadin.ui.AbstractField<?> field = FieldBuilder.instantiate(type);
396
397        // Configure the field
398        for (AnnotationApplier<?, ?> applier : applierList)
399            this.apply(applier, field);
400
401        // Done
402        return field;
403    }

411    private static <T> T instantiate(Class<T> type) {
412        Constructor<T> constructor;
413        try {
414            constructor = type.getDeclaredConstructor();
415        } catch (Exception e) {
416            throw new RuntimeException("cannot instantiate " + type + " because no zero-arg constructor could be found", e);
417        }
418        try {
419            constructor.setAccessible(true);
420        } catch (Exception e) {
421            // ignore
422        }
423        try {
424            return constructor.newInstance();
425        } catch (Exception e) {
426            throw new RuntimeException("cannot instantiate " + type + " using its zero-arg constructor", e);
427        }
428    }

430    /**
431     * Find all relevant annotations on the given method as well as on any supertype methods it overrides.
432     * The method must be a getter method taking no arguments. Annotations are ordered so that annotations
433     * on a method in type X appear before annotations on an overridden method in type Y, a supertype of X.
434     *
435     * @throws IllegalArgumentException if {@code method} is null
436     * @throws IllegalArgumentException if {@code method} has parameters
437     */
438    protected List<AnnotationApplier<?, ?>> buildApplierList(Method method) {
439
440        // Sanity check
441        if (method == null)
442            throw new IllegalArgumentException("null method");
443        if (method.getParameterTypes().length > 0)
444            throw new IllegalArgumentException("method takes parameters");
445
446        // Recurse
447        final ArrayList<AnnotationApplier<?, ?>> list = new ArrayList<>();
448        this.buildApplierList(method.getDeclaringClass(), method.getName(), list);
449        return list;
450    }
451
452    private void buildApplierList(Class<?> type, String methodName, List<AnnotationApplier<?, ?>> list) {
453
454        // Terminate recursion
455        if (type == null)
456            return;
457
458        // Check class
459        Method method;
460        try {
461            method = type.getMethod(methodName);
462        } catch (NoSuchMethodException e) {
463            method = null;
464        }
465        if (method != null)
466            list.addAll(this.buildDirectApplierList(method));
467
468        // Recurse on interfaces
469        for (Class<?> iface : type.getInterfaces())
470            this.buildApplierList(iface, methodName, list);
471
472        // Recurse on superclass
473        this.buildApplierList(type.getSuperclass(), methodName, list);
474    }

476    /**
477     * Find all relevant annotations declared directly on the given {@link Method}.
478     *
479     * @param method method to inspect
480     * @throws IllegalArgumentException if {@code method} is null
481     */
482    protected List<AnnotationApplier<?, ?>> buildDirectApplierList(Method method) {
483
484        // Sanity check
485        if (method == null)
486            throw new IllegalArgumentException("null method");
487
488        // Build list
489        final ArrayList<AnnotationApplier<?, ?>> list = new ArrayList<>();
490        for (Annotation annotation : method.getDeclaredAnnotations()) {
491            final AnnotationApplier<? extends Annotation, ?> applier = this.getAnnotationApplier(method, annotation);
492            if (applier != null)
493                list.add(applier);
494        }
495        return list;
496    }

498    /**
499     * Get the {@link AnnotationApplier} that applies the given annotation.
500     * Subclasses can add support for additioanl annotation types by overriding this method.
501     *
502     * @return corresponding {@link AnnotationApplier}, or null if annotation is unknown
503     */
504    protected AnnotationApplier<?, ?> getAnnotationApplier(Method method, Annotation annotation) {
505        if (annotation instanceof FieldBuilder.AbstractField)
506            return new AbstractFieldApplier(method, (FieldBuilder.AbstractField)annotation);
507        if (annotation instanceof FieldBuilder.AbstractSelect)
508            return new AbstractSelectApplier(method, (FieldBuilder.AbstractSelect)annotation);
509        if (annotation instanceof FieldBuilder.CheckBox)
510            return new CheckBoxApplier(method, (FieldBuilder.CheckBox)annotation);
511        if (annotation instanceof FieldBuilder.ComboBox)
512            return new ComboBoxApplier(method, (FieldBuilder.ComboBox)annotation);
513        if (annotation instanceof FieldBuilder.EnumComboBox)
514            return new EnumComboBoxApplier(method, (FieldBuilder.EnumComboBox)annotation);
515        if (annotation instanceof FieldBuilder.ListSelect)
516            return new ListSelectApplier(method, (FieldBuilder.ListSelect)annotation);
517        if (annotation instanceof FieldBuilder.DateField)
518            return new DateFieldApplier(method, (FieldBuilder.DateField)annotation);
519        if (annotation instanceof FieldBuilder.AbstractTextField)
520            return new AbstractTextFieldApplier(method, (FieldBuilder.AbstractTextField)annotation);
521        if (annotation instanceof FieldBuilder.TextField)
522            return new TextFieldApplier(method, (FieldBuilder.TextField)annotation);
523        if (annotation instanceof FieldBuilder.TextArea)
524            return new TextAreaApplier(method, (FieldBuilder.TextArea)annotation);
525        if (annotation instanceof FieldBuilder.PasswordField)
526            return new PasswordFieldApplier(method, (FieldBuilder.PasswordField)annotation);
527        return null;
528    }

533     * Class that knows how to apply annotation properties to a corresponding field.
534     */
535    protected abstract static class AnnotationApplier<A extends Annotation, F extends com.vaadin.ui.AbstractField<?>> {
536
537        protected final Method method;
538        protected final A annotation;
539        protected final Class<F> fieldType;
540
541        protected AnnotationApplier(Method method, A annotation, Class<F> fieldType) {
542            if (method == null)
543                throw new IllegalArgumentException("null method");
544            if (annotation == null)
545                throw new IllegalArgumentException("null annotation");
546            if (fieldType == null)
547                throw new IllegalArgumentException("null fieldType");
548            this.method = method;
549            this.annotation = annotation;
550            this.fieldType = fieldType;
551        }
552
553        public final Method getMethod() {
554            return this.method;
555        }
556
557        public final A getAnnotation() {
558            return this.annotation;
559        }
560
561        public final Class<F> getFieldType() {
562            return this.fieldType;
563        }
564
565        public abstract Class<? extends F> getActualFieldType();
566
567        public abstract void applyTo(F field);
568    }

570    /**
571     * Applies properties from a {@link FieldBuilder.AbstractField} annotation to a {@link com.vaadin.ui.AbstractField}.
572     */
573    private static class AbstractFieldApplier
574      extends AnnotationApplier<FieldBuilder.AbstractField, com.vaadin.ui.AbstractField<?>> {
575
576        @SuppressWarnings("unchecked")
577        public AbstractFieldApplier(Method method, FieldBuilder.AbstractField annotation) {
578            super(method, annotation, (Class<com.vaadin.ui.AbstractField<?>>)(Object)com.vaadin.ui.AbstractField.class);
579        }
580
581        @Override
582        @SuppressWarnings("unchecked")
583        public Class<? extends com.vaadin.ui.AbstractField<?>> getActualFieldType() {
584            return (Class<com.vaadin.ui.AbstractField<?>>)(Object)(this.annotation.type() != com.vaadin.ui.AbstractField.class ?
585              this.annotation.type() : null);
586        }
587
588        @Override
589        @SuppressWarnings("unchecked")
590        public void applyTo(com.vaadin.ui.AbstractField<?> field) {
591            if (this.annotation.width().length() > 0)
592                field.setWidth(this.annotation.width());
593            if (this.annotation.height().length() > 0)
594                field.setHeight(this.annotation.height());
595            if (this.annotation.caption().length() > 0)
596                field.setCaption(this.annotation.caption());
597            if (this.annotation.description().length() > 0)
598                field.setDescription(this.annotation.description());
599            field.setEnabled(this.annotation.enabled());
600            field.setImmediate(this.annotation.immediate());
601            field.setReadOnly(this.annotation.readOnly());
602            field.setBuffered(this.annotation.buffered());
603            field.setInvalidAllowed(this.annotation.invalidAllowed());
604            field.setInvalidCommitted(this.annotation.invalidCommitted());
605            field.setValidationVisible(this.annotation.validationVisible());
606            field.setRequired(this.annotation.required());
607            field.setTabIndex(this.annotation.tabIndex());
608            if (this.annotation.converter() != Converter.class)
609                field.setConverter(FieldBuilder.instantiate(this.annotation.converter()));
610            for (Class<? extends Validator> validatorType : this.annotation.validators())
611                field.addValidator(FieldBuilder.instantiate(validatorType));
612            for (String styleName : this.annotation.styleNames())
613                field.addStyleName(styleName);
614            if (this.annotation.conversionError().length() > 0)
615                field.setConversionError(this.annotation.conversionError());
616            if (this.annotation.requiredError().length() > 0)
617                field.setRequiredError(this.annotation.requiredError());
618        }
619    }

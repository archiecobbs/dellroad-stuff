024/**
025 * Support superclass for simple read-only, in-memory {@link Container} implementations where each
026 * {@link Item} in the container is backed by a Java object.
027 *
028 * <p>
029 * This {@link Container}'s {@link Property}s are defined via {@link PropertyDef}s, and a {@link PropertyExtractor}
030 * is used to actually extract the property values from each underlying object (alternately, subclasses can override
031 * {@link #getPropertyValue getPropertyValue()}). However, the easist way to configure the container {@link Property}s
032 * is to pass a {@link ProvidesProperty &#64;ProvidesProperty}-annotated Java class to the {@link #AbstractSimpleContainer(Class)}
033 * constructor.
034 * </p>
035 *
036 * <p>
037 * Use {@link #load load()} to load or reload the container.
038 * </p>
039 *
040 * @param <I> the item ID type
041 * @param <T> the type of the Java objects that back each {@link Item} in the container
042 *
043 * @see SimpleItem
044 */
045@SuppressWarnings("serial")
046public abstract class AbstractSimpleContainer<I, T> extends AbstractInMemoryContainer<I, String, BackedItem<T>>
047  implements PropertyExtractor<T>, Container.Filterable, Container.SimpleFilterable, Container.Sortable, Connectable {
048
049    private final HashMap<String, PropertyDef<?>> propertyMap = new HashMap<>();
050    private final HashMap<Object, BackedItem<T>> itemMap = new HashMap<>();
051
052    private PropertyExtractor<? super T> propertyExtractor;
053

056    /**
057     * Constructor.

100    /**
101     * Constructor.
102     *
103     * @param propertyExtractor used to extract properties from the underlying Java objects;
104     *  may be null but then container is not usable until one is configured via
105     * {@link #setPropertyExtractor setPropertyExtractor()}
106     * @param propertyDefs container property definitions; null is treated like the empty set
107     */
108    protected AbstractSimpleContainer(PropertyExtractor<? super T> propertyExtractor,
109      Collection<? extends PropertyDef<?>> propertyDefs) {
110        this.setItemSorter(new SimpleItemSorter());
111        this.setPropertyExtractor(propertyExtractor);
112        this.setProperties(propertyDefs);
113    }

115    /**
116     * Constructor.
117     *
118     * <p>
119     * Properties will be determined by the {@link ProvidesProperty &#64;ProvidesProperty} and
120     * {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods in the given class.
121     * </p>
122     *
123     * @param type class to introspect for annotated methods
124     * @throws IllegalArgumentException if {@code type} is null
125     * @throws IllegalArgumentException if {@code type} has two {@link ProvidesProperty &#64;ProvidesProperty}
126     *  or {@link ProvidesPropertySort &#64;ProvidesPropertySort} annotated methods for the same property
127     * @throws IllegalArgumentException if a {@link ProvidesProperty &#64;ProvidesProperty}-annotated method with no
128     *  {@linkplain ProvidesProperty#value property name specified} has a name which cannot be interpreted as a bean
129     *  property "getter" method
130     * @see ProvidesProperty
131     * @see ProvidesPropertySort
132     * @see ProvidesPropertyScanner
133     */
134    @SuppressWarnings({ "unchecked", "rawtypes" })
135    protected AbstractSimpleContainer(Class<? super T> type) {
136        // Why the JLS forces this stupid cast:
137        //  http://stackoverflow.com/questions/4902723/why-cant-a-java-type-parameter-have-a-lower-bound
138        final ProvidesPropertyScanner<? super T> propertyReader
139          = (ProvidesPropertyScanner<? super T>)new ProvidesPropertyScanner(type);
140        this.setItemSorter(new SimpleItemSorter());
141        this.setPropertyExtractor(propertyReader.getPropertyExtractor());
142        this.setProperties(propertyReader.getPropertyDefs());
143    }

165    /**
166     * Read the value of the property defined by {@code propertyDef} from the given object.
167     *
168     * <p>
169     * The implementation in {@link AbstractSimpleContainer} just delegates to the {@linkplain #setPropertyExtractor configured}
170     * {@link PropertyExtractor}; subclasses may override to customize property extraction.
171     * </p>
172     *
173     * @param obj Java object
174     * @param propertyDef definition of which property to read
175     * @throws NullPointerException if either parameter is null
176     * @throws IllegalStateException if no {@link PropertyExtractor} is configured for this container
177     */
178    @Override
179    public <V> V getPropertyValue(T obj, PropertyDef<V> propertyDef) {
180        if (this.propertyExtractor == null)
181            throw new IllegalStateException("no PropertyExtractor is configured for this container");
182        return this.propertyExtractor.getPropertyValue(obj, propertyDef);
183    }
184
185    /**
186     * Change the configured properties of this container.
187     *
188     * @param propertyDefs container property definitions; null is treated like the empty set
189     * @throws IllegalArgumentException if {@code propertyDefs} contains a property with a duplicate name
190     */
191    public void setProperties(Collection<? extends PropertyDef<?>> propertyDefs) {
192        if (propertyDefs == null)
193            propertyDefs = Collections.<PropertyDef<?>>emptySet();
194        this.propertyMap.clear();
195        for (PropertyDef<?> propertyDef : propertyDefs) {
196            if (this.propertyMap.put(propertyDef.getName(), propertyDef) != null)
197                throw new IllegalArgumentException("duplicate property name `" + propertyDef.getName() + "'");
198        }
199        this.fireContainerPropertySetChange();
200    }
201
202    /**
203     * Add or replace a configured property of this container.
204     *
205     * @param propertyDef new container property definitions
206     * @throws IllegalArgumentException if {@code propertyDef} is null
207     */
208    public void setProperty(PropertyDef<?> propertyDef) {
209        if (propertyDef == null)
210            throw new IllegalArgumentException("null propertyDef");
211        this.propertyMap.put(propertyDef.getName(), propertyDef);
212        this.fireContainerPropertySetChange();
213    }

215    /**
216     * Change this container's contents.
217     *
218     * @param contents new container contents
219     * @throws IllegalArgumentException if {@code contents} or any item in {@code contents} is null
220     */
221    public void load(Iterable<? extends T> contents) {
222
223        // Sanity check
224        if (contents == null)
225            throw new IllegalArgumentException("null contents");
226
227        // Reset item IDs
228        this.resetItemIds();
229        this.internalRemoveAllItems();
230
231        // Bulk load and register items with id's 0, 1, 2, ...
232        int index = 0;
233        for (T obj : contents) {
234            if (obj == null)
235                throw new IllegalArgumentException("null item in contents at index " + index);
236            final BackedItem<T> item = this.createBackedItem(obj, this.propertyMap.values(), this);
237            this.internalAddItemAtEnd(this.generateItemId(obj), item, false);
238            index++;
239        }
240
241        // Apply filters
242        this.doFilterContainer(!this.getFilters().isEmpty());
243
244        // Notify subclass
245        this.afterReload();
246
247        // Fire event
248        this.fireItemSetChange();
249    }
250
251    @Override
252    protected void internalRemoveAllItems() {
253        super.internalRemoveAllItems();
254        this.itemMap.clear();
255    }

257    /**
258     * Get the container item ID corresponding to the given underlying Java object which is wrapped by this container.
259     * Objects are tested for equality using {@link Object#equals Object.equals()}.
260     *
261     * <p>
262     * The implementation in {@link AbstractSimpleContainer} requires a linear search of the container.
263     * Some subclasses may provide a more efficient implementation.
264     * </p>
265     *
266     * <p>
267     * This method is not used by this class but is defined as a convenience for subclasses.
268     * </p>
269     *
270     * <p>
271     * Note: items that are filtered out will not be found.
272     *
273     * @param obj underlying container object
274     * @return item ID corresponding to {@code object}, or null if {@code object} is not found in this container
275     * @throws IllegalArgumentException if {@code object} is null
276     * @see #getItemIdForSame
277     */
278    public I getItemIdFor(T obj) {
279        if (obj == null)
280            throw new IllegalArgumentException("null object");
281        for (I itemId : this.getItemIds()) {
282            T candidate = this.getJavaObject(itemId);
283            if (obj.equals(candidate))
284                return itemId;
285        }
286        return null;
287    }

289    /**
290     * Get the container item ID corresponding to the given underlying Java object which is wrapped by this container.
291     * Objects are tested for equality using object equality, not {@link Object#equals Object.equals()}.
292     *
293     * <p>
294     * The implementation in {@link AbstractSimpleContainer} requires a linear search of the container.
295     * Some subclasses may provide a more efficient implementation.
296     * </p>
297     *
298     * <p>
299     * This method is not used by this class but is defined as a convenience for subclasses.
300     * </p>
301     *
302     * <p>
303     * Note: items that are filtered out will not be found.
304     *
305     * @param obj underlying container object
306     * @return item ID corresponding to {@code object}, or null if {@code object} is not found in this container
307     * @throws IllegalArgumentException if {@code object} is null
308     * @see #getItemIdFor
309     */
310    public I getItemIdForSame(T obj) {
311        if (obj == null)
312            throw new IllegalArgumentException("null object");
313        for (I itemId : this.getItemIds()) {
314            T candidate = this.getJavaObject(itemId);
315            if (obj == candidate)
316                return itemId;
317        }
318        return null;
319    }

377    @Override
378    public BackedItem<T> getUnfilteredItem(Object itemId) {
379        final T obj = this.getJavaObject(itemId);
380        if (obj == null)
381            return null;
382        BackedItem<T> item = this.itemMap.get(itemId);
383        if (item == null) {
384            item = this.createBackedItem(obj, this.propertyMap.values(), this);
385            this.itemMap.put(itemId, item);
386        }
387        return item;
388    }

489    @Override
490    public Collection<?> getSortableContainerPropertyIds() {
491        final ArrayList<String> propertyIds = new ArrayList<>(this.propertyMap.size());
492        for (Map.Entry<String, PropertyDef<?>> entry : this.propertyMap.entrySet()) {
493            if (this.propertyExtractor instanceof SortingPropertyExtractor) {
494                final SortingPropertyExtractor<? super T> sortingPropertyExtractor
495                  = (SortingPropertyExtractor<? super T>)this.propertyExtractor;
496                if (sortingPropertyExtractor.canSort(entry.getValue())) {
497                    propertyIds.add(entry.getKey());
498                    continue;
499                }
500            }
501            if (entry.getValue().isSortable())
502                propertyIds.add(entry.getKey());
503        }
504        return propertyIds;
505    }

509    /**
510     * {@link ItemSorter} implementation used by {@link AbstractSimpleContainer}.
511     */
512    private class SimpleItemSorter extends DefaultItemSorter {
513
514        @Override
515        @SuppressWarnings("unchecked")
516        protected int compareProperty(Object propertyId, boolean ascending, Item item1, Item item2) {
517
518            // Get property definition
519            final PropertyDef<?> propertyDef = AbstractSimpleContainer.this.propertyMap.get(propertyId);
520            if (propertyDef == null)
521                return super.compareProperty(propertyId, ascending, item1, item2);
522
523            // Ask SortingPropertyExtractor if we have one
524            if (AbstractSimpleContainer.this.propertyExtractor instanceof SortingPropertyExtractor) {
525                final SortingPropertyExtractor<? super T> sortingPropertyExtractor
526                  = (SortingPropertyExtractor<? super T>)AbstractSimpleContainer.this.propertyExtractor;
527                if (sortingPropertyExtractor.canSort(propertyDef)) {
528                    final T obj1 = ((BackedItem<T>)item1).getObject();
529                    final T obj2 = ((BackedItem<T>)item2).getObject();
530                    final int diff = sortingPropertyExtractor.sort(propertyDef, obj1, obj2);
531                    return ascending ? diff : -diff;
532                }
533            }
534
535            // Ask property definition
536            if (propertyDef.isSortable()) {
537                final int diff = this.sort(propertyDef, item1, item2);
538                return ascending ? diff : -diff;
539            }
540
541            // Defer to superclass
542            return super.compareProperty(propertyId, ascending, item1, item2);
543        }
544
545        // This method exists only to allow the generic parameter <V> to be bound
546        private <V> int sort(PropertyDef<V> propertyDef, Item item1, Item item2) {
547            return propertyDef.sort(propertyDef.read(item1), propertyDef.read(item2));
548        }
549    }

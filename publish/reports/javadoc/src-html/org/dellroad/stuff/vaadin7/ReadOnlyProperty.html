006package org.dellroad.stuff.vaadin7;
007
008import com.vaadin.data.Property;
009
010import java.util.EventObject;
011
012/**
013 * Support superclass for {@link Property}s with read-only values.
014 * Provides a somewhat more space-efficient implementation than Vaadin's {@link com.vaadin.data.util.AbstractProperty},
015 * which is useful when there are many instances in memory.
016 *
017 * @param <V> the type of the property
018 */
019@SuppressWarnings({ "serial", "deprecation" })
020public abstract class ReadOnlyProperty<V> implements
021  Property<V>, Property.ValueChangeNotifier, Property.ReadOnlyStatusChangeNotifier {
022
023    private Object listeners;                   // space efficient listener hackery
024
025    /**
026     * Change this instance's value.
027     *
028     * <p>
029     * The implementation in {@link ReadOnlyProperty} always throws {@link com.vaadin.data.Property.ReadOnlyException}.
030     */
031    @Override
032    public void setValue(V value) {
033        throw new Property.ReadOnlyException();
034    }
035
036    /**
037     * Determine if this instance is read-only.
038     *
039     * <p>
040     * The implementation in {@link ReadOnlyProperty} always returns true.
041     */
042    @Override
043    public boolean isReadOnly() {
044        return true;
045    }
046
047    /**
048     * Change this instance's read-only setting.
049     *
050     * <p>
051     * The implementation in {@link ReadOnlyProperty} throws {@link UnsupportedOperationException}
052     * if {@code readOnly} is false.
053     */
054    @Override
055    public void setReadOnly(boolean readOnly) {
056        if (!readOnly)
057            throw new UnsupportedOperationException();
058    }
059
060    /**
061     * Issue a {@link com.vaadin.data.Property.ValueChangeEvent} to all registered
062     * {@link com.vaadin.data.Property.ValueChangeListener}s.
063     */
064    protected void fireValueChange() {
065        if (this.listeners == null)
066            return;
067        final Property.ValueChangeEvent event = new ValueChangeEvent();
068        if (this.listeners instanceof Property.ValueChangeListener)
069            ((Property.ValueChangeListener)this.listeners).valueChange(event);
070        else {
071            for (Property.ValueChangeListener listener : (Property.ValueChangeListener[])this.listeners)
072                listener.valueChange(event);
073        }
074    }
075
076    /**
077     * Invoked when the first {@link com.vaadin.data.Property.ValueChangeListener} has been added.
078     *
079     * <p>
080     * The implementation in {@link ReadOnlyProperty} does nothing.
081     */
082    protected void firstListenerAdded() {
083    }
084
085    /**
086     * Invoked when the last {@link com.vaadin.data.Property.ValueChangeListener} has been removed.
087     *
088     * <p>
089     * The implementation in {@link ReadOnlyProperty} does nothing.
090     */
091    protected void lastListenerRemoved() {
092    }
093
094// Property.ValueChangeNotifier
095
096    @Override
097    public void addValueChangeListener(Property.ValueChangeListener listener) {
098        if (listener == null)
099            throw new IllegalArgumentException("null listener");
100        if (this.listeners == null) {
101            this.listeners = listener;
102            this.firstListenerAdded();
103        } else if (this.listeners instanceof Property.ValueChangeListener)
104            this.listeners = new Property.ValueChangeListener[] { (Property.ValueChangeListener)this.listeners, listener };
105        else {
106            final Property.ValueChangeListener[] oldArray = (Property.ValueChangeListener[])this.listeners;
107            final Property.ValueChangeListener[] newArray = new Property.ValueChangeListener[oldArray.length + 1];
108            System.arraycopy(oldArray, 0, newArray, 0, oldArray.length);
109            newArray[oldArray.length] = listener;
110            this.listeners = newArray;
111        }
112    }
113
114    @Override
115    public void removeValueChangeListener(Property.ValueChangeListener listener) {
116        if (listener == null || this.listeners == null)
117            return;
118        if (this.listeners instanceof Property.ValueChangeListener) {
119            if (!this.listeners.equals(listener))
120                return;
121            this.listeners = null;
122            this.lastListenerRemoved();
123            return;
124        }
125        final Property.ValueChangeListener[] oldArray = (Property.ValueChangeListener[])this.listeners;
126        int index = -1;
127        for (int i = 0; i < oldArray.length; i++) {
128            if (oldArray[i].equals(listener)) {
129                index = i;
130                break;
131            }
132        }
133        if (index == -1)
134            return;
135        switch (oldArray.length) {
136        case 0:
137        case 1:
138            throw new RuntimeException("internal error");
139        case 2:
140            this.listeners = oldArray[1 - index];
141            break;
142        default:
143            final Property.ValueChangeListener[] newArray = new Property.ValueChangeListener[oldArray.length - 1];
144            System.arraycopy(oldArray, 0, newArray, 0, index);
145            System.arraycopy(oldArray, index + 1, newArray, index, oldArray.length - index - 1);
146            this.listeners = newArray;
147            break;
148        }
149    }
150
151    @Override
152    @SuppressWarnings("deprecated")
153    public void addListener(Property.ValueChangeListener listener) {
154        this.addValueChangeListener(listener);
155    }
156
157    @Override
158    @SuppressWarnings("deprecated")
159    public void removeListener(Property.ValueChangeListener listener) {
160        this.removeValueChangeListener(listener);
161    }
162
163// Property.ReadOnlyStatusChangeNotifier
164
165    @Override
166    public void addReadOnlyStatusChangeListener(Property.ReadOnlyStatusChangeListener listener) {
167        // nothing to do - read-only status never changes
168    }
169
170    @Override
171    public void removeReadOnlyStatusChangeListener(Property.ReadOnlyStatusChangeListener listener) {
172        // nothing to do - read-only status never changes
173    }
174
175    @Override
176    @SuppressWarnings("deprecated")
177    public void addListener(Property.ReadOnlyStatusChangeListener listener) {
178        this.addReadOnlyStatusChangeListener(listener);
179    }
180
181    @Override
182    @SuppressWarnings("deprecated")
183    public void removeListener(Property.ReadOnlyStatusChangeListener listener) {
184        this.removeReadOnlyStatusChangeListener(listener);
185    }
186
187// ValueChangeEvent
188
189    private class ValueChangeEvent extends EventObject implements Property.ValueChangeEvent {
190
191        ValueChangeEvent() {
192            super(ReadOnlyProperty.this);
193        }
194
195        @Override
196        @SuppressWarnings("rawtypes")
197        public Property/*<?>*/ getProperty() {
198            return (Property)this.getSource();
199        }
200    }
201}

013/**
014 * A specialization of {@link AbstractQueryContainer} that doesn't require up-front knowledge of the underlying data's size.
015 *
016 * <p>
017 * Subclasses only need support retrieving a contiguous window of fixed size from the underlying data
018 * via {@link #queryWindow queryWindow()}; when the requested window goes beyond the end of the underlying data,
019 * a short or empty result is returned. Based on just this information, this class maintains an estimate of
020 * the size of the underlying data. Each time that size estimate changes, {@link #handleSizeChange} is invoked
021 * to schedule a (non-reentrant) property set change notification. Once the end of the underlying data is reached,
022 * the size is known.
023 *
024 * <p>
025 * If the actual size of the underlying data is constant, this class will eventually find it. If the
026 * actual size of the underlying data can change (either up or down), this class will adapt accordingly,
027 * but only when it learns of the new size through an invocation of {@link #queryWindow queryWindow()};
028 * as this depends on how the container is used, this may not occur for a long time.
029 *
030 * <p>
031 * When used to back a Vaadin table, the user will see a table that automatically grows as the user scrolls
032 * downward, until the actual end of the data is detected.
033 *
034 * @param <T> the type of the Java objects that back each {@link com.vaadin.data.Item} in the container
035 * @see AbstractQueryContainer
036 */
037@SuppressWarnings("serial")
038public abstract class AbstractUnsizedContainer<T> extends AbstractQueryContainer<T> {
039
040    public static final int DEFAULT_WINDOW_SIZE = 50;
041
042    private final int windowSize;
043
044    private long size;
045    private boolean sizeIsKnown;

232    @Override
233    protected QueryList<T> query(long hint) {
234
235        // Clip hint value
236        hint = Math.min(hint, this.size - 1);
237        hint = Math.max(hint, 0);
238
239        // Query underlying data for the window of items centered at "hint"
240        final long offset = Math.max(0, hint - this.windowSize / 2);
241        final List<? extends T> window = this.queryWindow(offset, this.windowSize);
242        final int querySize = window.size();
243
244        // Handle the case where we got an empty window - so the underlying data must have shrunk
245        if (querySize <= 0) {
246
247            // If our starting offset was zero, then we know the size must be zero
248            if (offset == 0) {
249                this.sizeIsKnown = true;
250                this.size = 0;
251            } else {
252
253                // Guess at what the smaller size is and return a QueryList that will throw an InvalidQueryListException,
254                // thereby trigging another probe with a smaller offset. This cycle will repeat until the size is found.
255                this.sizeIsKnown = false;
256                this.size = this.getSmallerEstimate(offset);
257                return new AlwaysInvalidQueryList<T>(this.size);
258            }
259        }
260
261        // Handle the case where we overlapped the end; now we know the exact size
262        if (querySize < this.windowSize) {                  // we overlapped the end; now we know the exact size
263            this.size = offset + querySize;
264            this.sizeIsKnown = true;
265            return new WindowQueryList<T>(offset, window, this.size);
266        }
267
268        // Handle the case where we are somewhere in the middle
269        if (!this.sizeIsKnown) {
270
271            // Get lower bound on size estimate
272            final long lowerBound = offset + querySize;
273
274            // If current size is less than lowerBound + 1, need to expand (the +1 is to leave room to trigger a requery)
275            if (this.size < lowerBound + 1) {
276
277                // Get estimate of new larger size
278                long largerSize = this.getLargerEstimate(lowerBound);
279
280                // Ensure that it is at least large enough to avoid causing another, redundant subsequent resize. This also
281                // ensures that it is strictly larger than the last item in our window so we can always trigger another query.
282                largerSize = Math.max(largerSize, lowerBound + this.windowSize);
283
284                // Increase size estimate (maybe)
285                this.size = Math.max(this.size, largerSize);
286            }
287        } else {                                            // we are somewhere in the middle and size is known
288
289            // Detect if the size has grown
290            this.size = Math.max(this.size, offset + querySize);
291        }
292
293        // Return QueryList based on the window of data we found
294        return new WindowQueryList<T>(offset, window, this.size);
295    }

297    /**
298     * Get the window size configured at construction time.
299     */
300    public int getWindowSize() {
301        return this.windowSize;
302    }

304    /**
305     * Get the current size estimate for the underlying data.
306     */
307    public long getCurrentSizeEstimate() {
308        return this.size;
309    }

311    /**
312     * Handle the case where the underlying data's size has suddenly shrunk, so we need to estimate the new size.
313     * All we know is that the actual new size is {@code upperBound} or less.
314     * This method should guess at the new value and, when called repeatedly, should converge rapidly to zero.
315     *
316     * <p>
317     * Note: this situation will not occur if the underlying data's size never decreases.
318     *
319     * <p>
320     * The implementation in {@link AbstractUnsizedContainer} returns {@code upperBound * 0.75}.
321     *
322     * @param upperBound an upper bound on the size of the underlying data
323     * @return estimate of the actual size of the underlying data
324     */
325    protected long getSmallerEstimate(long upperBound) {
326        return (upperBound >> 1) + (upperBound >> 2);
327    }
328
329    /**
330     * Estimate the size of the underlying data given that {@code lowerBound} is a lower bound.
331     * This effectively determines how much data will appear to be "beyond" the current window.
332     *
333     * <p>
334     * The implementation in {@link AbstractUnsizedContainer} returns {@code lowerBound * 1.25}.
335     *
336     * @param lowerBound a lower bound on the size of the underlying data
337     * @return estimate of the actual size of the underlying data
338     */
339    protected long getLargerEstimate(long lowerBound) {
340        return lowerBound + (lowerBound >> 2);
341    }
342
343    @Override
344    protected void fireItemSetChange(Container.ItemSetChangeEvent event) {
345        this.sizeIsKnown = false;
346        super.fireItemSetChange(event);
347    }
348
349    /**
350     * Query the underlying data for a window of items in the given range. This should return the "window" of underlying
351     * data items starting at offset {@code offset} and having at least length {@code length}, or else however many remain.
352     * If {@code offset} is greater than or equal to the size of the underlying data, an empty list should be returned.
353     *
354     * @param offset starting offset for window
355     * @param length window size, always greater than zero
356     * @return list containing at least {@code length} items in the window starting at offset {@code offset}; or less than
357     *  {@code length} items if {@code offset + length} is greater than the size of the underlying data
358     */
359    protected abstract List<? extends T> queryWindow(long offset, int length);
360
361    /**
362     * Emit a property set change notification.
363     *
364     * <p>
365     * Subclasses are required to implement this so that size changes are detected.
366     *
367     * <p>
368     * Note: to avoid re-entrancy problems, this method should not send out any notifications itself;
369     * instead, it must schedule notifications to be delivered later (perhaps in a different thread).
370     */
371    @Override
372    protected abstract void handleSizeChange();

006package org.dellroad.stuff.vaadin7;
007
008import com.vaadin.data.Container;
009import com.vaadin.data.Item;
010import com.vaadin.data.Property;
011import com.vaadin.data.util.ObjectProperty;
012import com.vaadin.ui.Table;
013
014import java.util.Comparator;
015
016/**
017 * Defines a Vaadin property, having a name, which is also the property ID, and its type.
018 * This class provides a mechanism for the explicit naming and identification of Vaadin properties.
019 *
020 * <p>
021 * For a given instance, <code>def.{@link #getPropertyId getPropertyId()}</code> is the property ID
022 * and you can access the value using <code>def.{@link #read read(item)}</code>.
023 *
024 * <p>
025 * Example:
026 * <blockquote><pre>
027 *  PropertyDef&lt;Integer&gt; def = new PropertyDef&lt;Integer&gt;("age", Integer.class, -1);
028 *  def.addTo(container);
029 *  def.addTo(item, property);
030 *  ...
031 *  int age = def.read(item);
032 *  ...
033 *  Property prop = this.get(container, itemId);
034 * </pre></blockquote>
035 *
036 * @param <T> property's value type
037 */
038public class PropertyDef<T> {
039
040    /**
041     * Comparator that sorts instances by name.
042     */
043    public static final Comparator<PropertyDef<?>> SORT_BY_NAME = new Comparator<PropertyDef<?>>() {
044        @Override
045        public int compare(PropertyDef<?> p1, PropertyDef<?> p2) {
046            return p1.getName().compareTo(p2.getName());
047        }
048    };
049
050    private final String name;
051    private final Class<T> type;
052    private final T defaultValue;
053
054    /**
055     * Convenience contructor.
056     *
057     * <p>
058     * Equivalent to:
059     *  <blockquote><pre>
060     *  PropertyDef(name, type, null);
061     *  </pre></blockquote>
062     */
063    public PropertyDef(String name, Class<T> type) {
064        this(name, type, null);
065    }
066
067    /**
068     * Primary constructor.
069     *
070     * @param name property name; also serves as the property ID
071     * @param type property type
072     * @param defaultValue default value for the property; may be null
073     */
074    public PropertyDef(String name, Class<T> type, T defaultValue) {
075        if (name == null)
076            throw new IllegalArgumentException("null name");
077        if (type == null)
078            throw new IllegalArgumentException("null type");
079        this.name = name;
080        this.type = type;
081        this.defaultValue = defaultValue;
082    }
083
084    /**
085     * Get the name of this property. This is also used as the property ID.
086     *
087     * @return property name, never null
088     */
089    public String getName() {
090        return this.name;
091    }
092
093    /**
094     * Get the ID of this property. Returns the same thing as {@link #getName()}.
095     *
096     * @return property name, never null
097     */
098    public String getPropertyId() {
099        return this.getName();
100    }
101
102    /**
103     * Get the type of the property value that this instance represents.
104     *
105     * @return property type, never null
106     */
107    public Class<T> getType() {
108        return this.type;
109    }
110
111    /**
112     * Get the default value for this property.
113     *
114     * @return property default value
115     */
116    public T getDefaultValue() {
117        return this.defaultValue;
118    }
119
120    /**
121     * Create a simple {@link ObjectProperty} using the given value.
122     *
123     * @param value property value
124     * @param readOnly whether property should be read-only
125     * @return new property
126     */
127    public ObjectProperty<T> createProperty(T value, boolean readOnly) {
128        return new ObjectProperty<T>(value, this.getType(), readOnly);
129    }
130
131    /**
132     * Create a read/write {@link ObjectProperty} using the given value.
133     *
134     * <p>
135     * Equivalent to:
136     *  <blockquote><pre>
137     *  {@link #createProperty(Object, boolean) createProperty()}(value, false);
138     *  </pre></blockquote>
139     *
140     * @param value property value
141     * @return new property
142     */
143    public ObjectProperty<T> createProperty(T value) {
144        return this.createProperty(value, false);
145    }
146
147    /**
148     * Create a read/write {@link ObjectProperty} using the default value.
149     *
150     * <p>
151     * Equivalent to:
152     *  <blockquote><pre>
153     *  {@link #createProperty(Object) createProperty()}(def.getDefaultValue());
154     *  </pre></blockquote>
155     *
156     * @return new property
157     */
158    public ObjectProperty<T> createProperty() {
159        return this.createProperty(this.getDefaultValue());
160    }
161
162    /**
163     * Get the property that this instance represents from the given {@link Item}.
164     *
165     * @return property, or null if not found
166     * @throws ClassCastException if the property found has a different type than this instance
167     */
168    public Property<T> get(Item item) {
169        return this.cast(item.getItemProperty(this.getPropertyId()));
170    }
171
172    /**
173     * Get the property that this instance represents from the given {@link Container}.
174     *
175     * @param container the container containing the items
176     * @param itemId the ID of the item containing the property
177     * @return property, or null if not found
178     * @throws ClassCastException if the property found has a different type than this instance
179     */
180    public Property<T> get(Container container, Object itemId) {
181        return this.cast(container.getContainerProperty(itemId, this.getPropertyId()));
182    }
183
184    /**
185     * Verify that the given property has the same Java type as this property definition.
186     *
187     * <p>
188     * This essentially verifies that <code>property.getType() == this.getType()</code>.
189     *
190     * @param property the property to verify; may be null
191     * @return null if {@code property} is null, otherwise {@code property}
192     * @throws ClassCastException if {@code property} has a different type than this definition
193     */
194    @SuppressWarnings("unchecked")
195    public Property<T> cast(Property<?> property) {
196        if (property == null)
197            return null;
198        if (property.getType() != this.getType()) {
199            throw new ClassCastException("property type " + property.getType().getName()
200              + " != definition type " + this.getType().getName());
201        }
202        return (Property<T>)property;
203    }
204
205    /**
206     * Add a property represented by this instance to the given {@link Container}.
207     *
208     * @param container the container to add the property to
209     * @return true if the operation succeeded, false if not
210     * @throws UnsupportedOperationException if the operation is not supported
211     */
212    public boolean addTo(Container container) {
213        return container.addContainerProperty(this.getPropertyId(), this.getType(), this.getDefaultValue());
214    }
215
216    /**
217     * Add a property represented by this instance to the given {@link Item}.
218     *
219     * @param item the item to add the property to
220     * @param property the property to be added to the item and identified by this instance's name
221     * @return true if the operation succeeded, false if not
222     * @throws UnsupportedOperationException if the operation is not supported
223     */
224    public boolean addTo(Item item, Property<T> property) {
225        return item.addItemProperty(this.getPropertyId(), property);
226    }
227
228    /**
229     * Add a property represented by this instance to the given {@link Table}.
230     *
231     * @param table the table to add the property to
232     * @return true if the operation succeeded, false if not
233     * @throws UnsupportedOperationException if the operation is not supported
234     */
235    public boolean addTo(Table table) {
236        return table.addContainerProperty(this.getPropertyId(), this.getType(), this.getDefaultValue());
237    }
238
239    /**
240     * Read the property that this instance represents from the given {@link Item}.
241     *
242     * @return property value, or null if not found
243     * @throws ClassCastException if the property in {@code item} has the wrong type
244     */
245    public T read(Item item) {
246        Property<T> property = this.get(item);
247        if (property == null)
248            return null;
249        return this.type.cast(property.getValue());
250    }
251
252    /**
253     * Determine whether this instance supports sorting property values.
254     *
255     * <p>
256     * The implementation in {@link PropertyDef} returns true if this instance's type implements {@link Comparable}.
257     *
258     * @see #sort sort()
259     */
260    public boolean isSortable() {
261        return Comparable.class.isAssignableFrom(this.type);
262    }
263
264    /**
265     * Sort two values of this property. Optional operation.
266     *
267     * <p>
268     * The implementation in {@link PropertyDef} sorts null values first, then delegates to {@link Comparable}
269     * if the values implement it, or else throws {@link UnsupportedOperationException}.
270     *
271     * @param value1 first value, possibly null
272     * @param value2 second value, possibly null
273     * @return negative, zero, or positive based on comparing {@code value1} to {@code value2}
274     * @throws UnsupportedOperationException if this instance does not support sorting property values
275     */
276    @SuppressWarnings("unchecked")
277    public int sort(T value1, T value2) {
278        if ((value1 == null) != (value2 == null))
279            return value1 == null ? -1 : 1;
280        if (value1 == null && value2 == null)
281            return 0;
282        try {
283            return ((Comparable<T>)value1).compareTo(value2);
284        } catch (ClassCastException e) {
285            throw new UnsupportedOperationException(e);
286        }
287    }
288
289    @Override
290    public int hashCode() {
291        return this.name.hashCode()
292          ^ this.type.hashCode()
293          ^ (this.defaultValue != null ? this.defaultValue.hashCode() : 0);
294    }
295
296    @Override
297    public boolean equals(Object obj) {
298        if (obj == null || obj.getClass() != this.getClass())
299            return false;
300        final PropertyDef<?> that = (PropertyDef<?>)obj;
301        return this.name.equals(that.name)
302          && this.type == that.type
303          && (this.defaultValue != null ? this.defaultValue.equals(that.defaultValue) : that.defaultValue == null);
304    }
305
306    @Override
307    public String toString() {
308        return this.getClass().getSimpleName() + "[name=\"" + this.name + "\",type="
309          + this.type.getName() + ",defaultValue=" + this.defaultValue + "]";
310    }
311}

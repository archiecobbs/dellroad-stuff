008/**
009 * Simple read-only {@link com.vaadin.data.Property} implementation backed by a Java object.
010 *
011 * <p>
012 * The {@link com.vaadin.data.Property} is defined using a {@link PropertyDef} along with a {@link PropertyExtractor}
013 * that is capable of reading the property's value from the underlying Java object.
014 * </p>
015 *
016 * @param <T> the type of the underlying Java object
017 * @param <V> the type of the property
018 * @see SimpleContainer
019 * @see SimpleItem
020 */
021@SuppressWarnings("serial")
022public class SimpleProperty<T, V> extends ReadOnlyProperty<V> implements BackedProperty<T, V> {
023
024    private final T object;
025    private final PropertyDef<V> propertyDef;
026    private final PropertyExtractor<? super T> propertyExtractor;
027
028    /**
029     * Constructor.
030     *
031     * @param object underlying Java object
032     * @param propertyDef property definition
033     * @param propertyExtractor extracts the property value from {@code object}
034     * @throws IllegalArgumentException if any parameter is null
035     */
036    public SimpleProperty(T object, PropertyDef<V> propertyDef, PropertyExtractor<? super T> propertyExtractor) {
037        if (object == null)
038            throw new IllegalArgumentException("null object");
039        if (propertyDef == null)
040            throw new IllegalArgumentException("null propertyDef");
041        if (propertyExtractor == null)
042            throw new IllegalArgumentException("null propertyExtractor");
043        this.object = object;
044        this.propertyDef = propertyDef;
045        this.propertyExtractor = propertyExtractor;
046    }
047
048    @Override
049    public T getObject() {
050        return this.object;
051    }
052
053    @Override
054    public Class<V> getType() {
055        return this.propertyDef.getType();
056    }
057
058    @Override
059    public V getValue() {
060        return this.propertyExtractor.getPropertyValue(this.object, this.propertyDef);
061    }
062}

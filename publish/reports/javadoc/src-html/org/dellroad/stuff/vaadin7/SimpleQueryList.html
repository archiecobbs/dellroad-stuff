006package org.dellroad.stuff.vaadin7;
007
008import java.util.List;
009
010/**
011 * Simple {@link QueryList} implementation using a normal {@link List}.
012 *
013 * @see AbstractQueryContainer
014 */
015public class SimpleQueryList<T> implements QueryList<T> {
016
017    private final List<? extends T> list;
018
019    /**
020     * Constructor.
021     *
022     * @param list backing list
023     * @throws IllegalArgumentException if {@code list} is null
024     */
025    public SimpleQueryList(List<? extends T> list) {
026        if (list == null)
027            throw new IllegalArgumentException("null list");
028        this.list = list;
029    }
030
031    @Override
032    public long size() {
033        return this.list.size();
034    }
035
036    @Override
037    public T get(long index) {
038        if (index > Integer.MAX_VALUE || index < Integer.MIN_VALUE)
039            throw new IndexOutOfBoundsException("index = " + index);
040        return this.list.get((int)index);
041    }
042}

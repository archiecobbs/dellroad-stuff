006package org.dellroad.stuff.vaadin7;
007
008import com.vaadin.server.SessionDestroyEvent;
009import com.vaadin.server.SessionDestroyListener;
010import com.vaadin.server.VaadinSession;
011
012import java.util.HashMap;
013
014import org.springframework.beans.factory.ObjectFactory;
015import org.springframework.beans.factory.config.BeanFactoryPostProcessor;
016import org.springframework.beans.factory.config.ConfigurableListableBeanFactory;
017import org.springframework.beans.factory.config.Scope;
018
019/**
020 * A Spring custom {@link Scope} for Vaadin applications.
021 *
022 * <p>
023 * This scopes beans to the lifetime of the {@link VaadinSession} (formerly known as "Vaadin application").
024 * Spring {@linkplain org.springframework.beans.factory.DisposableBean#destroy destroy-methods}
025 * will be invoked when the {@link VaadinSession} is closed.
026 * </p>
027 *
028 * <p>
029 * To enable this scope, simply add this bean to your application context as a singleton (it will register itself):
030 * <blockquote><pre>
031 *  &lt;!-- Enable the "vaadinApplication" custom scope --&gt;
032 *  &lt;bean class="org.dellroad.stuff.vaadin7.VaadinApplicationScope"/&gt;
033 * </pre></blockquote>
034 * Then declare scoped beans normally using the scope name {@code "vaadinApplication"}.
035 * </p>
036 */
037@SuppressWarnings("serial")
038public class VaadinApplicationScope implements Scope, BeanFactoryPostProcessor, SessionDestroyListener {
039
040    /**
041     * Key to the current {@link VaadinSession} instance. For use by {@link #resolveContextualObject}.
042     */
043    public static final String VAADIN_SERVICE_SESSION_KEY = "vaadinServiceSession";
044
045    /**
046     * The name of this scope (i.e., <code>{@value}</code>).
047     */
048    public static final String SCOPE_NAME = "vaadinApplication";
049
050    private final HashMap<VaadinSession, SessionBeanHolder> beanHolders = new HashMap<>();
051
052// BeanFactoryPostProcessor methods
053
054    @Override
055    public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) {
056        beanFactory.registerScope(VaadinApplicationScope.SCOPE_NAME, this);
057    }
058
059// SessionDestroyListener
060
061    @Override
062    public void sessionDestroy(SessionDestroyEvent event) {
063        final VaadinSession session = event.getSession();
064        SessionBeanHolder beanHolder;
065        synchronized (this) {
066            beanHolder = this.beanHolders.remove(session);
067        }
068        if (beanHolder != null)
069            beanHolder.close();
070    }
071
072// Scope methods
073
074    @Override
075    public synchronized Object get(String name, ObjectFactory<?> objectFactory) {
076        return this.getSessionBeanHolder(true).getBean(name, objectFactory);
077    }
078
079    @Override
080    public synchronized Object remove(String name) {
081        SessionBeanHolder beanHolder = this.getSessionBeanHolder(false);
082        return beanHolder != null ? beanHolder.remove(name) : null;
083    }
084
085    @Override
086    public synchronized void registerDestructionCallback(String name, Runnable callback) {
087        this.getSessionBeanHolder(true).registerDestructionCallback(name, callback);
088    }
089
090    @Override
091    public String getConversationId() {
092        VaadinSession session = VaadinSession.getCurrent();
093        if (session == null)
094            return null;
095        return session.getClass().getName() + "@" + System.identityHashCode(session);
096    }
097
098    @Override
099    public Object resolveContextualObject(String key) {
100        if (VAADIN_SERVICE_SESSION_KEY.equals(key))
101            return VaadinSession.getCurrent();
102        return null;
103    }
104
105// Internal methods
106
107    private synchronized SessionBeanHolder getSessionBeanHolder(boolean create) {
108        VaadinSession session = VaadinUtil.getCurrentSession();
109        VaadinUtil.addSessionDestroyListener(session, this);
110        SessionBeanHolder beanHolder = this.beanHolders.get(session);
111        if (beanHolder == null && create) {
112            beanHolder = new SessionBeanHolder();
113            this.beanHolders.put(session, beanHolder);
114        }
115        return beanHolder;
116    }
117
118// Bean holder class corresponding to a single Application instance
119
120    private static class SessionBeanHolder {
121
122        private final HashMap<String, Object> beans = new HashMap<>();
123        private final HashMap<String, Runnable> destructionCallbacks = new HashMap<>();
124
125        public Object getBean(String name, ObjectFactory<?> objectFactory) {
126            Object bean = this.beans.get(name);
127            if (bean == null) {
128                bean = objectFactory.getObject();
129                this.beans.put(name, bean);
130            }
131            return bean;
132        }
133
134        public Object remove(String name) {
135            this.destructionCallbacks.remove(name);
136            return this.beans.remove(name);
137        }
138
139        public void registerDestructionCallback(String name, Runnable callback) {
140            this.destructionCallbacks.put(name, callback);
141        }
142
143        public void close() {
144            for (Runnable callback : this.destructionCallbacks.values())
145                callback.run();
146            this.beans.clear();
147            this.destructionCallbacks.clear();
148        }
149    }
150}

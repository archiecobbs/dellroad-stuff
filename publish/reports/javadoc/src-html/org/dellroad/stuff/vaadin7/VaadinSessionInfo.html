006package org.dellroad.stuff.vaadin7;
007
008import com.vaadin.server.VaadinSession;
009
010/**
011 * Represents one {@link VaadinSession} in a {@link VaadinSessionContainer}.
012 */
013public class VaadinSessionInfo {
014
015    /**
016     * The {@link VaadinSession} associated with this instance.
017     */
018    protected final VaadinSession session;
019
020    /**
021     * Constructor.
022     *
023     * @throws IllegalStateException if there is no current {@link VaadinSession}
024     */
025    public VaadinSessionInfo() {
026        this.session = VaadinUtil.getCurrentSession();
027    }
028
029    /**
030     * Get the {@link VaadinSession} associated with this instance.
031     */
032    public VaadinSession getVaadinSession() {
033        return this.session;
034    }
035
036    /**
037     * Determine if the {@link VaadinSession} associated with this instance is also associated with the current thread.
038     */
039    public boolean isCurrentSession() {
040        return this.session == VaadinSession.getCurrent();
041    }
042
043    /**
044     * Update this instance with information from its corresponding {@link VaadinSession},
045     * which will be associated with the current thread and locked.
046     *
047     * <p>
048     * This method is invoked from {@link VaadinSessionContainer#doUpdate}; the implementation in
049     * {@link VaadinSessionInfo} does nothing.
050     *
051     * <p>
052     * When this method and {@link #makeUpdatesVisible} are overridden, consider declaring them both {@code synchronized},
053     * as they are invoked with different locks held yet will access the same information.
054     */
055    protected void updateInformation() {
056    }
057
058    /**
059     * Make the updates previously gathered by {@link #updateInformation} visible in this instance's {@link VaadinSessionContainer}.
060     * The  {@link VaadinSession} associated with this instance's {@link VaadinSessionContainer} will be associated with the
061     * current thread and locked.
062     *
063     * <p>
064     * This method is invoked from {@link VaadinSessionContainer#doUpdate}; the implementation in
065     * {@link VaadinSessionInfo} does nothing.
066     *
067     * <p>
068     * When this method and {@link #updateInformation} are used, consider declaring them both {@code synchronized},
069     * as they are invoked with different locks held yet will access the same information.
070     */
071    protected void makeUpdatesVisible() {
072    }
073}

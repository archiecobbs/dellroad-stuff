006package org.dellroad.stuff.vaadin7;
007
008import com.vaadin.data.Buffered;
009import com.vaadin.data.Property;
010import com.vaadin.data.Validator;
011import com.vaadin.ui.Field;
012import com.vaadin.ui.HorizontalLayout;
013
014import java.util.Collection;
015
016/**
017 * A {@link HorizontalLayout} that also exposes an internally wrapped {@link Field}. This is useful when you
018 * have an existing {@link Field} and want to display it differently and/or with additional {@link com.vaadin.ui.Component}s
019 * but still need it to function as a {@link Field} from a data perspective.
020 *
021 * <p>
022 * The layout is initially empty; use {@link #addComponent HorizontalLayout.addComponent()} to add content.
023 *
024 * @param <T> field type
025 */
026@SuppressWarnings({ "serial", "deprecation" })
027public class FieldLayout<T> extends HorizontalLayout implements Field<T> {
028
029    protected final Field<T> field;
030
031    /**
032     * Constructor.
033     *
034     * @param field the wrapped {@link Field} that this instance will derive its {@link Field} state from
035     */
036    public FieldLayout(Field<T> field) {
037        if (field == null)
038            throw new IllegalArgumentException("null field");
039        this.field = field;
040    }
041
042// Field
043
044    @Override
045    public String getRequiredError() {
046        return this.field.getRequiredError();
047    }
048
049    @Override
050    public void setRequiredError(String requiredError) {
051        this.field.setRequiredError(requiredError);
052    }
053
054    @Override
055    public boolean isRequired() {
056        return this.field.isRequired();
057    }
058
059    @Override
060    public void setRequired(boolean required) {
061        this.field.setRequired(required);
062    }
063
064// Component.Focusable
065
066    @Override
067    public void focus() {
068        this.field.focus();
069    }
070
071    @Override
072    public int getTabIndex() {
073        return this.field.getTabIndex();
074    }
075
076    @Override
077    public void setTabIndex(int index) {
078        this.field.setTabIndex(index);
079    }
080
081// BufferedValidatable
082
083    @Override
084    public boolean isInvalidCommitted() {
085        return this.field.isInvalidCommitted();
086    }
087
088    @Override
089    public void setInvalidCommitted(boolean invalidCommitted) {
090        this.field.setInvalidCommitted(invalidCommitted);
091    }
092
093// Buffered
094
095    @Override
096    public void commit() throws Buffered.SourceException {
097        this.field.commit();
098    }
099
100    @Override
101    public void discard() throws Buffered.SourceException {
102        this.field.discard();
103    }
104
105    @Override
106    public void setBuffered(boolean buffered) {
107        this.field.setBuffered(buffered);
108    }
109
110    @Override
111    public boolean isBuffered() {
112        return this.field.isBuffered();
113    }
114
115    @Override
116    public boolean isModified() {
117        return this.field.isModified();
118    }
119
120// Validatable
121
122    @Override
123    public void addValidator(Validator validator) {
124        this.field.addValidator(validator);
125    }
126
127    @Override
128    public void removeValidator(Validator validator) {
129        this.field.removeValidator(validator);
130    }
131
132    @Override
133    public void removeAllValidators() {
134        this.field.removeAllValidators();
135    }
136
137    @Override
138    public Collection<Validator> getValidators() {
139        return this.field.getValidators();
140    }
141
142    @Override
143    public boolean isValid() {
144        return this.field.isValid();
145    }
146
147    @Override
148    public void validate() throws Validator.InvalidValueException {
149        this.field.validate();
150    }
151
152    @Override
153    public boolean isInvalidAllowed() {
154        return this.field.isInvalidAllowed();
155    }
156
157    @Override
158    public void setInvalidAllowed(boolean invalidAllowed) {
159        this.field.setInvalidAllowed(invalidAllowed);
160    }
161
162// Property
163
164    @Override
165    public Class<? extends T> getType() {
166        return this.field.getType();
167    }
168
169    @Override
170    public T getValue() {
171        return this.field.getValue();
172    }
173
174    @Override
175    public boolean isReadOnly() {
176        return this.field.isReadOnly();
177    }
178
179    @Override
180    public void setReadOnly(boolean readOnly) {
181        this.field.setReadOnly(readOnly);
182    }
183
184    @Override
185    public void setValue(T value) {
186        this.field.setValue(value);
187    }
188
189// Property.ValueChangeNotifier
190
191    @Override
192    public void addValueChangeListener(Property.ValueChangeListener listener) {
193        this.field.addValueChangeListener(listener);
194    }
195
196    @Override
197    public void removeValueChangeListener(Property.ValueChangeListener listener) {
198        this.field.removeValueChangeListener(listener);
199    }
200
201    @Override
202    public void addListener(Property.ValueChangeListener listener) {
203        this.field.addValueChangeListener(listener);
204    }
205
206    @Override
207    public void removeListener(Property.ValueChangeListener listener) {
208        this.field.removeValueChangeListener(listener);
209    }
210
211// Property.ValueChangeListener
212
213    @Override
214    public void valueChange(Property.ValueChangeEvent event) {
215        this.field.valueChange(event);
216    }
217
218// Property.Viewer
219
220    @Override
221    @SuppressWarnings("rawtypes")
222    public void setPropertyDataSource(Property newDataSource) {
223        this.field.setPropertyDataSource(newDataSource);
224    }
225
226    @Override
227    @SuppressWarnings("rawtypes")
228    public Property getPropertyDataSource() {
229        return this.field.getPropertyDataSource();
230    }
231}

006package org.dellroad.stuff.vaadin7;
007
008import com.vaadin.data.Property;
009import com.vaadin.data.util.AbstractProperty;
010import com.vaadin.data.util.converter.Converter;
011
012import java.util.Locale;
013
014/**
015 * Applies a {@link Converter} to a {@link Property} of one type (the "model" type) to produce a
016 * {@link Property} of another type (the "presentation" type).
017 *
018 * @param <P> the "presentation" type of this {@link Property}
019 * @param <M> the "model" type of the nested {@link Property}
020 */
021@SuppressWarnings("serial")
022public class ConvertedProperty<P, M> extends AbstractProperty<P> {
023
024    private final Property<M> property;
025    private final Converter<P, M> converter;
026    private final Locale locale;
027
028    private ValueChangeListener valueChangeListener;
029    private ReadOnlyStatusChangeListener readOnlyStatusChangeListener;
030
031    /**
032     * Primary constructor.
033     *
034     * @param property underlying property
035     * @param converter converts this property's values from/to the underlying property's values
036     * @param locale the {@link Locale} to provide to {@code converter}, or null for none
037     * @throws IllegalArgumentException if {@code type}, {@code property}, or {@code converter} is null
038     */
039    public ConvertedProperty(Property<M> property, Converter<P, M> converter, Locale locale) {
040        if (property == null)
041            throw new IllegalArgumentException("null property");
042        if (converter == null)
043            throw new IllegalArgumentException("null converter");
044        this.property = property;
045        this.converter = converter;
046        this.locale = locale;
047    }
048
049    /**
050     * Convenience constructor for when no {@link Locale} is required.
051     *
052     * <p>
053     * Equivalent to:
054     *  <blockquote><code>
055     *  ConvertedProperty(property, converter, null)
056     *  </code></blockquote>
057     */
058    public ConvertedProperty(Property<M> property, Converter<P, M> converter) {
059        this(property, converter, null);
060    }
061
062    @Override
063    public Class<? extends P> getType() {
064        return this.converter.getPresentationType();
065    }
066
067    @Override
068    public P getValue() {
069        return this.converter.convertToPresentation(this.property.getValue(), this.getType(), this.locale);
070    }
071
072    @Override
073    public void setValue(P value) {
074        this.property.setValue(this.converter.convertToModel(value, this.converter.getModelType(), this.locale));
075    }
076
077    @Override
078    public boolean isReadOnly() {
079        return this.property.isReadOnly();
080    }
081
082    @Override
083    public void setReadOnly(boolean readOnly) {
084        this.property.setReadOnly(readOnly);
085    }
086
087    @Override
088    public void addReadOnlyStatusChangeListener(Property.ReadOnlyStatusChangeListener listener) {
089        if (this.readOnlyStatusChangeListener == null && this.property instanceof Property.ReadOnlyStatusChangeNotifier) {
090            this.readOnlyStatusChangeListener = new ReadOnlyStatusChangeListener();
091            ((Property.ReadOnlyStatusChangeNotifier)this.property).addReadOnlyStatusChangeListener(
092              this.readOnlyStatusChangeListener);
093        }
094        super.addReadOnlyStatusChangeListener(listener);
095    }
096
097    @Override
098    public void addValueChangeListener(Property.ValueChangeListener listener) {
099        if (this.valueChangeListener == null && this.property instanceof Property.ValueChangeNotifier) {
100            this.valueChangeListener = new ValueChangeListener();
101            ((Property.ValueChangeNotifier)this.property).addValueChangeListener(this.valueChangeListener);
102        }
103        super.addValueChangeListener(listener);
104    }
105
106// Property.ValueChangeListener relay
107
108    @SuppressWarnings("serial")
109    private class ValueChangeListener implements Property.ValueChangeListener {
110        @Override
111        public void valueChange(Property.ValueChangeEvent event) {
112            ConvertedProperty.this.fireValueChange();
113        }
114    }
115
116// Property.ReadOnlyStatusChangeListener relay
117
118    @SuppressWarnings("serial")
119    private class ReadOnlyStatusChangeListener implements Property.ReadOnlyStatusChangeListener {
120        @Override
121        public void readOnlyStatusChange(Property.ReadOnlyStatusChangeEvent event) {
122            ConvertedProperty.this.fireReadOnlyStatusChange();
123        }
124    }
125}

006package org.dellroad.stuff.schema;
007
008import java.sql.Connection;
009import java.sql.SQLException;
010import java.util.Comparator;
011
012import org.dellroad.stuff.spring.BeanNameComparator;
013import org.springframework.beans.factory.BeanFactory;
014import org.springframework.beans.factory.BeanFactoryAware;
015import org.springframework.beans.factory.InitializingBean;
016import org.springframework.beans.factory.ListableBeanFactory;
017import org.springframework.dao.DataAccessException;
018import org.springframework.jdbc.BadSqlGrammarException;
019import org.springframework.jdbc.support.SQLErrorCodeSQLExceptionTranslator;
020
021/**
022 * {@link SQLSchemaUpdater} optimized for use with Spring.
023 * <ul>
024 * <li>{@link #apply(Connection, DatabaseAction) apply()} is overridden so Spring {@link DataAccessException}s are thrown.</li>
025 * <li>{@link #indicatesUninitializedDatabase indicatesUninitializedDatabase()} is overridden to examine exceptions
026 *  and more precisely using Spring's exception translation infrastructure to filter out false positives.</li>
027 * <li>{@link #getOrderingTieBreaker} is overridden to break ties by ordering updates in the same order
028 *  as they are defined in the bean factory.</li>
029 * <li>This class implements {@link InitializingBean} and verifies all required properties are set.</li>
030 * <li>If no updates are {@linkplain #setUpdates explicitly configured}, then all {@link SpringSQLSchemaUpdate}s found
031 *  in the containing bean factory are automatically configured.
032 * </ul>
033 *
034 * <p>
035 * An example of how this class can be combined with custom XML to define an updater, all its updates,
036 * and a {@link SchemaUpdatingDataSource} that automatically updates the database schema:
037 * <blockquote><pre>
038 *  &lt;beans xmlns="http://www.springframework.org/schema/beans"
039 *    <b>xmlns:dellroad-stuff="http://dellroad-stuff.googlecode.com/schema/dellroad-stuff"</b>
040 *    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
041 *    xmlns:p="http://www.springframework.org/schema/p"
042 *    xsi:schemaLocation="
043 *      http://www.springframework.org/schema/beans
044 *        http://www.springframework.org/schema/beans/spring-beans-3.0.xsd
045 *      <b>http://dellroad-stuff.googlecode.com/schema/dellroad-stuff
046 *        http://dellroad-stuff.googlecode.com/svn/wiki/schemas/dellroad-stuff-1.0.xsd</b>"&gt;
047 *
048 *     &lt;!-- DataSource that automatically updates the database schema --&gt;
049 *     <b>&lt;bean id="dataSource" class="org.dellroad.stuff.schema.SchemaUpdatingDataSource"
050 *       p:dataSource-ref="realDataSource" p:schemaUpdater-ref="schemaUpdater"/&gt;</b>
051 *
052 *     &lt;!--
053 *          Database updater bean. This is used on first access to the DataSource above. Notes:
054 *            - "databaseInitialization" is used to initialize the schema (first time only)
055 *            - "updateTableInitialization" is used to initialize the update table (first time only)
056 *            - In this example, we just use dellroad-stuff's update table initialization for MySQL
057 *            - The &lt;dellroad-stuff:sql-update&gt; beans below will be auto-detected
058 *     --&gt;
059 *     <b>&lt;bean id="schemaUpdater" class="org.dellroad.stuff.schema.SpringSQLSchemaUpdater"&gt;
060 *         &lt;property name="databaseInitialization"&gt;
061 *             &lt;dellroad-stuff:sql resource="classpath:databaseInit.sql"/&gt;
062 *         &lt;/property&gt;
063 *         &lt;property name="updateTableInitialization"&gt;
064 *             &lt;dellroad-stuff:sql resource="classpath:org/dellroad/stuff/schema/updateTable-mysql.sql"/&gt;
065 *         &lt;/property&gt;
066 *     &lt;/bean&gt;</b>
067 *
068 *      &lt;!-- Schema update to add the 'phone' column to the 'User' table --&gt;
069 *      <b>&lt;dellroad-stuff:sql-update id="addPhone"&gt;ALTER TABLE User ADD phone VARCHAR(64)&lt;/dellroad-stuff:sql-update&gt;</b>
070 *
071 *      &lt;!-- Schema update to run some complicated external SQL script --&gt;
072 *      <b>&lt;dellroad-stuff:sql-update id="majorChanges" depends-on="addPhone" resource="classpath:majorChanges.sql"/&gt;</b>
073 *
074 *      &lt;!-- Multiple SQL commands that will be automatically separated into distinct updates --&gt;
075 *      <b>&lt;dellroad-stuff:sql-update id="renameColumn"&gt;
076 *          ALTER TABLE User ADD newName VARCHAR(64);
077 *          ALTER TABLE User SET newName = oldName;
078 *          ALTER TABLE User DROP oldName;
079 *      &lt;/dellroad-stuff:sql-update&gt;</b>
080 *
081 *      &lt;!-- Add more schema updates over time as needed and everything just works... --&gt;
082 *
083 *  &lt;/beans&gt;
084 * </pre></blockquote>
085 *
086 * <p>
087 * In the case no schema updates are explicitly configured, it is required that this updater and all of its
088 * schema updates are defined in the same {@link ListableBeanFactory}.
089 */
090public class SpringSQLSchemaUpdater extends SQLSchemaUpdater implements BeanFactoryAware, InitializingBean {
091
092    private ListableBeanFactory beanFactory;
093
094    @Override
095    public void afterPropertiesSet() throws Exception {
096        if (this.getDatabaseInitialization() == null)
097            throw new Exception("no database initialization configured");
098        if (this.getUpdateTableInitialization() == null)
099            throw new Exception("no update table initialization configured");
100        if (this.getUpdates() == null) {
101            if (this.beanFactory == null) {
102                throw new IllegalArgumentException("no updates explicitly configured and the containing BeanFactory"
103                  + " is not a ListableBeanFactory: " + this.beanFactory);
104            }
105            this.setUpdates(this.beanFactory.getBeansOfType(SpringSQLSchemaUpdate.class).values());
106        }
107    }
108
109    @Override
110    public void setBeanFactory(BeanFactory beanFactory) {
111        if (beanFactory instanceof ListableBeanFactory)
112            this.beanFactory = (ListableBeanFactory)beanFactory;
113    }
114
115    /**
116     * Determine if an exception thrown during {@link #databaseNeedsInitialization} is consistent with
117     * an uninitialized database.
118     *
119     * <p>
120     * The implementation in {@link SpringSQLSchemaUpdater} looks for a {@link BadSqlGrammarException}.
121     */
122    @Override
123    protected boolean indicatesUninitializedDatabase(Connection c, SQLException e) throws SQLException {
124        return this.translate(e, c, null) instanceof BadSqlGrammarException;
125    }
126
127    /**
128     * Apply a {@link DatabaseAction} to a {@link Connection}.
129     *
130     * <p>
131     * The implementation in {@link SQLSchemaUpdater} invokes the action and delegates to
132     * {@link #translate(SQLException, Connection, String) translate()} to convert any {@link SQLException} thrown.
133     *
134     * @throws SQLException if an error occurs attempting to translate a thrown SQLException
135     * @throws DataAccessException if an error occurs accessing the database
136     * @see #translate(SQLException, Connection, String) translate()
137     */
138    @Override
139    protected void apply(Connection c, DatabaseAction<Connection> action) throws SQLException {
140        try {
141            super.apply(c, action);
142        } catch (SQLException e) {
143            String sql = action instanceof SQLCommand ? ((SQLCommand)action).getSQL() : null;
144            throw this.translate(e, c, sql);
145        }
146    }
147
148    /**
149     * Converts {@link SQLException}s into Spring {@link DataAccessException}s.
150     */
151    protected DataAccessException translate(SQLException e, Connection c, String sql) throws SQLException {
152        return new SQLErrorCodeSQLExceptionTranslator(c.getMetaData().getDatabaseProductName())
153          .translate("database access during schema update", sql, e);
154    }
155
156    /**
157     * Get the preferred ordering of two updates that do not have any predecessor constraints
158     * (including implied indirect constraints) between them.
159     *
160     * <p>
161     * In the case no schema updates are explicitly configured, the {@link Comparator} returned by the
162     * implementation in {@link SpringSQLSchemaUpdater} sorts updates in the same order that they appear
163     * in the containing {@link ListableBeanFactory}. Otherwise, the {@linkplain AbstractSchemaUpdater#getOrderingTieBreaker
164     * superclass method} is used.
165     */
166    @Override
167    protected Comparator<SchemaUpdate<Connection>> getOrderingTieBreaker() {
168        if (this.beanFactory == null)
169            return super.getOrderingTieBreaker();
170        final BeanNameComparator beanNameComparator = new BeanNameComparator(this.beanFactory);
171        return new Comparator<SchemaUpdate<Connection>>() {
172            @Override
173            public int compare(SchemaUpdate<Connection> update1, SchemaUpdate<Connection> update2) {
174                return beanNameComparator.compare(update1.getName(), update2.getName());
175            }
176        };
177    }
178}

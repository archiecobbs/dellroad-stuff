006package org.dellroad.stuff.schema;
007
008import java.util.Set;
009
010import org.dellroad.stuff.graph.TopologicalSorter.EdgeLister;
011
012/**
013 * {@link EdgeLister} implementation reflecting {@link SchemaUpdate} predecessor constraints.
014 *
015 * <p>
016 * Graph edges will exist from each update to its predecessors. Note, this is the reverse of the actual
017 * desired ordering of the updates.
018 *
019 * @param <T> database transaction type
020 */
021public class SchemaUpdateEdgeLister<T> implements EdgeLister<SchemaUpdate<T>> {
022
023    @Override
024    public Set<SchemaUpdate<T>> getOutEdges(SchemaUpdate<T> update) {
025        return update.getRequiredPredecessors();
026    }
027}

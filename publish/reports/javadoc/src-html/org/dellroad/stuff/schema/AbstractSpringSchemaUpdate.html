006package org.dellroad.stuff.schema;
007
008import java.util.HashSet;
009
010import org.springframework.beans.factory.BeanFactory;
011import org.springframework.beans.factory.BeanFactoryAware;
012import org.springframework.beans.factory.BeanNameAware;
013import org.springframework.beans.factory.BeanNotOfRequiredTypeException;
014import org.springframework.beans.factory.InitializingBean;
015import org.springframework.beans.factory.NoSuchBeanDefinitionException;
016import org.springframework.beans.factory.config.ConfigurableBeanFactory;
017
018/**
019 * Support superclass for {@link SchemaUpdate}s declared in Spring {@link BeanFactory}s that infer their
020 * names and required predecessors from their Spring bean attributes.
021 *
022 * <p>
023 * Instances infer their {@linkplain #getName name} and {@linkplain #getRequiredPredecessors required predecessors} from
024 * their Spring bean name (specified by the <code>id</code> XML attribute) and Spring dependencies (specified by
025 * the <code>depends-on</code> XML attribute), respectively.
026 *
027 * <p>
028 * Note: the use of <code>depends-on</code> is an abuse of Spring's dependency notation for convenience. Normally
029 * <code>depends-on</code> refers to bean intialization ordering, whereas this class uses it to refer to schema update ordering.
030 * Schema updates are not normally expected to have any initialization ordering requirements, so this abuse shouldn't matter.
031 * If they do, this class should not be used.
032 *
033 * <p>
034 * The containing {@link BeanFactory} must be a {@link ConfigurableBeanFactory} (normally this is always the case).
035 *
036 * @param <T> database transaction type
037 */
038public abstract class AbstractSpringSchemaUpdate<T> extends AbstractSchemaUpdate<T>
039  implements BeanNameAware, BeanFactoryAware, InitializingBean {
040
041    protected String beanName;
042    protected BeanFactory beanFactory;
043
044    @Override
045    public void setBeanName(String beanName) {
046        this.beanName = beanName;
047    }
048
049    @Override
050    public void setBeanFactory(BeanFactory beanFactory) {
051        this.beanFactory = beanFactory;
052    }
053
054    /**
055     * Configures the update name and required predecessors based on the Spring bean's name and
056     * {@link BeanFactory} dependencies.
057     */
058    @Override
059    public void afterPropertiesSet() throws Exception {
060        if (this.beanFactory == null)
061            throw new IllegalArgumentException("no BeanFactory configured");
062        if (this.beanName == null)
063            throw new IllegalArgumentException("no beanName configured");
064        this.setName(this.beanName);
065        this.setRequiredPredecessorsFromDependencies();
066    }
067
068    /**
069     * Infer required predecessors from Spring dependencies.
070     *
071     * @throws IllegalArgumentException if this instance's {@code beanFactory} is not yet configured,
072     *  or not a {@link ConfigurableBeanFactory}
073     */
074    @SuppressWarnings("unchecked")
075    protected void setRequiredPredecessorsFromDependencies() {
076
077        // Check factory type
078        if (!(this.beanFactory instanceof ConfigurableBeanFactory))
079            throw new IllegalArgumentException("BeanFactory is not a ConfigurableBeanFactory: " + this.beanFactory);
080        ConfigurableBeanFactory configurableBeanFactory = (ConfigurableBeanFactory)this.beanFactory;
081
082        // Find required predecessors defined as Spring dependencies
083        String[] predecessorNames = configurableBeanFactory.getDependenciesForBean(this.beanName);
084        HashSet<SchemaUpdate<T>> predecessors = new HashSet<SchemaUpdate<T>>(predecessorNames.length);
085        for (String predecessorName : predecessorNames) {
086            try {
087                predecessors.add((SchemaUpdate<T>)configurableBeanFactory.getBean(predecessorName, SchemaUpdate.class));
088            } catch (NoSuchBeanDefinitionException e) {
089                continue;
090            } catch (BeanNotOfRequiredTypeException e) {
091                continue;
092            }
093        }
094        this.setRequiredPredecessors(predecessors);
095    }
096}

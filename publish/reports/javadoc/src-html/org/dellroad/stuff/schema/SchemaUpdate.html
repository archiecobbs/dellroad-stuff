006package org.dellroad.stuff.schema;
007
008import java.util.List;
009import java.util.Set;
010
011/**
012 * A one-time database schema update. Instances typically perform changes to the database schema, though non-schema updates
013 * that just manipulate the data in the database are perfectly valid as well.
014 *
015 * <p>
016 * Each {@link SchemaUpdate} has a unique name among all updates ever applied to a single database, and zero
017 * or more required predecessors, which are other dependent updates that must be applied first.
018 * </p>
019 *
020 * <p>
021 * Once an update has been applied to a database, it must not be changed; otherwise, inconsistencies
022 * can exist between databases that were updated using the old version vs. databases that were
023 * updated using the new version. If an update has been applied but had the wrong behavior, instead of
024 * changing the update, it's better to create a new update that depends on the first as a predecessor
025 * and corrects the mistake.
026 * </p>
027 *
028 * <p>
029 * If you do have to change an update after it has been applied, then those databases that have
030 * already seen the previous version of the update must be manually corrected so they are in the
031 * same state that the new version of the update would have left them.
032 * </p>
033 *
034 * @param <T> database transaction type
035 */
036public interface SchemaUpdate<T> {
037
038    /**
039     * Get the unique name of this update. This name must be unique among all updates ever applied to the database
040     * and must never change once this update has been applied to any database.
041     *
042     * @return the name of this update; must not be the empty string
043     */
044    String getName();
045
046    /**
047     * Get the the other updates (if any) that must be applied <b>before</b> this update may be applied.
048     *
049     * @return set of zero or more other updates
050     * @see #getName
051     */
052    Set<SchemaUpdate<T>> getRequiredPredecessors();
053
054    /**
055     * Get the action(s) that comprise this update. Ideally, individual actions should be atomic database operations, i.e.,
056     * each one should either finish completely, or else leave the database in a state where it can be tried again.
057     * In any case, each action will be applied within its own transaction when transactions are supported by the database
058     * unless {@link #isSingleAction} returns true.
059     *
060     * @return a list of zero or more actions to apply
061     * @see #isSingleAction
062     */
063    List<? extends DatabaseAction<T>> getDatabaseActions();
064
065    /**
066     * Determine whether, if this instance contains multiple individual actions, should they be applied in a single
067     * transaction and recorded as a single update. Normally this is false. If true, partially completed updates
068     * can result if one of the action fails.
069     */
070    boolean isSingleAction();
071}

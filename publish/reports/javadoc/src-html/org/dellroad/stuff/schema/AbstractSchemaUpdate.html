006package org.dellroad.stuff.schema;
007
008import java.util.HashSet;
009import java.util.Set;
010
011/**
012 * Support superclass for {@link SchemaUpdate} implementations with standard bean property implementations.
013 *
014 * @param <T> database transaction type
015 */
016public abstract class AbstractSchemaUpdate<T> implements SchemaUpdate<T> {
017
018    private String name;
019    private Set<SchemaUpdate<T>> requiredPredecessors = new HashSet<SchemaUpdate<T>>();
020    private boolean singleAction;
021
022    @Override
023    public String getName() {
024        return this.name;
025    }
026
027    public void setName(String name) {
028        this.name = name;
029    }
030
031    @Override
032    public Set<SchemaUpdate<T>> getRequiredPredecessors() {
033        return this.requiredPredecessors;
034    }
035
036    public void setRequiredPredecessors(Set<SchemaUpdate<T>> requiredPredecessors) {
037        this.requiredPredecessors = requiredPredecessors;
038    }
039
040    @Override
041    public boolean isSingleAction() {
042        return this.singleAction;
043    }
044
045    public void setSingleAction(boolean singleAction) {
046        this.singleAction = singleAction;
047    }
048
049    @Override
050    public String toString() {
051        return this.getClass().getSimpleName() + "[" + this.getName() + "]";
052    }
053}

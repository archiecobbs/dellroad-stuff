006package org.dellroad.stuff.schema;
007
008import java.sql.Connection;
009import java.sql.SQLException;
010import java.sql.Statement;
011
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015/**
016 * An SQL {@link DatabaseAction} that executes a single SQL statement.
017 */
018public class SQLCommand implements DatabaseAction<Connection> {
019
020    protected final Logger log = LoggerFactory.getLogger(this.getClass());
021
022    private final String sql;
023
024    /**
025     * Constructor.
026     *
027     * @param sql the SQL to execute; must be a single statement
028     * @throws IllegalArgumentException if {@code sql} is null or contains only whitespace
029     */
030    public SQLCommand(String sql) {
031        if (sql == null)
032            throw new IllegalArgumentException("null sql");
033        sql = sql.trim();
034        if (sql.length() == 0)
035            throw new IllegalArgumentException("empty sql");
036        this.sql = sql;
037    }
038
039    public String getSQL() {
040        return this.sql;
041    }
042
043    /**
044     * Execute the SQL statement.
045     *
046     * <p>
047     * The implementation in {@link SQLCommand} creates a {@link Statement} and then executes the configured
048     * SQL command via {@link Statement#execute}. Subclasses may wish to override.
049     *
050     * @throws SQLException if an error occurs while accessing the database
051     */
052    @Override
053    public void apply(Connection c) throws SQLException {
054        Statement statement = c.createStatement();
055        String sep = this.sql.indexOf('\n') != -1 ? "\n" : " ";
056        this.log.info("executing SQL statement:" + sep + this.sql);
057        try {
058            statement.execute(this.sql);
059        } finally {
060            statement.close();
061        }
062    }
063}

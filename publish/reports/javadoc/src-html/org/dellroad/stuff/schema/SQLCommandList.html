006package org.dellroad.stuff.schema;
007
008import java.sql.Connection;
009import java.sql.SQLException;
010import java.util.ArrayList;
011import java.util.List;
012import java.util.regex.Pattern;
013
014/**
015 * Holds and executes a configured SQL script, possibly containing multiple statements.
016 *
017 * <p>
018 * If the script contains multiple SQL statements, individual statements will be executed individually, in order, by
019 * {@link #apply apply()}; however, this requires proper configuration of the {@linkplain #setSplitPattern split pattern}.
020 *
021 * <p>
022 * When using Spring, beans of this type can be created succintly using the <code>&lt;dellroad-stuff:sql&gt;</code> custom
023 * XML element. The {@linkplain #setSplitPattern split pattern} may be configured via the {@code split-pattern} attribute,
024 * and the SQL script is specified either directly via inline text or using the {@code resource} attribute. In the latter case,
025 * the character encoding can specified via the {@code charset} attribute (default is {@code "UTF-8"}).
026 *
027 * <p>
028 * For example:
029 * <blockquote><pre>
030 *  &lt;beans xmlns="http://www.springframework.org/schema/beans"
031 *    <b>xmlns:dellroad-stuff="http://dellroad-stuff.googlecode.com/schema/dellroad-stuff"</b>
032 *    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
033 *    xsi:schemaLocation="
034 *      http://www.springframework.org/schema/beans
035 *        http://www.springframework.org/schema/beans/spring-beans-3.0.xsd
036 *      <b>http://dellroad-stuff.googlecode.com/schema/dellroad-stuff
037 *        http://dellroad-stuff.googlecode.com/svn/wiki/schemas/dellroad-stuff-1.0.xsd</b>"&gt;
038 *
039 *      &lt;!-- SQL action that clears the audit log --&gt;
040 *      <b>&lt;dellroad-stuff:sql&gt;DELETE * FROM AUDIT_LOG&lt;/dellroad-stuff:sql&gt;</b>
041 *
042 *      &lt;!-- A more complicated, externally referenced, SQL script --&gt;
043 *      <b>&lt;dellroad-stuff:sql resource="classpath:reset.sql" split-pattern=";\n"/&gt;</b>
044 *
045 *      &lt;!-- other beans... --&gt;
046 *
047 *  &lt;/beans&gt;
048 * </pre></blockquote>
049 */
050public class SQLCommandList implements DatabaseAction<Connection> {
051
052    /**
053     * The default split pattern: <code>{@value}</code>.
054     */
055    public static final String DEFAULT_SPLIT_PATTERN = ";[ \\t\\r]*\\n\\s*";
056
057    private String sqlScript;
058    private String splitPattern = DEFAULT_SPLIT_PATTERN;
059
060    public SQLCommandList() {
061    }
062
063    public SQLCommandList(String sqlScript) {
064        this.setSQLScript(sqlScript);
065    }
066
067    /**
068     * Configure the SQL script. This is a required property.
069     *
070     * <p>
071     * For scripts in external resources, consider {@link org.dellroad.stuff.spring.ResourceReaderFactoryBean}
072     * or use the <code>&lt;dellroad-stuff:sql&gt;</code> element.
073     *
074     * @param sqlScript script containing one or more SQL statements; if more than one statement is present,
075     *  a {@link #setSplitPattern split pattern} must also be configured
076     * @see #setSplitPattern setSplitPattern()
077     */
078    public void setSQLScript(String sqlScript) {
079        this.sqlScript = sqlScript;
080    }
081
082    /**
083     * Set the <i>split pattern</i> used to split apart a script containing multiple SQL statements into individual statements.
084     *
085     * <p>
086     * The default value for this property is <code>";[ \t\r]*\n\s*"</code>, which should handle cases where
087     * SQL statements are terminated by semi-colons and each SQL statement starts on a new line.
088     *
089     * <p>
090     * If this is set to {@code null}, or the script does not contain any instances of the regular expression,
091     * the script is assumed to contain a single SQL statement. SQL statements are whitespace-trimmed and any
092     * "statements" that consist entirely of whitespace are ignored.
093     *
094     * @param splitPattern regular expression pattern used to split multiple SQL statements
095     * @throws java.util.regex.PatternSyntaxException if the pattern is not a valid Java regular expression
096     */
097    public void setSplitPattern(String splitPattern) {
098        if (splitPattern != null)
099            Pattern.compile(splitPattern);
100        this.splitPattern = splitPattern;
101    }
102
103    /**
104     * Applies each individual SQL command in the script. Commands are separated
105     * using the {@link #setSplitPattern split pattern}.
106     */
107    @Override
108    public void apply(Connection c) throws SQLException {
109        for (SQLCommand sqlCommand : this.split())
110            sqlCommand.apply(c);
111    }
112
113    /**
114     * Split the SQL script into individual statements and return them as {@link DatabaseAction}s.
115     *
116     * @return list of individual SQL statements
117     */
118    public List<SQLCommand> split() {
119        ArrayList<SQLCommand> list = new ArrayList<SQLCommand>();
120        for (String sql : this.splitSQL())
121            list.add(new SQLCommand(sql));
122        return list;
123    }
124
125    /**
126     * Split the {@linkplain #setSQLScript configured SQL script} into individual SQL statements
127     * using the configured {@linkplain #setSplitPattern split pattern}.
128     *
129     * @return an array of individual SQL statements
130     * @throws IllegalArgumentException if no SQL script is configured
131     */
132    public String[] splitSQL() {
133        if (this.sqlScript == null)
134            throw new IllegalArgumentException("no SQL script configured");
135        String[] sqls = this.splitPattern != null ? this.sqlScript.split(this.splitPattern) : new String[] { this.sqlScript };
136        ArrayList<String> list = new ArrayList<String>(sqls.length);
137        for (String sql : sqls) {
138            sql = sql.trim();
139            if (sql.length() == 0)
140                continue;
141            list.add(sql);
142        }
143        return list.toArray(new String[list.size()]);
144    }
145}

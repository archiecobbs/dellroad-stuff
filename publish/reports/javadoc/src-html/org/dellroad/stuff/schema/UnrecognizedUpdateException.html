006package org.dellroad.stuff.schema;

008/**
009 * Exception thrown by a {@link AbstractSchemaUpdater#initializeAndUpdateDatabase} when it encounters
010 * one or more recorded updates that are not recognized.
011 */
012@SuppressWarnings("serial")
013public class UnrecognizedUpdateException extends IllegalStateException {
014
015    public UnrecognizedUpdateException(String message) {
016        super(message);
017    }
018
019    public UnrecognizedUpdateException(String message, Throwable cause) {
020        super(message, cause);
021    }
022}

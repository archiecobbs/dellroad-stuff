006package org.dellroad.stuff.schema;
007
008import java.sql.Connection;
009import java.sql.SQLException;
010
011import javax.sql.DataSource;
012
013/**
014 * A {@link DataSource} that wraps an inner {@link DataSource} and automatically applies a configured
015 * {@link SQLCommandList} on first access.
016 *
017 * @see SQLCommandList
018 */
019public class UpdatingDataSource extends AbstractUpdatingDataSource {
020
021    private SQLCommandList action;
022    private boolean transactional = true;
023
024    /**
025     * Configure the {@link SQLCommandList} to be applied to the database on first access. Required property.
026     *
027     * @param action SQL command(s) to apply exactly once to the underlying {@link DataSource}
028     */
029    public void setSQLCommandList(SQLCommandList action) {
030        this.action = action;
031    }
032
033    /**
034     * Configure whether the {@link SQLCommandList} is applied transactionally or not.
035     * Default is {@code true}.
036     *
037     * @param transactional true for transactional application of SQL command(s)
038     */
039    public void setTransactional(boolean transactional) {
040        this.transactional = transactional;
041    }
042
043    @Override
044    protected void updateDataSource(DataSource dataSource) throws SQLException {
045
046        // Sanity check
047        if (this.action == null)
048            throw new IllegalArgumentException("no SQLCommandList configured");
049
050        // Get connection
051        Connection c = dataSource.getConnection();
052        boolean tx = this.transactional;
053        try {
054            try {
055
056                // Open transaction if so configured
057                if (tx) {
058                    c.setTransactionIsolation(Connection.TRANSACTION_SERIALIZABLE);
059                    c.setAutoCommit(false);
060                }
061
062                // Apply SQL command(s)
063                this.action.apply(c);
064
065                // Commit transaction
066                if (tx)
067                    c.commit();
068                tx = false;
069            } finally {
070                if (tx)
071                    c.rollback();
072            }
073        } finally {
074            c.close();
075        }
076    }
077}

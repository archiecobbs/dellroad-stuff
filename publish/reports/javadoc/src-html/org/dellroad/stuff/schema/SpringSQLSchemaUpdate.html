006package org.dellroad.stuff.schema;
007
008import java.sql.Connection;
009import java.util.ArrayList;
010import java.util.List;
011
012/**
013 * Spring-enabled SQL {@link SchemaUpdate}.
014 *
015 * <p>
016 * The {@link #setSQLCommandList sqlCommandList} property is required.
017 *
018 * <p>
019 * Instances can be created succintly in Spring using the <code>&lt;dellroad-stuff:sql-update&gt;</code> custom XML element,
020 * which works just like <code>&lt;dellroad-stuff:sql&gt;</code> except that it wraps the resulting {@link SQLCommandList}
021 * as a delegate inside an instance of this class.
022 *
023 * <p>
024 * For example:
025 * <blockquote><pre>
026 *  &lt;beans xmlns="http://www.springframework.org/schema/beans"
027 *    <b>xmlns:dellroad-stuff="http://dellroad-stuff.googlecode.com/schema/dellroad-stuff"</b>
028 *    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
029 *    xsi:schemaLocation="
030 *      http://www.springframework.org/schema/beans
031 *        http://www.springframework.org/schema/beans/spring-beans-3.0.xsd
032 *      <b>http://dellroad-stuff.googlecode.com/schema/dellroad-stuff
033 *        http://dellroad-stuff.googlecode.com/svn/wiki/schemas/dellroad-stuff-1.0.xsd</b>"&gt;
034 *
035 *      &lt;!-- Schema update to add the 'phone' column to the 'User' table --&gt;
036 *      <b>&lt;dellroad-stuff:sql-update id="addPhone"&gt;ALTER TABLE User ADD phone VARCHAR(64)&lt;/dellroad-stuff:sql-update&gt;</b>
037 *
038 *      &lt;!-- Schema update to run some complicated external SQL script --&gt;
039 *      <b>&lt;dellroad-stuff:sql-update id="majorChanges" depends-on="addPhone" resource="classpath:majorChanges.sql"/&gt;</b>
040 *
041 *      &lt;!-- more beans... --&gt;
042 *
043 *  &lt;/beans&gt;
044 * </pre></blockquote>
045 *
046 * <p>
047 * A multi-statement SQL script is normally treated as a set of individual updates. For example:
048 * <blockquote><pre>
049 *      <b>&lt;dellroad-stuff:sql-update id="renameColumn"&gt;
050 *          ALTER TABLE User ADD newName VARCHAR(64);
051 *          ALTER TABLE User SET newName = oldName;
052 *          ALTER TABLE User DROP oldName;
053 *      &lt;/dellroad-stuff:sql-update&gt;</b>
054 * </pre></blockquote>
055 * This will create three separate update beans named <code>renameColumn-00001</code>, <code>renameColumn-00002</code>, and
056 * <code>renameColumn-00003</code>. You can disable this behavior by adding the attribute <code>single-action="true"</code>,
057 * in which case all three of the statements will be executed together in the same transaction and recorded under the name
058 * <code>renameColumn</code>; this means that they must all complete successfully or you could end up with a partially
059 * completed update.
060 *
061 * <p>
062 * Note that if the nested SQL script only contains one SQL statement, any <code>single-action</code> attribute is
063 * ignored and the bean's given name (e.g., <code>renameColumn</code>) is always used as the name of the single update.
064 *
065 * @see SQLCommandList
066 */
067public class SpringSQLSchemaUpdate extends AbstractSpringSchemaUpdate<Connection> {
068
069    private SQLCommandList sqlCommandList;
070
071    @Override
072    public void afterPropertiesSet() throws Exception {
073        super.afterPropertiesSet();
074        if (this.sqlCommandList == null)
075            throw new Exception("no SQLCommandList configured");
076    }
077
078    /**
079     * Configure the {@link SQLCommandList}. This is a required property.
080     *
081     * @param sqlCommandList list of SQL statements that perform this update
082     * @see DatabaseAction
083     */
084    public void setSQLCommandList(SQLCommandList sqlCommandList) {
085        this.sqlCommandList = sqlCommandList;
086    }
087
088    public SQLCommandList getSQLCommandList() {
089        return this.sqlCommandList;
090    }
091
092    @Override
093    public List<DatabaseAction<Connection>> getDatabaseActions() {
094        return new ArrayList<DatabaseAction<Connection>>(this.getSQLCommandList().split());
095    }
096}

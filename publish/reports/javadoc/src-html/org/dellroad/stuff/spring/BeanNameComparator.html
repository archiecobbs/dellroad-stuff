006package org.dellroad.stuff.spring;
007
008import java.util.Comparator;
009import java.util.HashMap;
010
011import org.springframework.beans.factory.ListableBeanFactory;
012
013/**
014 * A {@link Comparator} that orders Spring bean names in the same order as the corresponding
015 * beans appear in a {@link ListableBeanFactory}.
016 *
017 * <p>
018 * Names that are not present in the configured {@link ListableBeanFactory} cause an exception.
019 */
020public class BeanNameComparator implements Comparator<String> {
021
022    private final HashMap<String, Integer> beanNameMap;
023    private final String factoryName;
024
025    public BeanNameComparator(ListableBeanFactory beanFactory) {
026        String[] beanNames = beanFactory.getBeanDefinitionNames();
027        this.beanNameMap = new HashMap<String, Integer>(beanNames.length);
028        for (int i = 0; i < beanNames.length; i++)
029            this.beanNameMap.put(beanNames[i], i);
030        this.factoryName = "" + beanFactory;
031    }
032
033    @Override
034    public int compare(String name1, String name2) {
035        Integer index1 = this.beanNameMap.get(name1);
036        Integer index2 = this.beanNameMap.get(name2);
037        if (index1 == null)
038            throw new IllegalArgumentException("failed to find bean `" + name1 + "' in bean factory " + this.factoryName);
039        if (index2 == null)
040            throw new IllegalArgumentException("failed to find bean `" + name2 + "' in bean factory " + this.factoryName);
041        return index1 - index2;
042    }
043}

006package org.dellroad.stuff.spring;
007
008import org.springframework.beans.factory.xml.NamespaceHandlerSupport;
009
010/**
011 * Spring {@link org.springframework.beans.factory.xml.NamespaceHandler NamespaceHandler}
012 * for the <code>dellroad-stuff</code> XML namespace.
013 *
014 * <p>
015 * This adds support for the following Spring custom XML tags:
016 * <ul>
017 *  <li><code>&lt;dellroad-stuff:sql/&gt;</code>, which defines a {@link org.dellroad.stuff.schema.SQLCommandList} bean</li>
018 *  <li><code>&lt;dellroad-stuff:sql-update/&gt;</code>, which wraps a {@link org.dellroad.stuff.schema.SQLCommandList}
019 *  in a {@link org.dellroad.stuff.schema.SpringSQLSchemaUpdate} bean</li>
020 * </ul>
021 */
022public class DellRoadStuffNamespaceHandler extends NamespaceHandlerSupport {
023
024    public static final String NAMESPACE_URI = "http://dellroad-stuff.googlecode.com/schema/dellroad-stuff";
025
026    public static final String SQL_ELEMENT_NAME = "sql";
027    public static final String SQL_UPDATE_ELEMENT_NAME = "sql-update";
028
029    @Override
030    public void init() {
031        this.registerBeanDefinitionParser(SQL_ELEMENT_NAME, new SQLBeanDefinitionParser());
032        this.registerBeanDefinitionParser(SQL_UPDATE_ELEMENT_NAME, new SQLUpdateBeanDefinitionParser());
033    }
034}

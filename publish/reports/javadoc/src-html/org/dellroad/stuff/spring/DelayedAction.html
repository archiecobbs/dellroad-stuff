006package org.dellroad.stuff.spring;
007
008import java.util.Date;
009import java.util.concurrent.ScheduledExecutorService;
010import java.util.concurrent.ScheduledFuture;
011import java.util.concurrent.TimeUnit;
012import java.util.concurrent.locks.Lock;
013
014import org.springframework.scheduling.TaskScheduler;
015
016/**
017 * Manages a delayed action without race conditions.
018 *
019 * <p>
020 * A "delayed action" is a some action that needs to get done by some time in the future.
021 * </p>
022 *
023 * <p>
024 * This class does two things:
025 * <ul>
026 *  <li>It collapses multiple attempts to schedule the action into a single scheduled action,
027 *  i.e., at most one outstanding scheduled action can exist at a time.</li>
028 *  <li>It provides a race-free and 100% reliable way to {@link #cancel} a future scheduled action, if any.</li>
029 * </ul>
030 * </p>
031 *
032 * <p>
033 * The action itself is defined by the subclass' implementation of {@link #run run()}.
034 * </p>
035 *
036 * <p>
037 * To avoid races, this class requires the user to supply a <i>locking object</i>. This may either be a normal
038 * Java object, in which case normal Java synchronization is used, or a {@link Lock} object. The locking object is
039 * used to serialize scheduling activity and action invocation. In other words, the locking object is locked during
040 * the execution of {@link #schedule schedule()}, {@link #cancel cancel()}, and {@link #run run()}.
041 * </p>
042 *
043 * <p>
044 * Therefore, any time the locking object is locked, the state of this {@link DelayedAction} instance is "frozen"
045 * in one of three states: not scheduled, scheduled, or executing (in the latter case, of course the thread doing the
046 * executing is the one holding the lock). Therefore, to completely avoid race conditions, user code must <i>itself</i>
047 * lock the locking object itself prior to invoking any methods in this class.
048 * </p>
049 *
050 * <p>
051 * Typically the most convenient locking object to use is the user's own {@code this} object, which can be locked using a
052 * {@code synchronized} method or block.
053 * </p>
054 *
055 * <p>
056 * Note: in the case that {@link #run run()} invokes {@link Object#wait Object.wait()} on the locking object, thereby
057 * temporarily releasing the lock, to any other methods in this class it will appear as if that execution has already
058 * completed.
059 */
060public abstract class DelayedAction implements Runnable {
061
062    private final Lock lock;
063    private final Object objLock;
064    private final TaskScheduler taskScheduler;
065    private final ScheduledExecutorService executorService;
066
067    private ScheduledFuture<?> future;
068    private Date futureDate;
069
070    /**
071     * Constructor utitilizing a {@link TaskScheduler} and normal Java object locking.
072     *
073     * @param lock locking object used to serialize activity, or null for {@code this}
074     * @param taskScheduler scheduler object
075     * @throws IllegalArgumentException if {@code taskScheduler} is null
076     */
077    protected DelayedAction(Object lock, TaskScheduler taskScheduler) {
078        this(null, lock, taskScheduler, null);
079        if (taskScheduler == null)
080            throw new IllegalArgumentException("null taskScheduler");
081    }
082
083    /**
084     * Constructor utitilizing a {@link ScheduledExecutorService} and normal Java object locking.
085     *
086     * @param lock locking object used to serialize activity, or null for {@code this}
087     * @param executorService scheduler object
088     * @throws IllegalArgumentException if {@code executorService} is null
089     */
090    protected DelayedAction(Object lock, ScheduledExecutorService executorService) {
091        this(null, lock, null, executorService);
092        if (executorService == null)
093            throw new IllegalArgumentException("null executorService");
094    }
095
096    /**
097     * Constructor utitilizing a {@link TaskScheduler} and a {@link Lock} for locking.
098     *
099     * @param lock locking object used to serialize activity
100     * @param taskScheduler scheduler object
101     * @throws IllegalArgumentException if {@code lock} is null
102     * @throws IllegalArgumentException if {@code taskScheduler} is null
103     */
104    protected DelayedAction(Lock lock, TaskScheduler taskScheduler) {
105        this(lock, null, taskScheduler, null);
106        if (lock == null)
107            throw new IllegalArgumentException("null lock");
108        if (taskScheduler == null)
109            throw new IllegalArgumentException("null taskScheduler");
110    }
111
112    /**
113     * Constructor utitilizing a {@link ScheduledExecutorService} and a {@link Lock} for locking.
114     *
115     * @param lock locking object used to serialize activity
116     * @param executorService scheduler object
117     * @throws IllegalArgumentException if {@code lock} is null
118     * @throws IllegalArgumentException if {@code executorService} is null
119     */
120    protected DelayedAction(Lock lock, ScheduledExecutorService executorService) {
121        this(lock, null, null, executorService);
122        if (lock == null)
123            throw new IllegalArgumentException("null lock");
124        if (executorService == null)
125            throw new IllegalArgumentException("null executorService");
126    }
127
128    private DelayedAction(Lock lock, Object objLock, TaskScheduler taskScheduler, ScheduledExecutorService executorService) {
129        this.lock = lock;
130        this.objLock = lock != null ? null : objLock != null ? objLock : this;
131        this.taskScheduler = taskScheduler;
132        this.executorService = executorService;
133    }
134
135    /**
136     * Schedule the delayed action for the given time.
137     *
138     * <p>
139     * More precisely:
140     * <ul>
141     *  <li>If an action currently executing, before doing anything else this method blocks until it completes;
142     *  if this behavior is undesirable, the caller can avoid this behavior by synchronizing on the locking object
143     *  prior to invoking this method.</li>
144     *  <li>If no action is scheduled, one is scheduled for the given time.</li>
145     *  <li>If an action is already scheduled, and the given time is on or after the scheduled time, nothing changes.</li>
146     *  <li>If an action is already scheduled, and the given time is prior to the scheduled time,
147     *  the action is rescheduled for the earlier time.</li>
148     * </ul>
149     * </p>
150     *
151     * <p>
152     * The net result is that, for any invocation, this method guarantees exactly one execution of the action will
153     * occur approximately on or before the given date; however, multiple invocations of this method prior to action
154     * execution can only ever result in a single "shared" action.
155     * </p>
156     *
157     * @param date scheduled execution time (at the latest)
158     * @throws IllegalArgumentException if {@code date} is null
159     * @throws org.springframework.core.task.TaskRejectedException
160     *  if the given task was not accepted for internal reasons (e.g. a pool overload handling policy
161     *  or a pool shutdown in progress)
162     */
163    public void schedule(final Date date) {
164        this.runLocked(new Runnable() {
165            @Override
166            public void run() {
167                DelayedAction.this.scheduleWhileLocked(date);
168            }
169        });
170    }
171    private void scheduleWhileLocked(final Date date) {
172
173        // Sanity check
174        if (date == null)
175            throw new IllegalArgumentException("null date");
176
177        // Already scheduled?
178        if (this.future != null) {
179
180            // Requested time is after scheduled time? Note: must be ">=", not ">" to ensure monotonically increasing Dates
181            if (date.compareTo(this.futureDate) >= 0)
182                return;
183
184            // Cancel it
185            this.cancel();
186        }
187
188        // Schedule it
189        this.future = this.schedule(new Runnable() {
190            @Override
191            public void run() {
192                DelayedAction.this.futureInvoked(date);
193            }
194        }, date);
195        this.futureDate = date;
196    }
197
198    /**
199     * Cancel the future scheduled action, if any.
200     *
201     * <p>
202     * More precisely:
203     * <ul>
204     *  <li>If an action currently executing, before doing anything else this method blocks until it completes;
205     *  if this behavior is undesirable, the caller can avoid this behavior by synchronizing on the locking object
206     *  prior to invoking this method.</li>
207     *  <li>If an action is scheduled but has not started yet, it is guaranteed not to run.</li>
208     *  <li>If no action is scheduled or executing, nothing changes.</li>
209     * </ul>
210     * </p>
211     */
212    public void cancel() {
213        this.runLocked(new Runnable() {
214            @Override
215            public void run() {
216                DelayedAction.this.cancelWhileLocked();
217            }
218        });
219    }
220    private void cancelWhileLocked() {
221
222        // Anything to do?
223        if (this.future == null)
224            return;
225
226        // Cancel future
227        this.future.cancel(false);
228        this.future = null;
229        this.futureDate = null;
230    }
231
232    /**
233     * Determine whether there is currently an outstanding scheduled action.
234     */
235    public boolean isScheduled() {
236        final boolean[] result = new boolean[1];
237        this.runLocked(new Runnable() {
238            @Override
239            public void run() {
240                result[0] = DelayedAction.this.future != null;
241            }
242        });
243        return result[0];
244    }
245
246    /**
247     * Get the scheduled time for the outstanding scheduled action, if any.
248     *
249     * @return oustanding action scheduled time, or null if there is no scheduled action
250     */
251    public Date getScheduledTime() {
252        final Date[] result = new Date[1];
253        this.runLocked(new Runnable() {
254            @Override
255            public void run() {
256                result[0] = DelayedAction.this.futureDate;
257            }
258        });
259        return result[0];
260    }
261
262    /**
263     * Schedule the given action using the task scheduler passed to the constructor.
264     * Use of this method does not change the state of this instance.
265     *
266     * @param action action to perform
267     * @param date when to perform it
268     * @throws IllegalArgumentException if either parameter is null
269     * @throws java.util.concurrent.RejectedExecutionException
270     *   if the given task was not accepted for internal reasons (e.g. a pool overload handling
271     *  policy or a pool shutdown in progress)
272     */
273    protected ScheduledFuture<?> schedule(Runnable action, Date date) {
274        if (action == null)
275            throw new IllegalArgumentException("null action");
276        if (date == null)
277            throw new IllegalArgumentException("null date");
278        if (this.taskScheduler != null)
279            return this.taskScheduler.schedule(action, date);
280        long now = System.currentTimeMillis();
281        long when = date.getTime();
282        if (when < now)
283            when = now;
284        return this.executorService.schedule(action, when - now, TimeUnit.MILLISECONDS);
285    }
286
287    // Do the action
288    private void futureInvoked(final Date date) {
289        this.runLocked(new Runnable() {
290            @Override
291            public void run() {
292                DelayedAction.this.futureInvokedWhileLocked(date);
293            }
294        });
295    }
296    private void futureInvokedWhileLocked(Date date) {
297
298        // Handle race condition where future.cancel() fails
299        if (this.futureDate != date)
300            return;
301
302        // Reset state
303        this.future = null;
304        this.futureDate = null;
305
306        // Perform action
307        this.run();
308    }
309
310    // Invoke action while locked
311    private void runLocked(Runnable action) {
312        if (this.objLock != null) {
313            synchronized (this.objLock) {
314                action.run();
315            }
316        } else {
317            this.lock.lock();
318            try {
319                action.run();
320            } finally {
321                this.lock.unlock();
322            }
323        }
324    }
325}

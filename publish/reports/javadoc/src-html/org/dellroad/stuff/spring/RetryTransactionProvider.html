006package org.dellroad.stuff.spring;
007
008import org.springframework.dao.support.PersistenceExceptionTranslator;
009
010/**
011 * Interface implemented by the {@code RetryTransactionAspect}, which implements the {@link RetryTransaction} functionality.
012 *
013 * @see RetryTransaction
014 */
015public interface RetryTransactionProvider {
016
017    /**
018     * Get the configured exception translator.
019     *
020     * @return exception translator used to determine which transactions are retryable
021     */
022    PersistenceExceptionTranslator getPersistenceExceptionTranslator();
023
024    /**
025     * Get the aspect-wide default for {@link RetryTransaction#maxRetries}.
026     *
027     * @return defatul maximum number of transaction retry attempts
028     */
029    int getMaxRetriesDefault();
030
031    /**
032     * Get the aspect-wide default for {@link RetryTransaction#initialDelay}.
033     *
034     * @return default initial delay between retry attempts in milliseconds
035     */
036    long getInitialDelayDefault();
037
038    /**
039     * Get the aspect-wide default for {@link RetryTransaction#maximumDelay}.
040     *
041     * @return default maximum delay between retry attempts in milliseconds
042     */
043    long getMaximumDelayDefault();
044
045    /**
046     * Get the current transaction attempt number.
047     *
048     * @return transaction attempt number, or zero if the aspect is not active in the current thread
049     */
050    int getAttemptNumber();
051}

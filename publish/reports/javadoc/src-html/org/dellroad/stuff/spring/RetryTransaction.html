006package org.dellroad.stuff.spring;
007
008import java.lang.annotation.Documented;
009import java.lang.annotation.ElementType;
010import java.lang.annotation.Inherited;
011import java.lang.annotation.Retention;
012import java.lang.annotation.RetentionPolicy;
013import java.lang.annotation.Target;
014
015/**
016 * An annotation for {@link org.springframework.transaction.annotation.Transactional @Transactional} methods that
017 * want to have transactions automatically retried when they fail due to a transient exception. A transient exception
018 * is one that Spring would translate into a
019 * {@link org.springframework.dao.TransientDataAccessException TransientDataAccessException}.
020 *
021 * <p>
022 * This automatic retry logic is very handy for solving the problem of transient deadlocks that can occur in complex Java/ORM
023 * applications. Due to the ORM layer hiding the details of the underlying data access patterns, it's often difficult
024 * to design Java/ORM applications such that transient deadlocks at the database layer can't occur. Since these
025 * deadlocks can often be dealt with simply by retrying the transaction, having retry logic automatically applied can
026 * eliminate this problem.
027 *
028 * <p>
029 * Note, beans involved in transactions should either be stateless, or be prepared to rollback any state changes on transaction
030 * failure; of course, this is true whether or not transactions are automatically being retried, but adding automatic retry
031 * can magnify pre-existing bugs of that nature.
032 *
033 * <p>
034 * The {@link RetryTransaction @RetryTransaction} annotation is ignored unless all of the following conditions are satisfied:
035 * <ul>
036 *  <li>
037 *      The method (and/or the containing type) must be annotated with both
038 *      {@link org.springframework.transaction.annotation.Transactional @Transactional}
039 *      and {@link RetryTransaction @RetryTransaction}
040 *  </li>
041 *  <li>
042 *      The {@link org.springframework.transaction.annotation.Transactional @Transactional} annotation must have
043 *      {@linkplain org.springframework.transaction.annotation.Transactional#propagation propagation} set to either
044 *      {@link org.springframework.transaction.TransactionDefinition#PROPAGATION_REQUIRED PROPAGATION_REQUIRED} or
045 *      {@link org.springframework.transaction.TransactionDefinition#PROPAGATION_REQUIRES_NEW PROPAGATION_REQUIRES_NEW}
046 *      (other propagation values do not involve creating new transactions).
047 *  </li>
048 *  <li>
049 *      In the case of
050 *      {@link org.springframework.transaction.TransactionDefinition#PROPAGATION_REQUIRED PROPAGATION_REQUIRED} propagation,
051 *      there must not be a transaction already open in the calling thread. In other words, the invoked method must
052 *      be the one responsible for creating the transaction.
053 *  </li>
054 *  <li>
055 *      The method's class must be woven (either at build time or runtime) using the
056 *      <a href="http://www.eclipse.org/aspectj/doc/released/faq.php#compiler">AspectJ compiler</a>
057 *      with the {@code RetryTransactionAspect} aspect (included in the <code>dellroad-stuff</code> JAR file).
058 *  </li>
059 *  <li>
060 *      The {@code RetryTransactionAspect} aspect must be configured with a
061 *      {@link org.springframework.dao.support.PersistenceExceptionTranslator PersistenceExceptionTranslator} appropriate for
062 *      the ORM layer being used. The simplest way to do this is to include the aspect in your Spring application context,
063 *      for example:
064 *      <pre>
065 *
066 *      &lt;bean class="org.dellroad.stuff.spring.RetryTransactionAspect" factory-method="aspectOf"
067 *        p:persistenceExceptionTranslator-ref="myJpaDialect"/&gt;
068 *      </pre>This also gives you the opportunity to change the default values for {@link #maxRetries}, {@link #initialDelay},
069 *      and {@link #maximumDelay}, which are applied when not explicitly overridden in the annotation, for example:
070 *      <pre>
071 *
072 *      &lt;bean class="org.dellroad.stuff.spring.RetryTransactionAspect" factory-method="aspectOf"
073 *        p:persistenceExceptionTranslator-ref="myJpaDialect" p:maxRetriesDefault="2"
074 *        p:initialDelayDefault="25" p:maximumDelayDefault="5000"/&gt;
075 *      </pre>
076 *  </li>
077 * </ul>
078 *
079 * <p>
080 * Logging behavior: Normal activity is logged at trace level, retries are logged at debug level, and errors are logged
081 * at error level.
082 *
083 * <p>
084 * Transactional code can determine the transaction attempt number using the {@link RetryTransactionProvider} interface
085 * implemented by the aspect. {@link RetryTransactionProvider#getAttemptNumber} method returns the current attempt number
086 * (1, 2, 3...), or zero if the current thread is not executing within activated retry logic:
087 *  <blockquote><pre>
088 *      import org.dellroad.stuff.spring.RetryTransactionProvider;
089 *      ...
090 *
091 *      &#64;Autowired
092 *      private RetryTransactionProvider retryTransactionProvider;
093 *      ...
094 *
095 *      &#64;RetryTransaction
096 *      &#64;Transactional
097 *      public void doSomething() {
098 *          ...
099 *          final int attempt = this.retryTransactionProvider.getAttempt();
100 *          ...
101 *      }
102 *   </pre></blockquote>
103 *
104 * @see org.springframework.transaction.annotation.Transactional
105 */
106@Documented
107@Inherited
108@Retention(RetentionPolicy.RUNTIME)
109@Target({ ElementType.TYPE, ElementType.METHOD })
110public @interface RetryTransaction {
111
112    /**
113     * Default {@linkplain #maxRetries maximum number of retry attempts}, used when the {@link #maxRetries maxRetries}
114     * value is not explicitly set in an instance of this annotation.
115     * This default value can be overridden by configuring the {@code maxRetriesDefault} property on the aspect itself.
116     */
117    int DEFAULT_MAX_RETRIES = 4;
118
119    /**
120     * Default {@linkplain #initialDelay initial delay}, in milliseconds, used when the {@link #initialDelay initialDelay}
121     * value is not explicitly set in an instance of this annotation.
122     * This default value can be overridden by configuring the {@code initialDelayDefault} property on the aspect itself.
123     */
124    long DEFAULT_INITIAL_DELAY = 100;
125
126    /**
127     * Default {@linkplain #maximumDelay maximum delay}, in milliseconds, used when the {@link #maximumDelay maximumDelay}
128     * value is not explicitly set in an instance of this annotation.
129     * This default value can be overridden by configuring the {@code maximumDelayDefault} property on the aspect itself.
130     */
131    long DEFAULT_MAXIMUM_DELAY = 30 * 1000;
132
133    /**
134     * The maximum number of transaction retry attempts.
135     *
136     * <p>
137     * If the transaction fails, it will be retried at most this many times.
138     * This limit applies to retries only; it does not apply to the very first attempt, which is always made.
139     * So a value of zero means at most one attempt.
140     *
141     * <p>
142     * If this property is not set explicitly, the default value of {@code -1} indicates that the aspect-wide default value
143     * ({@value #DEFAULT_MAX_RETRIES} by default), should be used.
144     *
145     * @return maximum number of transaction retry attempts
146     */
147    int maxRetries() default -1;
148
149    /**
150     * The initial delay between retry attempts in milliseconds.
151     * After the first transaction failure, we will pause for approximately this many milliseconds.
152     * For additional failures we apply a randomized exponential back-off, up to a maximum of {@link #maximumDelay}.
153     *
154     * <p>
155     * If this property is not set explicitly, the default value of {@code -1} indicates that the aspect-wide default value
156     * ({@value #DEFAULT_INITIAL_DELAY} milliseconds by default), should be used.
157     *
158     * @return initial delay between retry attempts in milliseconds
159     */
160    long initialDelay() default -1;
161
162    /**
163     * The maximum delay between retry attempts in milliseconds.
164     * After the first transaction failure, we will pause for approximately {@link #initialDelay} milliseconds.
165     * For additional failures we apply a randomized exponential back-off, up to a maximum of this value.
166     *
167     * <p>
168     * If this property is not set explicitly, the default value of {@code -1} indicates that the aspect-wide default value
169     * ({@value #DEFAULT_MAXIMUM_DELAY} milliseconds by default), should be used.
170     *
171     * @return maximum delay between retry attempts in milliseconds
172     */
173    long maximumDelay() default -1;
174}

006package org.dellroad.stuff.graph;
007
008import java.util.ArrayList;
009import java.util.Collection;
010import java.util.Collections;
011import java.util.Comparator;
012import java.util.HashMap;
013import java.util.List;
014import java.util.Set;
015
016/**
017 * Topological sorting utility class.
018 */
019public class TopologicalSorter<E> {
020
021    private final Collection<E> nodes;
022    private final EdgeLister<E> edgeLister;
023    private final Comparator<? super E> tieBreaker;
024
025    private HashMap<E, Boolean> visited;
026    private ArrayList<E> ordering;
027
028    /**
029     * Primary constructor.
030     *
031     * @param nodes partially ordered nodes to be sorted
032     * @param edgeLister provides the edges defining the partial order
033     * @param tieBreaker used to sort nodes that are not otherwise ordered,
034     *  or null to tie break based on the original ordering
035     */
036    public TopologicalSorter(Collection<E> nodes, EdgeLister<E> edgeLister, Comparator<? super E> tieBreaker) {
037        this.nodes = nodes;
038        this.edgeLister = edgeLister;
039        if (tieBreaker == null)
040            tieBreaker = getDefaultTieBreaker();
041        this.tieBreaker = tieBreaker;
042    }
043
044    /**
045     * Convenience constructor for when ties should be broken based on the original ordering.
046     *
047     * <p>
048     * Equivalent to:
049     *  <blockquote>
050     *  {@code TopologicalSorter(nodes, edgeLister, null);}
051     *  </blockquote>
052     *
053     * @param nodes set of nodes to sort
054     * @param edgeLister defines the edges between nodes
055     */
056    public TopologicalSorter(Collection<E> nodes, EdgeLister<E> edgeLister) {
057        this(nodes, edgeLister, null);
058    }
059
060    /**
061     * Produce a total ordering of the nodes consistent with the partial ordering
062     * implied by the edge lister and tie breaker provided to the constructor.
063     *
064     * <p>
065     * The returned list will have the property that if there is an edge from X to Y,
066     * then X will appear before Y in the list. If there is no edge (or sequence of edges) from X to Y
067     * in either direction, then X will appear before Y if the tie breaker sorts X before Y.
068     *
069     * @return sorted, mutable list of nodes
070     * @throws IllegalArgumentException if the partial ordering relation contains a cycle
071     */
072    public List<E> sort() {
073
074        // Order nodes according to reverse tie breaker ordering
075        ArrayList<E> startList = Collections.list(Collections.enumeration(this.nodes));
076        Collections.sort(startList, getTieBreaker(true));
077
078        // Perform depth-first search through nodes
079        this.visited = new HashMap<E, Boolean>(startList.size());
080        this.ordering = new ArrayList<E>(startList.size());
081        for (E node : startList)
082            visit(node, true);
083
084        // Reverse list
085        Collections.reverse(this.ordering);
086        return this.ordering;
087    }
088
089    /**
090     * Same as {@link #sort sort()} but treats all edges as reversed.
091     *
092     * <p>
093     * The returned list will have the property that if there is an edge from X to Y,
094     * then Y will appear before X in the list. If there is no edge (or sequence of edges) from X to Y
095     * in either direction, then X will appear before Y if the tie breaker sorts X before Y.
096     *
097     * @return sorted, mutable list of nodes
098     * @throws IllegalArgumentException if the partial ordering relation contains a cycle
099     */
100    public List<E> sortEdgesReversed() {
101
102        // Order nodes according to normal tie breaker ordering
103        ArrayList<E> startList = Collections.list(Collections.enumeration(this.nodes));
104        Collections.sort(startList, getTieBreaker(false));
105
106        // Perform depth-first search through nodes
107        this.visited = new HashMap<E, Boolean>(startList.size());
108        this.ordering = new ArrayList<E>(startList.size());
109        for (E node : startList)
110            visit(node, false);
111
112        // Done
113        return this.ordering;
114    }
115
116    private void visit(E node, boolean reverse) {
117
118        // Have we been here before?
119        Boolean state = this.visited.get(node);
120        if (state != null) {
121            if (!state.booleanValue())
122                throw new IllegalArgumentException("cycle in graph containing " + node);
123            return;
124        }
125        this.visited.put(node, false);
126
127        // Get all destination nodes of all out-edges
128        ArrayList<E> targets = Collections.list(Collections.enumeration(this.edgeLister.getOutEdges(node)));
129
130        // Sort them in reverse desired order and recurse
131        Collections.sort(targets, getTieBreaker(reverse));
132        for (E target : targets)
133            visit(target, reverse);
134
135        // Add this node to list in post-order and mark complete
136        this.ordering.add(node);
137        this.visited.put(node, true);
138    }
139
140    private Comparator<? super E> getDefaultTieBreaker() {
141        final HashMap<E, Integer> orderMap = new HashMap<E, Integer>(this.nodes.size());
142        int posn = 0;
143        for (E node : this.nodes)
144            orderMap.put(node, posn++);
145        return new Comparator<E>() {
146            public int compare(E node1, E node2) {
147                return orderMap.get(node1) - orderMap.get(node2);
148            }
149        };
150    }
151
152    private Comparator<? super E> getTieBreaker(boolean reverse) {
153        if (reverse)
154            return Collections.reverseOrder(this.tieBreaker);
155        return this.tieBreaker;
156    }
157
158    /**
159     * Implemented by classes that can enumerate the outgoing edges from a node in a graph.
160     */
161    public interface EdgeLister<E> {
162
163        /**
164         * Get the set of all nodes X for which there is an edge from {@code node} to X.
165         *
166         * @param node node in question
167         * @return set of outgoing edges from {@code node}
168         */
169        Set<E> getOutEdges(E node);
170    }
171}

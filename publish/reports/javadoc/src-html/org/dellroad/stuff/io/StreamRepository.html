006package org.dellroad.stuff.io;
007
008import java.io.IOException;
009import java.io.InputStream;
010import java.io.OutputStream;
011
012/**
013 * Interface for atomic storage and retrieval of an opaque byte array, with access being done
014 * via {@link InputStream} and {@link OutputStream} interfaces.
015 *
016 * <p>
017 * Conceptually, there is a single underlying byte array. Instances behave as if an atomic readable snapshot
018 * is made when {@link #getInputStream} is invoked, and the results of that snapshot are then available
019 * in the returned {@link InputStream}. Similarly, instances behave as if the underlying byte array is
020 * atomically updated when the output stream returned by {@link #getOutputStream} is successfully
021 * {@linkplain OutputStream#close closed}.
022 *
023 * <p>
024 * Instances are thread safe, and support multiple concurrent open input and output streams. In particular,
025 * two output streams may be {@linkplain OutputStream#close closed} at the same time by two different threads,
026 * and one will always win the race.
027 */
028public interface StreamRepository {
029
030    /**
031     * Get an input stream reading the current value of the underlying store.
032     *
033     * @throws IOException if an error occurs
034     */
035    InputStream getInputStream() throws IOException;
036
037    /**
038     * Get an output stream writing to the underlying store. The underlying store is not affected until the
039     * returned output stream is successfully {@linkplain OutputStream#close closed}, at which time it is
040     * atomically updated with the newly written content.
041     *
042     * <p>
043     * If the returned stream throws an {@link IOException} at any time, including during {@link OutputStream#close close()},
044     * or if {@link OutputStream#close close()} is never invoked, then no update to the underlying storage occurs.
045     * </p>
046     *
047     * @throws IOException if an error occurs
048     */
049    OutputStream getOutputStream() throws IOException;
050}

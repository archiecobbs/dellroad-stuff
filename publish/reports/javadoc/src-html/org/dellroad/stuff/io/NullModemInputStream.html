006package org.dellroad.stuff.io;
007
008import java.io.FilterInputStream;
009import java.io.IOException;
010import java.io.PipedInputStream;
011import java.io.PipedOutputStream;
012
013/**
014 * Presents an {@link java.io.InputStream} interface given a {@link WriteCallback} that can write to an
015 * {@link java.io.OutputStream}. A separate thread is created to perform the actual writing.
016 *
017 * @since 1.0.74
018 */
019public class NullModemInputStream extends FilterInputStream {
020
021    private final PipedOutputStream output;
022
023    /**
024     * Constructor.
025     *
026     * <p>
027     * The {@code writer}'s {@link WriteCallback#writeTo writeTo()} method will be invoked (once)
028     * asynchronously in a dedicated writer thread. The {@link java.io.OutputStream} provided to it will
029     * relay the bytes that are then read from this instance.
030     * </p>
031     *
032     * @param writer    {@link java.io.OutputStream} writer callback
033     * @param name      name for this instance; used to create the name of the background thread
034     */
035    public NullModemInputStream(final WriteCallback writer, String name) {
036        super(new PipedInputStream());
037
038        // Sanity check
039        if (writer == null)
040            throw new IllegalArgumentException("null writer");
041
042        // Create other end of pipe
043        try {
044            this.output = new PipedOutputStream(this.getPipedInputStream());
045        } catch (IOException e) {
046            throw new RuntimeException("unexpected exception", e);
047        }
048
049        // Launch writer thread
050        Thread thread = new WriterThread(writer, this.output, name);
051        thread.setDaemon(true);
052        thread.start();
053    }
054
055    /**
056     * Get the wrapped stream cast as a {@link PipedInputStream}.
057     */
058    protected PipedInputStream getPipedInputStream() {
059        return (PipedInputStream)this.in;
060    }
061
062    /**
063     * Ensure input stream is closed when this instance is no longer referenced.
064     *
065     * <p>
066     * This ensures the writer thread wakes up (and exits, avoiding a memory leak) when an instance of this class
067     * is created but never read from.
068     */
069    @Override
070    protected void finalize() throws Throwable {
071        try {
072            try {
073                this.getPipedInputStream().close();
074            } catch (IOException e) {
075                // ignore
076            }
077        } finally {
078            super.finalize();
079        }
080    }
081
082    /**
083     * Writer thread. This is designed to not hold a reference to the {@link NullModemInputStream}.
084     */
085    private static class WriterThread extends Thread {
086
087        private final WriteCallback writer;
088        private final PipedOutputStream output;
089
090        WriterThread(WriteCallback writer, PipedOutputStream output, String name) {
091            super(name);
092            this.writer = writer;
093            this.output = output;
094        }
095
096        @Override
097        public void run() {
098            try {
099                this.writer.writeTo(this.output);
100            } catch (IOException e) {
101                // ignore - reader will get another IOException because pipe is about to be broken
102            } finally {
103                try {
104                    this.output.close();
105                } catch (IOException e) {
106                    // ignore
107                }
108            }
109        }
110    }
111}

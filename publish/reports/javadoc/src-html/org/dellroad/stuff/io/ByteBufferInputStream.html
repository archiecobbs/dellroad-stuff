006package org.dellroad.stuff.io;
007
008import java.io.IOException;
009import java.io.InputStream;
010import java.nio.ByteBuffer;
011import java.nio.InvalidMarkException;
012
013/**
014 * Provides an {@link InputStream} view of a {@link ByteBuffer}.
015 *
016 * <p>
017 * Instances are thread safe.
018 */
019public class ByteBufferInputStream extends InputStream {
020
021    /**
022     * The underlying {@link ByteBuffer}.
023     */
024    protected final ByteBuffer buf;
025
026    private boolean closed;
027
028    /**
029     * Constructor.
030     *
031     * @param buf buffer
032     * @throws IllegalArgumentException if {@code buf} is null
033     */
034    public ByteBufferInputStream(ByteBuffer buf) {
035        if (buf == null)
036            throw new IllegalArgumentException("null buf");
037        this.buf = buf;
038    }
039
040    @Override
041    public synchronized int read() throws IOException {
042        if (this.closed)
043            throw new IOException("stream is closed");
044        if (!this.buf.hasRemaining())
045            return -1;
046        return this.buf.get() & 0xff;
047    }
048
049    @Override
050    public synchronized int read(byte[] data, int off, int len) throws IOException {
051        if (this.closed)
052            throw new IOException("stream is closed");
053        if (!this.buf.hasRemaining())
054            return -1;
055        len = Math.min(len, this.buf.remaining());
056        this.buf.get(data, off, len);
057        return len;
058    }
059
060    @Override
061    public synchronized long skip(long amount) throws IOException {
062        if (this.closed)
063            throw new IOException("stream is closed");
064        final int skip = (int)Math.min(amount, this.buf.remaining());
065        if (skip <= 0)
066            return 0;
067        this.buf.position(this.buf.position() + skip);
068        return skip;
069    }
070
071    @Override
072    public synchronized int available() throws IOException {
073        if (this.closed)
074            throw new IOException("stream is closed");
075        return this.buf.remaining();
076    }
077
078    @Override
079    public synchronized void mark(int readlimit) {
080        if (this.closed)
081            return;
082        this.buf.mark();
083    }
084
085    @Override
086    public synchronized void reset() throws IOException {
087        if (this.closed)
088            throw new IOException("stream is closed");
089        try {
090            this.buf.reset();
091        } catch (InvalidMarkException e) {
092            throw new IOException("no mark set", e);
093        }
094    }
095
096    @Override
097    public boolean markSupported() {
098        return true;
099    }
100
101    @Override
102    public synchronized void close() {
103        this.closed = true;
104    }
105}

006package org.dellroad.stuff.io;
007
008import java.io.IOException;
009
010/**
011 * Exception thrown by an {@link IdleTimeoutInputStream} when a read operation takes too long.
012 */
013@SuppressWarnings("serial")
014public class IdleTimeoutException extends IOException {
015
016    private final long timeout;
017
018    public IdleTimeoutException(long timeout) {
019        super("no input received after " + timeout + " milliseconds");
020        this.timeout = timeout;
021    }
022
023    /**
024     * Get the idle timeout value that was exceeded to cause this exception.
025     *
026     * @return idle timeout in milliseconds
027     */
028    public long getTimeout() {
029        return this.timeout;
030    }
031}

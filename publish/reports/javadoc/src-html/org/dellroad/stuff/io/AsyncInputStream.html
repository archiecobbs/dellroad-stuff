006package org.dellroad.stuff.io;
007
008import java.io.IOException;
009import java.io.InputStream;
010
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013
014/**
015 * Performs asynchonous reads on an {@link InputStream} and notifies of input events.
016 *
017 * <p>
018 * Reads are performed in a dedicated background thread, from which the configured listener is notified.
019 * The background thread runs until this instance is {@linkplain #close closed}, EOF or an exception is detected
020 * on the input, or a listener callback method throws an exception. A null listener may be supplied; in which case
021 * this class will just sink the {@link InputStream}.
022 * </p>
023 *
024 * <p>
025 * Instances of this class are thread-safe. The {@link #close} method may be safely invoked re-entrantly from the
026 * listener callback methods.
027 * </p>
028 */
029public class AsyncInputStream {
030
031    private static final int BUFFER_SIZE = 4096;
032
033    protected final Logger log = LoggerFactory.getLogger(getClass());
034
035    private final InputStream input;
036    private final String name;
037    private final Listener listener;
038
039    private boolean closed;                     // this instance has been close()'d
040
041    /**
042     * Constructor.
043     *
044     * <p>
045     * If {@code listener} is null, this instance effectively reads and discards all of the input in a background thread.
046     * </p>
047     *
048     * @param input     underlying input stream
049     * @param name      name for this instance; used to create the name of the background thread
050     * @param listener  callback object for input events, or null for none
051     * @throws IllegalArgumentException if any parameter is null
052     */
053    public AsyncInputStream(InputStream input, String name, Listener listener) {
054        if (input == null)
055            throw new IllegalArgumentException("null input");
056        if (name == null)
057            throw new IllegalArgumentException("name input");
058        this.input = input;
059        this.name = name;
060        this.listener = listener;
061        new Thread(this.name) {
062            @Override
063            public void run() {
064                AsyncInputStream.this.threadMain();
065            }
066        }.start();
067    }
068
069    /**
070     * Close this instance.
071     *
072     * <p>
073     * Does nothing if already closed.
074     */
075    public synchronized void close() {
076        if (this.closed)
077            return;
078        try {
079            this.input.close();
080        } catch (IOException e) {
081            // ignore; we assume main thread will awake in any case
082        }
083        this.closed = true;
084    }
085
086    /**
087     * Writer thread main entry point.
088     */
089    private void threadMain() {
090        try {
091            this.runLoop();
092        } catch (Throwable t) {
093            synchronized (this) {
094                if (this.closed)
095                    return;
096            }
097            try {
098                if (this.listener != null)
099                    this.listener.handleException(t);
100            } catch (Exception e) {
101                this.log.error(this.name + ": caught unexpected exception", e);
102            }
103        }
104    }
105
106    /**
107     * Async reader thread main loop.
108     */
109    private void runLoop() throws IOException {
110        byte[] buf = new byte[BUFFER_SIZE];
111        while (true) {
112            int r = this.input.read(buf);
113            if (r == -1) {
114                if (this.listener != null)
115                    this.listener.handleEOF();
116                break;
117            }
118            if (this.listener != null)
119                this.listener.handleInput(buf, 0, r);
120        }
121    }
122
123    /**
124     * Callback interface required by {@link AsyncInputStream}.
125     */
126    public interface Listener {
127
128        /**
129         * Handle new data read from the underlying input.
130         * This method must not write to buffer bytes outside of the defined region.
131         *
132         * @param buf data buffer
133         * @param off starting offset of data in buffer
134         * @param len number of bytes of data
135         */
136        void handleInput(byte[] buf, int off, int len);
137
138        /**
139         * Handle an exception detected on the underlying input.
140         * No further events will be delivered.
141         *
142         * <p>
143         * Typically the assocaited {@link AsyncInputStream} will be closed in this callback.
144         *
145         * @param e the exception received (usually {@link IOException} but could also be any other {@link RuntimeException})
146         */
147        void handleException(Throwable e);
148
149        /**
150         * Handle end-of-file detected on the underlying input.
151         * No further events will be delivered.
152         *
153         * <p>
154         * Typically the assocaited {@link AsyncInputStream} will be closed in this callback.
155         */
156        void handleEOF();
157    }
158}

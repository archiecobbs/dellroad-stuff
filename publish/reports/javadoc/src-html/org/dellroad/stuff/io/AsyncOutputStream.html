006package org.dellroad.stuff.io;

008import java.io.FilterOutputStream;
009import java.io.IOException;
010import java.io.OutputStream;
011import java.util.concurrent.atomic.AtomicInteger;
012
013import org.dellroad.stuff.java.CheckedExceptionWrapper;
014import org.dellroad.stuff.java.Predicate;
015import org.dellroad.stuff.java.TimedWait;
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;

019/**
020 * An {@link OutputStream} that performs writes using a background thread, so that
021 * write, flush, and close operations never block.
022 * <p/>
023 *
024 * <p>
025 * If the underlying output stream throws an {@link IOException} during any operation,
026 * this instance will re-throw the exception for all subsequent operations.
027 * </p>
028 *
029 * <p>
030 * Instances use an internal buffer whose size is configured at construction time;
031 * if the buffer overflows, a {@link BufferOverflowException} is thrown. Alternately,
032 * if a buffer size of zero is configured, the internal buffer will expand automatically as needed
033 * (up to 2<super>31</super> bytes).
034 * However, this creates a memory leak if the underlying {@link OutputStream} blocks indefinitely.
035 * </p>
036 *
037 * <p>
038 * Instances of this class are thread safe, and moreover writes are atomic: if multiple threads are writing
039 * at the same time the bytes written in any single method invocation are written contiguously to the
040 * underlying output.
041 * </p>
042 */
043public class AsyncOutputStream extends FilterOutputStream {
044
045    private static final AtomicInteger COUNTER = new AtomicInteger();
046    private static final int MIN_BUFFER_SIZE = 20;
047    private static final int MAX_BUFFER_SIZE = Integer.MAX_VALUE - 32;
048
049    protected final Logger log = LoggerFactory.getLogger(getClass());
050
051    private byte[] buf;                 // output buffer
052    private int count;                  // number of bytes in output buffer ready to be written
053    private int flushMark = -1;         // buffer byte at which a flush is requested, or -1 if none
054    private Thread thread;              // async writer thread
055    private IOException exception;      // exception caught by async thread
056    private boolean closed;             // this instance has been close()'d
057    private boolean expand;             // whether to auto-expand buffer as needed
058
059    /**
060     * Convenience constructor for when an auto-expanding buffer is desired and a default thread name is to be used.
061     *
062     * @param out     underlying output stream
063     * @throws IllegalArgumentException if {@code out} is null
064     */
065    public AsyncOutputStream(OutputStream out) {
066        this(out, 0, AsyncOutputStream.class.getSimpleName() + "-" + AsyncOutputStream.COUNTER.incrementAndGet());
067    }
068
069    /**
070     * Convenience constructor for when an auto-expanding buffer is desired.
071     *
072     * @param out     underlying output stream
073     * @param name    name for this instance; used to create the name of the background thread
074     * @throws IllegalArgumentException if {@code out} or {@code name} is null
075     */
076    public AsyncOutputStream(OutputStream out, String name) {
077        this(out, 0, name);
078    }
079
080    /**
081     * Constructor.
082     *
083     * @param out     underlying output stream
084     * @param bufsize maximum number of bytes we can buffer, or zero for an auto-expanding buffer that has no fixed limit
085     * @param name    name for this instance; used to create the name of the background thread
086     * @throws IllegalArgumentException if {@code out} or {@code name} is null
087     * @throws IllegalArgumentException if {@code bufsize} is negative
088     */
089    public AsyncOutputStream(OutputStream out, int bufsize, String name) {
090        super(out);
091        if (out == null)
092            throw new IllegalArgumentException("null output");
093        if (name == null)
094            throw new IllegalArgumentException("null name");
095        if (bufsize < 0)
096            throw new IllegalArgumentException("invalid bufsize " + bufsize);
097        this.expand = bufsize == 0;
098        this.buf = new byte[Math.min(Math.max(bufsize, MIN_BUFFER_SIZE), MAX_BUFFER_SIZE)];
099
100        // Start worker thread
101        this.thread = new Thread(name) {
102            @Override
103            public void run() {
104                AsyncOutputStream.this.threadMain();
105            }
106        };
107        this.thread.setDaemon(true);
108        this.thread.start();
109        synchronized (this) { }
110    }
111
112    /**
113     * Write data.
114     *
115     * <p>
116     * This method will never block. To effect a normal blocking write, use {@link #waitForSpace} first.
117     * </p>
118     *
119     * @param b byte to write (lower 8 bits)
120     * @throws IOException             if an exception has been thrown by the underlying stream
121     * @throws IOException             if this instance has been closed
122     * @throws BufferOverflowException if the buffer does not have room for the new byte
123     */
124    @Override
125    public void write(int b) throws IOException {
126        this.write(new byte[] { (byte)b }, 0, 1);
127    }
128
129    /**
130     * Write data.
131     *
132     * <p>
133     * This method will never block. To effect a normal blocking write, invoke {@link #waitForSpace} first.
134     * </p>
135     *
136     * @param data bytes to write
137     * @param off  starting offset in buffer
138     * @param len  number of bytes to write
139     * @throws IOException              if an exception has been thrown by the underlying stream
140     * @throws IOException              if this instance has been closed
141     * @throws BufferOverflowException  if the buffer does not have room for the new data
142     * @throws IllegalArgumentException if {@code len} is negative
143     */
144    @Override
145    public synchronized void write(byte[] data, int off, int len) throws IOException {
146
147        // Check exception conditions
148        this.checkExceptions();
149        if (len < 0)
150            throw new IllegalArgumentException("len = " + len);
151        if (len == 0)
152            return;
153        if (this.count + len > this.buf.length) {
154            if (!this.expand) {
155                throw new BufferOverflowException(this.count + " + " + len + " = " + (this.count + len)
156                  + " byte(s) would exceed the " + this.buf.length + " byte buffer");
157            }
158            this.resizeBuffer(Math.max(this.count + len, this.buf.length * 2));
159
160            final byte[] newBuf = new byte[Math.max(this.count + len, this.buf.length * 2)];
161            System.arraycopy(this.buf, 0, newBuf, 0, this.count);
162            this.buf = newBuf;
163        }
164
165        // Add data to buffer
166        System.arraycopy(data, off, this.buf, this.count, len);
167        this.count += len;
168
169        // Wakeup writer thread
170        this.notifyAll();
171    }
172
173    /**
174     * Flush output. This method will cause the underlying stream to be flushed once all of the data written to this
175     * instance at the time this method is invoked has been written to it.
176     *
177     * <p>
178     * If additional data is written and then a second flush is requested before the first flush has actually occurred,
179     * the first flush will be canceled and only the second flush will be applied. Normally this is not a problem because
180     * the act of writing more data and then flushing forces earlier data to be flushed as well.
181     * </p>
182     *
183     * <p>
184     * This method will never block. To block until the underlying flush operation completes, invoke {@link #waitForIdle}.
185     * </p>
186     *
187     * @throws IOException if this instance has been closed
188     * @throws IOException if an exception has been detected on the underlying stream
189     * @throws IOException if the current thread is interrupted; the nested exception will an {@link InterruptedException}
190     */
191    @Override
192    public synchronized void flush() throws IOException {
193        this.checkExceptions();
194        this.flushMark = this.count;
195        this.notifyAll();                               // wake up writer thread
196    }
197
198    /**
199     * Close this instance. This will (eventually) close the underlying output stream.
200     *
201     * <p>
202     * If this instance has already been closed, nothing happens.
203     * </p>
204     *
205     * <p>
206     * This method will never block. To block until the underlying close operation completes, invoke {@link #waitForIdle}.
207     * </p>
208     *
209     * @throws IOException if an exception has been detected on the underlying stream
210     */
211    @Override
212    public synchronized void close() throws IOException {
213        if (this.closed)
214            return;
215        this.closed = true;
216        this.notifyAll();                               // wake up writer thread
217    }
218
219    /**
220     * Get the exception thrown by the underlying output stream, if any.
221     *
222     * @return thrown exception, or {@code null} if none has been thrown by the underlying stream
223     */
224    public synchronized IOException getException() {
225        return this.exception;
226    }
227
228    /**
229     * Get the capacity of this instance's output buffer.
230     *
231     * <p>
232     * If a (fixed) non-zero value was given at construction time, this will return that value.
233     * </p>
234     *
235     * @return current output buffer capacity
236     */
237    public synchronized int getBufferSize() {
238        return this.buf.length;
239    }
240
241    /**
242     * Get the number of free bytes remaining in the output buffer.
243     *
244     * @return current number of available bytes in the output buffer
245     * @throws IOException              if this instance is or has been closed
246     * @throws IOException              if an exception has been detected on the underlying stream
247     * @see #waitForSpace
248     */
249    public synchronized int availableBufferSpace() throws IOException {
250        this.checkExceptions();
251        return this.buf.length - this.count;
252    }
253
254    /**
255     * Determine if there is outstanding work still to be performed (writes, flushes, and/or close operations)
256     * by the background thread.
257     *
258     * @throws IOException              if this instance is or has been closed
259     * @throws IOException              if an exception has been detected on the underlying stream
260     * @see #waitForIdle
261     */
262    public synchronized boolean isWorkOutstanding() throws IOException {
263        this.checkExceptions();
264        return this.threadHasWork();
265    }
266
267    /**
268     * Wait for buffer space availability.
269     *
270     * <p>
271     * If a zero buffer size was configured at construction time, indicating an auto-expanding buffer,
272     * this will return immediately.
273     * </p>
274     *
275     * @param numBytes amount of buffer space required
276     * @param timeout  maximum time to wait in milliseconds, or zero for infinite
277     * @return true if space was found, false if time expired
278     * @throws IOException              if this instance is or has been closed
279     * @throws IOException              if an exception has been detected on the underlying stream
280     * @throws IllegalArgumentException if {@code numBytes} is greater than the configured buffer size
281     * @throws IllegalArgumentException if {@code timeout} is negative
282     * @throws InterruptedException     if the current thread is interrupted
283     * @see #availableBufferSpace
284     */
285    public boolean waitForSpace(final int numBytes, long timeout) throws IOException, InterruptedException {
286        if (this.expand)
287            return true;
288        if (numBytes > this.buf.length)
289            throw new IllegalArgumentException("numBytes (" + numBytes + ") > buffer size (" + this.buf.length + ")");
290        return this.waitForPredicate(timeout, new Predicate() {
291            @Override
292            public boolean test() {
293                return AsyncOutputStream.this.buf.length - AsyncOutputStream.this.count >= numBytes;
294            }
295        });
296    }
297
298    /**
299     * Wait for all outstanding work to complete.
300     *
301     * @param timeout maximum time to wait in milliseconds, or zero for infinite
302     * @return true for success, false if time expired
303     * @throws IOException              if this instance is or has been closed
304     * @throws IOException              if an exception has been detected on the underlying stream
305     * @throws IllegalArgumentException if {@code timeout} is negative
306     * @throws InterruptedException     if the current thread is interrupted
307     * @see #isWorkOutstanding
308     */
309    public synchronized boolean waitForIdle(long timeout) throws IOException, InterruptedException {
310        return this.waitForPredicate(timeout, new Predicate() {
311            @Override
312            public boolean test() {
313                return !AsyncOutputStream.this.threadHasWork();
314            }
315        });
316    }
317
318    /**
319     * Check for exceptions.
320     *
321     * @throws IOException if this instance has been closed
322     * @throws IOException if an exception has been detected on the underlying stream
323     */
324    private void checkExceptions() throws IOException {
325        if (this.closed)
326            throw new IOException("instance has been closed");
327        if (this.exception != null)
328            throw new IOException("exception from underlying output stream", this.exception);
329    }
330
331    /**
332     * Determine if there is outstanding work still to be performed (writes, flushes, and/or close operations)
333     * by the background thread.
334     */
335    private boolean threadHasWork() {
336        return this.count > 0 || this.flushMark != -1 || this.closed;
337    }
338
339    /**
340     * Resize internal buffer.
341     */
342    private /*synchronized*/ void resizeBuffer(int size) {
343        assert Thread.holdsLock(this);
344        assert this.count <= size;
345        size = Math.min(size, MAX_BUFFER_SIZE);
346        size = Math.max(size, MIN_BUFFER_SIZE);
347        final byte[] newBuf = new byte[size];
348        System.arraycopy(this.buf, 0, newBuf, 0, this.count);
349        this.buf = newBuf;
350    }
351
352    /**
353     * Writer thread main entry point.
354     */
355    private void threadMain() {
356        try {
357            this.runLoop();
358        } catch (Throwable t) {
359            synchronized (this) {
360                this.exception = t instanceof IOException ? (IOException)t : new IOException("caught unexpected exception", t);
361                this.notifyAll();                       // wake up sleepers in waitForSpace() and waitForIdle()
362            }
363        } finally {
364            synchronized (this) {
365                this.thread = null;
366                this.notifyAll();                       // wake up sleepers in waitForIdle()
367            }
368        }
369    }
370
371    /**
372     * Async writer thread main loop.
373     */
374    private void runLoop() throws IOException, InterruptedException {
375        while (true) {
376
377            // Wait for something to do
378            synchronized (this) {
379                while (!this.threadHasWork())
380                    this.wait();                        // will be woken up by write(), flush(), or close()
381            }
382
383            // Determine what needs to be done
384            byte[] currbuf;
385            int wlen;
386            boolean flush;
387            boolean close;
388            synchronized (this) {
389                currbuf = this.buf;
390                wlen = this.count;
391                flush = this.flushMark == 0;
392                close = this.closed;
393            }
394
395            // First priority: any data to write?
396            if (wlen > 0) {
397
398                // Write data
399                this.out.write(currbuf, 0, wlen);
400
401                // Shift data in buffer
402                synchronized (this) {
403                    System.arraycopy(this.buf, wlen, this.buf, 0, this.count - wlen);
404                    this.count -= wlen;
405                    if (this.flushMark != -1)
406                        this.flushMark = Math.max(0, this.flushMark - wlen);
407                    if (this.count <= this.buf.length >> 7)
408                        this.resizeBuffer(this.count);
409                    this.notifyAll();                   // wake up sleepers in waitForSpace() and waitForIdle()
410                }
411                continue;
412            }
413
414            // Second priority: is a flush required?
415            if (flush) {
416
417                // Flush output
418                this.out.flush();
419
420                // Update flush mark
421                synchronized (this) {
422                    if (this.flushMark == 0) {
423                        this.flushMark = -1;
424                        this.notifyAll();               // wake up sleepers in waitForIdle()
425                    }
426                }
427                continue;
428            }
429
430            // Third priority:  is a close required?
431            if (close) {
432                this.out.close();
433                break;
434            }
435        }
436    }
437
438    /**
439     * Wait for some condition to become true. Of course somebody has to wake us up when it becomes true.
440     */
441    private synchronized boolean waitForPredicate(long timeout, final Predicate predicate)
442      throws IOException, InterruptedException {
443        try {
444            return TimedWait.wait(this, timeout, new Predicate() {
445                @Override
446                public boolean test() {
447                    try {
448                        checkExceptions();
449                    } catch (IOException e) {
450                        throw new CheckedExceptionWrapper(e);
451                    }
452                    return predicate.test();
453                }
454            });
455        } catch (CheckedExceptionWrapper e) {
456            throw (IOException)e.getException();
457        }
458    }

006package org.dellroad.stuff.io;
007
008import java.io.FilterOutputStream;
009import java.io.IOException;
010import java.io.PipedInputStream;
011import java.io.PipedOutputStream;
012
013/**
014 * Presents an {@link java.io.OutputStream} interface given a {@link ReadCallback} that can read from an
015 * {@link java.io.InputStream}. A separate thread is created to perform the actual reading.
016 *
017 * <p>
018 * If data is written beyond what the reader is willing to consume, an {@link IOException} is thrown.
019 * </p>
020 *
021 * @since 1.0.82
022 */
023public class NullModemOutputStream extends FilterOutputStream {
024
025    /**
026     * Constructor.
027     *
028     * <p>
029     * The {@code reader}'s {@link ReadCallback#readFrom readFrom()} method will be invoked (once)
030     * asynchronously in a dedicated reader thread. The {@link java.io.InputStream} provided to it will
031     * relay the bytes that are written to this instance.
032     * </p>
033     *
034     * @param reader    {@link java.io.InputStream} reader callback
035     * @param name      name for this instance; used to create the name of the background thread
036     */
037    public NullModemOutputStream(final ReadCallback reader, String name) {
038        super(new PipedOutputStream());
039
040        // Sanity check
041        if (reader == null)
042            throw new IllegalArgumentException("null reader");
043
044        // Create other end of pipe
045        PipedInputStream input;
046        try {
047            input = new PipedInputStream(this.getPipedOutputStream());
048        } catch (IOException e) {
049            throw new RuntimeException("unexpected exception", e);
050        }
051
052        // Launch reader thread
053        Thread thread = new ReaderThread(reader, input, name);
054        thread.setDaemon(true);
055        thread.start();
056    }
057
058    /**
059     * Get the wrapped stream cast as a {@link PipedOutputStream}.
060     */
061    protected PipedOutputStream getPipedOutputStream() {
062        return (PipedOutputStream)this.out;
063    }
064
065    /**
066     * Ensure output stream is closed when this instance is no longer referenced.
067     *
068     * <p>
069     * This ensures the reader thread wakes up (and exits, avoiding a memory leak) when an instance of this class
070     * is created but never read from.
071     */
072    @Override
073    protected void finalize() throws Throwable {
074        try {
075            try {
076                this.getPipedOutputStream().close();
077            } catch (IOException e) {
078                // ignore
079            }
080        } finally {
081            super.finalize();
082        }
083    }
084
085    /**
086     * Reader thread. This is designed to not hold a reference to the {@link NullModemOutputStream}.
087     */
088    private static class ReaderThread extends Thread {
089
090        private final ReadCallback reader;
091        private final PipedInputStream input;
092
093        ReaderThread(ReadCallback reader, PipedInputStream input, String name) {
094            super(name);
095            this.reader = reader;
096            this.input = input;
097        }
098
099        @Override
100        public void run() {
101            try {
102                this.reader.readFrom(this.input);
103            } catch (IOException e) {
104                // ignore - writer will get another IOException because pipe is about to be broken
105            } finally {
106                try {
107                    this.input.close();
108                } catch (IOException e) {
109                    // ignore
110                }
111            }
112        }
113    }
114}

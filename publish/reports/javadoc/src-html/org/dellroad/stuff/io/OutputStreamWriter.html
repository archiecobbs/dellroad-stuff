006package org.dellroad.stuff.io;
007
008import java.io.FilterOutputStream;
009import java.io.IOException;
010import java.io.OutputStream;
011
012/**
013 * Serializes zero or more {@link OutputStream}s inside a single underlying {@link OutputStream}. The results can be
014 * deserialized as equally many distinct {@link java.io.InputStream}s on the other end using an {@link InputStreamReader}.
015 * Each {@link OutputStream} may contain an arbitrary amount of data.
016 *
017 * <p>
018 * To use this class, invoke {@link #start} to start a new {@link OutputStream}, write to it by writing to this
019 * class normally, and then use {@link #stop} to close the current {@link OutputStream}. A new, subsequent {@link OutputStream}
020 * is created by invoking {@link #start} again.
021 * </p>
022 *
023 * <p>
024 * Each {@link OutputStream} written in this way will be read as distinct {@link java.io.InputStream} by the
025 * {@link InputStreamReader} at the other end.
026 * </p>
027 *
028 * <p>
029 * Instances of this class are thread safe.
030 * </p>
031 *
032 * @see InputStreamReader
033 */
034public class OutputStreamWriter extends FilterOutputStream {
035
036    private final RandomEscape randomEscape = new RandomEscape();
037
038    private int escape;
039    private boolean started;
040    private boolean closed;
041
042    /**
043     * Constructor.
044     *
045     * @param output the underlying {@link OutputStream} that will carry nested {@link OutputStream}s within it
046     */
047    public OutputStreamWriter(OutputStream output) {
048        super(output);
049        this.escape = this.randomEscape.next();
050    }
051
052    /**
053     * Start a new {@link OutputStream}.
054     *
055     * @throws IOException if this instance is closed
056     * @throws IOException if an {@link OutputStream} is already started
057     */
058    public synchronized void start() throws IOException {
059        if (this.closed)
060            throw new IOException("this instance is closed");
061        if (this.started)
062            throw new IOException("already started");
063        this.started = true;
064    }
065
066    /**
067     * End the current {@link OutputStream}. This flushes the underlying output.
068     * A new {@link OutputStream} will be created upon the next invocation of {@link #start}.
069     *
070     * @throws IOException if this instance is closed
071     * @throws IOException if no {@link OutputStream} is currently started
072     * @throws IOException if the underlying {@link OutputStream} throws an exception
073     */
074    public synchronized void stop() throws IOException {
075        if (this.closed)
076            throw new IOException("this instance is closed");
077        if (!this.started)
078            throw new IOException("not started");
079        this.started = false;
080        this.writeControl(InputStreamReader.CONTROL_SEPARATOR);
081        this.flush();
082    }
083
084    /**
085     * Close this instance. Does nothing if already closed.
086     * If there an {@link OutputStream} is already started when this method is invoked, it will be implicitly
087     * {@linkplain #stop stopped}.
088     *
089     * <p>
090     * This ends the current {@link OutputStream} and closes the underlying output.
091     * </p>
092     *
093     * @throws IOException if an there is an error closing the underlying {@link OutputStream}
094     */
095    @Override
096    public synchronized void close() throws IOException {
097        if (this.closed)
098            return;
099        if (this.started) {
100            this.started = false;
101            this.writeControl(InputStreamReader.CONTROL_SEPARATOR);
102        }
103        this.closed = true;
104        this.out.close();
105    }
106
107    @Override
108    public synchronized void write(int ch) throws IOException {
109        if (this.closed)
110            throw new IOException("this instance is closed");
111        if (!this.started)
112            this.started = true;
113        if ((ch & 0xff) == this.escape) {
114            this.writeControl(InputStreamReader.CONTROL_ESCAPE);
115            return;
116        }
117        this.out.write(ch);
118    }
119
120    private synchronized void writeControl(int control) throws IOException {
121        this.out.write(this.escape);
122        this.out.write(this.escape ^ control);
123        this.escape = this.randomEscape.next();
124    }
125}

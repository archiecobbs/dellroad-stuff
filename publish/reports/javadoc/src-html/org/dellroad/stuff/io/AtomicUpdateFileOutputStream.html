006package org.dellroad.stuff.io;
007
008import java.io.File;
009import java.io.FileNotFoundException;
010import java.io.FileOutputStream;
011import java.io.IOException;
012
013/**
014 * A {@link FileOutputStream} that atomically updates the target file.
015 *
016 * <p>
017 * Instances write to a temporary file until {@link #close} is invoked, at which time the temporary file
018 * gets {@linkplain File#renameTo renamed} to the target file. This rename operation is atomic on most systems
019 * (e.g., all UNIX variants). The result is that the target file always exists, and if opened at any time,
020 * will contain either the previous content or the new content, but never a mix of the two.
021 *
022 * <p>
023 * An open instance can be thought of as representing an open transaction to rewrite the file.
024 * The "transaction" is committed via {@link #close}, or may be rolled back via {@link #cancel}
025 * (this also deletes the temporary file).
026 *
027 * <p>
028 * Note: to guarantee that the new content will always be found in the future, even if there is a sudden system crash,
029 * the caller must also {@link java.nio.channels.FileChannel#force} this instance (before closing) and the containing
030 * directory (after closing).
031 */
032public class AtomicUpdateFileOutputStream extends FileOutputStream {
033
034    private final File targetFile;
035
036    private File tempFile;
037    private long timestamp;
038
039    /**
040     * Constructor.
041     *
042     * @param targetFile the ultimate destination for the output when {@linkplain #close closed}.
043     * @param tempFile temporary file that accumulates output until {@linkplain #close close}.
044     * @throws FileNotFoundException if {@code tempFile} cannot be opened for any reason
045     * @throws SecurityException if a security manager prevents writing to {@code tempFile}
046     * @throws NullPointerException if either parameter is null
047     */
048    public AtomicUpdateFileOutputStream(File targetFile, File tempFile) throws FileNotFoundException {
049        super(tempFile);
050        this.tempFile = tempFile;
051        if (targetFile == null)
052            throw new NullPointerException("null targetFile");
053        this.targetFile = targetFile;
054    }
055
056    /**
057     * Convenience constructor.
058     *
059     * <p>
060     * This constructor uses a temporary file within the same directory as {@code targetFile}.
061     *
062     * @param targetFile the ultimate destination for the output when {@linkplain #close closed}.
063     * @throws FileNotFoundException if {@code tempFile} cannot be opened for any reason
064     * @throws IOException if a temporary file could not be created
065     * @throws SecurityException if a security manager prevents writing to {@code tempFile}
066     * @throws NullPointerException if either parameter is null
067     */
068    public AtomicUpdateFileOutputStream(File targetFile) throws IOException {
069        this(targetFile, File.createTempFile("atomicupdate", null, targetFile.getParentFile()));
070    }
071
072    /**
073     * Get the target file.
074     *
075     * @return target file, never null
076     */
077    public synchronized File getTargetFile() {
078        return this.targetFile;
079    }
080
081    /**
082     * Get the temporary file.
083     *
084     * <p>
085     * If this instance has already been {@linkplain #close closed} (either successfully or not)
086     * or {@linkplain #cancel canceled}, this will return null.
087     *
088     * @return temporary file, or null if {@link #close} or {@link #cancel} has already been invoked
089     */
090    public synchronized File getTempFile() {
091        return this.tempFile;
092    }
093
094    /**
095     * Cancel this instance. This "aborts" the open "transaction", and deletes the temporary file.
096     *
097     * <p>
098     * Does nothing if {@link #close} or {@link #cancel} has already been invoked.
099     */
100    public synchronized void cancel() {
101
102        // Close output stream to release file descriptor
103        try {
104            super.close();
105        } catch (IOException e) {
106            // ignore
107        }
108
109        // Delete temporary file
110        if (this.tempFile != null) {
111            this.tempFile.delete();
112            this.tempFile = null;
113        }
114    }
115
116    /**
117     * Close this instance. This "commits" the open "transaction".
118     *
119     * <p>
120     * If successful, the configured {@code tempFile} will be {@linkplain File#renameTo renamed}
121     * to the configured destination file {@code targetFile}. In any case, after this method returns
122     * (either normally or abnormally), the temporary file will be deleted.
123     *
124     * @throws IOException if {@link #close} or {@link #cancel} has already been invoked
125     */
126    @Override
127    public synchronized void close() throws IOException {
128
129        // Close temporary file
130        super.close();
131
132        // Be idempotent
133        if (this.tempFile == null)
134            return;
135
136        // Read updated modification time
137        final long newTimestamp = this.tempFile.lastModified();
138
139        // Rename file, or delete it if that fails
140        try {
141            if (!this.tempFile.renameTo(this.targetFile)) {
142                throw new IOException("error renaming temporary file `" + this.tempFile.getName()
143                  + "' to `" + this.targetFile.getName() + "'");
144            }
145            this.tempFile = null;
146        } finally {
147            if (this.tempFile != null)          // exception thrown, cancel transaction
148                this.cancel();
149        }
150
151        // Update target file timestamp
152        this.timestamp = newTimestamp;
153    }
154
155    /**
156     * Get the last modification timestamp of the target file as it was at the time it was updated by this instance.
157     *
158     * <p>
159     * This method only works after {@link #close} has been successfully invoked, otherwise it returns zero.
160     *
161     * @return target file modification time, or zero if {@link #close} has not been successfully invoked
162     */
163    public synchronized long getTimestamp() {
164        return this.timestamp;
165    }
166
167    /**
168     * Ensure the temporary file is deleted in cases where this instance never got successfully closed.
169     */
170    @Override
171    protected void finalize() throws IOException {
172        try {
173            if (this.tempFile != null)
174                this.cancel();
175        } finally {
176            super.finalize();
177        }
178    }
179}

006package org.dellroad.stuff.io;
007
008import au.com.bytecode.opencsv.CSVWriter;
009
010import java.io.BufferedWriter;
011import java.io.IOException;
012import java.io.Writer;
013import java.util.Arrays;
014import java.util.Date;
015import java.util.HashSet;
016import java.util.LinkedHashSet;
017import java.util.Map;
018
019import org.dellroad.stuff.string.DateEncoder;
020
021/**
022 * CSV file output stream that ensures values are matched to the correct columns.
023 * This class requires the <a href="http://opencsv.sourceforge.net/">OpenCSV</a> library.
024 *
025 * @see <a href="http://opencsv.sourceforge.net/">OpenCSV</a>
026 */
027public class CSVOutput {
028
029    private final CSVWriter writer;
030    private final String[] columns;
031
032    /**
033     * Constructor.
034     *
035     * <p>
036     * The column headers will be written out automatically.
037     *
038     * @param writer destination for the CSV output
039     * @param columns CSV columns names in their desired order
040     * @throws IllegalArgumentException if {@code writer} is null
041     * @throws IllegalArgumentException if {@code columns} is null
042     * @throws IllegalArgumentException if {@code columns} contains a duplicate column name
043     */
044    public CSVOutput(Writer writer, String... columns) {
045        this(writer, Arrays.asList(columns));
046    }
047
048    /**
049     * Constructor.
050     *
051     * <p>
052     * The column headers will be written out automatically.
053     *
054     * @param writer destination for the CSV output
055     * @param columns CSV columns names, iterated in their desired order
056     * @throws IllegalArgumentException if {@code writer} is null
057     * @throws IllegalArgumentException if {@code columns} is null
058     * @throws IllegalArgumentException if {@code columns} contains a duplicate column name
059     */
060    public CSVOutput(Writer writer, Iterable<String> columns) {
061        this(new CSVWriter(new BufferedWriter(writer)), columns);
062        if (writer == null)
063            throw new IllegalArgumentException("null writer");
064    }
065
066    /**
067     * Constructor.
068     *
069     * <p>
070     * The column headers will be written out automatically.
071     *
072     * @param writer CSV output object
073     * @param columns CSV columns names, iterated in their desired order
074     * @throws IllegalArgumentException if {@code writer} is null
075     * @throws IllegalArgumentException if {@code columns} is null
076     * @throws IllegalArgumentException if {@code columns} contains a duplicate column name
077     */
078    public CSVOutput(CSVWriter writer, Iterable<String> columns) {
079        if (writer == null)
080            throw new IllegalArgumentException("null writer");
081        if (columns == null)
082            throw new IllegalArgumentException("null columns");
083        this.writer = writer;
084        LinkedHashSet<String> columnSet = new LinkedHashSet<String>();
085        for (String column : columns) {
086            if (!columnSet.add(column))
087                throw new IllegalArgumentException("duplicate column name `" + column + "'");
088        }
089        this.columns = columnSet.toArray(new String[columnSet.size()]);
090
091        // Output header
092        this.writer.writeNext(this.columns);
093    }
094
095    /**
096     * Output a CSV row.
097     *
098     * @param row mapping from column name to value; missing or values are treated as null
099     * @throws IllegalArgumentException if {@code row} contains an unknown column name
100     */
101    public void writeRow(Map<String, ?> row) {
102
103        // Sanity check column names
104        HashSet<String> unknowns = new HashSet<String>(row.keySet());
105        unknowns.removeAll(Arrays.asList(this.columns));
106        if (!unknowns.isEmpty())
107            throw new IllegalArgumentException("row contains unknown column(s): " + unknowns);
108
109        // Format columns
110        String[] values = new String[this.columns.length];
111        for (int i = 0; i < this.columns.length; i++)
112            values[i] = this.formatObject(this.columns[i], row.get(this.columns[i]));
113
114        // Output row
115        this.writer.writeNext(values);
116    }
117
118    /**
119     * Flush output.
120     *
121     * @throws IOException if an I/O error occurs
122     */
123    public void flush() throws IOException {
124        this.writer.flush();
125    }
126
127    /**
128     * Close this instance and the underlying output.
129     *
130     * @throws IOException if an I/O error occurs
131     */
132    public void close() throws IOException {
133        this.writer.close();
134    }
135
136    /**
137     * Format a CSV column value.
138     *
139     * <p>
140     * The implementation in {@link CSVOutput} applies the following logic:
141     * <ul>
142     * <li>{@code null} values are output as the empty string</li>
143     * <li>{@link Boolean} values are output as {@code 1} or {@code 0}</li>
144     * <li>{@link Date} values are output by delegating to {@link #formatDate formatDate()}</li>
145     * <li>All other objects are output using {@link String#valueOf}</li>
146     * </ul>
147     *
148     * <p>
149     * Subclasses should override as needed.
150     *
151     * @param columnName name of the column
152     * @param value column value; will be null if no value was present in the {@link Map} parameter to {@link #writeRow}
153     * @return formatted string
154     */
155    protected String formatObject(String columnName, Object value) {
156        if (value == null)
157            return "";
158        if (value instanceof Boolean)
159            return ((Boolean)value).booleanValue() ? "1" : "0";
160        if (value instanceof Date)
161            return this.formatDate(columnName, (Date)value);
162        return String.valueOf(value);
163    }
164
165    /**
166     * Format a {@link Date} value.
167     *
168     * <p>
169     * The implementation in {@link CSVOutput} delegates to {@link DateEncoder#encode DateEncoder.encode()}.
170     *
171     * <p>
172     * Subclasses should override as needed.
173     *
174     * @param columnName name of the column
175     * @param date column value
176     * @return formatted date string
177     * @throws IllegalArgumentException if {@code date} is null
178     */
179    protected String formatDate(String columnName, Date date) {
180        if (date == null)
181            throw new IllegalArgumentException("null date");
182        return DateEncoder.encode(date);
183    }
184}

006package org.dellroad.stuff.io;
007
008import java.io.File;
009import java.io.FileInputStream;
010import java.io.IOException;
011import java.io.InputStream;
012import java.nio.file.FileAlreadyExistsException;
013import java.nio.file.Files;
014import java.nio.file.Path;
015
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019/**
020 * Provides a file-based implementation of the {@link StreamRepository} interface with
021 * the added feature of automated backups.
022 *
023 * <p>
024 * Atomic updates are implemented using an {@link AtomicUpdateFileOutputStream}.
025 *
026 * <p>
027 * When backups are configured, the base file must be copied, not moved, to the first backup on update to avoid
028 * a small window where the base file doesn't exist. This class uses {@linkplain Files#createLink hard links} to perform
029 * this "copy" efficiently. This behavior can be altered by overriding {@link #copy copy()} on systems not supporting
030 * hard links.
031 */
032public class FileStreamRepository implements StreamRepository {
033
034    private final Logger log = LoggerFactory.getLogger(this.getClass());
035
036    private final File file;
037
038    private int numBackups;
039    private long timestamp;
040
041    /**
042     * Primary constructor.
043     *
044     * @param file the file that will store the stream content
045     * @param numBackups number of backup copies to keep
046     * @throws IllegalArgumentException if {@code file} is null
047     * @throws IllegalArgumentException if {@code numBackups} is negative
048     */
049    public FileStreamRepository(File file, int numBackups) {
050        if (file == null)
051            throw new IllegalArgumentException("null file");
052        if (numBackups < 0)
053            throw new IllegalArgumentException("negative numBackups");
054        this.file = file;
055        this.numBackups = numBackups;
056    }
057
058    /**
059     * Convenience constructor for the case where no backup copies are needed.
060     *
061     * <p>
062     * Equivalent to:
063     * <blockquote><code>
064     *  FileStreamRepository(file, 0);
065     * </code></blockquote>
066     *
067     * @param file the file that will store the stream content
068     */
069    public FileStreamRepository(File file) {
070        this(file, 0);
071    }
072
073    /**
074     * Get the configured {@link File}.
075     *
076     * @return the file that stores the stream content (same as given to constructor)
077     */
078    public final File getFile() {
079        return this.file;
080    }
081
082    /**
083     * Get the configured number of backup files.
084     *
085     * @return the number of backup files to maintain
086     */
087    public final int getNumBackups() {
088        return this.numBackups;
089    }
090
091    /**
092     * Change the number of backup files to maintain.
093     *
094     * <p>
095     * If the number of backups is reduced, the "extra" backup files are not touched.
096     *
097     * @param numBackups number of backup files
098     * @throws IllegalArgumentException if {@code numBackups} is less than zero
099     */
100    public void setNumBackups(int numBackups) {
101        this.numBackups = numBackups;
102    }
103
104    @Override
105    public InputStream getInputStream() throws IOException {
106        return new FileInputStream(this.file);
107    }
108
109    @Override
110    public AtomicUpdateFileOutputStream getOutputStream() throws IOException {
111        String tempName = this.file.getName();
112        while (tempName.length() < 3)
113            tempName += "z";
114        final File tempFile = File.createTempFile(tempName, null, this.file.getParentFile());
115        return new AtomicUpdateFileOutputStream(this.file, tempFile) {
116            @Override
117            public void close() throws IOException {
118                synchronized (FileStreamRepository.this) {
119
120                    // Rotate backups
121                    FileStreamRepository.this.rotateBackups();
122
123                    // Save and update file
124                    super.close();
125
126                    // Update timestamp
127                    FileStreamRepository.this.timestamp = this.getTimestamp();
128                }
129            }
130        };
131    }
132
133    /**
134     * Get the last modification timestamp of the target file as known to this instance.
135     *
136     * <p>
137     * This returns the modification timestamp as known by this instance; it does not ask the filesystem
138     * for the actual last modification timestamp. So if the two are different, then the file has
139     * been updated by some external process. In other words, this returns the modification timestamp
140     * of the underlying file as it was at the time of the most recent update.
141     *
142     * @return file's last modification timestamp as known by this instance,
143     *  or zero if the file has not yet been successfully written to
144     */
145    public synchronized long getFileTimestamp() {
146        return this.timestamp;
147    }
148
149    /**
150     * Generate a backup file name.
151     *
152     * <p>
153     * The implementation in {@link FileStreamRepository} returns a file with the same name as {@code file}
154     * plus a suffix <code>.1</code>,  <code>.2</code>, <code>.3</code>, etc. corresponding to {@code index}.
155     * Subclasses may override as desired.
156     *
157     * @param file the file that stores the current stream content (i.e., from {@link #getFile})
158     * @param index backup index, always greater than or equal to 1
159     * @return the {@code index}'th backup file for {@code file}
160     */
161    protected File getBackupFile(File file, int index) {
162        return new File(file.toString() + "." + index);
163    }
164
165    /**
166     * Copy, via hard link if possible, a file. If the two files are the same, nothing should be done.
167     *
168     * <p>
169     * The implementation in {@link FileStreamRepository} uses {@linkplain Files#createLink hard links}.
170     * Subclasses must override this method if the platform does not support hard links.
171     *
172     * @param src source file
173     * @param dst destination file
174     * @throws IOException if unsuccessful
175     */
176    protected void copy(File src, File dst) throws IOException {
177        final Path srcPath = src.toPath();
178        final Path dstPath = dst.toPath();
179        try {
180            if (dst.exists() && Files.isSameFile(srcPath, dstPath))
181                return;
182        } catch (IOException e) {
183            // ignore
184        }
185        try {
186            Files.createLink(dstPath, srcPath);
187        } catch (FileAlreadyExistsException e) {
188            dst.delete();
189            Files.createLink(dstPath, srcPath);
190        }
191    }
192
193    /**
194     * Rotate backup files.
195     */
196    private synchronized void rotateBackups() throws IOException {
197
198        // Generate file names: 0 -> "file", 1 -> "file.1", 2 -> "file.2", etc.
199        File[] files = new File[this.numBackups + 1];
200        files[0] = this.getFile();
201        for (int i = 0; i < this.numBackups; i++)
202            files[i + 1] = this.getBackupFile(this.getFile(), i + 1);
203
204        // Rotate backups
205        for (int i = this.numBackups - 1; i >= 0; i--) {
206            final File src = files[i];
207            final File dst = files[i + 1];
208
209            // For the first backup, create a copy so there's no window when the target file does not exist
210            if (i == 0) {
211                try {
212                    this.copy(src, dst);
213                } catch (IOException e) {
214                    this.log.warn("failed to copy `" + src + "' to backup `" + dst + "'", e);
215                }
216            } else
217                src.renameTo(dst);      // OK if this fails, that means the backup file doesn't exist yet
218        }
219    }
220}

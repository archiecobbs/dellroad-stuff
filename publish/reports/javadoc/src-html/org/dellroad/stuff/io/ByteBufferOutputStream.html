006package org.dellroad.stuff.io;
007
008import java.io.IOException;
009import java.io.OutputStream;
010import java.nio.BufferOverflowException;
011import java.nio.ByteBuffer;
012import java.nio.ReadOnlyBufferException;
013
014/**
015 * Provides an {@link OutputStream} view of a {@link ByteBuffer}.
016 *
017 * <p>
018 * Instances are thread safe.
019 * </p>
020 */
021public class ByteBufferOutputStream extends OutputStream {
022
023    /**
024     * The underlying {@link ByteBuffer}.
025     */
026    protected final ByteBuffer buf;
027
028    private boolean closed;
029
030    /**
031     * Constructor.
032     *
033     * @param buf buffer
034     * @throws IllegalArgumentException if {@code buf} is null
035     */
036    public ByteBufferOutputStream(ByteBuffer buf) {
037        if (buf == null)
038            throw new IllegalArgumentException("null buf");
039        this.buf = buf;
040    }
041
042    @Override
043    public synchronized void close() {
044        this.closed = true;
045    }
046
047    @Override
048    public synchronized void flush() throws IOException {
049        if (this.closed)
050            throw new IOException("stream is closed");
051    }
052
053    @Override
054    public synchronized void write(int value) throws IOException {
055        if (this.closed)
056            throw new IOException("stream is closed");
057        try {
058            this.buf.put((byte)value);
059        } catch (ReadOnlyBufferException | BufferOverflowException e) {
060            throw new IOException("exception from underlying ByteBuffer", e);
061        }
062    }
063
064    @Override
065    public synchronized void write(byte[] data, int off, int len) throws IOException {
066        if (this.closed)
067            throw new IOException("stream is closed");
068        try {
069            this.buf.put(data, off, len);
070        } catch (ReadOnlyBufferException | BufferOverflowException e) {
071            throw new IOException("exception from underlying ByteBuffer", e);
072        }
073    }
074}
